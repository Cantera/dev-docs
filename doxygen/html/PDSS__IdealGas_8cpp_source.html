79 void PDSS_IdealGas::constructPDSSFile(VPStandardStateTP* tp, size_t spindex,
80  const std::string& inputFile,
81  const std::string& id)
82 {
83  if (inputFile.size() == 0) {
84  throw CanteraError("PDSS_IdealGas::constructPDSSFile",
85  "input file is null");
86  }
87  std::string path = findInputFile(inputFile);
88  ifstream fin(path.c_str());
89  if (!fin) {
90  throw CanteraError("PDSS_IdealGas::constructPDSSFile","could not open "
91  +path+" for reading.");
92  }
93  /*
94  * The phase object automatically constructs an XML object.
95  * Use this object to store information.
96  */
97 
98  XML_Node* fxml = new XML_Node();
99  fxml->build(fin);
100  XML_Node* fxml_phase = findXMLPhase(fxml, id);
101  if (!fxml_phase) {
102  throw CanteraError("PDSS_IdealGas::constructPDSSFile",
103  "ERROR: Can not find phase named " +
104  id + " in file named " + inputFile);
105  }
106  constructPDSSXML(tp, spindex, *fxml_phase, id);
107  delete fxml;
108 }

215 void PDSS_IdealGas::setTemperature(doublereal temp)
216 {
217  m_temp = temp;
218  m_spthermo->update_one(m_spindex, temp,
219  m_cp0_R_ptr, m_h0_RT_ptr, m_s0_R_ptr);

221  m_V0_ptr[m_spindex] = GasConstant * m_temp / m_p0;
222 
223  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex];
224  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex];
225  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex] + log(m_pres/m_p0);
226  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
227  m_Vss_ptr[m_spindex] = GasConstant * m_temp / m_pres;
