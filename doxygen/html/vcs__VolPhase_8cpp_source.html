368 void vcs_VolPhase::setMoleFractionsState(const double totalMoles,
369  const double* const moleFractions,
370  const int vcsStateStatus)
371 {
372  if (totalMoles != 0.0) {
373  // There are other ways to set the mole fractions when VCS_STATECALC
374  // is set to a normal settting.
375  if (vcsStateStatus != VCS_STATECALC_TMP) {
376  throw CanteraError("vcs_VolPhase::setMolesFractionsState",
377  "inappropriate usage");
378  }
379  m_UpToDate = false;
380  m_vcsStateStatus = VCS_STATECALC_TMP;
381  if (m_existence == VCS_PHASE_EXIST_ZEROEDPHASE) {
382  throw CanteraError("vcs_VolPhase::setMolesFractionsState",
383  "inappropriate usage");
384  }
385  m_existence = VCS_PHASE_EXIST_YES;
386  } else {
387  m_UpToDate = true;
388  m_vcsStateStatus = vcsStateStatus;
389  m_existence = std::min(m_existence, VCS_PHASE_EXIST_NO);
390  }
391  double fractotal = 1.0;
392  v_totalMoles = totalMoles;
393  if (m_totalMolesInert > 0.0) {
394  if (m_totalMolesInert > v_totalMoles) {
395  throw CanteraError("vcs_VolPhase::setMolesFractionsState",
396  "inerts greater than total: {} {}",
397  v_totalMoles, m_totalMolesInert);
398  }
399  fractotal = 1.0 - m_totalMolesInert/v_totalMoles;
400  }
401  double sum = 0.0;
402  for (size_t k = 0; k < m_numSpecies; k++) {
403  Xmol_[k] = moleFractions[k];
404  sum += moleFractions[k];
405  }
406  if (sum == 0.0) {
407  throw CanteraError("vcs_VolPhase::setMolesFractionsState",
408  "inappropriate usage");
409  }
410  if (sum != fractotal) {
411  for (size_t k = 0; k < m_numSpecies; k++) {
412  Xmol_[k] *= (fractotal /sum);
413  }
414  }
415  _updateMoleFractionDependencies();
416 }

418 void vcs_VolPhase::setMolesFromVCS(const int stateCalc,
419  const double* molesSpeciesVCS)
420 {
421  v_totalMoles = m_totalMolesInert;
422 
423  if (molesSpeciesVCS == 0) {
424  AssertThrowMsg(m_owningSolverObject, "vcs_VolPhase::setMolesFromVCS",
425  "shouldn't be here");
426  if (stateCalc == VCS_STATECALC_OLD) {
427  molesSpeciesVCS = &m_owningSolverObject->m_molNumSpecies_old[0];
428  } else if (stateCalc == VCS_STATECALC_NEW) {
429  molesSpeciesVCS = &m_owningSolverObject->m_molNumSpecies_new[0];
430  } else {
431  throw CanteraError("vcs_VolPhase::setMolesFromVCS", "shouldn't be here");
432  }
433  } else if (m_owningSolverObject) {
434  if (stateCalc == VCS_STATECALC_OLD) {
435  if (molesSpeciesVCS != &m_owningSolverObject->m_molNumSpecies_old[0]) {
436  throw CanteraError("vcs_VolPhase::setMolesFromVCS", "shouldn't be here");
437  }
438  } else if (stateCalc == VCS_STATECALC_NEW) {
439  if (molesSpeciesVCS != &m_owningSolverObject->m_molNumSpecies_new[0]) {
440  throw CanteraError("vcs_VolPhase::setMolesFromVCS", "shouldn't be here");
441  }
442  }
443  }
444 
445  for (size_t k = 0; k < m_numSpecies; k++) {
446  if (m_speciesUnknownType[k] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
447  size_t kglob = IndSpecies[k];
448  v_totalMoles += std::max(0.0, molesSpeciesVCS[kglob]);
449  }
450  }
451  if (v_totalMoles > 0.0) {
452  for (size_t k = 0; k < m_numSpecies; k++) {
453  if (m_speciesUnknownType[k] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
454  size_t kglob = IndSpecies[k];
455  double tmp = std::max(0.0, molesSpeciesVCS[kglob]);
456  Xmol_[k] = tmp / v_totalMoles;
457  }
458  }
459  m_existence = VCS_PHASE_EXIST_YES;
460  } else {
461  // This is where we will start to store a better approximation
462  // for the mole fractions, when the phase doesn't exist.
463  // This is currently unimplemented.
464  m_existence = VCS_PHASE_EXIST_NO;
465  }
466 
467  // Update the electric potential if it is a solution variable in the
468  // equation system
469  if (m_phiVarIndex != npos) {
470  size_t kglob = IndSpecies[m_phiVarIndex];
471  if (m_numSpecies == 1) {
472  Xmol_[m_phiVarIndex] = 1.0;
473  } else {
474  Xmol_[m_phiVarIndex] = 0.0;
475  }
476  double phi = molesSpeciesVCS[kglob];
477  setElectricPotential(phi);
478  if (m_numSpecies == 1) {
479  m_existence = VCS_PHASE_EXIST_YES;
480  }
481  }
482  _updateMoleFractionDependencies();
483  if (m_totalMolesInert > 0.0) {
484  m_existence = VCS_PHASE_EXIST_ALWAYS;
485  }
486 
487  // If stateCalc is old and the total moles is positive, then we have a valid
488  // state. If the phase went away, it would be a valid starting point for
489  // F_k's. So, save the state.
490  if (stateCalc == VCS_STATECALC_OLD && v_totalMoles > 0.0) {
491  creationMoleNumbers_ = Xmol_;
492  }
493 
494  // Set flags indicating we are up to date with the VCS state vector.
495  m_UpToDate = true;
496  m_vcsStateStatus = stateCalc;
497 }

634 {

637  phaseTotalMoles = 1.0;
638  }
639 
640  // Evaluate the current base activity coefficients if necessary
641  if (!m_UpToDate_AC) {
642  _updateActCoeff();
643  }

649  double moles_j_base = phaseTotalMoles * Xmol_[j];
650  double* const np_lnActCoeffCol = np_dLnActCoeffdMolNumber.ptrColumn(j);
651  if (moles_j_base < 1.0E-200) {
652  moles_j_base = 1.0E-7 * moles_j_base + 1.0E-13 * phaseTotalMoles + 1.0E-150;

655  np_lnActCoeffCol[k] = np_lnActCoeffCol[k] * phaseTotalMoles / moles_j_base;
656  }
657  }
658 
659  double deltaMoles_j = 0.0;
660  // Make copies of ActCoeff and Xmol_ for use in taking differences
661  vector_fp ActCoeff_Base(ActCoeff);
662  vector_fp Xmol_Base(Xmol_);
663  double TMoles_base = phaseTotalMoles;
664 
665  // Loop over the columns species to be deltad
666  for (size_t j = 0; j < m_numSpecies; j++) {
667  // Calculate a value for the delta moles of species j. Note Xmol_[] and
668  // Tmoles are always positive or zero quantities.
669  double moles_j_base = phaseTotalMoles * Xmol_Base[j];
670  deltaMoles_j = 1.0E-7 * moles_j_base + 1.0E-13 * phaseTotalMoles + 1.0E-150;
671

673  // fractions based on this.
674  phaseTotalMoles = TMoles_base + deltaMoles_j;
675  for (size_t k = 0; k < m_numSpecies; k++) {
676  Xmol_[k] = Xmol_Base[k] * TMoles_base / phaseTotalMoles;
677  }
678  Xmol_[j] = (moles_j_base + deltaMoles_j) / phaseTotalMoles;

681  // setState_PX();

686  v_totalMoles = TMoles_base;

688  }
