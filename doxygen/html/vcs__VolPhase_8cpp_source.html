915 void vcs_VolPhase::_updateLnActCoeffJac()
916 {
917  double phaseTotalMoles = v_totalMoles;
918  if (phaseTotalMoles < 1.0E-14) {
919  phaseTotalMoles = 1.0;
920  }
921 
922  /*
923  * Evaluate the current base activity coefficients if necessary
924  */
925  if (!m_UpToDate_AC) {
926  _updateActCoeff();
927  }
928  if (!TP_ptr) {
929  return;
930  }
931  TP_ptr->getdlnActCoeffdlnN(m_numSpecies, &np_dLnActCoeffdMolNumber[0][0]);
932  for (size_t j = 0; j < m_numSpecies; j++) {
933  double moles_j_base = phaseTotalMoles * Xmol_[j];
934  double* const np_lnActCoeffCol = np_dLnActCoeffdMolNumber[j];
935  if (moles_j_base < 1.0E-200) {
936  moles_j_base = 1.0E-7 * moles_j_base + 1.0E-13 * phaseTotalMoles + 1.0E-150;
937  }
938  for (size_t k = 0; k < m_numSpecies; k++) {
939  np_lnActCoeffCol[k] = np_lnActCoeffCol[k] * phaseTotalMoles / moles_j_base;
940  }
941  }
942 
943  double deltaMoles_j = 0.0;
944  // Make copies of ActCoeff and Xmol_ for use in taking differences
945  std::vector<double> ActCoeff_Base(ActCoeff);
946  std::vector<double> Xmol_Base(Xmol_);
947  double TMoles_base = phaseTotalMoles;
948 
949  /*
950  * Loop over the columns species to be deltad
951  */
952  for (size_t j = 0; j < m_numSpecies; j++) {
953  /*
954  * Calculate a value for the delta moles of species j
955  * -> Note Xmol_[] and Tmoles are always positive or zero
956  * quantities.
957  */
958  double moles_j_base = phaseTotalMoles * Xmol_Base[j];
959  deltaMoles_j = 1.0E-7 * moles_j_base + 1.0E-13 * phaseTotalMoles + 1.0E-150;
960  /*
961  * Now, update the total moles in the phase and all of the
962  * mole fractions based on this.
963  */
964  phaseTotalMoles = TMoles_base + deltaMoles_j;
965  for (size_t k = 0; k < m_numSpecies; k++) {
966  Xmol_[k] = Xmol_Base[k] * TMoles_base / phaseTotalMoles;
967  }
968  Xmol_[j] = (moles_j_base + deltaMoles_j) / phaseTotalMoles;
969 
970  /*
971  * Go get new values for the activity coefficients.
972  * -> Note this calls setState_PX();
973  */
974  _updateMoleFractionDependencies();
975  _updateActCoeff();
976  /*
977  * Calculate the column of the matrix
978  */
979  double* const np_lnActCoeffCol = np_dLnActCoeffdMolNumber[j];
980  for (size_t k = 0; k < m_numSpecies; k++) {
981  double tmp;
982  tmp = (ActCoeff[k] - ActCoeff_Base[k]) /
983  ((ActCoeff[k] + ActCoeff_Base[k]) * 0.5 * deltaMoles_j);
984  if (fabs(tmp - np_lnActCoeffCol[k]) > 1.0E-4 * fabs(tmp) + fabs(np_lnActCoeffCol[k])) {
985  // printf(" we have an error\n");
986 
987  }
988  //tmp = lnActCoeffCol[k];
989 
990  }
991  /*
992  * Revert to the base case Xmol_, v_totalMoles
993  */
994  v_totalMoles = TMoles_base;
995  vcs_vdcopy(Xmol_, Xmol_Base, m_numSpecies);
996  }
997  /*
998  * Go get base values for the activity coefficients.
999  * -> Note this calls setState_TPX() again;
1000  * -> Just wanted to make sure that cantera is in sync
1001  * with VolPhase after this call.
1002  */
1003  setMoleFractions(VCS_DATA_PTR(Xmol_Base));
1004  _updateMoleFractionDependencies();
1005  _updateActCoeff();
1006 }

1011 /*
1012  *
1013  * This is essentially a scatter operation.
1014  *
1015  * The Jacobians are actually d( lnActCoeff) / d (MolNumber);
1016  * dLnActCoeffdMolNumber[j][k]
