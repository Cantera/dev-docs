51 void Domain1D::eval(size_t jg, doublereal* xg, doublereal* rg,
52  integer* mask, doublereal rdt)
53 {
54 
55  if (jg != npos && (jg + 1 < firstPoint() || jg > lastPoint() + 1)) {
56  return;
57  }
58 
59  // if evaluating a Jacobian, compute the steady-state residual
60  if (jg != npos) {
61  rdt = 0.0;
62  }
63 
64  // start of local part of global arrays
65  doublereal* x = xg + loc();
66  doublereal* rsd = rg + loc();
67  integer* diag = mask + loc();
68 
69  size_t jmin, jmax, jpt, j, i;
70  jpt = jg - firstPoint();
71 
72  if (jg == npos) { // evaluate all points
73  jmin = 0;
74  jmax = m_points - 1;
75  } else { // evaluate points for Jacobian
76  jmin = std::max<size_t>(jpt, 1) - 1;
77  jmax = std::min(jpt+1,m_points-1);
78  }
79 
80  for (j = jmin; j <= jmax; j++) {
81  if (j == 0 || j == m_points - 1) {
82  for (i = 0; i < m_nv; i++) {
83  rsd[index(i,j)] = residual(x,i,j);
84  diag[index(i,j)] = 0;
85  }
86  } else {
87  for (i = 0; i < m_nv; i++) {
88  rsd[index(i,j)] = residual(x,i,j)
89  - timeDerivativeFlag(i)*rdt*(value(x,i,j) - prevSoln(i,j));
90  diag[index(i,j)] = timeDerivativeFlag(i);
91  }
92  }
93  }

109 void Domain1D::restore(const XML_Node& dom, doublereal* soln, int loglevel)
110 {
111  vector_fp values;
112  vector<XML_Node*> nodes = dom.getChildren("floatArray");
113  for (size_t i = 0; i < nodes.size(); i++) {
114  string title = nodes[i]->attrib("title");
115  getFloatArray(*nodes[i], values, false);
116  if (values.size() != nComponents()) {
117  if (loglevel > 0) {
118  writelog("Warning: Domain1D::restore: Got an array of length " +
119  int2str(values.size()) + " when one of length " +
120  int2str(nComponents()) + " was expected. " +
121  "Tolerances for individual species may not be preserved.\n");
122  }
123  // The number of components will differ when restoring from a
124  // mechanism with a different number of species. Assuming that
125  // tolerances are the same for all species, we can just copy the
126  // tolerance from the last species.
127  if (!values.empty()) {
128  values.resize(nComponents(), values[values.size()-1]);
129  } else {
130  // If the tolerance vector is empty, just leave the defaults
131  // in place.
132  continue;
133  }
134  }
135  if (title == "abstol_transient") {
136  m_atol_ts = values;
137  } else if (title == "reltol_transient") {
138  m_rtol_ts = values;
139  } else if (title == "abstol_steady") {
140  m_atol_ss = values;
141  } else if (title == "reltol_steady") {
142  m_rtol_ss = values;
143  } else {
144  throw CanteraError("Domain1D::restore",
145  "Got an unexpected array, '" + title + "'");
146  }
