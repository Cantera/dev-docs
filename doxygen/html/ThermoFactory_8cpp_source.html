208 //! Gather a vector of pointers to XML_Nodes for a phase
209 /*!
210  * @param spDataNodeList Output vector of pointer to XML_Nodes which contain the species XML_Nodes for the
211  * species in the current phase.
212  * @param spNamesList Output Vector of strings, which contain the names of the species in the phase
213  * @param spRuleList Output Vector of ints, which contain the value of sprule for each species in the phase
214  * @param spArray_names Vector of pointers to the XML_Nodes which contains the names of the
215  * species in the phase
216  * @param spArray_dbases Input vector of pointers to species data bases.
217  * We search each data base for the required species names
218  * @param sprule Input vector of sprule values
219  */
220 static void formSpeciesXMLNodeList(std::vector<XML_Node*> &spDataNodeList,
221  std::vector<std::string> &spNamesList,
222  std::vector<int> &spRuleList,
223  const std::vector<XML_Node*> spArray_names,
224  const std::vector<XML_Node*> spArray_dbases,
225  const vector_int sprule)
226 {
227  // used to check that each species is declared only once
228  std::map<std::string, bool> declared;
229 
230  for (size_t jsp = 0; jsp < spArray_dbases.size(); jsp++) {
231  const XML_Node& speciesArray = *spArray_names[jsp];
232 
233  // Get the top XML for the database
234  const XML_Node* db = spArray_dbases[jsp];
235 
236  // Get the array of species name strings and then count them
237  std::vector<std::string> spnames;
238  getStringArray(speciesArray, spnames);
239  size_t nsp = spnames.size();
240 
241  // if 'all' is specified as the one and only species in the
242  // spArray_names field, then add all species
243  // defined in the corresponding database to the phase
244  if (nsp == 1 && spnames[0] == "all") {
245  std::vector<XML_Node*> allsp;
246  db->getChildren("species", allsp);
247  nsp = allsp.size();
248  spnames.resize(nsp);
249  for (size_t nn = 0; nn < nsp; nn++) {
250  string stemp = (*allsp[nn])["name"];
251  bool skip = false;
252  if (declared[stemp]) {
253  if (sprule[jsp] >= 10) {
254  skip = true;
255  } else {
256  throw CanteraError("ThermoFactory::formSpeciesXMLNodeList()",
257  "duplicate species: \"" + stemp + "\"");
258  }
259  }
260  if (!skip) {
261  declared[stemp] = true;
262  spNamesList.push_back(stemp);
263  spDataNodeList.push_back(allsp[nn]);
264  spRuleList.push_back(sprule[jsp]);
265  }
266  }
267  } else if (nsp == 1 && spnames[0] == "unique") {
268  std::vector<XML_Node*> allsp;
269  db->getChildren("species", allsp);
270  nsp = allsp.size();
271  spnames.resize(nsp);
272  for (size_t nn = 0; nn < nsp; nn++) {
273  string stemp = (*allsp[nn])["name"];
274  bool skip = false;
275  if (declared[stemp]) {
276  skip = true;
277  }
278  if (!skip) {
279  declared[stemp] = true;
280  spNamesList.push_back(stemp);
281  spDataNodeList.push_back(allsp[nn]);
282  spRuleList.push_back(sprule[jsp]);
283  }
284  }
285  } else {
286  std::map<std::string, XML_Node*> speciesNodes;
287  for (size_t k = 0; k < db->nChildren(); k++) {
288  XML_Node& child = db->child(k);
289  speciesNodes[child["name"]] = &child;
290  }
291  for (size_t k = 0; k < nsp; k++) {
292  string stemp = spnames[k];
293  bool skip = false;
294  if (declared[stemp]) {
295  if (sprule[jsp] >= 10) {
296  skip = true;
297  } else {
298  throw CanteraError("ThermoFactory::formSpeciesXMLNodeList()",
299  "duplicate species: \"" + stemp + "\"");
300  }
301  }
302  if (!skip) {
303  declared[stemp] = true;
304  // Find the species in the database by name.
305  std::map<std::string, XML_Node*>::iterator iter = speciesNodes.find(stemp);
306  if (iter == speciesNodes.end()) {
307  throw CanteraError("importPhase","no data for species, \""
308  + stemp + "\"");
309  }
310  spNamesList.push_back(stemp);
311  spDataNodeList.push_back(iter->second);
312  spRuleList.push_back(sprule[jsp]);
313  }
314  }
315  }
316  }

319 bool importPhase(XML_Node& phase, ThermoPhase* th,
320  SpeciesThermoFactory* spfactory)
321 {
322  // Check the the supplied XML node in fact represents a phase.
323  if (phase.name() != "phase") {
324  throw CanteraError("importPhase",
325  "Current const XML_Node named, " + phase.name() +
326  ", is not a phase element.");
327  }
328 
329  /*
330  * In this section of code, we get the reference to the
331  * phase xml tree within the ThermoPhase object. Then,
332  * we clear it and fill it with the current information that
333  * we are about to use to construct the object. We will then
334  * be able to resurrect the information later by calling xml().
335  */
336  XML_Node& phaseNode_XML = th->xml();
337  phaseNode_XML.clear();
338  phase.copy(&phaseNode_XML);
339 
340  // set the id attribute of the phase to the 'id' attribute in the XML tree.
341  th->setID(phase.id());
342  th->setName(phase.id());
343 
344  // Number of spatial dimensions. Defaults to 3 (bulk phase)
345  if (phase.hasAttrib("dim")) {
346  int idim = intValue(phase["dim"]);
347  if (idim < 1 || idim > 3)
348  throw CanteraError("importPhase",
349  "phase, " + th->id() +
350  ", has unphysical number of dimensions: " + phase["dim"]);
351  th->setNDim(idim);
352  } else {
353  th->setNDim(3); // default
354  }
355 
356  // Set equation of state parameters. The parameters are
357  // specific to each subclass of ThermoPhase, so this is done
358  // by method setParametersFromXML in each subclass.
359  const XML_Node& eos = phase.child("thermo");
360  if (phase.hasChild("thermo")) {
361  th->setParametersFromXML(eos);
362  } else {
363  throw CanteraError("importPhase",
364  " phase, " + th->id() +
365  ", XML_Node does not have a \"thermo\" XML_Node");
366  }
367 
368  VPStandardStateTP* vpss_ptr = 0;
369  int ssConvention = th->standardStateConvention();
370  if (ssConvention == cSS_CONVENTION_VPSS) {
371  vpss_ptr = dynamic_cast <VPStandardStateTP*>(th);
372  if (vpss_ptr == 0) {
373  throw CanteraError("importPhase",
374  "phase, " + th->id() + ", was VPSS, but dynamic cast failed");
375  }
376  }
377 
378  // if no species thermo factory was supplied, use the default one.
379  if (!spfactory) {
380  spfactory = SpeciesThermoFactory::factory();
381  }
382 
383  /***************************************************************
384  * Add the elements.
385  ***************************************************************/
386  if (ssConvention != cSS_CONVENTION_SLAVE) {
387  th->addElementsFromXML(phase);
388  }
389 
390  /***************************************************************
391  * Add the species.
392  *
393  * Species definitions may be imported from multiple
394  * sources. For each one, a speciesArray element must be
395  * present.
396  ***************************************************************/
397  vector<XML_Node*> sparrays;
398  phase.getChildren("speciesArray", sparrays);
399  if (ssConvention != cSS_CONVENTION_SLAVE) {
400  if (sparrays.empty()) {
401  throw CanteraError("importPhase",
402  "phase, " + th->id() + ", has zero \"speciesArray\" XML nodes.\n"
403  + " There must be at least one speciesArray nodes "
404  "with one or more species");
405  }
406  }
407  vector<XML_Node*> dbases;
408  vector_int sprule(sparrays.size(),0);
409 
410  // loop over the speciesArray elements
411  for (size_t jsp = 0; jsp < sparrays.size(); jsp++) {
412 
413  const XML_Node& speciesArray = *sparrays[jsp];
414 
415  // If the speciesArray element has a child element
416  //
417  // <skip element="undeclared">
418  //
419  // then set sprule[jsp] to 1, so that any species with an undeclared
420  // element will be quietly skipped when importing species. Additionally,
421  // if the skip node has the following attribute:
422  //
423  // <skip species="duplicate">
424  //
425  // then duplicate species names will not cause Cantera to throw an
426  // exception. Instead, the duplicate entry will be discarded.
427  if (speciesArray.hasChild("skip")) {
428  const XML_Node& sk = speciesArray.child("skip");
429  string eskip = sk["element"];
430  if (eskip == "undeclared") {
431  sprule[jsp] = 1;
432  }
433  string dskip = sk["species"];
434  if (dskip == "duplicate") {
435  sprule[jsp] += 10;
436  }
437  }
438 
439  // Get a pointer to the node containing the species
440  // definitions for the species declared in this
441  // speciesArray element. This may be in the local file
442  // containing the phase element, or may be in another
443  // file.
444  XML_Node* db = get_XML_Node(speciesArray["datasrc"], &phase.root());
445  if (db == 0) {
446  throw CanteraError("importPhase()",
447  " Can not find XML node for species database: "
448  + speciesArray["datasrc"]);
449  }
450 
451  // add this node to the list of species database nodes.
452  dbases.push_back(db);
453  }
454 
455  // Now, collect all the species names and all the XML_Node * pointers
456  // for those species in a single vector. This is where we decide what
457  // species are to be included in the phase.
458  // The logic is complicated enough that we put it in a separate routine.
459  std::vector<XML_Node*> spDataNodeList;
460  std::vector<std::string> spNamesList;
461  std::vector<int> spRuleList;
462  formSpeciesXMLNodeList(spDataNodeList, spNamesList, spRuleList,
463  sparrays, dbases, sprule);
464 
465  // Decide whether the the phase has a variable pressure ss or not
466  SpeciesThermo* spth = 0;
467  VPSSMgr* vp_spth = 0;
468  if (ssConvention == cSS_CONVENTION_TEMPERATURE) {
469  // Create a new species thermo manager. Function
470  // 'newSpeciesThermoMgr' looks at the species in the database
471  // to see what thermodynamic property parameterizations are
472  // used, and selects a class that can handle the
473  // parameterizations found.
474  spth = newSpeciesThermoMgr(spDataNodeList);
475 
476  // install it in the phase object
477  th->setSpeciesThermo(spth);
478  } else if (ssConvention == cSS_CONVENTION_SLAVE) {
479  /*
480  * No species thermo manager for this type
481  */
482  } else if (ssConvention == cSS_CONVENTION_VPSS) {
483  vp_spth = newVPSSMgr(vpss_ptr, &phase, spDataNodeList);
484  vpss_ptr->setVPSSMgr(vp_spth);
485  spth = vp_spth->SpeciesThermoMgr();
486  th->setSpeciesThermo(spth);
487  } else {
488  throw CanteraError("importPhase()", "unknown convention");
489  }
490 
491 
492  size_t k = 0;
493 
494  size_t nsp = spDataNodeList.size();
495  if (ssConvention == cSS_CONVENTION_SLAVE) {
496  if (nsp > 0) {
497  throw CanteraError("importPhase()", "For Slave standard states, number of species must be zero: "
498  + int2str(nsp));
499  }
500  }
501  for (size_t i = 0; i < nsp; i++) {
502  XML_Node* s = spDataNodeList[i];
503  AssertTrace(s != 0);
504  bool ok = installSpecies(k, *s, *th, spth, spRuleList[i],
505  &phase, vp_spth, spfactory);
506  if (ok) {
507  th->saveSpeciesData(k, s);
508  ++k;
509  }
510  }
511 
512  if (ssConvention == cSS_CONVENTION_SLAVE) {
513  th->installSlavePhases(&phase);
514  }
515 
516  // Done adding species. Perform any required subclass-specific
517  // initialization.
518  th->initThermo();
519 
520  // Perform any required subclass-specific initialization
521  // that requires the XML phase object
522  std::string id = "";
523  th->initThermoXML(phase, id);
524 
525  return true;
526 }

528 bool installSpecies(size_t k, const XML_Node& s, thermo_t& th,
529  SpeciesThermo* spthermo_ptr, int rule,
530  XML_Node* phaseNode_ptr,
531  VPSSMgr* vpss_ptr,
532  SpeciesThermoFactory* factory)
533 {
534  std::string xname = s.name();
535  if (xname != "species") {
536  throw CanteraError("installSpecies",
537  "Unexpected XML name of species XML_Node: " + xname);
538  }
539  // get the composition of the species
540  const XML_Node& a = s.child("atomArray");
541  map<string,string> comp;
542  getMap(a, comp);
543 
544  // check that all elements in the species exist in 'p'. If rule != 0,
545  // quietly skip this species if some elements are undeclared; otherwise,
546  // throw an exception
547  map<string,string>::const_iterator _b = comp.begin();
548  for (; _b != comp.end(); ++_b) {
549  if (th.elementIndex(_b->first) == npos) {
550  if (rule == 0) {
551  throw CanteraError("installSpecies",
552  "Species " + s["name"] +
553  " contains undeclared element " + _b->first);
554  } else {
555  return false;
556  }
557  }
558  }
559 
560  // construct a vector of atom numbers for each element in phase th. Elements
561  // not declared in the species (i.e., not in map comp) will have zero
562  // entries in the vector.
563  size_t nel = th.nElements();
564  vector_fp ecomp(nel, 0.0);
565  for (size_t m = 0; m < nel; m++) {
566  std::string& es = comp[th.elementName(m)];
567  if (!es.empty()) {
568  ecomp[m] = fpValueCheck(es);
569  }
570  }
571 
572 
573  // get the species charge, if any. Note that the charge need
574  // not be explicitly specified if special element 'E'
575  // (electron) is one of the elements.
576  doublereal chrg = 0.0;
577  if (s.hasChild("charge")) {
578  chrg = getFloat(s, "charge");
579  }
580 
581  // get the species size, if any. (This is used by surface
582  // phases to represent how many sites a species occupies.)
583  doublereal sz = 1.0;
584  if (s.hasChild("size")) {
585  sz = getFloat(s, "size");
586  }
587 
588  // add the species to phase th
589  th.addUniqueSpecies(s["name"], &ecomp[0], chrg, sz);
590 
591  if (vpss_ptr) {
592  VPStandardStateTP* vp_ptr = dynamic_cast<VPStandardStateTP*>(&th);
593  factory->installVPThermoForSpecies(k, s, vp_ptr, vpss_ptr, spthermo_ptr,
594  phaseNode_ptr);
595  } else {
596  // install the thermo parameterization for this species into
597  // the species thermo manager for phase th
598  factory->installThermoForSpecies(k, s, &th, *spthermo_ptr, phaseNode_ptr);
599  }
600 
601  return true;
602 }
