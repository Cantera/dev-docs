201 /*!
202  * @param spDataNodeList Output vector of pointer to XML_Nodes which contain the species XML_Nodes for the
203  * species in the current phase.
204  * @param spNamesList Output Vector of strings, which contain the names of the species in the phase
205  * @param spRuleList Output Vector of ints, which contain the value of sprule for each species in the phase
206  * @param spArray_names Vector of pointers to the XML_Nodes which contains the names of the
207  * species in the phase
208  * @param spArray_dbases Input vector of pointers to species data bases.
209  * We search each data base for the required species names
210  * @param sprule Input vector of sprule values
211  */
212 static void formSpeciesXMLNodeList(std::vector<XML_Node*> &spDataNodeList,
213  std::vector<std::string> &spNamesList,
214  vector_int &spRuleList,
215  const std::vector<XML_Node*> spArray_names,
216  const std::vector<XML_Node*> spArray_dbases,
217  const vector_int sprule)
218 {
219  // used to check that each species is declared only once
220  std::map<std::string, bool> declared;
221 
222  for (size_t jsp = 0; jsp < spArray_dbases.size(); jsp++) {
223  const XML_Node& speciesArray = *spArray_names[jsp];
224 
225  // Get the top XML for the database
226  const XML_Node* db = spArray_dbases[jsp];
227 
228  // Get the array of species name strings and then count them
229  std::vector<std::string> spnames;
230  getStringArray(speciesArray, spnames);
231  size_t nsp = spnames.size();
232 
233  // if 'all' is specified as the one and only species in the
234  // spArray_names field, then add all species
235  // defined in the corresponding database to the phase
236  if (nsp == 1 && spnames[0] == "all") {
237  std::vector<XML_Node*> allsp = db->getChildren("species");
238  nsp = allsp.size();
239  spnames.resize(nsp);
240  for (size_t nn = 0; nn < nsp; nn++) {
241  string stemp = (*allsp[nn])["name"];
242  bool skip = false;
243  if (declared[stemp]) {
244  if (sprule[jsp] >= 10) {
245  skip = true;
246  } else {
247  throw CanteraError("ThermoFactory::formSpeciesXMLNodeList()",
248  "duplicate species: \"" + stemp + "\"");
249  }
250  }
251  if (!skip) {
252  declared[stemp] = true;
253  spNamesList.push_back(stemp);
254  spDataNodeList.push_back(allsp[nn]);
255  spRuleList.push_back(sprule[jsp]);
256  }
257  }
258  } else if (nsp == 1 && spnames[0] == "unique") {
259  std::vector<XML_Node*> allsp = db->getChildren("species");
260  nsp = allsp.size();
261  spnames.resize(nsp);
262  for (size_t nn = 0; nn < nsp; nn++) {
263  string stemp = (*allsp[nn])["name"];
264  bool skip = false;
265  if (declared[stemp]) {
266  skip = true;
267  }
268  if (!skip) {
269  declared[stemp] = true;
270  spNamesList.push_back(stemp);
271  spDataNodeList.push_back(allsp[nn]);
272  spRuleList.push_back(sprule[jsp]);
273  }
274  }
275  } else {
276  std::map<std::string, XML_Node*> speciesNodes;
277  for (size_t k = 0; k < db->nChildren(); k++) {
278  XML_Node& child = db->child(k);
279  speciesNodes[child["name"]] = &child;
280  }
281  for (size_t k = 0; k < nsp; k++) {
282  string stemp = spnames[k];
283  bool skip = false;
284  if (declared[stemp]) {
285  if (sprule[jsp] >= 10) {
286  skip = true;
287  } else {
288  throw CanteraError("ThermoFactory::formSpeciesXMLNodeList()",
289  "duplicate species: \"" + stemp + "\"");
290  }
291  }
292  if (!skip) {
293  declared[stemp] = true;
294  // Find the species in the database by name.
295  auto iter = speciesNodes.find(stemp);
296  if (iter == speciesNodes.end()) {
297  throw CanteraError("importPhase","no data for species, \""
298  + stemp + "\"");
299  }
300  spNamesList.push_back(stemp);
301  spDataNodeList.push_back(iter->second);
302  spRuleList.push_back(sprule[jsp]);
303  }
304  }
305  }
306  }

309 void importPhase(XML_Node& phase, ThermoPhase* th)
310 {
311  // Check the the supplied XML node in fact represents a phase.
312  if (phase.name() != "phase") {
313  throw CanteraError("importPhase",
314  "Current const XML_Node named, " + phase.name() +
315  ", is not a phase element.");
316  }
317 
318  /*
319  * In this section of code, we get the reference to the
320  * phase XML tree within the ThermoPhase object. Then,
321  * we clear it and fill it with the current information that
322  * we are about to use to construct the object. We will then
323  * be able to resurrect the information later by calling xml().
324  */
325  th->setXMLdata(phase);
326 
327  // set the id attribute of the phase to the 'id' attribute in the XML tree.
328  th->setID(phase.id());
329  th->setName(phase.id());
330 
331  // Number of spatial dimensions. Defaults to 3 (bulk phase)
332  if (phase.hasAttrib("dim")) {
333  int idim = intValue(phase["dim"]);
334  if (idim < 1 || idim > 3) {
335  throw CanteraError("importPhase",
336  "phase, " + th->id() +
337  ", has unphysical number of dimensions: " + phase["dim"]);
338  }
339  th->setNDim(idim);
340  } else {
341  th->setNDim(3); // default
342  }
343 
344  // Set equation of state parameters. The parameters are
345  // specific to each subclass of ThermoPhase, so this is done
346  // by method setParametersFromXML in each subclass.
347  const XML_Node& eos = phase.child("thermo");
348  if (phase.hasChild("thermo")) {
349  th->setParametersFromXML(eos);
350  } else {
351  throw CanteraError("importPhase",
352  " phase, " + th->id() +
353  ", XML_Node does not have a \"thermo\" XML_Node");
354  }
355 
356  VPStandardStateTP* vpss_ptr = 0;
357  int ssConvention = th->standardStateConvention();
358  if (ssConvention == cSS_CONVENTION_VPSS) {
359  vpss_ptr = dynamic_cast <VPStandardStateTP*>(th);
360  if (vpss_ptr == 0) {
361  throw CanteraError("importPhase",
362  "phase, " + th->id() + ", was VPSS, but dynamic cast failed");
363  }
364  }
365 
366  /***************************************************************
367  * Add the elements.
368  ***************************************************************/
369  if (ssConvention != cSS_CONVENTION_SLAVE) {
370  installElements(*th, phase);
371  }
372 
373  /***************************************************************
374  * Add the species.
375  *
376  * Species definitions may be imported from multiple
377  * sources. For each one, a speciesArray element must be
378  * present.
379  ***************************************************************/
380  vector<XML_Node*> sparrays = phase.getChildren("speciesArray");
381  if (ssConvention != cSS_CONVENTION_SLAVE && sparrays.empty()) {
382  throw CanteraError("importPhase",
383  "phase, " + th->id() + ", has zero \"speciesArray\" XML nodes.\n"
384  + " There must be at least one speciesArray nodes "
385  "with one or more species");
386  }
387  vector<XML_Node*> dbases;
388  vector_int sprule(sparrays.size(),0);
389 
390  // loop over the speciesArray elements
391  for (size_t jsp = 0; jsp < sparrays.size(); jsp++) {
392  const XML_Node& speciesArray = *sparrays[jsp];
393 
394  // If the speciesArray element has a child element
395  //
396  // <skip element="undeclared">
397  //
398  // then set sprule[jsp] to 1, so that any species with an undeclared
399  // element will be quietly skipped when importing species. Additionally,
400  // if the skip node has the following attribute:
401  //
402  // <skip species="duplicate">
403  //
404  // then duplicate species names will not cause Cantera to throw an
405  // exception. Instead, the duplicate entry will be discarded.
406  if (speciesArray.hasChild("skip")) {
407  const XML_Node& sk = speciesArray.child("skip");
408  string eskip = sk["element"];
409  if (eskip == "undeclared") {
410  sprule[jsp] = 1;
411  }
412  string dskip = sk["species"];
413  if (dskip == "duplicate") {
414  sprule[jsp] += 10;
415  }
416  }
417 
418  // Get a pointer to the node containing the species
419  // definitions for the species declared in this
420  // speciesArray element. This may be in the local file
421  // containing the phase element, or may be in another
422  // file.
423  XML_Node* db = get_XML_Node(speciesArray["datasrc"], &phase.root());
424  if (db == 0) {
425  throw CanteraError("importPhase()",
426  " Can not find XML node for species database: "
427  + speciesArray["datasrc"]);
428  }
429 
430  // add this node to the list of species database nodes.
431  dbases.push_back(db);
432  }
433 
434  // Now, collect all the species names and all the XML_Node * pointers
435  // for those species in a single vector. This is where we decide what
436  // species are to be included in the phase.
437  // The logic is complicated enough that we put it in a separate routine.
438  std::vector<XML_Node*> spDataNodeList;
439  std::vector<std::string> spNamesList;
440  vector_int spRuleList;
441  formSpeciesXMLNodeList(spDataNodeList, spNamesList, spRuleList,
442  sparrays, dbases, sprule);
443 
444  // Decide whether the the phase has a variable pressure ss or not
445  if (ssConvention == cSS_CONVENTION_VPSS) {
446  VPSSMgr* vp_spth = newVPSSMgr(vpss_ptr, &phase, spDataNodeList);
447  vpss_ptr->setVPSSMgr(vp_spth);
448  th->setSpeciesThermo(vp_spth->SpeciesThermoMgr());
449  }
450 
451  size_t nsp = spDataNodeList.size();
452  if (ssConvention == cSS_CONVENTION_SLAVE && nsp > 0) {
453  throw CanteraError("importPhase()", "For Slave standard states, "
454  "number of species must be zero: {}", nsp);
455  }
456  for (size_t k = 0; k < nsp; k++) {
457  XML_Node* s = spDataNodeList[k];
458  AssertTrace(s != 0);
459  if (spRuleList[k]) {
460  th->ignoreUndefinedElements();
461  }
462  th->addSpecies(newSpecies(*s));
463  if (vpss_ptr) {
464  vpss_ptr->createInstallPDSS(k, *s, &phase);
465  }
466  th->saveSpeciesData(k, s);
467  }
468 
469  if (ssConvention == cSS_CONVENTION_SLAVE) {
470  th->installSlavePhases(&phase);
471  }
472 
473  // Done adding species. Perform any required subclass-specific
474  // initialization.
475  th->initThermo();
476 
477  // Perform any required subclass-specific initialization
478  // that requires the XML phase object
479  std::string id = "";
480  th->initThermoXML(phase, id);
481 }

483 void installElements(Phase& th, const XML_Node& phaseNode)
484 {
485  // get the declared element names
486  if (!phaseNode.hasChild("elementArray")) {
487  throw CanteraError("installElements",
488  "phase XML node doesn't have \"elementArray\" XML Node");
489  }
490  XML_Node& elements = phaseNode.child("elementArray");
491  vector<string> enames;
492  getStringArray(elements, enames);
493 
494  // // element database defaults to elements.xml
495  string element_database = "elements.xml";
496  if (elements.hasAttrib("datasrc")) {
497  element_database = elements["datasrc"];
498  }
499 
500  XML_Node* doc = get_XML_File(element_database);
501  XML_Node* dbe = &doc->child("elementData");
502 
503  XML_Node& root = phaseNode.root();
504  XML_Node* local_db = 0;
505  if (root.hasChild("elementData")) {
506  local_db = &root.child("elementData");
507  }
508 
509  for (size_t i = 0; i < enames.size(); i++) {
510  // Find the element data
511  XML_Node* e = 0;
512  if (local_db) {
513  e = local_db->findByAttr("name",enames[i]);
514  }
515  if (!e) {
516  e = dbe->findByAttr("name",enames[i]);
517  }
518  if (!e) {
519  throw CanteraError("addElementsFromXML","no data for element "
520  +enames[i]);
521  }
522 
523  // Add the element
524  doublereal weight = 0.0;
525  if (e->hasAttrib("atomicWt")) {
526  weight = fpValue(e->attrib("atomicWt"));
527  }
528  int anum = 0;
529  if (e->hasAttrib("atomicNumber")) {
530  anum = intValue(e->attrib("atomicNumber"));
531  }
532  string symbol = e->attrib("name");
533  doublereal entropy298 = ENTROPY298_UNKNOWN;
534  if (e->hasChild("entropy298")) {
535  XML_Node& e298Node = e->child("entropy298");
536  if (e298Node.hasAttrib("value")) {
537  entropy298 = fpValueCheck(e298Node["value"]);
538  }
539  }
540  if (weight != 0.0) {
541  th.addElement(symbol, weight, anum, entropy298);
542  } else {
543  th.addElement(symbol);
544  }
545  }
