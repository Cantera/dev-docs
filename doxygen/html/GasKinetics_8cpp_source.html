114 void GasKinetics::update_rates_T()
115 {
116  doublereal T = thermo().temperature();
117  doublereal P = thermo().pressure();
118  m_logStandConc = log(thermo().standardConcentration());
119  doublereal logT = log(T);
120 
121  if (T != m_temp) {
122  if (!m_rfn.empty()) {
123  m_rates.update(T, logT, &m_rfn[0]);
124  }
125 
126  if (!m_rfn_low.empty()) {
127  m_falloff_low_rates.update(T, logT, &m_rfn_low[0]);
128  m_falloff_high_rates.update(T, logT, &m_rfn_high[0]);
129  }
130  if (!falloff_work.empty()) {
131  m_falloffn.updateTemp(T, &falloff_work[0]);
132  }
133  updateKc();
134  m_ROP_ok = false;
135  }
136 
137  if (T != m_temp || P != m_pres) {
138  if (m_plog_rates.nReactions()) {
139  m_plog_rates.update(T, logT, &m_rfn[0]);
140  m_ROP_ok = false;
141  }
142 
143  if (m_cheb_rates.nReactions()) {
144  m_cheb_rates.update(T, logT, &m_rfn[0]);
145  m_ROP_ok = false;
146  }
147  }
148  m_pres = P;
149  m_temp = T;
150 }

182 void GasKinetics::updateKc()
183 {
184  thermo().getStandardChemPotentials(&m_grt[0]);
185  fill(m_rkcn.begin(), m_rkcn.end(), 0.0);
186 
187  // compute Delta G^0 for all reversible reactions
188  m_rxnstoich.getRevReactionDelta(m_ii, &m_grt[0], &m_rkcn[0]);
189 
190  doublereal rrt = 1.0/(GasConstant * thermo().temperature());
191  for (size_t i = 0; i < m_nrev; i++) {
192  size_t irxn = m_revindex[i];
193  m_rkcn[irxn] = std::min(exp(m_rkcn[irxn]*rrt - m_dn[irxn]*m_logStandConc),
194  BigNumber);
195  }
196 
197  for (size_t i = 0; i != m_nirrev; ++i) {
198  m_rkcn[ m_irrev[i] ] = 0.0;

336 void GasKinetics::processFalloffReactions()
337 {
338  // use m_ropr for temporary storage of reduced pressure
339  vector_fp& pr = m_ropr;
340 
341  for (size_t i = 0; i < m_nfall; i++) {
342  pr[i] = concm_falloff_values[i] * m_rfn_low[i] / (m_rfn_high[i] + SmallNumber);
343  AssertFinite(pr[i], "GasKinetics::processFalloffReactions",
344  "pr[" + int2str(i) + "] is not finite.");
345  }
346 
347  double* work = (falloff_work.empty()) ? 0 : &falloff_work[0];
348  m_falloffn.pr_to_falloff(&pr[0], work);
349 
350  for (size_t i = 0; i < m_nfall; i++) {
351  if (m_rxntype[m_fallindx[i]] == FALLOFF_RXN) {
352  pr[i] *= m_rfn_high[i];
353  } else { // CHEMACT_RXN
354  pr[i] *= m_rfn_low[i];
355  }
356  }
357 
358  scatter_copy(pr.begin(), pr.begin() + m_nfall,
359  m_ropf.begin(), m_fallindx.begin());
360 }

362 void GasKinetics::updateROP()
363 {
364  update_rates_C();
365  update_rates_T();
366 
367  if (m_ROP_ok) {
368  return;
369  }
370 
371  // copy rate coefficients into ropf
372  copy(m_rfn.begin(), m_rfn.end(), m_ropf.begin());
373 
374  // multiply ropf by enhanced 3b conc for all 3b rxns
375  if (!concm_3b_values.empty()) {
376  m_3b_concm.multiply(&m_ropf[0], &concm_3b_values[0]);
377  }
378 
379  if (m_nfall) {
380  processFalloffReactions();
381  }
382 
383  // multiply by perturbation factor
384  multiply_each(m_ropf.begin(), m_ropf.end(), m_perturb.begin());
385 
386  // copy the forward rates to the reverse rates
387  copy(m_ropf.begin(), m_ropf.end(), m_ropr.begin());
388 
389  // for reverse rates computed from thermochemistry, multiply
390  // the forward rates copied into m_ropr by the reciprocals of
391  // the equilibrium constants
392  multiply_each(m_ropr.begin(), m_ropr.end(), m_rkcn.begin());
393 
394  // multiply ropf by concentration products
395  m_rxnstoich.multiplyReactants(&m_conc[0], &m_ropf[0]);
396  //m_reactantStoich.multiply(m_conc.begin(), ropf.begin());
397 
398  // for reversible reactions, multiply ropr by concentration
399  // products
400  m_rxnstoich.multiplyRevProducts(&m_conc[0], &m_ropr[0]);
401  //m_revProductStoich.multiply(m_conc.begin(), ropr.begin());
402 
403  for (size_t j = 0; j != m_ii; ++j) {
404  m_ropnet[j] = m_ropf[j] - m_ropr[j];
405  }
406 
407  for (size_t i = 0; i < m_rfn.size(); i++) {
408  AssertFinite(m_rfn[i], "GasKinetics::updateROP",
409  "m_rfn[" + int2str(i) + "] is not finite.");
410  AssertFinite(m_ropf[i], "GasKinetics::updateROP",
411  "m_ropf[" + int2str(i) + "] is not finite.");
412  AssertFinite(m_ropr[i], "GasKinetics::updateROP",
413  "m_ropr[" + int2str(i) + "] is not finite.");
414  }
415 
416  m_ROP_ok = true;
417 }

582 void GasKinetics::installReagents(const ReactionData& r)
583 {
584  m_ropf.push_back(0.0); // extend by one for new rxn
585  m_ropr.push_back(0.0);
586  m_ropnet.push_back(0.0);
587  size_t n, ns, m;
588  doublereal nsFlt;
589  doublereal reactantGlobalOrder = 0.0;
590  doublereal productGlobalOrder = 0.0;
591  size_t rnum = reactionNumber();
592 
593  std::vector<size_t> rk;
594  size_t nr = r.reactants.size();
595  for (n = 0; n < nr; n++) {
596  nsFlt = r.rstoich[n];
597  reactantGlobalOrder += nsFlt;
598  ns = (size_t) nsFlt;
599  if ((doublereal) ns != nsFlt) {
600  ns = std::max<size_t>(ns, 1);
601  }
602  if (r.rstoich[n] != 0.0) {
603  m_rrxn[r.reactants[n]][rnum] += r.rstoich[n];
604  }
605  for (m = 0; m < ns; m++) {
606  rk.push_back(r.reactants[n]);
607  }
608  }
609  m_reactants.push_back(rk);
610 
611  std::vector<size_t> pk;
612  size_t np = r.products.size();
613  for (n = 0; n < np; n++) {
614  nsFlt = r.pstoich[n];
615  productGlobalOrder += nsFlt;
616  ns = (size_t) nsFlt;
617  if ((double) ns != nsFlt) {
618  ns = std::max<size_t>(ns, 1);
619  }
620  if (r.pstoich[n] != 0.0) {
621  m_prxn[r.products[n]][rnum] += r.pstoich[n];
622  }
623  for (m = 0; m < ns; m++) {
624  pk.push_back(r.products[n]);
625  }
626  }
627  m_products.push_back(pk);
628  m_rkcn.push_back(0.0);
629  m_rxnstoich.add(reactionNumber(), r);
630 
631  if (r.reversible) {
632  m_dn.push_back(productGlobalOrder - reactantGlobalOrder);
633  m_revindex.push_back(reactionNumber());
634  m_nrev++;
635  } else {
636  m_dn.push_back(productGlobalOrder - reactantGlobalOrder);
637  m_irrev.push_back(reactionNumber());
638  m_nirrev++;
639  }
