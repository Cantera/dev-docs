29 void GasKinetics::update_rates_T()
30 {
31  doublereal T = thermo().temperature();
32  doublereal P = thermo().pressure();
33  m_logStandConc = log(thermo().standardConcentration());
34  doublereal logT = log(T);
35 
36  if (T != m_temp) {
37  if (!m_rfn.empty()) {
38  m_rates.update(T, logT, m_rfn.data());
39  }
40 
41  if (!m_rfn_low.empty()) {
42  m_falloff_low_rates.update(T, logT, m_rfn_low.data());
43  m_falloff_high_rates.update(T, logT, m_rfn_high.data());
44  }
45  if (!falloff_work.empty()) {
46  m_falloffn.updateTemp(T, falloff_work.data());
47  }
48  updateKc();
49  m_ROP_ok = false;
50  }
51 
52  if (T != m_temp || P != m_pres) {
53  if (m_plog_rates.nReactions()) {
54  m_plog_rates.update(T, logT, m_rfn.data());
55  m_ROP_ok = false;
56  }
57 
58  if (m_cheb_rates.nReactions()) {
59  m_cheb_rates.update(T, logT, m_rfn.data());
60  m_ROP_ok = false;
61  }
62  }
63  m_pres = P;
64  m_temp = T;
65 }

97 void GasKinetics::updateKc()
98 {
99  thermo().getStandardChemPotentials(m_grt.data());
100  fill(m_rkcn.begin(), m_rkcn.end(), 0.0);
101 
102  // compute Delta G^0 for all reversible reactions
103  getRevReactionDelta(m_grt.data(), m_rkcn.data());
104 
105  doublereal rrt = 1.0 / thermo().RT();
106  for (size_t i = 0; i < m_revindex.size(); i++) {
107  size_t irxn = m_revindex[i];
108  m_rkcn[irxn] = std::min(exp(m_rkcn[irxn]*rrt - m_dn[irxn]*m_logStandConc),
109  BigNumber);
110  }
111 
112  for (size_t i = 0; i != m_irrev.size(); ++i) {
113  m_rkcn[ m_irrev[i] ] = 0.0;
114  }

161 void GasKinetics::updateROP()
162 {
163  update_rates_C();
164  update_rates_T();
165  if (m_ROP_ok) {
166  return;
167  }
168 
169  // copy rate coefficients into ropf
170  m_ropf = m_rfn;
171 
172  // multiply ropf by enhanced 3b conc for all 3b rxns
173  if (!concm_3b_values.empty()) {
174  m_3b_concm.multiply(m_ropf.data(), concm_3b_values.data());
175  }
176 
177  if (m_falloff_high_rates.nReactions()) {
178  processFalloffReactions();
179  }
180 
181  // multiply by perturbation factor
182  multiply_each(m_ropf.begin(), m_ropf.end(), m_perturb.begin());
183 
184  // copy the forward rates to the reverse rates
185  m_ropr = m_ropf;
186 
187  // for reverse rates computed from thermochemistry, multiply the forward
188  // rates copied into m_ropr by the reciprocals of the equilibrium constants
189  multiply_each(m_ropr.begin(), m_ropr.end(), m_rkcn.begin());
190 
191  // multiply ropf by concentration products
192  m_reactantStoich.multiply(m_conc.data(), m_ropf.data());
193 
194  // for reversible reactions, multiply ropr by concentration products
195  m_revProductStoich.multiply(m_conc.data(), m_ropr.data());
196 
197  for (size_t j = 0; j != nReactions(); ++j) {
198  m_ropnet[j] = m_ropf[j] - m_ropr[j];
199  }
200 
201  for (size_t i = 0; i < m_rfn.size(); i++) {
202  AssertFinite(m_rfn[i], "GasKinetics::updateROP",
203  "m_rfn[{}] is not finite.", i);
204  AssertFinite(m_ropf[i], "GasKinetics::updateROP",
205  "m_ropf[{}] is not finite.", i);
206  AssertFinite(m_ropr[i], "GasKinetics::updateROP",
207  "m_ropr[{}] is not finite.", i);
208  }
209  m_ROP_ok = true;
210 }

237 bool GasKinetics::addReaction(shared_ptr<Reaction> r)
238 {

269 void GasKinetics::addFalloffReaction(FalloffReaction& r)
270 {
271  // install high and low rate coeff calculators and extend the high and low
272  // rate coeff value vectors
273  size_t nfall = m_falloff_high_rates.nReactions();
274  m_falloff_high_rates.install(nfall, r.high_rate);
275  m_rfn_high.push_back(0.0);
276  m_falloff_low_rates.install(nfall, r.low_rate);
277  m_rfn_low.push_back(0.0);
278 
279  // add this reaction number to the list of falloff reactions
280  m_fallindx.push_back(nReactions()-1);
281  m_rfallindx[nReactions()-1] = nfall;
282 
283  // install the enhanced third-body concentration calculator
284  map<size_t, double> efficiencies;
285  for (const auto& eff : r.third_body.efficiencies) {
286  size_t k = kineticsSpeciesIndex(eff.first);
287  if (k != npos) {
288  efficiencies[k] = eff.second;
289  } else if (!m_skipUndeclaredThirdBodies) {
290  throw CanteraError("GasKinetics::addTFalloffReaction", "Found "
291  "third-body efficiency for undefined species '" + eff.first +
292  "' while adding reaction '" + r.equation() + "'");
293  }
294  }
295  m_falloff_concm.install(nfall, efficiencies,
296  r.third_body.default_efficiency);
297 
298  // install the falloff function calculator for this reaction
299  m_falloffn.install(nfall, r.reaction_type, r.falloff);
300 }
