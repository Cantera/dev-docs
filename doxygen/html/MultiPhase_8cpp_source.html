113 void MultiPhase::
114 addPhase(ThermoPhase* p, doublereal moles)
115 {
116  if (m_init) {
117  throw CanteraError("addPhase",
118  "phases cannot be added after init() has been called.");
119  }
120 
121  // save the pointer to the phase object
122  m_phase.push_back(p);
123 
124  // store its number of moles
125  m_moles.push_back(moles);
126  m_temp_OK.push_back(true);
127 
128  // update the number of phases and the total number of
129  // species
130  m_np = m_phase.size();
131  m_nsp += p->nSpecies();
132 
133  // determine if this phase has new elements
134  // for each new element, add an entry in the map
135  // from names to index number + 1:
136 
137  string ename;
138  // iterate over the elements in this phase
139  index_t m, nel = p->nElements();
140  for (m = 0; m < nel; m++) {
141  ename = p->elementName(m);
142 
143  // if no entry is found for this element name, then
144  // it is a new element. In this case, add the name
145  // to the list of names, increment the element count,
146  // and add an entry to the name->(index+1) map.
147  if (m_enamemap.find(ename) == m_enamemap.end()) {
148  m_enamemap[ename] = m_nel + 1;
149  m_enames.push_back(ename);
150  m_atomicNumber.push_back(p->atomicNumber(m));
151 
152  // Element 'E' (or 'e') is special. Note its location.
153  if (ename == "E" || ename == "e") {
154  m_eloc = m_nel;
155  }
156 
157  m_nel++;
158  }
159  }
160 
161  // If the mixture temperature hasn't been set, then set the
162  // temperature and pressure to the values for the phase being
163  // added. There is no good way to do this. However, this will be overridden later.
164  if (m_temp == 298.15 && p->temperature() > 2.0E-3) {
165  m_temp = p->temperature();
166  m_press = p->pressure();
167  }
168 
169  // If this is a solution phase, update the minimum and maximum
170  // mixture temperatures. Stoichiometric phases are excluded,
171  // since a mixture may define multiple stoichiometric phases,
172  // each of which has thermo data valid only over a limited
173  // range. For example, a mixture might be defined to contain a
174  // phase representing water ice and one representing liquid
175  // water, only one of which should be present if the mixture
176  // represents an equilibrium state.
177  if (p->nSpecies() > 1) {
178  double t = p->minTemp();
179  if (t > m_Tmin) {
180  m_Tmin = t;
181  }
182  t = p->maxTemp();
183  if (t < m_Tmax) {
184  m_Tmax = t;
185  }
186  }
187 }

189 // Process phases and build atomic composition array. This method
190 // must be called after all phases are added, before doing
191 // anything else with the mixture. After init() has been called,
192 // no more phases may be added.
193 void MultiPhase::init()
194 {
195  if (m_init) {
196  return;
197  }
198  index_t ip, kp, k = 0, nsp, m;
199  size_t mlocal;
200  string sym;
201 
202  // allocate space for the atomic composition matrix
203  m_atoms.resize(m_nel, m_nsp, 0.0);
204  m_moleFractions.resize(m_nsp, 0.0);
205  m_elemAbundances.resize(m_nel, 0.0);
206 
207  // iterate over the elements
208  // -> fill in m_atoms(m,k), m_snames(k), m_spphase(k),
209  // m_sptart(ip)
210  for (m = 0; m < m_nel; m++) {
211  sym = m_enames[m];
212  k = 0;
213  // iterate over the phases
214  for (ip = 0; ip < m_np; ip++) {
215  ThermoPhase* p = m_phase[ip];
216  nsp = p->nSpecies();
217  mlocal = p->elementIndex(sym);
218  for (kp = 0; kp < nsp; kp++) {
219  if (mlocal != npos) {
220  m_atoms(m, k) = p->nAtoms(kp, mlocal);
221  }
222  if (m == 0) {
223  m_snames.push_back(p->speciesName(kp));
224  if (kp == 0) {
225  m_spstart.push_back(m_spphase.size());
226  }
227  m_spphase.push_back(ip);
228  }
229  k++;
230  }
231  }
232  }
233 
234  if (m_eloc != npos) {
235  doublereal esum;
236  for (k = 0; k < m_nsp; k++) {
237  esum = 0.0;
238  for (m = 0; m < m_nel; m++) {
239  if (m != m_eloc) {
240  esum += m_atoms(m,k) * m_atomicNumber[m];
241  }
242  }
243  //m_atoms(m_eloc, k) += esum;
244  }
245  }
246 
247  /// set the initial composition within each phase to the
248  /// mole fractions stored in the phase objects
249  m_init = true;
250 
251  uploadMoleFractionsFromPhases();
252 
253  updatePhases();
254 }

341 /// \f[ Q_p = N_p \sum_k F z_k X_k \f]

366 //====================================================================================================================
367 // Get chemical potentials of species with valid thermo
368 // data. This method is designed for use in computing chemical
369 // equilibrium by Gibbs minimization. For solution phases (more
370 // than one species), this does the same thing as
371 // getChemPotentials. But for stoichiometric phases, this writes
372 // into array \a mu the user-specified value \a not_mu instead of
373 // the chemical potential if the temperature is outside the range
374 // for which the thermo data for the one species in the phase are
375 // valid. The need for this arises since many condensed phases
376 // have thermo data fit only for the temperature range for which
377 // they are stable. For example, in the NASA database, the fits
378 // for H2O(s) are only done up to 0 C, the fits for H2O(L) are
379 // only done from 0 C to 100 C, etc. Using the polynomial fits outside
380 // the range for which the fits were done can result in spurious
381 // chemical potentials, and can lead to condensed phases
382 // appearing when in fact they should be absent.
383 //
384 // By setting \a not_mu to a large positive value, it is possible
385 // to force routines which seek to minimize the Gibbs free energy
386 // of the mixture to zero out any phases outside the temperature
387 // range for which their thermo data are valid.
388 //
389 // If this method is called with \a standard set to true, then
390 // the composition-independent standard chemical potentials are
391 // returned instead of the composition-dependent chemical
392 // potentials.
393 //
394 void MultiPhase::getValidChemPotentials(doublereal not_mu,
395  doublereal* mu, bool standard) const
396 {
397  index_t i, loc = 0;
398 
399  updatePhases();
400  // iterate over the phases
401  for (i = 0; i < m_np; i++) {
402  if (tempOK(i) || m_phase[i]->nSpecies() > 1) {
403  if (!standard) {
404  m_phase[i]->getChemPotentials(mu + loc);
405  } else {
406  m_phase[i]->getStandardChemPotentials(mu + loc);
407  }
408  } else {
409  fill(mu + loc, mu + loc + m_phase[i]->nSpecies(), not_mu);
410  }
411  loc += m_phase[i]->nSpecies();
412  }
413 }

632 // Internal routine to calculate the element abundance vector
633 void MultiPhase::calcElemAbundances() const
634 {
635  index_t loc = 0;
636  index_t eGlobal;
637  index_t ik, kGlobal;
638  doublereal spMoles;
639  for (eGlobal = 0; eGlobal < m_nel; eGlobal++) {

675  int n;
676  doublereal hnow, herr = 1.0;
677  doublereal snow, serr = 1.0, s0;
678  doublereal Tlow = -1.0, Thigh = -1.0;
679  doublereal Hlow = Undef, Hhigh = Undef, tnew;
680  doublereal dta=0.0, dtmax, cpb;
681  MultiPhaseEquil* e = 0;
682 
683  if (!m_init) {
684  init();
685  }
686  if (loglevel > 0) {
687  beginLogGroup("MultiPhase::equilibrate", loglevel);
688  }
689 
690  if (XY == TP) {
691  if (loglevel > 0) {
692  addLogEntry("problem type","fixed T,P");
693  addLogEntry("Temperature",temperature());
694  addLogEntry("Pressure", pressure());
695  }
696 
697  // create an equilibrium manager
698  e = new MultiPhaseEquil(this);
699  try {
700  e->equilibrate(XY, err, maxsteps, loglevel);
701  } catch (CanteraError& err) {
702  err.save();
703  if (loglevel > 0) {
704  endLogGroup();
705  }
706  delete e;
707  e = 0;
708  throw err;
709  }
710  goto done;
711  }
712 
713  else if (XY == HP) {
714  h0 = enthalpy();
715  Tlow = 0.5*m_Tmin; // lower bound on T
716  Thigh = 2.0*m_Tmax; // upper bound on T
717  if (loglevel > 0) {
718  addLogEntry("problem type","fixed H,P");
719  addLogEntry("H target",fp2str(h0));
720  }
721  for (n = 0; n < maxiter; n++) {
722 
723  // if 'strt' is false, the current composition will be used as
724  // the starting estimate; otherwise it will be estimated
725  // if (e) {
726  // cout << "e should be zero, but it is not!" << endl;
727  // delete e;
728  // }
729  e = new MultiPhaseEquil(this, strt);
730  // start with a loose error tolerance, but tighten it as we get
731  // close to the final temperature
732  if (loglevel > 0) {
733  beginLogGroup("iteration "+int2str(n));
734  }
735 
736  try {
737  e->equilibrate(TP, err, maxsteps, loglevel);
738  hnow = enthalpy();
739  // the equilibrium enthalpy monotonically increases with T;
740  // if the current value is below the target, the we know the
741  // current temperature is too low. Set
742  if (hnow < h0) {
743  if (m_temp > Tlow) {
744  Tlow = m_temp;
745  Hlow = hnow;
746  }
747  }
748  // the current enthalpy is greater than the target; therefore the
749  // current temperature is too high.
750  else {
751  if (m_temp < Thigh) {
752  Thigh = m_temp;
753  Hhigh = hnow;
754  }
755  }
756  if (Hlow != Undef && Hhigh != Undef) {
757  cpb = (Hhigh - Hlow)/(Thigh - Tlow);
758  dt = (h0 - hnow)/cpb;
759  dta = fabs(dt);
760  dtmax = 0.5*fabs(Thigh - Tlow);
761  if (dta > dtmax) {
762  dt *= dtmax/dta;
763  }
764  } else {
765  tnew = sqrt(Tlow*Thigh);
766  dt = tnew - m_temp;
767  //cpb = cp();
768  }
769 
770  herr = fabs((h0 - hnow)/h0);
771  if (loglevel > 0) {
772  addLogEntry("T",fp2str(temperature()));
773  addLogEntry("H",fp2str(hnow));
774  addLogEntry("H rel error",fp2str(herr));
775  addLogEntry("lower T bound",fp2str(Tlow));
776  addLogEntry("upper T bound",fp2str(Thigh));
777  endLogGroup(); // iteration
778  }
779 
780 
781  if (herr < err) { // || dta < 1.0e-4) {
782  if (loglevel > 0) {
783  addLogEntry("T iterations",int2str(n));
784  addLogEntry("Final T",fp2str(temperature()));
785  addLogEntry("H rel error",fp2str(herr));
786  }
787  goto done;
788  }
789  tnew = m_temp + dt;
790  if (tnew < 0.0) {
791  tnew = 0.5*m_temp;
792  }
793  //dta = fabs(tnew - m_temp);
794  setTemperature(tnew);
795 
796  // if the size of Delta T is not too large, use
797  // the current composition as the starting estimate
798  if (dta < 100.0) {
799  strt = false;
800  }
801 
802  }
803 
804  catch (CanteraError& err) {
805  err.save();
806  if (!strt) {
807  if (loglevel > 0)
808  addLogEntry("no convergence",
809  "try estimating starting composition");
810  strt = true;
811  } else {
812  tnew = 0.5*(m_temp + Thigh);
813  if (fabs(tnew - m_temp) < 1.0) {
814  tnew = m_temp + 1.0;
815  }
816  setTemperature(tnew);
817  if (loglevel > 0)
818  addLogEntry("no convergence",
819  "trying T = "+fp2str(m_temp));
820  }
821  if (loglevel > 0) {
822  endLogGroup();
823  }
824  }
825  delete e;
826  e = 0;
827  }
828  if (loglevel > 0) {
829  addLogEntry("reached max number of T iterations",int2str(maxiter));
830  endLogGroup();
831  }
832  throw CanteraError("MultiPhase::equilibrate",
833  "No convergence for T");
834  } else if (XY == SP) {
835  s0 = entropy();
836  Tlow = 1.0; // m_Tmin; // lower bound on T
837  Thigh = 1.0e6; // m_Tmax; // upper bound on T
838  if (loglevel > 0) {
839  addLogEntry("problem type","fixed S,P");
840  addLogEntry("S target",fp2str(s0));
841  addLogEntry("min T",fp2str(Tlow));
842  addLogEntry("max T",fp2str(Thigh));
843  }
844  for (n = 0; n < maxiter; n++) {
845  if (e) {
846  delete e;
847  }
848  e = new MultiPhaseEquil(this, strt);
849  if (loglevel > 0) {
850  beginLogGroup("iteration "+int2str(n));
851  }
852 
853  try {
854  e->equilibrate(TP, err, maxsteps, loglevel);
855  snow = entropy();
856  if (snow < s0) {
857  if (m_temp > Tlow) {
858  Tlow = m_temp;
859  }
860  } else {
861  if (m_temp < Thigh) {
862  Thigh = m_temp;
863  }
864  }
865  serr = fabs((s0 - snow)/s0);
866  if (loglevel > 0) {
867  addLogEntry("T",fp2str(temperature()));
868  addLogEntry("S",fp2str(snow));
869  addLogEntry("S rel error",fp2str(serr));
870  endLogGroup();
871  }
872  dt = (s0 - snow)*m_temp/cp();
873  dtmax = 0.5*fabs(Thigh - Tlow);
874  dtmax = (dtmax > 500.0 ? 500.0 : dtmax);
875  dta = fabs(dt);
876  if (dta > dtmax) {
877  dt *= dtmax/dta;
878  }
879  if (herr < err || dta < 1.0e-4) {
880  if (loglevel > 0) {
881  addLogEntry("T iterations",int2str(n));
882  addLogEntry("Final T",fp2str(temperature()));
883  addLogEntry("S rel error",fp2str(serr));
884  }
885  goto done;
886  }
887  tnew = m_temp + dt;
888  setTemperature(tnew);
889 
890  // if the size of Delta T is not too large, use
891  // the current composition as the starting estimate
892  if (dta < 100.0) {
893  strt = false;
894  }
895  }
896 
897  catch (CanteraError& err) {
898  err.save();
899  if (!strt) {
900  if (loglevel > 0) {
901  addLogEntry("no convergence",
902  "setting strt to True");
903  }
904  strt = true;
905  } else {
906  tnew = 0.5*(m_temp + Thigh);
907  setTemperature(tnew);
908  if (loglevel > 0) {
909  addLogEntry("no convergence",
910  "trying T = "+fp2str(m_temp));
911  }
912  }
913  if (loglevel > 0) {
914  endLogGroup();
915  }
916  }
917  delete e;
918  e = 0;
919  }
920  if (loglevel > 0) {
921  addLogEntry("reached max number of T iterations",int2str(maxiter));
922  endLogGroup();
923  }
924  throw CanteraError("MultiPhase::equilibrate",
925  "No convergence for T");
926  } else if (XY == TV) {
927  addLogEntry("problem type","fixed T, V");
928  // doublereal dt = 1.0e3;
929  doublereal v0 = volume();
930  doublereal dVdP;
931  int n;
932  bool start = true;
933  doublereal vnow, pnow, verr;
934  for (n = 0; n < maxiter; n++) {
935  pnow = pressure();
936  MultiPhaseEquil e(this, start);
937  start = false;
938  beginLogGroup("iteration "+int2str(n));
939 
940  e.equilibrate(TP, err, maxsteps, loglevel);
941  vnow = volume();
942  verr = fabs((v0 - vnow)/v0);
943  addLogEntry("P",fp2str(pressure()));
944  addLogEntry("V rel error",fp2str(verr));
945  endLogGroup();
946 
947  if (verr < err) {
948  addLogEntry("P iterations",int2str(n));
949  addLogEntry("Final P",fp2str(pressure()));
950  addLogEntry("V rel error",fp2str(verr));
951  goto done;
952  }
953  // find dV/dP
954  setPressure(pnow*1.01);
955  dVdP = (volume() - vnow)/(0.01*pnow);
956  setPressure(pnow + 0.5*(v0 - vnow)/dVdP);
957  }
958  }
959 
960  else {
961  if (loglevel > 0) {
962  endLogGroup();
963  }
964  throw CanteraError("MultiPhase::equilibrate","unknown option");
965  }
966  return -1.0;
967 done:
968  delete e;
969  e = 0;
970  if (loglevel > 0) {
971  endLogGroup();
972  }
973  return err;
