30 /*!
31  * The Arrhenius expression is
32  * \f[
33  * k = A T^(b) exp (-E_a / RT)
34  * \f]
35  *
36  * @param node XML_Node to be read
37  * @param A Output pre-exponential factor. The units are variable.
38  * @param b output temperature power
39  * @param E Output activation energy in units of Kelvin
40  */

42  doublereal& A, doublereal& b, doublereal& E)

44  /* parse the children for the A, b, and E components.
45  */
46  A = getFloat(node, "A", "toSI");
47  b = getFloat(node, "b");
48  E = getFloat(node, "E", "actEnergy");
49  E /= GasConstant;

275 // Return the pure species value for this transport property evaluated
276 // from the Arrhenius expression
277 /*
278  * In general the Arrhenius expression is
279  *
280  * \f[
281  * \mu = A T^n \exp(- E / R T).
282  * \f]
283  *
284  * Note that for viscosity, the convention is such that
285  * a positive activation energy corresponds to the typical
286  * case of a positive argument to the exponential so that
287  * the Arrhenius expression is
288  *
289  * \f[
290  * \mu = A T^n \exp(+ E / R T).
291  * \f]
292  *
293  * Any temperature and composition dependence will be
294  * adjusted internally according to the information provided.
295  */
296 doublereal LTPspecies_Arrhenius::getSpeciesTransProp()
297 {
298 
299  doublereal t = m_thermo->temperature();
300  //m_coeffs[0] holds A
301  //m_coeffs[1] holds n
302  //m_coeffs[2] holds Tact
303  //m_coeffs[3] holds log(A)
304  if (t != m_temp) {
305  m_prop = 0;
306  m_logProp = 0;
307  m_temp = t;
308  m_logt = log(m_temp);
309  //For viscosity the sign convention on positive activation energy is swithced
310  if (m_property == TP_VISCOSITY) {
311  m_logProp = m_coeffs[3] + m_coeffs[1] * m_logt + m_coeffs[2] / m_temp ;
312  } else {
313  m_logProp = m_coeffs[3] + m_coeffs[1] * m_logt - m_coeffs[2] / m_temp ;
314  }
315  m_prop = exp(m_logProp);
316  }
317  return m_prop;
318 
319  }
