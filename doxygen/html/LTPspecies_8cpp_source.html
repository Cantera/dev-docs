31 //! Parses the XML element called Arrhenius.
32 /*!
33  * The Arrhenius expression is
34  * \f[
35  * k = A T^(b) exp (-E_a / RT)
36  * \f]
37  *
38  * @param node XML_Node to be read
39  * @param A Output pre-exponential factor. The units are variable.
40  * @param b output temperature power
41  * @param E Output activation energy in units of Kelvin
42  */
43 static void getArrhenius(const XML_Node& node,
44  doublereal& A, doublereal& b, doublereal& E)
45 {
46  // parse the children for the A, b, and E components.
47  A = getFloat(node, "A", "toSI");
48  b = getFloat(node, "b");
49  E = getFloat(node, "E", "actEnergy") / GasConstant;
50 }

142 doublereal LTPspecies_Arrhenius::getSpeciesTransProp()
143 {
144  doublereal t = m_thermo->temperature();
145  //m_coeffs[0] holds A
146  //m_coeffs[1] holds n
147  //m_coeffs[2] holds Tact
148  //m_coeffs[3] holds log(A)
149  if (t != m_temp) {
150  m_prop = 0;
151  m_logProp = 0;
152  m_temp = t;
153  m_logt = log(m_temp);
154  //For viscosity the sign convention on positive activation energy is swithced
155  if (m_property == TP_VISCOSITY) {
156  m_logProp = m_coeffs[3] + m_coeffs[1] * m_logt + m_coeffs[2] / m_temp;
157  } else {
158  m_logProp = m_coeffs[3] + m_coeffs[1] * m_logt - m_coeffs[2] / m_temp;
159  }
160  m_prop = exp(m_logProp);
161  }
162  return m_prop;
163 }

189 doublereal LTPspecies_Poly::getSpeciesTransProp()
190 {
191  doublereal t = m_thermo->temperature();
192  if (t != m_temp) {
193  m_prop = 0.0;
194  m_temp = t;
195  double tempN = 1.0;
196  for (int i = 0; i < (int) m_coeffs.size() ; i++) {
197  m_prop += m_coeffs[i] * tempN;
198  tempN *= m_temp;
199  }
200  }
201  return m_prop;
202 }

228 doublereal LTPspecies_ExpT::getSpeciesTransProp()
229 {
230  doublereal t = m_thermo->temperature();
231  if (t != m_temp) {
232  m_temp=t;
233  m_prop = m_coeffs[0];
234  doublereal tempN = 1.0;
235  doublereal tmp = 0.0;
236  for (int i = 1; i < (int) m_coeffs.size() ; i++) {
237  tempN *= m_temp;
238  tmp += m_coeffs[i] * tempN;
239  }
240  m_prop *= exp(tmp);
241  }
242  return m_prop;
