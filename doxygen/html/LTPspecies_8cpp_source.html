29 //! Parses the XML element called Arrhenius.
30 /*!
31  * The Arrhenius expression is
32  * \f[
33  * k = A T^(b) exp (-E_a / RT)
34  * \f]
35  *
36  * @param node XML_Node to be read
37  * @param A Output pre-exponential factor. The units are variable.
38  * @param b output temperature power
39  * @param E Output activation energy in units of Kelvin
40  */
41 static void getArrhenius(const XML_Node& node,
42  doublereal& A, doublereal& b, doublereal& E)
43 {
44  // parse the children for the A, b, and E components.
45  A = getFloat(node, "A", "toSI");
46  b = getFloat(node, "b");
47  E = getFloat(node, "E", "actEnergy");
48  E /= GasConstant;
49 }

143 LTPspecies_Arrhenius::LTPspecies_Arrhenius(const XML_Node& propNode, const std::string name,
144  TransportPropertyType tp_ind, const thermo_t* thermo) :
145  LTPspecies(&propNode, name, tp_ind, thermo)
146 {
147  m_model = LTP_TD_ARRHENIUS;
148  m_temp = 0.0;
149  m_prop = 0.0;
150 
151  doublereal A_k, n_k, Tact_k;
152  getArrhenius(propNode, A_k, n_k, Tact_k);
153  if (A_k <= 0.0) {
154  throw LTPError("negative or zero " + propNode.name());
155  }
156  m_coeffs.push_back(A_k);
157  m_coeffs.push_back(n_k);
158  m_coeffs.push_back(Tact_k);
159  m_coeffs.push_back(log(A_k));

184 doublereal LTPspecies_Arrhenius::getSpeciesTransProp()
185 {
186  doublereal t = m_thermo->temperature();
187  //m_coeffs[0] holds A
188  //m_coeffs[1] holds n
189  //m_coeffs[2] holds Tact
190  //m_coeffs[3] holds log(A)
191  if (t != m_temp) {
192  m_prop = 0;
193  m_logProp = 0;
194  m_temp = t;
195  m_logt = log(m_temp);
196  //For viscosity the sign convention on positive activation energy is swithced
197  if (m_property == TP_VISCOSITY) {
198  m_logProp = m_coeffs[3] + m_coeffs[1] * m_logt + m_coeffs[2] / m_temp;
199  } else {
200  m_logProp = m_coeffs[3] + m_coeffs[1] * m_logt - m_coeffs[2] / m_temp;
201  }
202  m_prop = exp(m_logProp);
203  }
204  return m_prop;
205 }

238 doublereal LTPspecies_Poly::getSpeciesTransProp()
239 {
240  doublereal t = m_thermo->temperature();
241  if (t != m_temp) {
242  m_prop = 0.0;
243  m_temp = t;
244  double tempN = 1.0;
245  for (int i = 0; i < (int) m_coeffs.size() ; i++) {
246  m_prop += m_coeffs[i] * tempN;
247  tempN *= m_temp;
248  }
249  }
250  return m_prop;

285 doublereal LTPspecies_ExpT::getSpeciesTransProp()
286 {
287  doublereal t = m_thermo->temperature();
288  if (t != m_temp) {
289  m_temp=t;
290  m_prop = m_coeffs[0];
291  doublereal tempN = 1.0;
