222 void PDSS_ConstVol::setPressure(doublereal p)
223 {
224  m_pres = p;
225  doublereal del_pRT = (m_pres - m_p0) / (GasConstant * m_temp);
226  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] + del_pRT * m_Vss_ptr[m_spindex];
227  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
228 }

230 void PDSS_ConstVol::setTemperature(doublereal temp)
231 {
232  m_temp = temp;
233  m_spthermo->update_one(m_spindex, temp,
234  m_cp0_R_ptr, m_h0_RT_ptr, m_s0_R_ptr);
235  m_g0_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] - m_s0_R_ptr[m_spindex];
236 
237  doublereal del_pRT = (m_pres - m_p0) / (GasConstant * m_temp);
238 
239  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] + del_pRT * m_Vss_ptr[m_spindex];
240  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex];
241  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex];
242  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
