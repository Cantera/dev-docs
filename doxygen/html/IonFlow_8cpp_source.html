97 void IonFlow::frozenIonMethod(const double* x, size_t j0, size_t j1)
98 {
99  for (size_t j = j0; j < j1; j++) {
100  double wtm = m_wtm[j];
101  double rho = density(j);
102  double dz = z(j+1) - z(j);
103  double sum = 0.0;
104  for (size_t k : m_kNeutral) {
105  m_flux(k,j) = m_wt[k]*(rho*m_diff[k+m_nsp*j]/wtm);
106  m_flux(k,j) *= (X(x,k,j) - X(x,k,j+1))/dz;
107  sum -= m_flux(k,j);
108  }
109 
110  // correction flux to insure that \sum_k Y_k V_k = 0.
111  for (size_t k : m_kNeutral) {
112  m_flux(k,j) += sum*Y(x,k,j);
113  }
114 
115  // flux for ions
116  // Set flux to zero to prevent some fast charged species (e.g. electron)
117  // to run away
118  for (size_t k : m_kCharge) {
119  m_flux(k,j) = 0;
120  }
121  }

124 void IonFlow::chargeNeutralityModel(const double* x, size_t j0, size_t j1)
125 {
126  for (size_t j = j0; j < j1; j++) {
127  double wtm = m_wtm[j];
128  double rho = density(j);
129  double dz = z(j+1) - z(j);
130 
131  // mixture-average diffusion
132  for (size_t k = 0; k < m_nsp; k++) {
133  m_flux(k,j) = m_wt[k]*(rho*m_diff[k+m_nsp*j]/wtm);
134  m_flux(k,j) *= (X(x,k,j) - X(x,k,j+1))/dz;
135  }
136 
137  // ambipolar diffusion
138  double sum_chargeFlux = 0.0;
139  double sum = 0.0;
140  for (size_t k : m_kCharge) {
141  double Xav = 0.5 * (X(x,k,j+1) + X(x,k,j));
142  int q_k = m_speciesCharge[k];
143  sum_chargeFlux += m_speciesCharge[k] / m_wt[k] * m_flux(k,j);
144  // The mobility is used because it is more general than
145  // using diffusion coefficient and Einstein relation
146  sum += m_mobility[k+m_nsp*j] * Xav * q_k * q_k;
147  }
148  for (size_t k : m_kCharge) {
149  double Xav = 0.5 * (X(x,k,j+1) + X(x,k,j));
150  double drift;
151  int q_k = m_speciesCharge[k];
152  drift = q_k * q_k * m_mobility[k+m_nsp*j] * Xav / sum;
153  drift *= -sum_chargeFlux * m_wt[k] / q_k;
154  m_flux(k,j) += drift;
155  }
156 
157  // correction flux
158  double sum_flux = 0.0;
159  for (size_t k = 0; k < m_nsp; k++) {
160  sum_flux -= m_flux(k,j); // total net flux
161  }
162  double sum_ion = 0.0;
163  for (size_t k : m_kCharge) {
164  sum_ion += Y(x,k,j);
165  }
166  // The portion of correction for ions is taken off
167  for (size_t k : m_kNeutral) {
168  m_flux(k,j) += Y(x,k,j) / (1-sum_ion) * sum_flux;
169  }
170  }
171 }

173 void IonFlow::poissonEqnMethod(const double* x, size_t j0, size_t j1)
174 {
175  for (size_t j = j0; j < j1; j++) {
176  double wtm = m_wtm[j];
177  double rho = density(j);
178  double dz = z(j+1) - z(j);
179 
180  // mixture-average diffusion
181  double sum = 0.0;
182  for (size_t k = 0; k < m_nsp; k++) {
183  m_flux(k,j) = m_wt[k]*(rho*m_diff[k+m_nsp*j]/wtm);
184  m_flux(k,j) *= (X(x,k,j) - X(x,k,j+1))/dz;
185  sum -= m_flux(k,j);
186  }
187 
188  // ambipolar diffusion
189  double E_ambi = E(x,j);
190  for (size_t k : m_kCharge) {
191  double Yav = 0.5 * (Y(x,k,j) + Y(x,k,j+1));
192  double drift = rho * Yav * E_ambi
193  * m_speciesCharge[k] * m_mobility[k+m_nsp*j];
194  m_flux(k,j) += drift;
195  }
196 
197  // correction flux
198  double sum_flux = 0.0;
199  for (size_t k = 0; k < m_nsp; k++) {
200  sum_flux -= m_flux(k,j); // total net flux
201  }
202  double sum_ion = 0.0;
203  for (size_t k : m_kCharge) {
204  sum_ion += Y(x,k,j);
205  }
206  // The portion of correction for ions is taken off
207  for (size_t k : m_kNeutral) {
208  m_flux(k,j) += Y(x,k,j) / (1-sum_ion) * sum_flux;
209  }
210  }

233 void IonFlow::evalResidual(double* x, double* rsd, int* diag,
234  double rdt, size_t jmin, size_t jmax)
235 {
236  StFlow::evalResidual(x, rsd, diag, rdt, jmin, jmax);
237  if (m_stage != 3) {
238  return;
239  }
240 
241  for (size_t j = jmin; j <= jmax; j++) {
242  if (j == 0) {
243  rsd[index(c_offset_P, j)] = m_inletVoltage - phi(x,j);
244  diag[index(c_offset_P, j)] = 0;
245  } else if (j == m_points - 1) {
246  rsd[index(c_offset_P, j)] = m_outletVoltage - phi(x,j);
247  diag[index(c_offset_P, j)] = 0;
248  } else {
249  //-----------------------------------------------
250  // Poisson's equation
251  //
252  // dE/dz = e/eps_0 * sum(q_k*n_k)
253  //
254  // E = -dV/dz
255  //-----------------------------------------------
256  double chargeDensity = 0.0;
257  for (size_t k : m_kCharge) {
258  chargeDensity += m_speciesCharge[k] * ElectronCharge * ND(x,k,j);
259  }
260  rsd[index(c_offset_P, j)] = dEdz(x,j) - chargeDensity / epsilon_0;
261  diag[index(c_offset_P, j)] = 0;
262 
263  // This method is used when you disable energy equation
264  // but still maintain the velocity profile
265  if (!m_do_velocity[j]) {
266  rsd[index(c_offset_U, j)] = u(x,j) - u_fixed(j);
267  diag[index(c_offset_U, j)] = 0;
268  }
269  }
270  }
