139 // Static function to look up an atomic weight
140 /*
141  * This static function looks up the argument string in the
142  * database above and returns the associated molecular weight.
143  * The data are from the periodic table.
144  *
145  * Note: The idea behind this function is to provide a unified
146  * source for the element atomic weights. This helps to
147  * ensure that mass is conserved.
148  *
149  * @param s String, Only the first 3 characters are significant
150  *
151  * @return
152  * Return value contains the atomic weight of the element
153  * If a match for the string is not found, a value of -1.0 is
154  * returned.
155  *
156  * @exception CanteraError
157  * If a match is not found, a CanteraError is thrown as well
158  */
159 doublereal Elements::LookupWtElements(const std::string& ename)
160 {
161  int num = sizeof(aWTable) / sizeof(struct awData);
162  string s3 = ename.substr(0,3);
163  for (int i = 0; i < num; i++) {
164  if (s3 == aWTable[i].name) {
165  return aWTable[i].atomicWeight;
166  }
167  }
168  throw CanteraError("LookupWtElements", "element not found");
169  return -1.0;
170 }

392 /*
393  * addUniqueElement():
394  *
395  * Add a unique element to the set. This routine will not allow
396  * duplicate elements to be input.
397  *
398  * @param symbol symbol string
399  * @param weight atomic weight in kg/kmol.
400  *
401  *
402  * The default weight is a special value, which will cause the
403  * routine to look up the actual weight via a string lookup.
404  */
405 void Elements::addUniqueElement(const std::string& symbol, doublereal weight,
406  int atomicNumber_, doublereal entropy298,
407  int elem_type)
408 {
409  if (weight == -12345.0) {
410  weight = LookupWtElements(symbol);
411  if (weight < 0.0) {
412  throw ElementsFrozen("addElement");
413  }
414  }
415  /*
416  * First decide if this element has been previously added
417  * by conducting a string search. If it unique, add it to
418  * the list.
419  */
420  int ifound = 0;
421  int i = 0;
422  for (vector<string>::const_iterator it = m_elementNames.begin();
423  it < m_elementNames.end(); ++it, ++i) {
424  if (*it == symbol) {
425  ifound = 1;
426  break;
427  }
428  }
429  if (!ifound) {
430  if (m_elementsFrozen) {
431  throw ElementsFrozen("addElement");
432  return;
433  }
434  m_atomicWeights.push_back(weight);
435  m_elementNames.push_back(symbol);
436  m_atomicNumbers.push_back(atomicNumber_);
437  m_entropy298.push_back(entropy298);
438  if (symbol == "E") {
439  m_elem_type.push_back(CT_ELEM_TYPE_ELECTRONCHARGE);
440  } else {
441  m_elem_type.push_back(elem_type);
442  }
443  m_mm++;
444  } else {
445  if (m_atomicWeights[i] != weight) {
446  throw CanteraError("AddUniqueElement",
447  "Duplicate Elements (" + symbol + ") have different weights");
448  }
449  }
450 }
