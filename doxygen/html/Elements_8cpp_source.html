138 doublereal LookupWtElements(const std::string& ename)
139 {
140  int num = sizeof(aWTable) / sizeof(struct awData);
141  string s3 = ename.substr(0,3);
142  for (int i = 0; i < num; i++) {
143  if (s3 == aWTable[i].name) {
144  return aWTable[i].atomicWeight;
145  }
146  }
147  throw CanteraError("LookupWtElements", "element not found");
148  return -1.0;
149 }
