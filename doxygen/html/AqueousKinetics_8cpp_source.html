129 void AqueousKinetics::updateKc()
130 {
131  doublereal rt = GasConstant * m_temp;
132 
133  thermo().getStandardChemPotentials(&m_grt[0]);
134  fill(m_rkcn.begin(), m_rkcn.end(), 0.0);
135  for (size_t k = 0; k < thermo().nSpecies(); k++) {
136  doublereal logStandConc_k = thermo().logStandardConc(k);
137  m_grt[k] -= rt * logStandConc_k;
138  }
139 
140  // compute Delta G^0 for all reversible reactions
141  m_rxnstoich.getRevReactionDelta(m_ii, &m_grt[0], &m_rkcn[0]);
142 
143  //doublereal logStandConc = m_kdata->m_logStandConc;
144  doublereal rrt = 1.0/(GasConstant * thermo().temperature());
145  for (size_t i = 0; i < m_nrev; i++) {
146  size_t irxn = m_revindex[i];
147  m_rkcn[irxn] = exp(m_rkcn[irxn]*rrt);
148  }
149 
150  for (size_t i = 0; i != m_nirrev; ++i) {
151  m_rkcn[ m_irrev[i] ] = 0.0;

155 void AqueousKinetics::getEquilibriumConstants(doublereal* kc)
156 {
157  _update_rates_T();
158 
159  thermo().getStandardChemPotentials(&m_grt[0]);
160  fill(m_rkcn.begin(), m_rkcn.end(), 0.0);
161  doublereal rt = GasConstant * m_temp;
162  for (size_t k = 0; k < thermo().nSpecies(); k++) {
163  doublereal logStandConc_k = thermo().logStandardConc(k);
164  m_grt[k] -= rt * logStandConc_k;
165  }
166 
167  // compute Delta G^0 for all reactions
168  m_rxnstoich.getReactionDelta(m_ii, &m_grt[0], &m_rkcn[0]);
169 
170  doublereal rrt = 1.0/(GasConstant * thermo().temperature());
171  for (size_t i = 0; i < m_ii; i++) {
172  kc[i] = exp(-m_rkcn[i]*rrt);
173  }
174 
175  // force an update of T-dependent properties, so that m_rkcn will
176  // be updated before it is used next.
177  m_temp = 0.0;

385 void AqueousKinetics::installReagents(const ReactionData& r)
386 {
387 
388  m_ropf.push_back(0.0); // extend by one for new rxn
389  m_ropr.push_back(0.0);
390  m_ropnet.push_back(0.0);
391  size_t n, ns, m;
392  doublereal nsFlt;
393  doublereal reactantGlobalOrder = 0.0;
394  doublereal productGlobalOrder = 0.0;
395  size_t rnum = reactionNumber();
396 
397  std::vector<size_t> rk;
398  size_t nr = r.reactants.size();
399  for (n = 0; n < nr; n++) {
400  nsFlt = r.rstoich[n];
401  reactantGlobalOrder += nsFlt;
402  ns = (size_t) nsFlt;
403  if ((doublereal) ns != nsFlt) {
404  if (ns < 1) {
405  ns = 1;
406  }
407  }
408  if (r.rstoich[n] != 0.0) {
409  m_rrxn[r.reactants[n]][rnum] += r.rstoich[n];
410  }
411  for (m = 0; m < ns; m++) {
412  rk.push_back(r.reactants[n]);
413  }
414  }
415  m_reactants.push_back(rk);
416 
417  std::vector<size_t> pk;
418  size_t np = r.products.size();
419  for (n = 0; n < np; n++) {
420  nsFlt = r.pstoich[n];
421  productGlobalOrder += nsFlt;
422  ns = (size_t) nsFlt;
423  if ((double) ns != nsFlt) {
424  if (ns < 1) {
425  ns = 1;
426  }
427  }
428  if (r.pstoich[n] != 0.0) {
429  m_prxn[r.products[n]][rnum] += r.pstoich[n];
430  }
431  for (m = 0; m < ns; m++) {
432  pk.push_back(r.products[n]);
433  }
434  }
435  m_products.push_back(pk);
436 
437  m_rkcn.push_back(0.0);
438 
439  m_rxnstoich.add(reactionNumber(), r);
440 
441  if (r.reversible) {
442  m_dn.push_back(productGlobalOrder - reactantGlobalOrder);
443  m_revindex.push_back(reactionNumber());
444  m_nrev++;
445  } else {
446  m_dn.push_back(productGlobalOrder - reactantGlobalOrder);
447  m_irrev.push_back(reactionNumber());
448  m_nirrev++;
449  }
450 }
