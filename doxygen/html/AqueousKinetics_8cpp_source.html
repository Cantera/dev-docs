49 void AqueousKinetics::updateKc()
50 {
51  thermo().getStandardChemPotentials(m_grt.data());
52  fill(m_rkcn.begin(), m_rkcn.end(), 0.0);
53  for (size_t k = 0; k < thermo().nSpecies(); k++) {
54  doublereal logStandConc_k = thermo().logStandardConc(k);
55  m_grt[k] -= GasConstant * m_temp * logStandConc_k;
56  }
57 
58  // compute Delta G^0 for all reversible reactions
59  getRevReactionDelta(m_grt.data(), m_rkcn.data());
60 
61  doublereal rrt = 1.0/(GasConstant * thermo().temperature());
62  for (size_t i = 0; i < m_revindex.size(); i++) {
63  size_t irxn = m_revindex[i];
64  m_rkcn[irxn] = exp(m_rkcn[irxn]*rrt);
65  }
66 
67  for (size_t i = 0; i != m_irrev.size(); ++i) {
68  m_rkcn[ m_irrev[i] ] = 0.0;
69  }
70 }
