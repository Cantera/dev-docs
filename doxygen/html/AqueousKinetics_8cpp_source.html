123 void AqueousKinetics::updateKc()
124 {
125  doublereal rt = GasConstant * m_temp;
126 
127  thermo().getStandardChemPotentials(&m_grt[0]);
128  fill(m_rkcn.begin(), m_rkcn.end(), 0.0);
129  for (size_t k = 0; k < thermo().nSpecies(); k++) {
130  doublereal logStandConc_k = thermo().logStandardConc(k);
131  m_grt[k] -= rt * logStandConc_k;
132  }
133 
134  // compute Delta G^0 for all reversible reactions
135  m_rxnstoich.getRevReactionDelta(m_ii, &m_grt[0], &m_rkcn[0]);
136 
137  doublereal rrt = 1.0/(GasConstant * thermo().temperature());
138  for (size_t i = 0; i < m_nrev; i++) {
139  size_t irxn = m_revindex[i];
140  m_rkcn[irxn] = exp(m_rkcn[irxn]*rrt);
141  }
142 
143  for (size_t i = 0; i != m_nirrev; ++i) {
144  m_rkcn[ m_irrev[i] ] = 0.0;
145  }
146 }

372 void AqueousKinetics::installReagents(const ReactionData& r)
373 {
374 
375  m_ropf.push_back(0.0); // extend by one for new rxn
376  m_ropr.push_back(0.0);
377  m_ropnet.push_back(0.0);
378  size_t n, ns, m;
379  doublereal nsFlt;
380  doublereal reactantGlobalOrder = 0.0;
381  doublereal productGlobalOrder = 0.0;
382  size_t rnum = reactionNumber();
383 
384  std::vector<size_t> rk;
385  size_t nr = r.reactants.size();
386  for (n = 0; n < nr; n++) {
387  nsFlt = r.rstoich[n];
388  reactantGlobalOrder += nsFlt;
389  ns = (size_t) nsFlt;
390  if ((doublereal) ns != nsFlt) {
391  ns = std::max<size_t>(ns, 1);
392  }
393  if (r.rstoich[n] != 0.0) {
394  m_rrxn[r.reactants[n]][rnum] += r.rstoich[n];
395  }
396  for (m = 0; m < ns; m++) {
397  rk.push_back(r.reactants[n]);
398  }
399  }
400  m_reactants.push_back(rk);
401 
402  std::vector<size_t> pk;
403  size_t np = r.products.size();
404  for (n = 0; n < np; n++) {
405  nsFlt = r.pstoich[n];
406  productGlobalOrder += nsFlt;
407  ns = (size_t) nsFlt;
408  if ((double) ns != nsFlt) {
409  ns = std::max<size_t>(ns, 1);
410  }
411  if (r.pstoich[n] != 0.0) {
412  m_prxn[r.products[n]][rnum] += r.pstoich[n];
413  }
414  for (m = 0; m < ns; m++) {
415  pk.push_back(r.products[n]);
416  }
417  }
418  m_products.push_back(pk);
419 
420  m_rkcn.push_back(0.0);
421 
422  m_rxnstoich.add(reactionNumber(), r);
423 
424  if (r.reversible) {
425  m_dn.push_back(productGlobalOrder - reactantGlobalOrder);
426  m_revindex.push_back(reactionNumber());
427  m_nrev++;
428  } else {
429  m_dn.push_back(productGlobalOrder - reactantGlobalOrder);
430  m_irrev.push_back(reactionNumber());
431  m_nirrev++;
432  }
433 }
