36 size_t vcs_optMax(const double* x, const double* xSize, size_t j, size_t n)
37 {
38  size_t largest = j;
39  double big = x[j];
40  if (xSize) {
41  assert(xSize[j] > 0.0);
42  big *= xSize[j];
43  for (size_t i = j + 1; i < n; ++i) {
44  assert(xSize[i] > 0.0);
45  if ((x[i] * xSize[i]) > big) {
46  largest = i;
47  big = x[i] * xSize[i];
48  }
49  }
50  } else {
51  for (size_t i = j + 1; i < n; ++i) {
52  if (x[i] > big) {
53  largest = i;
54  big = x[i];
55  }
56  }
57  }
58  return largest;
