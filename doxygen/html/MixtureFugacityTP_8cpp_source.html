376 void MixtureFugacityTP::setState_TP(doublereal t, doublereal pres)
377 {
378  /*
379  * A pretty tricky algorithm is needed here, due to problems involving
380  * standard states of real fluids. For those cases you need
381  * to combine the T and P specification for the standard state, or else
382  * you may venture into the forbidden zone, especially when nearing the
383  * triple point.
384  * Therefore, we need to do the standard state thermo calc with the
385  * (t, pres) combo.
386  */
387  getMoleFractions(DATA_PTR(moleFractions_));
388 
389 
390  Phase::setTemperature(t);
391  _updateReferenceStateThermo();
392  // Depends on the mole fractions and the temperature
393  updateMixingExpressions();
394  // setPressure(pres);
395  m_Pcurrent = pres;
396  // double mmw = meanMolecularWeight();
397 
398  if (forcedState_ == FLUID_UNDEFINED) {
399  double rhoNow = Phase::density();
400  double rho = densityCalc(t, pres, iState_, rhoNow);
401  if (rho > 0.0) {
402  Phase::setDensity(rho);
403  m_Pcurrent = pres;
404  iState_ = phaseState(true);
405  } else {
406  if (rho < -1.5) {
407  rho = densityCalc(t, pres, FLUID_UNDEFINED , rhoNow);
408  if (rho > 0.0) {
409  Phase::setDensity(rho);
410  m_Pcurrent = pres;
411  iState_ = phaseState(true);
412  } else {
413  throw CanteraError("MixtureFugacityTP::setState_TP()", "neg rho");
414  }
415  } else {
416  throw CanteraError("MixtureFugacityTP::setState_TP()", "neg rho");
417  }
418  }
419 
420 
421 
422  } else if (forcedState_ == FLUID_GAS) {
423  // Normal density calculation
424  if (iState_ < FLUID_LIQUID_0) {
425  double rhoNow = Phase::density();
426  double rho = densityCalc(t, pres, iState_, rhoNow);
427  if (rho > 0.0) {
428  Phase::setDensity(rho);
429  m_Pcurrent = pres;
430  iState_ = phaseState(true);
431  if (iState_ >= FLUID_LIQUID_0) {
432  throw CanteraError("MixtureFugacityTP::setState_TP()", "wrong state");
433  }
434  } else {
435  throw CanteraError("MixtureFugacityTP::setState_TP()", "neg rho");
436  }
437 
438  }
439 
440 
441  } else if (forcedState_ > FLUID_LIQUID_0) {
442  if (iState_ >= FLUID_LIQUID_0) {
443  double rhoNow = Phase::density();
444  double rho = densityCalc(t, pres, iState_, rhoNow);
445  if (rho > 0.0) {
446  Phase::setDensity(rho);
447  m_Pcurrent = pres;
448  iState_ = phaseState(true);
449  if (iState_ == FLUID_GAS) {
450  throw CanteraError("MixtureFugacityTP::setState_TP()", "wrong state");
451  }
452  } else {
453  throw CanteraError("MixtureFugacityTP::setState_TP()", "neg rho");
454  }
455 
456  }
457  }

538 doublereal MixtureFugacityTP::densityCalc(doublereal TKelvin, doublereal presPa,
539  int phase, doublereal rhoguess)
540 {
541  double tcrit = critTemperature();
542  doublereal mmw = meanMolecularWeight();
543  // double pcrit = critPressure();
544  // doublereal deltaGuess = 0.0;
545  if (rhoguess == -1.0) {
546  if (phase != -1) {
547  if (TKelvin > tcrit) {
548  rhoguess = presPa * mmw / (GasConstant * TKelvin);
549  } else {
550  if (phase == FLUID_GAS || phase == FLUID_SUPERCRIT) {
551  rhoguess = presPa * mmw / (GasConstant * TKelvin);
552  } else if (phase >= FLUID_LIQUID_0) {
553  double lqvol = liquidVolEst(TKelvin, presPa);
554  rhoguess = mmw / lqvol;
555  }
556  }
557  } else {
558  /*
559  * Assume the Gas phase initial guess, if nothing is
560  * specified to the routine
561  */
562  rhoguess = presPa * mmw / (GasConstant * TKelvin);
563  }
564 
565  }
566 
567  double molarVolBase = mmw / rhoguess;
568  double molarVolLast = molarVolBase;
569  double vc = mmw / critDensity();
570  /*
571  * molar volume of the spinodal at the current temperature and mole fractions. this will
572  * be updated as we go.
573  */
574  double molarVolSpinodal = vc;
575  doublereal pcheck = 1.0E-30 + 1.0E-8 * presPa;
576  doublereal presBase, dpdVBase, delMV;
577  bool conv = false;
578  /*
579  * We start on one side of the vc and stick with that side
580  */
581  bool gasSide = molarVolBase > vc;
582  if (gasSide) {
583  molarVolLast = (GasConstant * TKelvin)/presPa;
584  } else {
585  molarVolLast = liquidVolEst(TKelvin, presPa);
586  }
587 
588  /*
589  * OK, now we do a small solve to calculate the molar volume given the T,P value.
590  * The algorithm is taken from dfind()
591  */
592  for (int n = 0; n < 200; n++) {
593 
594  /*
595  * Calculate the predicted reduced pressure, pred0, based on the
596  * current tau and dd.
597  */
598 
599  /*
600  * Calculate the derivative of the predicted pressure
601  * wrt the molar volume.
602  * This routine also returns the pressure, presBase
603  */
604  dpdVBase = dpdVCalc(TKelvin, molarVolBase, presBase);
605 
606  /*
607  * If dpdV is positve, then we are in the middle of the
608  * 2 phase region and beyond the spinodal stability curve. We need to adjust
609  * the initial guess outwards and start a new iteration.
610  */
611  if (dpdVBase >= 0.0) {
612  if (TKelvin > tcrit) {
613  throw CanteraError("", "confused");
614  }
615  /*
616  * TODO Spawn a calculation for the value of the spinodal point that is
617  * very accurate. Answer the question as to wethera solution is
618  * possible on the current side of the vapor dome.
619  */
620  if (gasSide) {
621  if (molarVolBase >= vc) {
622  molarVolSpinodal = molarVolBase;
623  molarVolBase = 0.5 * (molarVolLast + molarVolSpinodal);
624  } else {
625  molarVolBase = 0.5 * (molarVolLast + molarVolSpinodal);
626  }
627  } else {
628  if (molarVolBase <= vc) {
629  molarVolSpinodal = molarVolBase;
630  molarVolBase = 0.5 * (molarVolLast + molarVolSpinodal);
631  } else {
632  molarVolBase = 0.5 * (molarVolLast + molarVolSpinodal);
633  }
634  }
635  continue;
636  }
637 
638  /*
639  * Check for convergence
640  */
641  if (fabs(presBase-presPa) < pcheck) {
642  conv = true;
643  break;
644  }
645 
646  /*
647  * Dampen and crop the update
648  */
649  doublereal dpdV = dpdVBase;
650  if (n < 10) {
651  dpdV = dpdVBase * 1.5;
652  }
653  // if (dpdV > -0.001) dpdV = -0.001;
654 
655  /*
656  * Formulate the update to the molar volume by
657  * Newton's method. Then, crop it to a max value
658  * of 0.1 times the current volume
659  */
660  delMV = - (presBase - presPa) / dpdV;
661  if (!gasSide || delMV < 0.0) {
662  if (fabs(delMV) > 0.2 * molarVolBase) {
663  delMV = delMV / fabs(delMV) * 0.2 * molarVolBase;
664  }
665  }
666  /*
667  * Only go 1/10 the way towards the spinodal at any one time.
668  */
669  if (TKelvin < tcrit) {
670  if (gasSide) {
671  if (delMV < 0.0) {
672  if (-delMV > 0.5 * (molarVolBase - molarVolSpinodal)) {
673  delMV = - 0.5 * (molarVolBase - molarVolSpinodal);
674  }
675  }
676  } else {
677  if (delMV > 0.0) {
678  if (delMV > 0.5 * (molarVolSpinodal - molarVolBase)) {
679  delMV = 0.5 * (molarVolSpinodal - molarVolBase);
680  }
681  }
682  }
683  }
684  /*
685  * updated the molar volume value
686  */
687  molarVolLast = molarVolBase;
688  molarVolBase += delMV;
689 
690 
691  if (fabs(delMV/molarVolBase) < 1.0E-14) {
692  conv = true;
693  break;
694  }
695 
696  /*
697  * Check for negative molar volumes
698  */
699  if (molarVolBase <= 0.0) {
700  molarVolBase = std::min(1.0E-30, fabs(delMV*1.0E-4));
701  }
702 
703  }
704 
705 
706  /*
707  * Check for convergence, and return 0.0 if it wasn't achieved.
708  */
709  double densBase = 0.0;
710  if (! conv) {
711  molarVolBase = 0.0;
712  throw CanteraError("MixtureFugacityTP::densityCalc()", "Process didnot converge");
713  } else {
714  densBase = mmw / molarVolBase;
715  }
716  return densBase;

831 doublereal MixtureFugacityTP::calculatePsat(doublereal TKelvin, doublereal& molarVolGas,
832  doublereal& molarVolLiquid)
833 {
834  /*
835  * The algorithm for this routine has undergone quite a bit of work. It probably needs more work.
836  * However, it seems now to be fairly robust.
837  * The key requirement is to find an initial pressure where both the liquid and the gas exist. This
838  * is not as easy as it sounds, and it gets exceedingly hard as the critical temperature is approached
839  * from below.
840  * Once we have this initial state, then we seek to equilibrate the gibbs free energies of the
841  * gas and liquid and use the formula
842  *
843  * dp = VdG
844  *
845  * to create an update condition for deltaP using
846  *
847  * - (Gliq - Ggas) = (Vliq - Vgas) (deltaP)
848  *
849  * @TODO Suggestions for the future would be to switch it to an algorithm that uses the gas molar volume
850  * and the liquid molar volumes as the fundamental unknowns.
851  */
852 
853  // we need this because this is a non-const routine that is public
854  setTemperature(TKelvin);
855  double tcrit = critTemperature();
856  double RhoLiquid, RhoGas;
857  double RhoLiquidGood, RhoGasGood;
858  double densSave = density();
859  double tempSave = temperature();
860  double pres;
861  doublereal mw = meanMolecularWeight();
862  if (TKelvin < tcrit) {
863 
864  pres = psatEst(TKelvin);
865  // trial value = Psat from correlation
866  doublereal volLiquid = liquidVolEst(TKelvin, pres);
867  RhoLiquidGood = mw / volLiquid;
868  RhoGasGood = pres * mw / (GasConstant * TKelvin);
869  doublereal delGRT = 1.0E6;
870  doublereal liqGRT, gasGRT;
871  int stab;
872  doublereal presLast = pres;
873 
874  /*
875  * First part of the calculation involves finding a pressure at which the
876  * gas and the liquid state coexists.
877  */
878  doublereal presLiquid = 0.;
879  doublereal presGas;
880  doublereal presBase = pres;
881  bool foundLiquid = false;
882  bool foundGas = false;
883 
884  doublereal densLiquid = densityCalc(TKelvin, presBase, FLUID_LIQUID_0, RhoLiquidGood);
885  if (densLiquid > 0.0) {
886  foundLiquid = true;
887  presLiquid = pres;
888  RhoLiquidGood = densLiquid;
889  }
890  if (!foundLiquid) {
891  for (int i = 0; i < 50; i++) {
892  pres = 1.1 * pres;
893  densLiquid = densityCalc(TKelvin, pres, FLUID_LIQUID_0, RhoLiquidGood);
894  if (densLiquid > 0.0) {
895  foundLiquid = true;
896  presLiquid = pres;
897  RhoLiquidGood = densLiquid;
898  break;
899  }
900  }
901  }
902 
903  pres = presBase;
904  doublereal densGas = densityCalc(TKelvin, pres, FLUID_GAS, RhoGasGood);
905  if (densGas <= 0.0) {
906  foundGas = false;
907  } else {
908  foundGas = true;
909  presGas = pres;
910  RhoGasGood = densGas;
911  }
912  if (!foundGas) {
913  for (int i = 0; i < 50; i++) {
914  pres = 0.9 * pres;
915  densGas = densityCalc(TKelvin, pres, FLUID_GAS, RhoGasGood);
916  if (densGas > 0.0) {
917  foundGas = true;
918  presGas = pres;
919  RhoGasGood = densGas;
920  break;
921  }
922  }
923  }
924 
925  if (foundGas && foundLiquid) {
926  if (presGas != presLiquid) {
927  pres = 0.5 * (presLiquid + presGas);
928  bool goodLiq;
929  bool goodGas;
930  for (int i = 0; i < 50; i++) {
931  densLiquid = densityCalc(TKelvin, pres, FLUID_LIQUID_0, RhoLiquidGood);
932  if (densLiquid <= 0.0) {
933  goodLiq = false;
934  } else {
935  goodLiq = true;
936  RhoLiquidGood = densLiquid;
937  presLiquid = pres;
938  }
939  densGas = densityCalc(TKelvin, pres, FLUID_GAS, RhoGasGood);
940  if (densGas <= 0.0) {
941  goodGas = false;
942  } else {
943  goodGas = true;
944  RhoGasGood = densGas;
945  presGas = pres;
946  }
947  if (goodGas && goodLiq) {
948  break;
949  }
950  if (!goodLiq && !goodGas) {
951  pres = 0.5 * (pres + presLiquid);
952  }
953  if (goodLiq || goodGas) {
954  pres = 0.5 * (presLiquid + presGas);
955  }
956  }
957  }
958  }
959  if (!foundGas || !foundLiquid) {
960  printf("error coundn't find a starting pressure\n");
961  return 0.0;
962  }
963  if (presGas != presLiquid) {
964  printf("error coundn't find a starting pressure\n");
965  return 0.0;
966  }
967 
968  pres = presGas;
969  presLast = pres;
970  RhoGas = RhoGasGood;
971  RhoLiquid = RhoLiquidGood;
972 
973 
974  /*
975  * Now that we have found a good pressure we can proceed with the algorithm.
976  */
977 
978  for (int i = 0; i < 20; i++) {
979 
980  stab = corr0(TKelvin, pres, RhoLiquid, RhoGas, liqGRT, gasGRT);
981  if (stab == 0) {
982  presLast = pres;
983  delGRT = liqGRT - gasGRT;
984  doublereal delV = mw * (1.0/RhoLiquid - 1.0/RhoGas);
985  doublereal dp = - delGRT * GasConstant * TKelvin / delV;
986 
987  if (fabs(dp) > 0.1 * pres) {
988  if (dp > 0.0) {
989  dp = 0.1 * pres;
990  } else {
991  dp = -0.1 * pres;
992  }
993  }
994  pres += dp;
995 
996  } else if (stab == -1) {
997  delGRT = 1.0E6;
998  if (presLast > pres) {
999  pres = 0.5 * (presLast + pres);
1000  } else {
1001  // we are stuck here - try this
1002  pres = 1.1 * pres;
1003  }
1004  } else if (stab == -2) {
1005  if (presLast < pres) {
1006  pres = 0.5 * (presLast + pres);
1007  } else {
1008  // we are stuck here - try this
1009  pres = 0.9 * pres;
1010  }
1011  }
1012  molarVolGas = mw / RhoGas;
1013  molarVolLiquid = mw / RhoLiquid;
1014 
1015 
1016  if (fabs(delGRT) < 1.0E-8) {
1017  // converged
1018  break;
1019  }
1020  }
1021 
1022  molarVolGas = mw / RhoGas;
1023  molarVolLiquid = mw / RhoLiquid;
1024  // Put the fluid in the desired end condition
1025  setState_TR(tempSave, densSave);
1026 
1027  return pres;
1028 
1029 
1030  } else {
1031  pres = critPressure();
1032  setState_TP(TKelvin, pres);
1033  RhoGas = density();
1034  molarVolGas = mw / RhoGas;
1035  molarVolLiquid = molarVolGas;
1036  setState_TR(tempSave, densSave);
1037  }
1038  return pres;
1039 }
