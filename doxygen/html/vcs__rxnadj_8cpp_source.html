495  if (m_phaseID[k] == m_phaseID[j]) {
496  double np = m_tPhaseMoles_old[m_phaseID[k]];
497  if (np > 0.0) {
498  s += sc_irxn[k] * sc_irxn[j] * m_np_dLnActCoeffdMolNum(j,k) / np;
499  }
500  }
501  }
502  if (kph == m_phaseID[j]) {
503  s += sc_irxn[j] * (m_np_dLnActCoeffdMolNum(j,kspec) + m_np_dLnActCoeffdMolNum(kspec,j)) / np_kspec;
504  }
505  }
506  }

544 double VCS_SOLVE::vcs_line_search(const size_t irxn, const double dx_orig, char* const ANOTE)
545 {
546  int its = 0;
547  size_t kspec = m_indexRxnToSpecies[irxn];
548  const int MAXITS = 10;
549  double dx = dx_orig;
550  double* sc_irxn = m_stoichCoeffRxnMatrix.ptrColumn(irxn);
551  vector_fp& molNumBase = m_molNumSpecies_old;
552  vector_fp& acBase = m_actCoeffSpecies_old;
553  vector_fp& ac = m_actCoeffSpecies_new;
554 
555  // Calculate the deltaG value at the dx = 0.0 point
556  vcs_setFlagsVolPhases(false, VCS_STATECALC_OLD);
557  double deltaGOrig = deltaG_Recalc_Rxn(VCS_STATECALC_OLD, irxn, &molNumBase[0], &acBase[0], &m_feSpecies_old[0]);
558  double forig = fabs(deltaGOrig) + 1.0E-15;
559  if (deltaGOrig > 0.0) {
560  if (dx_orig > 0.0) {
561  dx = 0.0;
562  if (ANOTE) {
563  sprintf(ANOTE, "Rxn reduced to zero step size in line search: dx>0 dg > 0");
564  }
565  return dx;
566  }
567  } else if (deltaGOrig < 0.0) {
568  if (dx_orig < 0.0) {
569  dx = 0.0;
570  if (ANOTE) {
571  sprintf(ANOTE, "Rxn reduced to zero step size in line search: dx<0 dg < 0");
572  }
573  return dx;
574  }
575  } else if (deltaGOrig == 0.0) {
576  return 0.0;
577  }
578  if (dx_orig == 0.0) {
579  return 0.0;
580  }
581 
582  m_molNumSpecies_new = m_molNumSpecies_old;
583  double molSum = molNumBase[kspec];
584  m_molNumSpecies_new[kspec] = molNumBase[kspec] + dx_orig;
585  for (size_t k = 0; k < m_numComponents; k++) {
586  m_molNumSpecies_new[k] = molNumBase[k] + sc_irxn[k] * dx_orig;
587  molSum += molNumBase[k];
588  }
589  vcs_setFlagsVolPhases(false, VCS_STATECALC_NEW);
590 
591  double deltaG1 = deltaG_Recalc_Rxn(VCS_STATECALC_NEW, irxn, &m_molNumSpecies_new[0],
592  &ac[0], &m_feSpecies_new[0]);
593 
594  // If deltaG hasn't switched signs when going the full distance then we are
595  // heading in the appropriate direction, and we should accept the current
596  // full step size
597  if (deltaG1 * deltaGOrig > 0.0) {
598  dx = dx_orig;
599  goto finalize;
600  }
601 
602  // If we have decreased somewhat, the deltaG return after finding a better
603  // estimate for the line search.
604  if (fabs(deltaG1) < 0.8 * forig) {
605  if (deltaG1 * deltaGOrig < 0.0) {
606  double slope = (deltaG1 - deltaGOrig) / dx_orig;
607  dx = -deltaGOrig / slope;
608  } else {
609  dx = dx_orig;
610  }
611  goto finalize;
612  }
613 
614  dx = dx_orig;
615  for (its = 0; its < MAXITS; its++) {
616  // Calculate the approximation to the total Gibbs free energy at
617  // the dx *= 0.5 point
618  dx *= 0.5;
619  m_molNumSpecies_new[kspec] = molNumBase[kspec] + dx;
620  for (size_t k = 0; k < m_numComponents; k++) {
621  m_molNumSpecies_new[k] = molNumBase[k] + sc_irxn[k] * dx;
622  }
623  vcs_setFlagsVolPhases(false, VCS_STATECALC_NEW);
624  double deltaG = deltaG_Recalc_Rxn(VCS_STATECALC_NEW, irxn, &m_molNumSpecies_new[0],
625  &ac[0], &m_feSpecies_new[0]);
626 
627  // If deltaG hasn't switched signs when going the full distance then we
628  // are heading in the appropriate direction, and we should accept the
629  // current step
630  if (deltaG * deltaGOrig > 0.0) {
631  goto finalize;
632  }
633 
634  // If we have decreased somewhat, the deltaG return after finding
635  // a better estimate for the line search.
636  if (fabs(deltaG) / forig < (1.0 - 0.1 * dx / dx_orig)) {
637  if (deltaG * deltaGOrig < 0.0) {
638  double slope = (deltaG - deltaGOrig) / dx;
639  dx = -deltaGOrig / slope;
640  }
641  goto finalize;
642  }
643  }
644 
645 finalize:
646  vcs_setFlagsVolPhases(false, VCS_STATECALC_NEW);
647  if (its >= MAXITS) {
648  sprintf(ANOTE, "Rxn reduced to zero step size from %g to %g (MAXITS)", dx_orig, dx);
649  return dx;
650  }
651  if (dx != dx_orig) {
652  sprintf(ANOTE, "Line Search reduced step size from %g to %g", dx_orig, dx);
653  }
654  return dx;
655 }
