119 void IdealSolnGasVPSS::calcDensity()
120 {
121  /*
122  * Calculate the molarVolume of the solution (m**3 kmol-1)
123  */
124  if (m_idealGas) {
125  double dens = (m_Pcurrent * meanMolecularWeight()
126  /(GasConstant * temperature()));
127  Phase::setDensity(dens);
128  } else {
129  const doublereal* const dtmp = moleFractdivMMW();
130  const vector_fp& vss = m_VPSS_ptr->getStandardVolumes();
131  double dens = 1.0 / dot(vss.begin(), vss.end(), dtmp);
132  /*
133  * Set the density in the parent State object directly,
134  * by calling the Phase::setDensity() function.
135  */
136  Phase::setDensity(dens);
137  }
138 }

152 {

263 void IdealSolnGasVPSS::setToEquilState(const doublereal* mu_RT)
264 {
265  updateStandardStateThermo();
266  const vector_fp& grt = m_VPSS_ptr->Gibbs_RT_ref();
267 
268  /*
269  * Within the method, we protect against inf results if the
270  * exponent is too high.
271  *
272  * If it is too low, we set
273  * the partial pressure to zero. This capability is needed
274  * by the elemental potential method.
275  */
276  doublereal pres = 0.0;
277  double m_p0 = m_VPSS_ptr->refPressure();
278  for (size_t k = 0; k < m_kk; k++) {
279  double tmp = -grt[k] + mu_RT[k];
280  if (tmp < -600.) {
281  m_pp[k] = 0.0;
282  } else if (tmp > 500.0) {
283  double tmp2 = tmp / 500.;
284  tmp2 *= tmp2;
285  m_pp[k] = m_p0 * exp(500.) * tmp2;
286  } else {
287  m_pp[k] = m_p0 * exp(tmp);
288  }
289  pres += m_pp[k];
290  }
291  // set state
292  setState_PX(pres, &m_pp[0]);
293 }

300 void IdealSolnGasVPSS::initThermoXML(XML_Node& phaseNode, const std::string& id_)
301 {
302  IdealSolnGasVPSS::initLengths();
303 
304  if (phaseNode.hasChild("thermo")) {
305  XML_Node& thermoNode = phaseNode.child("thermo");
306  std::string model = thermoNode["model"];
307  if (model == "IdealGasVPSS") {
308  m_idealGas = 1;
309  } else if (model == "IdealSolnVPSS") {
310  m_idealGas = 0;
311  } else {
312  throw CanteraError("IdealSolnGasVPSS::initThermoXML",
313  "Unknown thermo model : " + model);
314  }
315  }
316 
317  /*
318  * Form of the standard concentrations. Must have one of:
319  *
320  * <standardConc model="unity" />
321  * <standardConc model="molar_volume" />
322  * <standardConc model="solvent_volume" />
323  */
324  if (phaseNode.hasChild("standardConc")) {
325  if (m_idealGas) {
326  throw CanteraError("IdealSolnGasVPSS::initThermoXML",
327  "standardConc node for ideal gas");
328  }
329  XML_Node& scNode = phaseNode.child("standardConc");
330  string formStringa = scNode.attrib("model");
331  string formString = lowercase(formStringa);
332  if (formString == "unity") {
333  m_formGC = 0;
334  } else if (formString == "molar_volume") {
335  m_formGC = 1;
336  } else if (formString == "solvent_volume") {
337  m_formGC = 2;
338  } else {
339  throw CanteraError("initThermoXML",
340  "Unknown standardConc model: " + formStringa);
341  }
342  } else {
343  if (!m_idealGas) {
344  throw CanteraError("initThermoXML",
345  "Unspecified standardConc model");
346  }
347  }
348 
349  VPStandardStateTP::initThermoXML(phaseNode, id_);
350 }
