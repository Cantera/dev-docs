127 void IdealSolnGasVPSS::calcDensity()
128 {
129  /*
130  * Calculate the molarVolume of the solution (m**3 kmol-1)
131  */
132  if (m_idealGas) {
133  double dens = (m_Pcurrent * meanMolecularWeight()
134  /(GasConstant * temperature()));
135  Phase::setDensity(dens);
136  } else {
137  const doublereal* const dtmp = moleFractdivMMW();
138  const vector_fp& vss = m_VPSS_ptr->getStandardVolumes();
139  double invDens = dot(vss.begin(), vss.end(), dtmp);
140  /*
141  * Set the density in the parent State object directly,
142  * by calling the Phase::setDensity() function.
143  */
144  double dens = 1.0/invDens;
145  Phase::setDensity(dens);
146  }
147 }

311 void IdealSolnGasVPSS::setToEquilState(const doublereal* mu_RT)
312 {
313  double tmp, tmp2;
314  updateStandardStateThermo();
315  const vector_fp& grt = m_VPSS_ptr->Gibbs_RT_ref();
316 
317  /*
318  * Within the method, we protect against inf results if the
319  * exponent is too high.
320  *
321  * If it is too low, we set
322  * the partial pressure to zero. This capability is needed
323  * by the elemental potential method.
324  */
325  doublereal pres = 0.0;
326  double m_p0 = m_VPSS_ptr->refPressure();
327  for (size_t k = 0; k < m_kk; k++) {
328  tmp = -grt[k] + mu_RT[k];
329  if (tmp < -600.) {
330  m_pp[k] = 0.0;
331  } else if (tmp > 500.0) {
332  tmp2 = tmp / 500.;
333  tmp2 *= tmp2;
334  m_pp[k] = m_p0 * exp(500.) * tmp2;
335  } else {
336  m_pp[k] = m_p0 * exp(tmp);
337  }
338  pres += m_pp[k];
339  }
340  // set state
341  setState_PX(pres, &m_pp[0]);
342 }

350 void IdealSolnGasVPSS::initThermoXML(XML_Node& phaseNode, const std::string& id_)
351 {
352  IdealSolnGasVPSS::initLengths();
353 
354  if (phaseNode.hasChild("thermo")) {
355  XML_Node& thermoNode = phaseNode.child("thermo");
356  std::string model = thermoNode["model"];
357  if (model == "IdealGasVPSS") {
358  m_idealGas = 1;
359  } else if (model == "IdealSolnVPSS") {
360  m_idealGas = 0;
361  } else {
362  throw CanteraError("IdealSolnGasVPSS::initThermoXML",
363  "Unknown thermo model : " + model);
364  }
365  }
366 
367  /*
368  * Form of the standard concentrations. Must have one of:
369  *
370  * <standardConc model="unity" />
371  * <standardConc model="molar_volume" />
372  * <standardConc model="solvent_volume" />
373  */
374  if (phaseNode.hasChild("standardConc")) {
375  if (m_idealGas) {
376  throw CanteraError("IdealSolnGasVPSS::initThermoXML",
377  "standardConc node for ideal gas");
378  }
379  XML_Node& scNode = phaseNode.child("standardConc");
380  string formStringa = scNode.attrib("model");
381  string formString = lowercase(formStringa);
382  if (formString == "unity") {
383  m_formGC = 0;
384  } else if (formString == "molar_volume") {
385  m_formGC = 1;
386  } else if (formString == "solvent_volume") {
387  m_formGC = 2;
388  } else {
389  throw CanteraError("initThermoXML",
390  "Unknown standardConc model: " + formStringa);
391  }
392  } else {
393  if (!m_idealGas) {
394  throw CanteraError("initThermoXML",
395  "Unspecified standardConc model");
396  }
397  }
398 
399  VPStandardStateTP::initThermoXML(phaseNode, id_);
400 }
