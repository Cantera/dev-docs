96 void BulkKinetics::getRevRateConstants(doublereal* krev, bool doIrreversible)
97 {

99  * go get the forward rate constants. -> note, we don't
100  * really care about speed or redundancy in these

118 bool BulkKinetics::addReaction(shared_ptr<Reaction> r)
119 {
120  bool added = Kinetics::addReaction(r);
121  if (!added) {
122  return false;
123  }
124  double dn = 0.0;
125  for (Composition::const_iterator iter = r->products.begin();
126  iter != r->products.end();
127  ++iter) {
128  dn += iter->second;
129  }
130  for (Composition::const_iterator iter = r->reactants.begin();
131  iter != r->reactants.end();
132  ++iter) {
133  dn -= iter->second;
134  }
135 
136  m_dn.push_back(dn);
137 
138  if (r->reversible) {
139  m_revindex.push_back(nReactions()-1);
140  } else {
141  m_irrev.push_back(nReactions()-1);
142  }
143  return true;
144 }
