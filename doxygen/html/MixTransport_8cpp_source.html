117 // Returns the mixture thermal conductivity (W/m /K)
118 /*
119  * The thermal conductivity is computed from the following mixture rule:
120  * \f[
121  * \lambda = 0.5 \left( \sum_k X_k \lambda_k + \frac{1}{\sum_k X_k/\lambda_k} \right)
122  * \f]
123  *
124  * It's used to compute the flux of energy due to a thermal gradient
125  *
126  * \f[
127  * j_T = - \lambda \nabla T
128  * \f]
129  *
130  * The flux of energy has units of energy (kg m2 /s2) per second per area.
131  *
132  * The units of lambda are W / m K which is equivalent to kg m / s^3 K.
133  *
134  * @return Returns the mixture thermal conductivity, with units of W/m/K
135  */
136 doublereal MixTransport::thermalConductivity()
137 {
138  update_T();
139  update_C();
140 
141  if (!m_spcond_ok) {
142  updateCond_T();
143  }
144  if (!m_condmix_ok) {
145  doublereal sum1 = 0.0, sum2 = 0.0;
146  for (size_t k = 0; k < m_nsp; k++) {
147  sum1 += m_molefracs[k] * m_cond[k];
148  sum2 += m_molefracs[k] / m_cond[k];
149  }
150  m_lambda = 0.5*(sum1 + 1.0/sum2);
151  m_condmix_ok = true;
152  }
153  return m_lambda;
154 }

171 // Get the species diffusive mass fluxes wrt to the mass averaged velocity,
172 // given the gradients in mole fraction and temperature
173 /*
174  * Units for the returned fluxes are kg m-2 s-1.
175  *
176  *
177  * The diffusive mass flux of species \e k is computed from
178  * \f[
179  * \vec{j}_k = -n M_k D_k \nabla X_k.
180  * \f]
181  *
182  * @param ndim Number of dimensions in the flux expressions
183  * @param grad_T Gradient of the temperature
184  * (length = ndim)
185  * @param ldx Leading dimension of the grad_X array
186  * (usually equal to m_nsp but not always)
187  * @param grad_X Gradients of the mole fraction
188  * Flat vector with the m_nsp in the inner loop.
189  * length = ldx * ndim
190  * @param ldf Leading dimension of the fluxes array
191  * (usually equal to m_nsp but not always)
192  * @param fluxes Output of the diffusive mass fluxes
193  * Flat vector with the m_nsp in the inner loop.
194  * length = ldx * ndim
195  */
196 void MixTransport::getSpeciesFluxes(size_t ndim, const doublereal* const grad_T,
197  size_t ldx, const doublereal* const grad_X,
198  size_t ldf, doublereal* const fluxes)
199 {
200  update_T();
201  update_C();
202 
203  getMixDiffCoeffs(DATA_PTR(m_spwork));
204 
205  const vector_fp& mw = m_thermo->molecularWeights();
206  const doublereal* y = m_thermo->massFractions();
207  doublereal rhon = m_thermo->molarDensity();
208 
209  vector_fp sum(ndim,0.0);
210  for (size_t n = 0; n < ndim; n++) {
211  for (size_t k = 0; k < m_nsp; k++) {
212  fluxes[n*ldf + k] = -rhon * mw[k] * m_spwork[k] * grad_X[n*ldx + k];
213  sum[n] += fluxes[n*ldf + k];
214  }
215  }
216  // add correction flux to enforce sum to zero
217  for (size_t n = 0; n < ndim; n++) {
218  for (size_t k = 0; k < m_nsp; k++) {
219  fluxes[n*ldf + k] -= y[k]*sum[n];
220  }
221  }
222 }

273 void MixTransport::updateCond_T()
274 {
275  if (m_mode == CK_Mode) {
276  for (size_t k = 0; k < m_nsp; k++) {
277  m_cond[k] = exp(dot4(m_polytempvec, m_condcoeffs[k]));
278  }
279  } else {
280  for (size_t k = 0; k < m_nsp; k++) {
281  m_cond[k] = m_sqrt_t * dot5(m_polytempvec, m_condcoeffs[k]);
