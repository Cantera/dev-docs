116  x = -x;
117  }
118  double y = m_gamma;
119 
120  double c3 = log(Pi*m_eps/2.0);
121  double tau = sqrt(-log(y) - c3);
122  double b = (tau + x)/y;
123  double t = b*y;
124  double f1, f2, f3;
125  const double c0 = 2.0/(Pi*exp(0.0));
126  const double c1 = 1.0/SqrtTwo;
127  const double c2 = 2.0/SqrtPi;
128 
129  if (y > c0/m_eps) {
130  return 0.0;
131  }
132  double f0, ef0;
133  while (1 > 0) {
134  f0 = Pi*Pi/(t*t);
135  ef0 = exp(-f0);
136  f1 = c2*y*ef0;
137  f2 = fabs(y*y - Pi*Pi/(t*t));
138  f3 = 1.0 - ef0*ef0;
139  t *= c1;
140  if (f1/(f2*f3) < 0.5*m_eps) {
141  break;
142  }
143  }
144  double h = t/y;
145  int N = int(0.5 + b/h);
146  double S = 0.0;
147  double u = h/2;
148  for (int i = 0; i < N; i++) {
149  S += (1.0 + exp(-4.0*x*y*u))*exp(-pow(y*u-x,2))/(u*u+1.0);
150  u += h;
151  }
152  double Q = h*S/Pi;
153  double C = 0.0;
154  if (y*y < Pi/h) {
155  C = 2.0*exp(y*y - x*x)*cos(2*x*y)/(1.0 + exp(2*Pi/h));
156  } else {
157  return 0.0;
158  }
159  return Q + C;
160 }
