80  SingleSpeciesTP(),
81  chemPot_(0.0)
82 {
83 
84  std::string pname = Ename + "Fixed";
85  setID(pname);
86  setName(pname);
87  setNDim(3);
88  addUniqueElement(Ename, -12345.);
89  freezeElements();
90  vector_fp ecomp(nElements(), 0.0);
91  ecomp[0] = 1.0;
92  double chrg = 0.0;

96  double c[4];
97  c[0] = 298.15;
98  c[1] = val;
99  c[2] = 0.0;
100  c[3] = 0.0;

102  initThermo();
103  m_p0 = OneAtm;
104  m_tlast = 298.15;
105  setChemicalPotential(val);
106 
107  // Create an XML_Node entry for this species
108  XML_Node* s = new XML_Node("species", 0);
109  s->addAttribute("name", pname);
110  std::string aaS = Ename + ":1";
111  s->addChild("atomArray", aaS);
112  XML_Node& tt = s->addChild("thermo");
113  XML_Node& ss = tt.addChild("Simple");
114  ss.addAttribute("Pref", "1 bar");
115  ss.addAttribute("Tmax", "5000.");
