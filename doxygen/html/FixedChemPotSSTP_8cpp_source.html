239 void FixedChemPotSSTP::initThermoXML(XML_Node& phaseNode, const std::string& id_)
240 {
241  /*
242  * Find the Thermo XML node
243  */
244  if (!phaseNode.hasChild("thermo")) {
245  throw CanteraError("FixedChemPotSSTP::initThermoXML", "no thermo XML node");
246  }
247  XML_Node& tnode = phaseNode.child("thermo");
248  std::string model = tnode["model"];
249  if (model != "StoichSubstance" && model != "FixedChemPot" && model != "StoichSubstanceSSTP") {
250  throw CanteraError("FixedChemPotSSTP::initThermoXML()",
251  "thermo model attribute must be FixedChemPot or StoichSubstance or StoichSubstanceSSTP");
252  }
253 
254  SingleSpeciesTP::initThermoXML(phaseNode, id_);
255  if (model == "FixedChemPot") {
256  double val = getFloatDefaultUnits(tnode, "chemicalPotential", "J/kmol");
257  chemPot_ = val;
258  } else {
259  _updateThermo();
260  chemPot_ = (m_h0_RT[0] - m_s0_R[0]) * GasConstant * temperature();
261  }
262 }
