76 void MultiSpeciesThermo::install_STIT(size_t index,
77  shared_ptr<SpeciesThermoInterpType> stit_ptr)
78 {
79  if (!stit_ptr) {
80  throw CanteraError("MultiSpeciesThermo::install_STIT",
81  "null pointer");
82  }
83  AssertThrowMsg(m_speciesLoc.find(index) == m_speciesLoc.end(),
84  "MultiSpeciesThermo::install_STIT",
85  "Index position isn't null, duplication of assignment: {}", index);
86  int type = stit_ptr->reportType();
87  m_speciesLoc[index] = {type, m_sp[type].size()};
88  m_sp[type].emplace_back(index, stit_ptr);
89  if (m_sp[type].size() == 1) {
90  m_tpoly[type].resize(stit_ptr->temperaturePolySize());
91  }
92 
93  // Calculate max and min T
94  m_tlow_max = std::max(stit_ptr->minTemp(), m_tlow_max);
95  m_thigh_min = std::min(stit_ptr->maxTemp(), m_thigh_min);
96  markInstalled(index);
97 }

99 void MultiSpeciesThermo::modifySpecies(size_t index,
100  shared_ptr<SpeciesThermoInterpType> spthermo)
101 {
102  if (!spthermo) {
103  throw CanteraError("MultiSpeciesThermo::modifySpecies",
104  "null pointer");
105  }
106  if (m_speciesLoc.find(index) == m_speciesLoc.end()) {
107  throw CanteraError("MultiSpeciesThermo::modifySpecies",
108  "Species with this index not previously added: {}",
109  index);
110  }
111  int type = spthermo->reportType();
112  if (m_speciesLoc[index].first != type) {
113  throw CanteraError("MultiSpeciesThermo::modifySpecies",
114  "Type of parameterization changed: {} != {}", type,
115  m_speciesLoc[index].first);
116  }
117  if (spthermo->minTemp() > m_tlow_max) {
118  throw CanteraError("MultiSpeciesThermo::modifySpecies",
119  "Cannot increase minimum temperature for phase from {} to {}",
120  m_tlow_max, spthermo->minTemp());
121  }
122  if (spthermo->maxTemp() < m_thigh_min) {
123  throw CanteraError("MultiSpeciesThermo::modifySpecies",
124  "Cannot increase minimum temperature for phase from {} to {}",
125  m_thigh_min, spthermo->maxTemp());
126  }
127 
128  m_sp[type][m_speciesLoc[index].second] = {index, spthermo};
129 }

147 void MultiSpeciesThermo::update(doublereal t, doublereal* cp_R,
148  doublereal* h_RT, doublereal* s_R) const
149 {
150  auto iter = m_sp.begin();
151  auto jter = m_tpoly.begin();
152  for (; iter != m_sp.end(); iter++, jter++) {
153  const std::vector<index_STIT>& species = iter->second;
154  double* tpoly = &jter->second[0];
155  species[0].second->updateTemperaturePoly(t, tpoly);
156  for (size_t k = 0; k < species.size(); k++) {
157  size_t i = species[k].first;
158  species[k].second->updateProperties(tpoly, cp_R+i, h_RT+i, s_R+i);
159  }
160  }
161 }
