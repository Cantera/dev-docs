72 // The tortuosity factor models the effective increase in the diffusive transport length.
73 /*
74  * This method returns \f$ 1/\tau^2 \f$ in the description of the flux
75  *
76  * \f$ C_T D_i \nabla X_i / \tau^2 \f$.
77  *
78  */
79 doublereal TortuosityMaxwell::tortuosityFactor(doublereal porosity)
80 {
81  return McMillanFactor(porosity) / porosity;
82 }

84 // The McMillan number is the ratio of the flux-like variable to the value it would have without porous flow.
85 /*
86  * The McMillan number combines the effect of tortuosity
87  * and volume fraction of the transported phase. The net flux
88  * observed is then the product of the McMillan number and the
89  * non-porous transport rate. For a conductivity in a non-porous
90  * media, \f$ \kappa_0 \f$, the conductivity in the porous media
91  * would be \f$ \kappa = (\rm McMillan) \kappa_0 \f$.
92  */
93 doublereal TortuosityMaxwell::McMillanFactor(doublereal porosity)
94 {
95  return 1 + 3 * (1.0 - porosity) * (relativeConductivities_ - 1.0) / (relativeConductivities_ + 2);
96 }
