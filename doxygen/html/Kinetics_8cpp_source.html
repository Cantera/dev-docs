141 void Kinetics::assignShallowPointers(const std::vector<thermo_t*> & tpVector)
142 {
143  size_t ns = tpVector.size();
144  if (ns != m_thermo.size()) {
145  throw CanteraError(" Kinetics::assignShallowPointers",
146  " Number of ThermoPhase objects arent't the same");
147  }
148  for (size_t i = 0; i < ns; i++) {
149  ThermoPhase* ntp = tpVector[i];
150  ThermoPhase* otp = m_thermo[i];
151  if (ntp->id() != otp->id()) {
152  throw CanteraError(" Kinetics::assignShallowPointers",
153  " id() of the ThermoPhase objects isn't the same");
154  }
155  if (ntp->eosType() != otp->eosType()) {
156  throw CanteraError(" Kinetics::assignShallowPointers",
157  " eosType() of the ThermoPhase objects isn't the same");
158  }
159  if (ntp->nSpecies() != otp->nSpecies()) {
160  throw CanteraError(" Kinetics::assignShallowPointers",
161  " Number of ThermoPhase objects isn't the same");
162  }
163  m_thermo[i] = tpVector[i];
164  }

252 void Kinetics::addPhase(thermo_t& thermo)
253 {
254  // if not the first thermo object, set the start position
255  // to that of the last object added + the number of its species
256  if (m_thermo.size() > 0) {
257  m_start.push_back(m_start.back()
258  + m_thermo.back()->nSpecies());
259  }
260  // otherwise start at 0
261  else {
262  m_start.push_back(0);
263  }
264 
265  // the phase with lowest dimensionality is assumed to be the
266  // phase/interface at which reactions take place
267  if (thermo.nDim() <= m_mindim) {
268  m_mindim = thermo.nDim();
269  m_rxnphase = nPhases();
270  }
271 
272  // there should only be one surface phase
273  int ptype = -100;
274  if (type() == cEdgeKinetics) {
275  ptype = cEdge;
276  } else if (type() == cInterfaceKinetics) {
277  ptype = cSurf;
278  }
279  if (thermo.eosType() == ptype) {
280  m_surfphase = nPhases();
281  m_rxnphase = nPhases();
282  }
283  m_thermo.push_back(&thermo);
284  m_phaseindex[m_thermo.back()->id()] = nPhases();
285 }
