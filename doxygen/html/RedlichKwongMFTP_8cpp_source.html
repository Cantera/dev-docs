345 {
346  doublereal TKelvin = temperature();
347  doublereal rt = TKelvin * GasConstant;
348  doublereal mv = molarVolume();
349  doublereal sqt = sqrt(TKelvin);
350  doublereal vpb = mv + m_b_current;
351  doublereal vmb = mv - m_b_current;
352 
353  for (size_t k = 0; k < m_kk; k++) {
354  m_pp[k] = 0.0;
355  for (size_t i = 0; i < m_kk; i++) {
356  size_t counter = k + m_kk*i;
357  m_pp[k] += moleFractions_[i] * a_vec_Curr_[counter];
358  }
359  }
360  doublereal pres = pressure();
361 
362  for (size_t k = 0; k < m_kk; k++) {
363  ac[k] = (- rt * log(pres * mv / rt)
364  + rt * log(mv / vmb)
365  + rt * b_vec_Curr_[k] / vmb
366  - 2.0 * m_pp[k] / (m_b_current * sqt) * log(vpb/mv)
367  + m_a_current * b_vec_Curr_[k] / (m_b_current * m_b_current * sqt) * log(vpb/mv)
368  - m_a_current / (m_b_current * sqt) * (b_vec_Curr_[k]/vpb)
369  );
370  }

1306 int RedlichKwongMFTP::NicholsSolve(double TKelvin, double pres, doublereal a, doublereal b,
1307  doublereal Vroot[3]) const
1308 {
1309  Vroot[0] = 0.0;
1310  Vroot[1] = 0.0;
1311  Vroot[2] = 0.0;
1312  if (TKelvin <= 0.0) {
1313  throw CanteraError("RedlichKwongMFTP::NicholsSolve()", "neg temperature");
1314  }
1315  /*
1316  * Derive the coefficients of the cubic polynomial to solve.
1317  */
1318  doublereal an = 1.0;
1319  doublereal bn = - GasConstant * TKelvin / pres;
1320  doublereal sqt = sqrt(TKelvin);
1321  doublereal cn = - (GasConstant * TKelvin * b / pres - a/(pres * sqt) + b * b);
1322  doublereal dn = - (a * b / (pres * sqt));
1323 
1324  double tmp = a * omega_b / (b * omega_a * GasConstant);
1325  double pp = 2./3.;
1326  double tc = pow(tmp, pp);
1327  double pc = omega_b * GasConstant * tc / b;
1328  double vc = omega_vc * GasConstant * tc / pc;
1329  // Derive the center of the cubic, x_N
1330  doublereal xN = - bn /(3 * an);
1331 
1332 
1333  // Derive the value of delta**2. This is a key quantity that determines the number of turning points
1334  doublereal delta2 = (bn * bn - 3 * an * cn) / (9 * an * an);
1335  doublereal delta = 0.0;
1336 
1337  // Calculate a couple of ratios
1338  doublereal ratio1 = 3.0 * an * cn / (bn * bn);
1339  doublereal ratio2 = pres * b / (GasConstant * TKelvin);
1340  if (fabs(ratio1) < 1.0E-7) {
1341  //printf("NicholsSolve(): Alternative solution (p = %g T = %g)\n", pres, TKelvin);
1342  doublereal ratio3 = a / (GasConstant * sqt) * pres / (GasConstant * TKelvin);
1343  if (fabs(ratio2) < 1.0E-5 && fabs(ratio3) < 1.0E-5) {
1344  doublereal zz = 1.0;
1345  for (int i = 0; i < 10; i++) {
1346  doublereal znew = zz / (zz - ratio2) - ratio3 / (zz + ratio1);
1347  doublereal deltaz = znew - zz;
1348  zz = znew;
1349  if (fabs(deltaz) < 1.0E-14) {
1350  break;
1351  }
1352  }
1353  doublereal v = zz * GasConstant * TKelvin / pres;
1354  Vroot[0] = v;
1355  return 1;
1356  }
1357  }
1358 
1359  int nSolnValues;
1360 
1361  double h2 = 4. * an * an * delta2 * delta2 * delta2;
1362  if (delta2 > 0.0) {
1363  delta = sqrt(delta2);
1364  }
1365 
1366  doublereal h = 2.0 * an * delta * delta2;
1367 
1368  doublereal yN = 2.0 * bn * bn * bn / (27.0 * an * an) - bn * cn / (3.0 * an) + dn;
1369 
1370  doublereal desc = yN * yN - h2;
1371 
1372  if (fabs(fabs(h) - fabs(yN)) < 1.0E-10) {
1373  if (desc != 0.0) {
1374  // this is for getting to other cases
1375  printf("NicholsSolve(): numerical issues\n");
1376  throw CanteraError("NicholsSolve()", "numerical issues");
1377  }
1378  desc = 0.0;
1379  }
1380 
1381  if (desc < 0.0) {
1382  nSolnValues = 3;
1383  } else if (desc == 0.0) {
1384  nSolnValues = 2;
1385  // We are here as p goes to zero.
1386  } else if (desc > 0.0) {
1387  nSolnValues = 1;
1388  }
1389 
1390  /*
1391  * One real root -> have to determine whether gas or liquid is the root
1392  */
1393  if (desc > 0.0) {
1394  doublereal tmpD = sqrt(desc);
1395  doublereal tmp1 = (- yN + tmpD) / (2.0 * an);
1396  doublereal sgn1 = 1.0;
1397  if (tmp1 < 0.0) {
1398  sgn1 = -1.0;
1399  tmp1 = -tmp1;
1400  }
1401  doublereal tmp2 = (- yN - tmpD) / (2.0 * an);
1402  doublereal sgn2 = 1.0;
1403  if (tmp2 < 0.0) {
1404  sgn2 = -1.0;
1405  tmp2 = -tmp2;
1406  }
1407  doublereal p1 = pow(tmp1, 1./3.);
1408  doublereal p2 = pow(tmp2, 1./3.);
1409 
1410  doublereal alpha = xN + sgn1 * p1 + sgn2 * p2;
1411  Vroot[0] = alpha;
1412  Vroot[1] = 0.0;
1413  Vroot[2] = 0.0;
1414 
1415  tmp = an * Vroot[0] * Vroot[0] * Vroot[0] + bn * Vroot[0] * Vroot[0] + cn * Vroot[0] + dn;
1416 
1417  } else if (desc < 0.0) {
1418  doublereal tmp = - yN/h;
1419 
1420  doublereal val = acos(tmp);
1421  doublereal theta = val / 3.0;
1422 
1423  doublereal oo = 2. * Pi / 3.;
1424  doublereal alpha = xN + 2. * delta * cos(theta);
1425 
1426  doublereal beta = xN + 2. * delta * cos(theta + oo);
1427 
1428  doublereal gamma = xN + 2. * delta * cos(theta + 2.0 * oo);
1429 
1430 
1431  Vroot[0] = beta;
1432  Vroot[1] = gamma;
1433  Vroot[2] = alpha;
1434 
1435  for (int i = 0; i < 3; i++) {
1436  tmp = an * Vroot[i] * Vroot[i] * Vroot[i] + bn * Vroot[i] * Vroot[i] + cn * Vroot[i] + dn;
1437  if (fabs(tmp) > 1.0E-4) {
1438  for (int j = 0; j < 3; j++) {
1439  if (j != i) {
1440  if (fabs(Vroot[i] - Vroot[j]) < 1.0E-4 * (fabs(Vroot[i]) + fabs(Vroot[j]))) {
1441  writelog("RedlichKwongMFTP::NicholsSolve(T = " + fp2str(TKelvin) + ", p = " +
1442  fp2str(pres) + "): WARNING roots have merged: " +
1443  fp2str(Vroot[i]) + ", " + fp2str(Vroot[j]));
1444  writelogendl();
1445  }
1446  }
1447  }
1448  }
1449  }
1450  } else if (desc == 0.0) {
1451  if (yN == 0.0 && h == 0.0) {
1452  Vroot[0] = xN;
1453  Vroot[1] = xN;
1454  Vroot[2] = xN;
1455  } else {
1456  // need to figure out whether delta is pos or neg
1457  if (yN > 0.0) {
1458  tmp = pow(yN/(2*an), 1./3.);
1459  if (fabs(tmp - delta) > 1.0E-9) {
1460  throw CanteraError("RedlichKwongMFTP::NicholsSolve()", "unexpected");
1461  }
1462  Vroot[1] = xN + delta;
1463  Vroot[0] = xN - 2.0*delta; // liquid phase root
1464  } else {
1465  tmp = pow(yN/(2*an), 1./3.);
1466  if (fabs(tmp - delta) > 1.0E-9) {
1467  throw CanteraError("RedlichKwongMFTP::NicholsSolve()", "unexpected");
1468  }
1469  delta = -delta;
1470  Vroot[0] = xN + delta;
1471  Vroot[1] = xN - 2.0*delta; // gas phase root
1472  }
1473  }
1474  for (int i = 0; i < 2; i++) {
1475  tmp = an * Vroot[i] * Vroot[i] * Vroot[i] + bn * Vroot[i] * Vroot[i] + cn * Vroot[i] + dn;
1476  }
1477  }
1478 
1479  /*
1480  * Unfortunately, there is a heavy amount of roundoff error due to bad conditioning in this
1481  */
1482  double res, dresdV = 0.0;
1483  for (int i = 0; i < nSolnValues; i++) {
1484  for (int n = 0; n < 20; n++) {
1485  res = an * Vroot[i] * Vroot[i] * Vroot[i] + bn * Vroot[i] * Vroot[i] + cn * Vroot[i] + dn;
1486  if (fabs(res) < 1.0E-14) {
1487  break;
1488  }
1489  dresdV = 3.0 * an * Vroot[i] * Vroot[i] + 2.0 * bn * Vroot[i] + cn;
1490  double del = - res / dresdV;
1491 
1492  Vroot[i] += del;
1493  if (fabs(del) / (fabs(Vroot[i]) + fabs(del)) < 1.0E-14) {
1494  break;
1495  }
1496  double res2 = an * Vroot[i] * Vroot[i] * Vroot[i] + bn * Vroot[i] * Vroot[i] + cn * Vroot[i] + dn;
1497  if (fabs(res2) < fabs(res)) {
1498  continue;
1499  } else {
1500  Vroot[i] -= del;
1501  Vroot[i] += 0.1 * del;
1502  }
1503  }
1504  if ((fabs(res) > 1.0E-14) && (fabs(res) > 1.0E-14 * fabs(dresdV) * fabs(Vroot[i]))) {
1505  writelog("RedlichKwongMFTP::NicholsSolve(T = " + fp2str(TKelvin) + ", p = " +
1506  fp2str(pres) + "): WARNING root didn't converge V = " + fp2str(Vroot[i]));
1507  writelogendl();
1508  }
1509  }
1510 
1511  if (nSolnValues == 1) {
1512  if (TKelvin > tc) {
1513  if (Vroot[0] < vc) {
1514  nSolnValues = -1;
1515  }
1516  } else {
1517  if (Vroot[0] < xN) {
1518  nSolnValues = -1;
1519  }
1520  }
1521 
1522  } else {
1523  if (nSolnValues == 2) {
1524  if (delta > 0.0) {
1525  nSolnValues = -2;
1526  }
1527  }
1528  }
1529  return nSolnValues;
1530 }
