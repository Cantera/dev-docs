148 //! Compute the reference-state property of one species
149 /*!
150  * Given temperature T in K, this method updates the values of
151  * the non-dimensional heat capacity at constant pressure,
152  * enthalpy, and entropy, at the reference pressure, Pref
153  * of one of the species. The species index is used
154  * to reference into the cp_R, h_RT, and s_R arrays.
155  *
156  * @param temp Temperature (Kelvin)
157  * @param cp_R Vector of Dimensionless heat capacities.
158  * (length m_kk).
159  * @param h_RT Vector of Dimensionless enthalpies.
160  * (length m_kk).
161  * @param s_R Vector of Dimensionless entropies.
162  * (length m_kk).
163  */
164 void STITbyPDSS::updatePropertiesTemp(const doublereal temp,
165  doublereal* cp_R,
166  doublereal* h_RT,
167  doublereal* s_R) const
168 {
169  //m_vpssmgr_ptr->setState_T(temp);
170  m_PDSS_ptr->setTemperature(temp);
171  AssertThrowMsg(m_speciesIndex != npos, "STITbyPDSS::updatePropertiesTemp",
172  "object was probably not installed correctly");
173  h_RT[m_speciesIndex] = m_PDSS_ptr->enthalpy_RT_ref();
174  cp_R[m_speciesIndex] = m_PDSS_ptr->cp_R_ref();
175  s_R[m_speciesIndex] = m_PDSS_ptr->entropy_R_ref();
176 }
