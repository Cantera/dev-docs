353 {
354  _updateStandardStateThermo();
355  /*
356  * Update the molality array, m_molalities()

384 void DebyeHuckel::getChemPotentials(doublereal* mu) const
385 {
386  double xx;
387  /*
388  * First get the standard chemical potentials in
389  * molar form.
390  * -> this requires updates of standard state as a function
391  * of T and P
392  */
393  getStandardChemPotentials(mu);
394  /*
395  * Update the activity coefficients
396  * This also updates the internal molality array.
397  */
398  s_update_lnMolalityActCoeff();
399  doublereal RT = GasConstant * temperature();
400  double xmolSolvent = moleFraction(m_indexSolvent);
401  for (size_t k = 0; k < m_kk; k++) {
402  if (m_indexSolvent != k) {
403  xx = std::max(m_molalities[k], SmallNumber);
404  mu[k] += RT * (log(xx) + m_lnActCoeffMolal[k]);
405  }
406  }
407  xx = std::max(xmolSolvent, SmallNumber);
408  mu[m_indexSolvent] +=
409  RT * (log(xx) + m_lnActCoeffMolal[m_indexSolvent]);
410 }

412 void DebyeHuckel::getPartialMolarEnthalpies(doublereal* hbar) const
413 {
414  /*
415  * Get the nondimensional standard state enthalpies
416  */
417  getEnthalpy_RT(hbar);
418  /*
419  * Dimensionalize it.
420  */
421  double T = temperature();
422  double RT = GasConstant * T;
423  for (size_t k = 0; k < m_kk; k++) {
424  hbar[k] *= RT;
425  }
426  /*
427  * Check to see whether activity coefficients are temperature
428  * dependent. If they are, then calculate the their temperature
429  * derivatives and add them into the result.
430  */
431  double dAdT = dA_DebyedT_TP();
432  if (dAdT != 0.0) {
433  /*
434  * Update the activity coefficients, This also update the
435  * internally stored molalities.
436  */
437  s_update_lnMolalityActCoeff();
438  s_update_dlnMolalityActCoeff_dT();
439  double RTT = GasConstant * T * T;
440  for (size_t k = 0; k < m_kk; k++) {
441  hbar[k] -= RTT * m_dlnActCoeffMolaldT[k];
442  }
443  }

446 void DebyeHuckel::getPartialMolarEntropies(doublereal* sbar) const
447 {
448  /*
449  * Get the standard state entropies at the temperature
450  * and pressure of the solution.
451  */
452  getEntropy_R(sbar);
453  /*
454  * Dimensionalize the entropies
455  */
456  doublereal R = GasConstant;
457  for (size_t k = 0; k < m_kk; k++) {
458  sbar[k] *= R;
459  }
460  /*
461  * Update the activity coefficients, This also update the
462  * internally stored molalities.
463  */
464  s_update_lnMolalityActCoeff();
465  /*
466  * First we will add in the obvious dependence on the T
467  * term out front of the log activity term
468  */
469  doublereal mm;
470  for (size_t k = 0; k < m_kk; k++) {
471  if (k != m_indexSolvent) {
472  mm = std::max(SmallNumber, m_molalities[k]);
473  sbar[k] -= R * (log(mm) + m_lnActCoeffMolal[k]);
474  }
475  }
476  double xmolSolvent = moleFraction(m_indexSolvent);
477  mm = std::max(SmallNumber, xmolSolvent);
478  sbar[m_indexSolvent] -= R *(log(mm) + m_lnActCoeffMolal[m_indexSolvent]);
479  /*
480  * Check to see whether activity coefficients are temperature
481  * dependent. If they are, then calculate the their temperature
482  * derivatives and add them into the result.
483  */
484  double dAdT = dA_DebyedT_TP();
485  if (dAdT != 0.0) {
486  s_update_dlnMolalityActCoeff_dT();
487  double RT = R * temperature();
488  for (size_t k = 0; k < m_kk; k++) {
489  sbar[k] -= RT * m_dlnActCoeffMolaldT[k];
490  }
491  }

1233 double DebyeHuckel::_osmoticCoeffHelgesonFixedForm() const
1234 {
1235  const double a0 = 1.454;
1236  const double b0 = 0.02236;
1237  const double c0 = 9.380E-3;
1238  const double d0 = -5.362E-4;
1239  double Is = m_IionicMolalityStoich;
1240  if (Is <= 0.0) {
1241  return 0.0;
1242  }
1243  double Is2 = Is * Is;
1244  double bhat = 1.0 + a0 * sqrt(Is);
1245  double func = bhat - 2.0 * log(bhat) - 1.0/bhat;
1246  double v1 = m_A_Debye / (a0 * a0 * a0 * Is) * func;
1247  double oc = 1.0 - v1 + b0 * Is / 2.0 + 2.0 * c0 * Is2 / 3.0
1248  + 3.0 * d0 * Is2 * Is / 4.0;
1249  return oc;
1250 }
