20 double VCS_SOLVE::vcs_nondim_Farad(int mu_units, double TKelvin) const
21 {
22  if (TKelvin <= 0.0) {
23  TKelvin = 293.15;
24  }
25  switch (mu_units) {
26  case VCS_UNITS_MKS:
27  case VCS_UNITS_KJMOL:
28  case VCS_UNITS_KCALMOL:
29  return Cantera::ElectronCharge * Cantera::Avogadro /
30  (TKelvin * Cantera::GasConstant);
31  case VCS_UNITS_UNITLESS:
32  return Cantera::ElectronCharge * Cantera::Avogadro;
33  case VCS_UNITS_KELVIN:
34  return Cantera::ElectronCharge * Cantera::Avogadro/ TKelvin;
35  default:
36  throw CanteraError("vcs_nondim_Farad",
37  "unknown units: " + int2str(mu_units));
38  }
39 }

63 void VCS_SOLVE::vcs_nondim_TP()
64 {
65  if (m_unitsState == VCS_DIMENSIONAL_G) {
66  m_unitsState = VCS_NONDIMENSIONAL_G;
67  double tf = 1.0 / vcs_nondimMult_TP(m_VCS_UnitsFormat, m_temperature);
68  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
69  /*
70  * Modify the standard state and total chemical potential data,
71  * FF(I), to make it dimensionless, i.e., mu / RT.
72  * Thus, we may divide it by the temperature.
73  */
74  m_SSfeSpecies[i] *= tf;
75  m_deltaGRxn_new[i] *= tf;
76  m_deltaGRxn_old[i] *= tf;
77  m_feSpecies_old[i] *= tf;
78  }
79 
80  m_Faraday_dim = vcs_nondim_Farad(m_VCS_UnitsFormat, m_temperature);
81 
82  /*
83  * Scale the total moles if necessary:
84  * First find out the total moles
85  */
86  double tmole_orig = vcs_tmoles();
87 
88  /*
89  * Then add in the total moles of elements that are goals. Either one
90  * or the other is specified here.
91  */
92  double esum = 0.0;
93  for (size_t i = 0; i < m_numElemConstraints; ++i) {
94  if (m_elType[i] == VCS_ELEM_TYPE_ABSPOS) {
95  esum += fabs(m_elemAbundancesGoal[i]);
96  }
97  }
98  tmole_orig += esum;
99 
100  /*
101  * Ok now test out the bounds on the total moles that this program can
102  * handle. These are a bit arbitrary. However, it would seem that any
103  * reasonable input would be between these two numbers below.
104  */
105  if (tmole_orig < 1.0E-200 || tmole_orig > 1.0E200) {
106  throw CanteraError("VCS_SOLVE::vcs_nondim_TP",
107  "Total input moles ," + fp2str(tmole_orig) +
108  "is outside the range handled by vcs.\n");
109  }
110 
111  // Determine the scale of the problem
112  if (tmole_orig > 1.0E4) {
113  m_totalMoleScale = tmole_orig / 1.0E4;
114  } else if (tmole_orig < 1.0E-4) {
115  m_totalMoleScale = tmole_orig / 1.0E-4;
116  } else {
117  m_totalMoleScale = 1.0;
118  }
119 
120  if (m_totalMoleScale != 1.0) {
121  if (m_VCS_UnitsFormat == VCS_UNITS_MKS) {
122  if (DEBUG_MODE_ENABLED && m_debug_print_lvl >= 2) {
123  plogf(" --- vcs_nondim_TP() called: USING A MOLE SCALE OF %g until further notice", m_totalMoleScale);
124  plogendl();
125  }
126  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
127  if (m_speciesUnknownType[i] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
128  m_molNumSpecies_old[i] *= (1.0 / m_totalMoleScale);
129  }
130  }
131  for (size_t i = 0; i < m_numElemConstraints; ++i) {
132  m_elemAbundancesGoal[i] *= (1.0 / m_totalMoleScale);
133  }
134 
135  for (size_t iph = 0; iph < m_numPhases; iph++) {
136  TPhInertMoles[iph] *= (1.0 / m_totalMoleScale);
137  if (TPhInertMoles[iph] != 0.0) {
138  vcs_VolPhase* vphase = m_VolPhaseList[iph];
139  vphase->setTotalMolesInert(TPhInertMoles[iph]);
140  }
141  }
142  }
143  vcs_tmoles();
144  }
145  }
146 }

148 void VCS_SOLVE::vcs_redim_TP(void)
149 {
150  if (m_unitsState != VCS_DIMENSIONAL_G) {
151  m_unitsState = VCS_DIMENSIONAL_G;
152  double tf = vcs_nondimMult_TP(m_VCS_UnitsFormat, m_temperature);
153  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
154  /*
155  * Modify the standard state and total chemical potential data,
156  * FF(I), to make it have units, i.e. mu = RT * mu_star
157  */
158  m_SSfeSpecies[i] *= tf;
159  m_deltaGRxn_new[i] *= tf;
160  m_deltaGRxn_old[i] *= tf;
161  m_feSpecies_old[i] *= tf;
162  }
163  m_Faraday_dim *= tf;
164  }
165  if (m_totalMoleScale != 1.0) {
166  if (m_VCS_UnitsFormat == VCS_UNITS_MKS) {
167  if (DEBUG_MODE_ENABLED && m_debug_print_lvl >= 2) {
168  plogf(" --- vcs_redim_TP() called: getting rid of mole scale of %g", m_totalMoleScale);
169  plogendl();
170  }
171  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
172  if (m_speciesUnknownType[i] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
173  m_molNumSpecies_old[i] *= m_totalMoleScale;
174  }
175  }
176  for (size_t i = 0; i < m_numElemConstraints; ++i) {
177  m_elemAbundancesGoal[i] *= m_totalMoleScale;
178  }
179

181  TPhInertMoles[iph] *= m_totalMoleScale;
