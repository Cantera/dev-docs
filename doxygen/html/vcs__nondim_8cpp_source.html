18 double VCS_SOLVE::vcs_nondim_Farad(int mu_units, double TKelvin) const
19 {
20  if (TKelvin <= 0.0) {
21  TKelvin = 293.15;
22  }
23  switch (mu_units) {
24  case VCS_UNITS_MKS:
25  case VCS_UNITS_KJMOL:
26  case VCS_UNITS_KCALMOL:
27  return ElectronCharge * Avogadro / (TKelvin * GasConstant);
28  case VCS_UNITS_UNITLESS:
29  return ElectronCharge * Avogadro;
30  case VCS_UNITS_KELVIN:
31  return ElectronCharge * Avogadro/ TKelvin;
32  default:
33  throw CanteraError("vcs_nondim_Farad", "unknown units: {}", mu_units);
34  }
35 }

37 double VCS_SOLVE::vcs_nondimMult_TP(int mu_units, double TKelvin) const
38 {
39  if (TKelvin <= 0.0) {
40  TKelvin = 293.15;
41  }
42  switch (mu_units) {
43  case VCS_UNITS_KCALMOL:
44  return TKelvin * GasConst_cal_mol_K * 1e-3;
45  case VCS_UNITS_UNITLESS:
46  return 1.0;
47  case VCS_UNITS_KJMOL:
48  return TKelvin * GasConstant * 1e-6;
49  case VCS_UNITS_KELVIN:
50  return TKelvin;
51  case VCS_UNITS_MKS:
52  return TKelvin * GasConstant;
53  default:
54  throw CanteraError("vcs_nondimMult_TP", "unknown units: {}", mu_units);
55  }

58 void VCS_SOLVE::vcs_nondim_TP()
59 {
60  if (m_unitsState == VCS_DIMENSIONAL_G) {
61  m_unitsState = VCS_NONDIMENSIONAL_G;
62  double tf = 1.0 / vcs_nondimMult_TP(m_VCS_UnitsFormat, m_temperature);
63  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
64  // Modify the standard state and total chemical potential data,
65  // FF(I), to make it dimensionless, i.e., mu / RT. Thus, we may
66  // divide it by the temperature.
67  m_SSfeSpecies[i] *= tf;
68  m_deltaGRxn_new[i] *= tf;
69  m_deltaGRxn_old[i] *= tf;
70  m_feSpecies_old[i] *= tf;
71  }
72 
73  m_Faraday_dim = vcs_nondim_Farad(m_VCS_UnitsFormat, m_temperature);
74 
75  // Scale the total moles if necessary: First find out the total moles
76  double tmole_orig = vcs_tmoles();
77 
78  // Then add in the total moles of elements that are goals. Either one or
79  // the other is specified here.
80  double esum = 0.0;
81  for (size_t i = 0; i < m_numElemConstraints; ++i) {
82  if (m_elType[i] == VCS_ELEM_TYPE_ABSPOS) {
83  esum += fabs(m_elemAbundancesGoal[i]);
84  }
85  }
86  tmole_orig += esum;
87 
88  // Ok now test out the bounds on the total moles that this program can
89  // handle. These are a bit arbitrary. However, it would seem that any
90  // reasonable input would be between these two numbers below.
91  if (tmole_orig < 1.0E-200 || tmole_orig > 1.0E200) {
92  throw CanteraError("VCS_SOLVE::vcs_nondim_TP",
93  "Total input moles, {} is outside the range handled by vcs.\n",
94  tmole_orig);
95  }
96 
97  // Determine the scale of the problem
98  if (tmole_orig > 1.0E4) {
99  m_totalMoleScale = tmole_orig / 1.0E4;
100  } else if (tmole_orig < 1.0E-4) {
101  m_totalMoleScale = tmole_orig / 1.0E-4;
102  } else {
103  m_totalMoleScale = 1.0;
104  }
105 
106  if (m_totalMoleScale != 1.0) {
107  if (m_VCS_UnitsFormat == VCS_UNITS_MKS) {
108  if (m_debug_print_lvl >= 2) {
109  plogf(" --- vcs_nondim_TP() called: USING A MOLE SCALE OF %g until further notice", m_totalMoleScale);
110  plogendl();
111  }
112  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
113  if (m_speciesUnknownType[i] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
114  m_molNumSpecies_old[i] *= (1.0 / m_totalMoleScale);
115  }
116  }
117  for (size_t i = 0; i < m_numElemConstraints; ++i) {
118  m_elemAbundancesGoal[i] *= (1.0 / m_totalMoleScale);
119  }
120 
121  for (size_t iph = 0; iph < m_numPhases; iph++) {
122  TPhInertMoles[iph] *= (1.0 / m_totalMoleScale);
123  if (TPhInertMoles[iph] != 0.0) {
124  vcs_VolPhase* vphase = m_VolPhaseList[iph];
125  vphase->setTotalMolesInert(TPhInertMoles[iph]);
126  }
127  }
128  }
129  vcs_tmoles();
130  }
131  }
132 }

134 void VCS_SOLVE::vcs_redim_TP()
135 {
136  if (m_unitsState != VCS_DIMENSIONAL_G) {
137  m_unitsState = VCS_DIMENSIONAL_G;
138  double tf = vcs_nondimMult_TP(m_VCS_UnitsFormat, m_temperature);
139  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
140 
141  // Modify the standard state and total chemical potential data,
142  // FF(I), to make it have units, i.e. mu = RT * mu_star
143  m_SSfeSpecies[i] *= tf;
144  m_deltaGRxn_new[i] *= tf;
145  m_deltaGRxn_old[i] *= tf;
146  m_feSpecies_old[i] *= tf;
147  }
148  m_Faraday_dim *= tf;
149  }
150  if (m_totalMoleScale != 1.0) {
151  if (m_VCS_UnitsFormat == VCS_UNITS_MKS) {
152  if (m_debug_print_lvl >= 2) {
153  plogf(" --- vcs_redim_TP() called: getting rid of mole scale of %g", m_totalMoleScale);
154  plogendl();
155  }
156  for (size_t i = 0; i < m_numSpeciesTot; ++i) {
157  if (m_speciesUnknownType[i] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
158  m_molNumSpecies_old[i] *= m_totalMoleScale;
159  }
160  }
161  for (size_t i = 0; i < m_numElemConstraints; ++i) {
162  m_elemAbundancesGoal[i] *= m_totalMoleScale;
163  }
164 
165  for (size_t iph = 0; iph < m_numPhases; iph++) {
166  TPhInertMoles[iph] *= m_totalMoleScale;
167  if (TPhInertMoles[iph] != 0.0) {
168  vcs_VolPhase* vphase = m_VolPhaseList[iph];
169  vphase->setTotalMolesInert(TPhInertMoles[iph]);
170  }
171  }
172  vcs_tmoles();
173  }
174  }
