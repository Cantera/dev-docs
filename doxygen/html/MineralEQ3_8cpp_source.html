222 void MineralEQ3::initThermoXML(XML_Node& phaseNode, const std::string& id_)
223 {
224  /*
225  * Find the Thermo XML node
226  */
227  if (!phaseNode.hasChild("thermo")) {
228  throw CanteraError("HMWSoln::initThermoXML",
229  "no thermo XML node");
230  }
231 
232  const XML_Node* xsp = speciesData()[0];
233 
234  XML_Node* aStandardState = 0;
235  if (xsp->hasChild("standardState")) {
236  aStandardState = &xsp->child("standardState");
237  } else {
238  throw CanteraError("MineralEQ3::initThermoXML",
239  "no standard state mode");
240  }
241  doublereal volVal = 0.0;
242  if (aStandardState->attrib("model") != "constantVolume") {
243  throw CanteraError("MineralEQ3::initThermoXML",
244  "wrong standard state mode");
245  }
246  if (aStandardState->hasChild("V0_Pr_Tr")) {
247  XML_Node& aV = aStandardState->child("V0_Pr_Tr");
248  double Afactor = toSI("cm3/gmol");
249  if (aV.hasAttrib("units")) {
250  Afactor = toSI(aV.attrib("units"));
251  }
252  volVal = ctml::getFloat(*aStandardState, "V0_Pr_Tr");
253  m_V0_pr_tr= volVal;
254  volVal *= Afactor;
255  m_speciesSize[0] = volVal;
256  } else {
257  throw CanteraError("MineralEQ3::initThermoXML",
258  "wrong standard state mode");
259  }
260  setDensity(molecularWeight(0) / volVal);
261 
262  const XML_Node& MinEQ3node = xsp->child("thermo").child("MinEQ3");
263 
264  m_deltaG_formation_pr_tr =

266  m_deltaH_formation_pr_tr =

284 doublereal MineralEQ3::LookupGe(const std::string& elemName)
285 {
286  size_t iE = elementIndex(elemName);
287  if (iE == npos) {
288  throw CanteraError("PDSS_HKFT::LookupGe", "element " + elemName + " not found");
289  }
290  doublereal geValue = entropyElement298(iE);
291  if (geValue == ENTROPY298_UNKNOWN) {
292  throw CanteraError("PDSS_HKFT::LookupGe",
293  "element " + elemName + " does not have a supplied entropy298");
294  }
295  geValue *= (-298.15);

299 void MineralEQ3::convertDGFormation()
300 {
301  /*
302  * Ok let's get the element compositions and conversion factors.
303  */
304 
305  doublereal totalSum = 0.0;
306  for (size_t m = 0; m < nElements(); m++) {
307  double na = nAtoms(0, m);
308  if (na > 0.0) {
309  totalSum += na * LookupGe(elementName(m));
310  }
311  }
312  // Ok, now do the calculation. Convert to joules kmol-1
313  doublereal dg = m_deltaG_formation_pr_tr * 4.184 * 1.0E3;
314  //! Store the result into an internal variable.
315  m_Mu0_pr_tr = dg + totalSum;
316 
317  double Hcalc = m_Mu0_pr_tr + 298.15 * m_Entrop_pr_tr * 4184.0;
318  double DHjmol = m_deltaH_formation_pr_tr * 4184.0;
319 
320  // If the discrepancy is greater than 100 cal gmol-1, print an error
321  if (fabs(Hcalc -DHjmol) > 10.* 1.0E6 * 4.184) {
322  throw CanteraError("installMinEQ3asShomateThermoFromXML()",
323  "DHjmol is not consistent with G and S" +
324  fp2str(Hcalc) + " vs " + fp2str(DHjmol));
325  }
326 }
