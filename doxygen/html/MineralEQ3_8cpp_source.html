131 void MineralEQ3::initThermoXML(XML_Node& phaseNode, const std::string& id_)
132 {
133  // Find the Thermo XML node
134  if (!phaseNode.hasChild("thermo")) {
135  throw CanteraError("HMWSoln::initThermoXML",
136  "no thermo XML node");
137  }
138 
139  const XML_Node* xsp = speciesData()[0];
140 
141  XML_Node* aStandardState = 0;
142  if (xsp->hasChild("standardState")) {
143  aStandardState = &xsp->child("standardState");
144  } else {
145  throw CanteraError("MineralEQ3::initThermoXML",
146  "no standard state mode");
147  }
148  doublereal volVal = 0.0;
149  if (aStandardState->attrib("model") != "constantVolume") {
150  throw CanteraError("MineralEQ3::initThermoXML",
151  "wrong standard state mode");
152  }
153  if (aStandardState->hasChild("V0_Pr_Tr")) {
154  XML_Node& aV = aStandardState->child("V0_Pr_Tr");
155  double Afactor = toSI("cm3/gmol");
156  if (aV.hasAttrib("units")) {
157  Afactor = toSI(aV.attrib("units"));
158  }
159  volVal = getFloat(*aStandardState, "V0_Pr_Tr");
160  m_V0_pr_tr= volVal;
161  volVal *= Afactor;
162  } else {
163  throw CanteraError("MineralEQ3::initThermoXML",
164  "wrong standard state mode");
165  }
166  setDensity(molecularWeight(0) / volVal);
167 
168  const XML_Node& MinEQ3node = xsp->child("thermo").child("MinEQ3");
169 
170  m_deltaG_formation_pr_tr =
171  getFloat(MinEQ3node, "DG0_f_Pr_Tr", "actEnergy") / actEnergyToSI("cal/gmol");
172  m_deltaH_formation_pr_tr =
173  getFloat(MinEQ3node, "DH0_f_Pr_Tr", "actEnergy") / actEnergyToSI("cal/gmol");
174  m_Entrop_pr_tr = getFloat(MinEQ3node, "S0_Pr_Tr", "toSI") / toSI("cal/gmol/K");
175  m_a = getFloat(MinEQ3node, "a", "toSI") / toSI("cal/gmol/K");
176  m_b = getFloat(MinEQ3node, "b", "toSI") / toSI("cal/gmol/K2");
177  m_c = getFloat(MinEQ3node, "c", "toSI") / toSI("cal-K/gmol");
178 
179  convertDGFormation();
180 }

190 doublereal MineralEQ3::LookupGe(const std::string& elemName)
191 {
192  size_t iE = elementIndex(elemName);
193  if (iE == npos) {
194  throw CanteraError("PDSS_HKFT::LookupGe", "element " + elemName + " not found");
195  }
196  doublereal geValue = entropyElement298(iE);
197  if (geValue == ENTROPY298_UNKNOWN) {
198  throw CanteraError("PDSS_HKFT::LookupGe",
199  "element " + elemName + " does not have a supplied entropy298");
200  }
201  geValue *= (-298.15);
202  return geValue;
203 }

205 void MineralEQ3::convertDGFormation()
206 {
207  // Ok let's get the element compositions and conversion factors.
208  doublereal totalSum = 0.0;
209  for (size_t m = 0; m < nElements(); m++) {
210  double na = nAtoms(0, m);
211  if (na > 0.0) {
212  totalSum += na * LookupGe(elementName(m));
213  }
214  }
215  // Ok, now do the calculation. Convert to joules kmol-1
216  doublereal dg = m_deltaG_formation_pr_tr * toSI("cal/gmol");
217  //! Store the result into an internal variable.
218  m_Mu0_pr_tr = dg + totalSum;
219 
220  double Hcalc = m_Mu0_pr_tr + 298.15 * m_Entrop_pr_tr * toSI("cal/gmol");
221  double DHjmol = m_deltaH_formation_pr_tr * toSI("kal/gmol");
222 
223  // If the discrepancy is greater than 100 cal gmol-1, print an error
224  if (fabs(Hcalc -DHjmol) > 100 * toSI("cal/gmol")) {
225  throw CanteraError("installMinEQ3asShomateThermoFromXML()",
226  "DHjmol is not consistent with G and S: {} vs {}",
227  Hcalc, DHjmol);
228  }
229 }
