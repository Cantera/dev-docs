312 /*
313  * Get the nondimensional Enthalpy functions for the species
314  * at their standard states at the current
315  * <I>T</I> and <I>P</I> of the solution.
316  * Molar enthalpy. Units: J/kmol. For an incompressible,
317  * stoichiometric substance, the internal energy is
318  * independent of pressure, and therefore the molar enthalpy
319  * is \f[ \hat h(T, P) = \hat u(T) + P \hat v \f], where the
320  * molar specific volume is constant.
321  */
322 void MineralEQ3::getEnthalpy_RT(doublereal* hrt) const
323 {
324  getEnthalpy_RT_ref(hrt);
325  doublereal RT = GasConstant * temperature();
326  doublereal presCorrect = (m_press - m_p0) / molarDensity();
327  hrt[0] += presCorrect / RT;
328 }

361 /*
362  * Molar internal energy (J/kmol).
363  * For an incompressible,
364  * stoichiometric substance, the molar internal energy is
365  * independent of pressure. Since the thermodynamic properties
366  * are specified by giving the standard-state enthalpy, the
367  * term \f$ P_0 \hat v\f$ is subtracted from the specified molar
368  * enthalpy to compute the molar internal energy.
369  */
370 void MineralEQ3::getIntEnergy_RT(doublereal* urt) const
371 {
372  _updateThermo();
373  doublereal RT = GasConstant * temperature();
374  doublereal PV = m_p0 / molarDensity();
375  urt[0] = m_h0_RT[0] - PV / RT;
376 }

481 void MineralEQ3::initThermoXML(XML_Node& phaseNode, const std::string& id)
482 {
483  /*
484  * Find the Thermo XML node
485  */
486  if (!phaseNode.hasChild("thermo")) {
487  throw CanteraError("HMWSoln::initThermoXML",
488  "no thermo XML node");

491  std::vector<const XML_Node*> xspecies = speciesData();
492  const XML_Node* xsp = xspecies[0];
493 
494  XML_Node* aStandardState = 0;
495  if (xsp->hasChild("standardState")) {
496  aStandardState = &xsp->child("standardState");
497  } else {
498  throw CanteraError("MineralEQ3::initThermoXML",
499  "no standard state mode");
500  }
501  doublereal volVal = 0.0;
502  string smodel = (*aStandardState)["model"];
503  if (smodel != "constantVolume") {
504  throw CanteraError("MineralEQ3::initThermoXML",

506  }
507  if (aStandardState->hasChild("V0_Pr_Tr")) {
508  XML_Node& aV = aStandardState->child("V0_Pr_Tr");
509  string Aunits = "";
510  double Afactor = toSI("cm3/gmol");
511  if (aV.hasAttrib("units")) {
512  Aunits = aV.attrib("units");
513  Afactor = toSI(Aunits);

515  volVal = ctml::getFloat(*aStandardState, "V0_Pr_Tr");
516  m_V0_pr_tr= volVal;
517  volVal *= Afactor;
518  m_speciesSize[0] = volVal;
519  } else {
520  throw CanteraError("MineralEQ3::initThermoXML",
521  "wrong standard state mode");
522  }
523  doublereal rho = molecularWeight(0) / volVal;
524  setDensity(rho);

554 doublereal MineralEQ3::LookupGe(const std::string& elemName)
555 {
556  size_t iE = elementIndex(elemName);

558  throw CanteraError("PDSS_HKFT::LookupGe", "element " + elemName + " not found");
559  }
