172 void MineralEQ3::initThermoXML(XML_Node& phaseNode, const std::string& id_)
173 {
174  // Find the Thermo XML node
175  if (!phaseNode.hasChild("thermo")) {
176  throw CanteraError("HMWSoln::initThermoXML",
177  "no thermo XML node");
178  }
179 
180  const XML_Node* xsp = speciesData()[0];
181 
182  XML_Node* aStandardState = 0;
183  if (xsp->hasChild("standardState")) {
184  aStandardState = &xsp->child("standardState");
185  } else {
186  throw CanteraError("MineralEQ3::initThermoXML",
187  "no standard state mode");
188  }
189  doublereal volVal = 0.0;
190  if (aStandardState->attrib("model") != "constantVolume") {
191  throw CanteraError("MineralEQ3::initThermoXML",
192  "wrong standard state mode");
193  }
194  if (aStandardState->hasChild("V0_Pr_Tr")) {
195  XML_Node& aV = aStandardState->child("V0_Pr_Tr");
196  double Afactor = toSI("cm3/gmol");
197  if (aV.hasAttrib("units")) {
198  Afactor = toSI(aV.attrib("units"));
199  }
200  volVal = getFloat(*aStandardState, "V0_Pr_Tr");
201  m_V0_pr_tr= volVal;
202  volVal *= Afactor;
203  m_speciesSize[0] = volVal;
204  } else {
205  throw CanteraError("MineralEQ3::initThermoXML",
206  "wrong standard state mode");
207  }
208  setDensity(molecularWeight(0) / volVal);
209 
210  const XML_Node& MinEQ3node = xsp->child("thermo").child("MinEQ3");
211 
212  m_deltaG_formation_pr_tr =
213  getFloatDefaultUnits(MinEQ3node, "DG0_f_Pr_Tr", "cal/gmol", "actEnergy");
214  m_deltaH_formation_pr_tr =
215  getFloatDefaultUnits(MinEQ3node, "DH0_f_Pr_Tr", "cal/gmol", "actEnergy");
216  m_Entrop_pr_tr = getFloatDefaultUnits(MinEQ3node, "S0_Pr_Tr", "cal/gmol/K");
217  m_a = getFloatDefaultUnits(MinEQ3node, "a", "cal/gmol/K");
218  m_b = getFloatDefaultUnits(MinEQ3node, "b", "cal/gmol/K2");
219  m_c = getFloatDefaultUnits(MinEQ3node, "c", "cal-K/gmol");
220 
221  convertDGFormation();
222 }

232 doublereal MineralEQ3::LookupGe(const std::string& elemName)
233 {
234  size_t iE = elementIndex(elemName);
235  if (iE == npos) {
236  throw CanteraError("PDSS_HKFT::LookupGe", "element " + elemName + " not found");
237  }
238  doublereal geValue = entropyElement298(iE);
239  if (geValue == ENTROPY298_UNKNOWN) {
240  throw CanteraError("PDSS_HKFT::LookupGe",
241  "element " + elemName + " does not have a supplied entropy298");
242  }
243  geValue *= (-298.15);
244  return geValue;
245 }

247 void MineralEQ3::convertDGFormation()
248 {
249  // Ok let's get the element compositions and conversion factors.
250  doublereal totalSum = 0.0;
251  for (size_t m = 0; m < nElements(); m++) {
252  double na = nAtoms(0, m);
253  if (na > 0.0) {
254  totalSum += na * LookupGe(elementName(m));
255  }
256  }
257  // Ok, now do the calculation. Convert to joules kmol-1
258  doublereal dg = m_deltaG_formation_pr_tr * 4.184 * 1.0E3;
259  //! Store the result into an internal variable.
260  m_Mu0_pr_tr = dg + totalSum;
261 
262  double Hcalc = m_Mu0_pr_tr + 298.15 * m_Entrop_pr_tr * 4184.0;
263  double DHjmol = m_deltaH_formation_pr_tr * 4184.0;
264 
265  // If the discrepancy is greater than 100 cal gmol-1, print an error
266  if (fabs(Hcalc -DHjmol) > 10.* 1.0E6 * 4.184) {
267  throw CanteraError("installMinEQ3asShomateThermoFromXML()",
268  "DHjmol is not consistent with G and S: {} vs {}",
269  Hcalc, DHjmol);
270  }
271 }
