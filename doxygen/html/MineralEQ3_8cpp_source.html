324 doublereal MineralEQ3::LookupGe(const std::string& elemName)
325 {
326  size_t iE = elementIndex(elemName);
327  if (iE == npos) {
328  throw CanteraError("PDSS_HKFT::LookupGe", "element " + elemName + " not found");
329  }
330  doublereal geValue = entropyElement298(iE);
