236 {
237  /*
238  * Find the Thermo XML node
239  */
240  if (!phaseNode.hasChild("thermo")) {
241  throw CanteraError("HMWSoln::initThermoXML",

244 
245  std::vector<const XML_Node*> xspecies = speciesData();
246  const XML_Node* xsp = xspecies[0];
247 
248  XML_Node* aStandardState = 0;
249  if (xsp->hasChild("standardState")) {
250  aStandardState = &xsp->child("standardState");
251  } else {
252  throw CanteraError("MineralEQ3::initThermoXML",
253  "no standard state mode");
254  }
255  doublereal volVal = 0.0;
256  string smodel = (*aStandardState)["model"];
257  if (smodel != "constantVolume") {

274  throw CanteraError("MineralEQ3::initThermoXML",

305 doublereal MineralEQ3::LookupGe(const std::string& elemName)
306 {
307  size_t iE = elementIndex(elemName);
308  if (iE == npos) {
309  throw CanteraError("PDSS_HKFT::LookupGe", "element " + elemName + " not found");
310  }
311  doublereal geValue = entropyElement298(iE);
312  if (geValue == ENTROPY298_UNKNOWN) {
313  throw CanteraError("PDSS_HKFT::LookupGe",
314  "element " + elemName + " does not have a supplied entropy298");
315  }
316  geValue *= (-298.15);
317  return geValue;

320 void MineralEQ3::convertDGFormation()
321 {
322  /*
323  * Ok let's get the element compositions and conversion factors.
324  */
325  doublereal na;
326  doublereal ge;
327  string ename;
328 
329  doublereal totalSum = 0.0;
330  for (size_t m = 0; m < nElements(); m++) {
331  na = nAtoms(0, m);
332  if (na > 0.0) {
333  ename = elementName(m);
334  ge = LookupGe(ename);
335  totalSum += na * ge;
336  }
337  }
338  // Ok, now do the calculation. Convert to joules kmol-1
339  doublereal dg = m_deltaG_formation_pr_tr * 4.184 * 1.0E3;
340  //! Store the result into an internal variable.
341  m_Mu0_pr_tr = dg + totalSum;
342 
343  double Hcalc = m_Mu0_pr_tr + 298.15 * m_Entrop_pr_tr * 4184.0;
344  double DHjmol = m_deltaH_formation_pr_tr * 4184.0;
345 
346  // If the discrepancy is greater than 100 cal gmol-1, print an error
347  if (fabs(Hcalc -DHjmol) > 10.* 1.0E6 * 4.184) {
348  throw CanteraError("installMinEQ3asShomateThermoFromXML()",
349  "DHjmol is not consistent with G and S" +
350  fp2str(Hcalc) + " vs " + fp2str(DHjmol));
351  }
352 }
