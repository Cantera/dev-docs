60 VPSSMgr& VPSSMgr::operator=(const VPSSMgr& right)
61 {
62  if (&right == this) {
63  return *this;
64  }
65  m_kk = right.m_kk;
66  /*
67  * What we are doing here is to make a shallow copy of the VPStandardStateTP
68  * pointer in the "new" VPSSMgr object using the value from the "old"
69  * VPSSMgr object. This is not appropriate if we are making a copy of a ThermoPhase
70  * object and the VPSSMgr objects are owned by the ThermoPhase object.
71  *
72  * The new object will want to have a different value of m_vptp_ptr than the
73  * value this is being copied here. It will want to refer to the copy of the
74  * VPStandardStateTP object being made that will own the new VPSSMgr object.
75  * However, the assignment object is not the place to carry out this fixup.
76  *
77  * We will have to "fix" up the shallow copies later.
78  */
79  m_vptp_ptr = right.m_vptp_ptr;
