276 void IdealGasPhase::setToEquilState(const doublereal* mu_RT)
277 {
278  const vector_fp& grt = gibbs_RT_ref();
279 
280  // Within the method, we protect against inf results if the exponent is too
281  // high.
282  //
283  // If it is too low, we set the partial pressure to zero. This capability is
284  // needed by the elemental potential method.
285  doublereal pres = 0.0;
286  for (size_t k = 0; k < m_kk; k++) {
287  double tmp = -grt[k] + mu_RT[k];
288  if (tmp < -600.) {
289  m_pp[k] = 0.0;
290  } else if (tmp > 300.0) {
291  double tmp2 = tmp / 300.;
292  tmp2 *= tmp2;
293  m_pp[k] = m_p0 * exp(300.) * tmp2;
294  } else {
295  m_pp[k] = m_p0 * exp(tmp);
296  }
297  pres += m_pp[k];
298  }
299  // set state
300  setState_PX(pres, &m_pp[0]);
301 }

303 void IdealGasPhase::_updateThermo() const
304 {
305  static const int cacheId = m_cache.getId();
306  CachedScalar cached = m_cache.getScalar(cacheId);
307  doublereal tnow = temperature();
308 
309  // If the temperature has changed since the last time these
310  // properties were computed, recompute them.
311  if (cached.state1 != tnow) {
312  m_spthermo->update(tnow, &m_cp0_R[0], &m_h0_RT[0], &m_s0_R[0]);
313  cached.state1 = tnow;
314 
315  // update the species Gibbs functions
316  for (size_t k = 0; k < m_kk; k++) {
317  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
318  }
319  m_logc0 = log(m_p0 / RT());
320  }
321 }
