315 int Sim1D::refine(int loglevel)
316 {
317  int ianalyze, np = 0;
318  vector_fp znew, xnew;
319  doublereal xmid, zmid;
320  std::vector<size_t> dsize;
321 
322  for (size_t n = 0; n < nDomains(); n++) {
323  Domain1D& d = domain(n);
324  Refiner& r = d.refiner();
325 
326  // determine where new points are needed
327  ianalyze = r.analyze(d.grid().size(), d.grid().data(), &m_x[start(n)]);
328  if (ianalyze < 0) {
329  return ianalyze;
330  }
331 
332  if (loglevel > 0) {
333  r.show();
334  }
335 
336  np += r.nNewPoints();
337  size_t comp = d.nComponents();
338 
339  // loop over points in the current grid
340  size_t npnow = d.nPoints();
341  size_t nstart = znew.size();
342  for (size_t m = 0; m < npnow; m++) {
343  if (r.keepPoint(m)) {
344  // add the current grid point to the new grid
345  znew.push_back(d.grid(m));
346 
347  // do the same for the solution at this point
348  for (size_t i = 0; i < comp; i++) {
349  xnew.push_back(value(n, i, m));
350  }
351 
352  // now check whether a new point is needed in the interval to
353  // the right of point m, and if so, add entries to znew and xnew
354  // for this new point
355  if (r.newPointNeeded(m) && m + 1 < npnow) {
356  // add new point at midpoint
357  zmid = 0.5*(d.grid(m) + d.grid(m+1));
358  znew.push_back(zmid);
359  np++;
360 
361  // for each component, linearly interpolate
362  // the solution to this point
363  for (size_t i = 0; i < comp; i++) {
364  xmid = 0.5*(value(n, i, m) + value(n, i, m+1));
365  xnew.push_back(xmid);
366  }
367  }
368  } else {
369  debuglog("refine: discarding point at "+fp2str(d.grid(m))+"\n", loglevel);
370  }
371  }
372  dsize.push_back(znew.size() - nstart);
373  }
374 
375  // At this point, the new grid znew and the new solution vector xnew have
376  // been constructed, but the domains themselves have not yet been modified.
377  // Now update each domain with the new grid.
378 
379  size_t gridstart = 0, gridsize;
380  for (size_t n = 0; n < nDomains(); n++) {
381  Domain1D& d = domain(n);
382  gridsize = dsize[n];
383  d.setupGrid(gridsize, &znew[gridstart]);
384  gridstart += gridsize;
385  }
386 
387  // Replace the current solution vector with the new one
388  m_x = xnew;
389 
390  // resize the work array
391  m_xnew.resize(xnew.size());
392 
393  resize();
394  finalize();
395  return np;
396 }

398 int Sim1D::setFixedTemperature(doublereal t)
399 {
400  int np = 0;
401  vector_fp znew, xnew;
402  doublereal xmid;
403  doublereal zfixed,interp_factor;
404  doublereal z1 = 0.0, z2 = 0.0, t1,t2;
405  size_t n, m, i;
406  size_t m1 = 0;
407  std::vector<size_t> dsize;
408 
409  for (n = 0; n < nDomains(); n++) {
410  bool addnewpt=false;
411  Domain1D& d = domain(n);
412  size_t comp = d.nComponents();
413 
414  // loop over points in the current grid to determine where new point is
415  // needed.
416  FreeFlame* d_free = dynamic_cast<FreeFlame*>(&domain(n));
417  size_t npnow = d.nPoints();
418  size_t nstart = znew.size();
419  if (d_free) {
420  for (m = 0; m < npnow-1; m++) {
421  if (value(n,2,m) == t) {
422  zfixed = d.grid(m);
423  d_free->m_zfixed = zfixed;
424  d_free->m_tfixed = t;
425  addnewpt = false;
426  break;
427  } else if ((value(n,2,m)<t) && (value(n,2,m+1)>t)) {
428  z1 = d.grid(m);
429  m1 = m;
430  z2 = d.grid(m+1);
431  t1 = value(n,2,m);
432  t2 = value(n,2,m+1);
433 
434  zfixed = (z1-z2)/(t1-t2)*(t-t2)+z2;
435  d_free->m_zfixed = zfixed;
436  d_free->m_tfixed = t;
437  addnewpt = true;
438  break;
439  //copy solution domain and push back values
440  }
441  }
442  }
443 
444  for (m = 0; m < npnow; m++) {
445  // add the current grid point to the new grid
446  znew.push_back(d.grid(m));
447 
448  // do the same for the solution at this point
449  for (i = 0; i < comp; i++) {
450  xnew.push_back(value(n, i, m));
451  }
452  if (m==m1 && addnewpt) {
453  //add new point at zfixed
454  znew.push_back(zfixed);
455  np++;
456  interp_factor = (zfixed-z2) / (z1-z2);
457  // for each component, linearly interpolate
458  // the solution to this point
459  for (i = 0; i < comp; i++) {
460  xmid = interp_factor*(value(n, i, m) - value(n, i, m+1)) + value(n,i,m+1);
461  xnew.push_back(xmid);
462  }
463  }
464  }
465  dsize.push_back(znew.size() - nstart);
466  }
467 
468  // At this point, the new grid znew and the new solution vector xnew have
469  // been constructed, but the domains themselves have not yet been modified.
470  // Now update each domain with the new grid.
471  size_t gridstart = 0, gridsize;
472  for (n = 0; n < nDomains(); n++) {
473  Domain1D& d = domain(n);
474  gridsize = dsize[n];
475  d.setupGrid(gridsize, &znew[gridstart]);
476  gridstart += gridsize;
477  }
478 
479  // Replace the current solution vector with the new one
480  m_x = xnew;
481 
482  // resize the work array
483  m_xnew = xnew;
484  resize();
485  finalize();
486  return np;
