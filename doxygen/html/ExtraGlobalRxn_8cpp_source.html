64 //! add the species into the list of products or reactants
65 /*!
66  * Note this function gets called for both the product and reactant sides. However, it's only
67  * called for one side or another.
68  *
69  * @param kkinspec kinetic species index of the product
70  * @param
71  */
72 static void addV(int kkinspec, double ps, std::vector<int>& m_Products,
73  std::vector<doublereal>& m_ProductStoich)
74 {
75  int nsize = m_Products.size();
76  for (int i = 0; i < nsize; i++) {
77  if (m_Products[i] == kkinspec) {
78  m_ProductStoich[i] += ps;
79  return;
80  }
81  }
82  m_Products.push_back(kkinspec);
83  m_ProductStoich.push_back(ps);
84 }

110 void ExtraGlobalRxn::setupElemRxnVector(double* RxnVector,
111  int specialSpecies)
112 {
113  int i;
114  int kkinspec;
115  for (size_t i = 0; i < (size_t) m_nRxns; i++) {
116  m_ElemRxnVector[i] = RxnVector[i];
117  }
118  for (size_t i = 0; i < (size_t) m_nRxns; i++) {
119  if (RxnVector[i] > 0.0) {
120  for (kkinspec = 0; kkinspec < m_nKinSpecies; kkinspec++) {
121  double ps = m_kinetics->productStoichCoeff(kkinspec, i);
122  if (ps > 0.0) {
123  addV(kkinspec, RxnVector[i]* ps, m_Products, m_ProductStoich);
124  addV(kkinspec, RxnVector[i]* ps, m_NetSpecies, m_netStoich);
125  }
126  double rs = m_kinetics->reactantStoichCoeff(kkinspec, i);
127  if (rs > 0.0) {
128  addV(kkinspec, RxnVector[i] * rs, m_Reactants, m_ReactantStoich);
129  addV(kkinspec, -RxnVector[i] * rs, m_NetSpecies, m_netStoich);
130  }
131  }
132  } else if (RxnVector[i] < 0.0) {
133  for (kkinspec = 0; kkinspec < m_nKinSpecies; kkinspec++) {
134  double ps = m_kinetics->productStoichCoeff(kkinspec, i);
135  if (ps > 0.0) {
136  addV(kkinspec,- RxnVector[i]* ps, m_Reactants, m_ReactantStoich);
137  addV(kkinspec, RxnVector[i]* ps, m_NetSpecies, m_netStoich);
138  }
139  double rs = m_kinetics->reactantStoichCoeff(kkinspec, i);
140  if (rs > 0.0) {
141  addV(kkinspec, -RxnVector[i] * rs, m_Products, m_ProductStoich);
142  addV(kkinspec, -RxnVector[i] * rs, m_NetSpecies, m_netStoich);
143  }
144  }
145  }
146  }
147 Recheck:
148  for (i = 0; i < static_cast<int>(m_Products.size()); i++) {
149  if (m_ProductStoich[i] == 0.0) {
150  erase_vi(m_Products, i);
151  erase_vd(m_ProductStoich, i);
152  goto Recheck ;
153  }
154  }
155  for (i = 0; i < static_cast<int>(m_Reactants.size()); i++) {
156  if (m_ReactantStoich[i] == 0.0) {
157  erase_vi(m_Reactants, i);
158  erase_vd(m_ReactantStoich, i);
159  goto Recheck ;
160  }
161  }
162  for (i = 0; i < static_cast<int>(m_NetSpecies.size()); i++) {
163  if (m_netStoich[i] == 0.0) {
164  erase_vi(m_NetSpecies, i);
165  erase_vd(m_netStoich, i);
166  goto Recheck ;
167  }
168  }
169 
170  for (i = 0; i < static_cast<int>(m_Products.size()); i++) {
171  int ik = m_Products[i];
172  for (int j = 0; j < static_cast<int>(m_Reactants.size()); j++) {
173  int jk = m_Reactants[j];
174  if (ik == jk) {
175  if (m_ProductStoich[i] == m_ReactantStoich[j]) {
176  erase_vi(m_Products, i);
177  erase_vd(m_ProductStoich, i);
178  erase_vi(m_Reactants, j);
179  erase_vd(m_ReactantStoich, j);
180  } else if (m_ProductStoich[i] > m_ReactantStoich[j]) {
181  m_ProductStoich[i] -= m_ReactantStoich[j];
182  erase_vi(m_Reactants, j);
183  erase_vd(m_ReactantStoich, j);
184  } else {
185  m_ReactantStoich[j] -= m_ProductStoich[i];
186  erase_vi(m_Products, i);
187  erase_vd(m_ProductStoich, i);
188  }
189  // We just screwed up the indexing -> restart.
190  goto Recheck ;
191  }
192 
193  }
194  }
195  m_nProducts = m_Products.size();
196  m_nReactants = m_Reactants.size();
197  m_nNetSpecies = m_NetSpecies.size();
198 
199  /*
200  * Section to assign the special species
201  */
202  m_SpecialSpecies = specialSpecies;
203  if (specialSpecies == -1) {
204  m_SpecialSpecies = m_Products[0];
205  }
206  bool ifound = false;
207  for (i = 0; i < (int) m_NetSpecies.size(); i++) {
208  int ik = m_NetSpecies[i];
209  if (ik == m_SpecialSpecies) {
210  if (m_netStoich[i] > 0.0) {
211  m_SpecialSpeciesProduct = true;
212  } else {
213  m_SpecialSpeciesProduct = false;
214  }
215  m_SS_index = i;
216  ifound = true;
217  break;
218  }
219  }
220  if (!ifound) {
221  throw CanteraError(":setupElemRxnVector",
222  "Special species not a reactant or product: "
223  + int2str(m_SpecialSpecies));
224  }
225 
226  m_ok = true;

229 std::string ExtraGlobalRxn::reactionString()

236  if (stoich != 1.0) {
237  istoich = (int) stoich;

239  rs += int2str(istoich) + " ";

241  rs += fp2str(stoich) + " ";

243  }
244  string sName = m_kinetics->kineticsSpeciesName(kkinspecies);
245  rs += sName;
246  if (k < (m_nReactants-1)) {
247  rs += " + ";

249  }
