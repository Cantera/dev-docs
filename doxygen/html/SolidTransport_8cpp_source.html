137 doublereal SolidTransport::electricalConductivity()
138 {
139  if (m_nmobile == 0) {
140  // LTPspecies method
141  return m_electConductivity->getSpeciesTransProp();
142  } else {
143  getMobilities(&m_work[0]);
144  int nsp = m_thermo->nSpecies();
145  doublereal sum = 0.0;
146  for (int k = 0; k < nsp; k++) {
147  sum += m_thermo->charge(k) * m_thermo->moleFraction(k) * m_work[k];
148  }
149  return sum * m_thermo->molarDensity();
150  }
