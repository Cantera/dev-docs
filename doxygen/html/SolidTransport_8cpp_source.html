74 doublereal SolidTransport::electricalConductivity()
75 {
76  if (m_nmobile == 0) {
77  // LTPspecies method
78  return m_electConductivity->getSpeciesTransProp();
79  } else {
80  getMobilities(&m_work[0]);
81  doublereal sum = 0.0;
82  for (size_t k = 0; k < m_thermo->nSpecies(); k++) {
83  sum += m_thermo->charge(k) * m_thermo->moleFraction(k) * m_work[k];
84  }
85  return sum * m_thermo->molarDensity();
86  }
87 }
