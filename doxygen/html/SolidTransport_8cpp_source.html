168 
169  // Returns the electron conductivity of the phase
170  /*
171  * The thermo phase needs to be updated (temperature) prior to calling this.
172  * The ionConductivity calculation is handled by subclasses of
173  * LTPspecies as specified in the input file.
174  *
175  * There is also a legacy multicomponent diffusion approach to electrical conductivity.
176  *
177  */
178  doublereal SolidTransport::electricalConductivity() {
179  if ( m_nmobile == 0 ) {
180  // LTPspecies method
181  return m_electConductivity->getSpeciesTransProp();
182  } else {
183  getMobilities(&m_work[0]);
184  int nsp = m_thermo->nSpecies();
185  doublereal sum = 0.0;
186  for (int k = 0; k < nsp; k++) {
187  sum += m_thermo->charge(k) * m_thermo->moleFraction(k) * m_work[k];
188  }
189  return sum * m_thermo->molarDensity();
190  }
191  }

246  * Compute the mobilities of the species from the diffusion coefficients,
247  * using the Einstein relation.
248  */
249  void SolidTransport::getMobilities(doublereal* const mobil) {
250  getMixDiffCoeffs(mobil);
251  doublereal t = m_thermo->temperature();
252  doublereal c1 = ElectronCharge / (Boltzmann * t);
253  for (size_t k = 0; k < m_thermo->nSpecies(); k++) {
254  mobil[k] *= c1;
255  }
256 
257  }
