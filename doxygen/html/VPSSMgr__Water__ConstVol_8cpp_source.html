166 void VPSSMgr_Water_ConstVol::_updateRefStateThermo() const
167 {
168  m_p0 = m_waterSS->pref_safe(m_tlast);
169  m_spthermo->update(m_tlast, &m_cp0_R[0], &m_h0_RT[0], &m_s0_R[0]);
170  for (size_t k = 0; k < m_kk; k++) {
171  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
172  PDSS* kPDSS = m_vptp_ptr->providePDSS(k);
173  kPDSS->setTemperature(m_tlast);
174  }
175  doublereal RT = GasConstant * m_tlast;
176  m_waterSS->setState_TP(m_tlast, m_p0);
177  m_h0_RT[0] = (m_waterSS->enthalpy_mole())/ RT;
178  m_s0_R[0] = (m_waterSS->entropy_mole()) / GasConstant;
179  m_cp0_R[0] = (m_waterSS->cp_mole()) / GasConstant;
180  m_g0_RT[0] = (m_hss_RT[0] - m_sss_R[0]);
181  m_V0[0] = m_vptp_ptr->molecularWeight(0) / (m_waterSS->density());
182  m_waterSS->setState_TP(m_tlast, m_plast);
183 }

187 void VPSSMgr_Water_ConstVol::_updateStandardStateThermo()
188 {
189 
190  doublereal RT = GasConstant * m_tlast;
191  doublereal del_pRT = (m_plast - OneAtm) / (RT);
192 
193  for (size_t k = 1; k < m_kk; k++) {
194  m_hss_RT[k] = m_h0_RT[k] + del_pRT * m_Vss[k];
195  m_cpss_R[k] = m_cp0_R[k];
196  m_sss_R[k] = m_s0_R[k];
197  m_gss_RT[k] = m_hss_RT[k] - m_sss_R[k];
198  // m_Vss[k] constant
199  PDSS* kPDSS = m_vptp_ptr->providePDSS(k);
200  kPDSS->setState_TP(m_tlast, m_plast);
201  }
202  // Do the water
203  m_waterSS->setState_TP(m_tlast, m_plast);
204  m_hss_RT[0] = (m_waterSS->enthalpy_mole())/ RT;
205  m_sss_R[0] = (m_waterSS->entropy_mole()) / GasConstant;
206  m_cpss_R[0] = (m_waterSS->cp_mole()) / GasConstant;
207  m_gss_RT[0] = (m_hss_RT[0] - m_sss_R[0]);
208  m_Vss[0] = (m_vptp_ptr->molecularWeight(0) / m_waterSS->density());
209 }

259 PDSS*
260 VPSSMgr_Water_ConstVol::createInstallPDSS(size_t k, const XML_Node& speciesNode,
261  const XML_Node* const phaseNode_ptr)
262 {
263 
264  PDSS* kPDSS = 0;
265  // Will have to do something for water
266  // -> make sure it's species 0
267  // -> make sure it's designated as a real water EOS
268  if (k == 0) {
269  string xn = speciesNode["name"];
270  if (xn != "H2O(L)") {
271  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
272  "h2o wrong name: " + xn);
273  }
274  const XML_Node* ss = speciesNode.findByName("standardState");
275  std::string model = (*ss)["model"];
276  if (model != "waterIAPWS" && model != "waterPDSS") {
277  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
278  "wrong SS mode: " + model);
279  }
280  delete m_waterSS;
281  m_waterSS = new PDSS_Water(m_vptp_ptr, 0);
282  GeneralSpeciesThermo* genSpthermo = dynamic_cast<GeneralSpeciesThermo*>(m_spthermo);
283  if (!genSpthermo) {
284  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
285  "failed dynamic cast");
286  }
287  genSpthermo->installPDSShandler(k, m_waterSS, this);
288  kPDSS = m_waterSS;
289  } else {
290 
291  VPSSMgr::installSTSpecies(k, speciesNode, phaseNode_ptr);
292 
293  const XML_Node* ss = speciesNode.findByName("standardState");
294  if (!ss) {
295  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
296  "no standardState Node for species " + speciesNode.name());
297  }
298  std::string model = (*ss)["model"];
299  if (model != "constant_incompressible") {
300  throw CanteraError("VPSSMgr_Water_ConstVol::initThermoXML",
301  "standardState model for species isn't "
302  "constant_incompressible: " + speciesNode.name());
303  }
304  if (m_Vss.size() < k+1) {
305  m_Vss.resize(k+1, 0.0);
306  }
307  m_Vss[k] = ctml::getFloat(*ss, "molarVolume", "toSI");
308 
309  // instantiate a new kPDSS object
310  kPDSS = new PDSS_ConstVol(m_vptp_ptr, k, speciesNode, *phaseNode_ptr, true);
311  }
312  return kPDSS;
313 }
