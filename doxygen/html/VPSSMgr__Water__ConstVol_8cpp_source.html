160 void VPSSMgr_Water_ConstVol::_updateRefStateThermo() const
161 {
162  m_p0 = m_waterSS->pref_safe(m_tlast);
163  m_spthermo->update(m_tlast, &m_cp0_R[0], &m_h0_RT[0], &m_s0_R[0]);
164  for (size_t k = 0; k < m_kk; k++) {
165  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
166  PDSS* kPDSS = m_vptp_ptr->providePDSS(k);
167  kPDSS->setTemperature(m_tlast);
168  }
169  doublereal RT = GasConstant * m_tlast;
170  m_waterSS->setState_TP(m_tlast, m_p0);
171  m_h0_RT[0] = (m_waterSS->enthalpy_mole())/ RT;
172  m_s0_R[0] = (m_waterSS->entropy_mole()) / GasConstant;
173  m_cp0_R[0] = (m_waterSS->cp_mole()) / GasConstant;
174  m_g0_RT[0] = (m_hss_RT[0] - m_sss_R[0]);
175  m_V0[0] = m_vptp_ptr->molecularWeight(0) / (m_waterSS->density());
176  m_waterSS->setState_TP(m_tlast, m_plast);
177 }

179 void VPSSMgr_Water_ConstVol::_updateStandardStateThermo()
180 {
181  doublereal RT = GasConstant * m_tlast;
182  doublereal del_pRT = (m_plast - OneAtm) / (RT);
183 
184  for (size_t k = 1; k < m_kk; k++) {
185  m_hss_RT[k] = m_h0_RT[k] + del_pRT * m_Vss[k];
186  m_cpss_R[k] = m_cp0_R[k];
187  m_sss_R[k] = m_s0_R[k];
188  m_gss_RT[k] = m_hss_RT[k] - m_sss_R[k];
189  // m_Vss[k] constant
190  PDSS* kPDSS = m_vptp_ptr->providePDSS(k);
191  kPDSS->setState_TP(m_tlast, m_plast);
192  }
193  // Do the water
194  m_waterSS->setState_TP(m_tlast, m_plast);
195  m_hss_RT[0] = (m_waterSS->enthalpy_mole())/ RT;
196  m_sss_R[0] = (m_waterSS->entropy_mole()) / GasConstant;
197  m_cpss_R[0] = (m_waterSS->cp_mole()) / GasConstant;
198  m_gss_RT[0] = (m_hss_RT[0] - m_sss_R[0]);
199  m_Vss[0] = (m_vptp_ptr->molecularWeight(0) / m_waterSS->density());
200 }

249 PDSS*
250 VPSSMgr_Water_ConstVol::createInstallPDSS(size_t k, const XML_Node& speciesNode,
251  const XML_Node* const phaseNode_ptr)
252 {
253  PDSS* kPDSS = 0;
254  // Will have to do something for water
255  // -> make sure it's species 0
256  // -> make sure it's designated as a real water EOS
257  if (k == 0) {
258  string xn = speciesNode["name"];
259  if (xn != "H2O(L)") {
260  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
261  "h2o wrong name: " + xn);
262  }
263  const XML_Node* ss = speciesNode.findByName("standardState");
264  std::string model = (*ss)["model"];
265  if (model != "waterIAPWS" && model != "waterPDSS") {
266  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
267  "wrong SS mode: " + model);
268  }
269  delete m_waterSS;
270  m_waterSS = new PDSS_Water(m_vptp_ptr, 0);
271  GeneralSpeciesThermo* genSpthermo = dynamic_cast<GeneralSpeciesThermo*>(m_spthermo);
272  if (!genSpthermo) {
273  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
274  "failed dynamic cast");
275  }
276  genSpthermo->installPDSShandler(k, m_waterSS, this);
277  kPDSS = m_waterSS;
278  } else {
279 
280  VPSSMgr::installSTSpecies(k, speciesNode, phaseNode_ptr);
281 
282  const XML_Node* ss = speciesNode.findByName("standardState");
283  if (!ss) {
284  throw CanteraError("VPSSMgr_Water_ConstVol::installSpecies",
285  "no standardState Node for species " + speciesNode.name());
286  }
287  std::string model = (*ss)["model"];
288  if (model != "constant_incompressible") {
289  throw CanteraError("VPSSMgr_Water_ConstVol::initThermoXML",
290  "standardState model for species isn't "
291  "constant_incompressible: " + speciesNode.name());
292  }
293  if (m_Vss.size() < k+1) {
294  m_Vss.resize(k+1, 0.0);
295  }
296  m_Vss[k] = ctml::getFloat(*ss, "molarVolume", "toSI");
297 
298  // instantiate a new kPDSS object
299  kPDSS = new PDSS_ConstVol(m_vptp_ptr, k, speciesNode, *phaseNode_ptr, true);
300  }
301  return kPDSS;
302 }
