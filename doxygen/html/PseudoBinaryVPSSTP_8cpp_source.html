101 void PseudoBinaryVPSSTP::calcPseudoBinaryMoleFractions() const
102 {
103  size_t k;
104  doublereal sumCat;
105  doublereal sumAnion;
106  doublereal sum = 0.0;
107  switch (PBType_) {
108  case PBTYPE_PASSTHROUGH:
109  for (k = 0; k < m_kk; k++) {
110  PBMoleFractions_[k] = moleFractions_[k];
111  }
112  break;
113  case PBTYPE_SINGLEANION:
114  sumCat = 0.0;
115  sumAnion = 0.0;
116  for (k = 0; k < m_kk; k++) {
117  moleFractionsTmp_[k] = moleFractions_[k];
118  }
119  for (k = 0; k < cationList_.size(); k++) {
120  sumCat += moleFractions_[cationList_[k]];
121  }
122  sumAnion = moleFractions_[anionList_[k]];
123  PBMoleFractions_[0] = sumCat -sumAnion;
124  moleFractionsTmp_[indexSpecialSpecies_] -= PBMoleFractions_[0];
125 
126 
127  for (k = 0; k < numCationSpecies_; k++) {
128  PBMoleFractions_[1+k] = moleFractionsTmp_[cationList_[k]];
129  }
130 
131  for (k = 0; k < numPassThroughSpecies_; k++) {
132  PBMoleFractions_[neutralPBindexStart + k] =
133  moleFractions_[cationList_[k]];
134  }
135 
136  sum = std::max(0.0, PBMoleFractions_[0]);
137  for (k = 1; k < numPBSpecies_; k++) {
138  sum += PBMoleFractions_[k];
139  }
140  for (k = 0; k < numPBSpecies_; k++) {
141  PBMoleFractions_[k] /= sum;
142  }
143 
144  break;
145  case PBTYPE_SINGLECATION:
146  throw CanteraError("eosType", "Unknown type");
147 
148  break;
149 
150  case PBTYPE_MULTICATIONANION:
151  throw CanteraError("eosType", "Unknown type");
152 
153  break;
154  default:
155  throw CanteraError("eosType", "Unknown type");
156  break;
157 
158  }
159 }
