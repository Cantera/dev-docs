174 void PseudoBinaryVPSSTP::calcPseudoBinaryMoleFractions() const
175 {
176  size_t k;
177  doublereal sumCat;
178  doublereal sumAnion;
179  doublereal sum = 0.0;
180  switch (PBType_) {
181  case PBTYPE_PASSTHROUGH:
182  for (k = 0; k < m_kk; k++) {
183  PBMoleFractions_[k] = moleFractions_[k];
184  }
185  break;
186  case PBTYPE_SINGLEANION:
187  sumCat = 0.0;
188  sumAnion = 0.0;
189  for (k = 0; k < m_kk; k++) {
190  moleFractionsTmp_[k] = moleFractions_[k];
191  }
192  for (k = 0; k < cationList_.size(); k++) {
193  sumCat += moleFractions_[cationList_[k]];
194  }
195  sumAnion = moleFractions_[anionList_[k]];
196  PBMoleFractions_[0] = sumCat -sumAnion;
197  moleFractionsTmp_[indexSpecialSpecies_] -= PBMoleFractions_[0];
198 
199 
200  for (k = 0; k < numCationSpecies_; k++) {
201  PBMoleFractions_[1+k] = moleFractionsTmp_[cationList_[k]];
202  }
203 
204  for (k = 0; k < numPassThroughSpecies_; k++) {
205  PBMoleFractions_[neutralPBindexStart + k] =
206  moleFractions_[cationList_[k]];
207  }
208 
209  sum = std::max(0.0, PBMoleFractions_[0]);
210  for (k = 1; k < numPBSpecies_; k++) {
211  sum += PBMoleFractions_[k];
212  }
213  for (k = 0; k < numPBSpecies_; k++) {
214  PBMoleFractions_[k] /= sum;
215  }
216 
217  break;
218  case PBTYPE_SINGLECATION:
219  throw CanteraError("eosType", "Unknown type");
220 
221  break;
222 
223  case PBTYPE_MULTICATIONANION:
224  throw CanteraError("eosType", "Unknown type");
225 
226  break;
227  default:
228  throw CanteraError("eosType", "Unknown type");
229  break;
230 
231  }
232 }
