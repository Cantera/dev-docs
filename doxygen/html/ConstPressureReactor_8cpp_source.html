19 {

46 void ConstPressureReactor::updateState(doublereal* y)
47 {
48  // The components of y are [0] the total mass, [1] the total enthalpy,
49  // [2...K+2) are the mass fractions of each species, and [K+2...] are the
50  // coverages of surface species on each wall.
51  m_mass = y[0];
52  m_thermo->setMassFractions_NoNorm(y+2);
53  if (m_energy) {
54  m_thermo->setState_HP(y[1]/m_mass, m_pressure, 1.0e-4);
55  } else {
56  m_thermo->setPressure(m_pressure);
57  }
58  m_vol = m_mass / m_thermo->density();
59  updateSurfaceState(y + m_nsp + 2);
60 
61  // save parameters needed by other connected reactors
62  m_enthalpy = m_thermo->enthalpy_mass();
63  m_intEnergy = m_thermo->intEnergy_mass();
64  m_thermo->saveState(m_state);
65 }

67 void ConstPressureReactor::evalEqs(doublereal time, doublereal* y,
68  doublereal* ydot, doublereal* params)
69 {
70  double dmdt = 0.0; // dm/dt (gas phase)
71  double* dYdt = ydot + 2;
72 
73  m_thermo->restoreState(m_state);
74  applySensitivity(params);
75  evalWalls(time);
76  double mdot_surf = evalSurfaces(time, ydot + m_nsp + 2);
77  dmdt += mdot_surf;
78 
79  const vector_fp& mw = m_thermo->molecularWeights();
80  const doublereal* Y = m_thermo->massFractions();
81 
82  if (m_chem) {
83  m_kin->getNetProductionRates(&m_wdot[0]); // "omega dot"
84  }
85 
86  for (size_t k = 0; k < m_nsp; k++) {
87  // production in gas phase and from surfaces
88  dYdt[k] = (m_wdot[k] * m_vol + m_sdot[k]) * mw[k] / m_mass;
89  // dilution by net surface mass flux
90  dYdt[k] -= Y[k] * mdot_surf / m_mass;
91  }
92 
93  // external heat transfer
94  double dHdt = - m_Q;
95 
96  // add terms for outlets
97  for (size_t i = 0; i < m_outlet.size(); i++) {
98  double mdot_out = m_outlet[i]->massFlowRate(time); // mass flow out of system
99  dmdt -= mdot_out;
100  dHdt -= mdot_out * m_enthalpy;
101  }
102 
103  // add terms for inlets
104  for (size_t i = 0; i < m_inlet.size(); i++) {
105  double mdot_in = m_inlet[i]->massFlowRate(time);
106  dmdt += mdot_in; // mass flow into system
107  for (size_t n = 0; n < m_nsp; n++) {
108  double mdot_spec = m_inlet[i]->outletSpeciesMassFlowRate(n);
109  // flow of species into system and dilution by other species
110  dYdt[n] += (mdot_spec - mdot_in * Y[n]) / m_mass;
111  }
112  dHdt += mdot_in * m_inlet[i]->enthalpy_mass();
113  }
114 
115  ydot[0] = dmdt;
116  if (m_energy) {
117  ydot[1] = dHdt;
118  } else {
119  ydot[1] = 0.0;
120  }
121 
122  // reset sensitivity parameters
123  resetSensitivity(params);
124 }
