83 void WaterSSTP::initThermoXML(XML_Node& phaseNode, const std::string& id)
84 {
85  // Do initializations that don't depend on knowing the XML file
86  initThermo();
87 
88  // Calculate the molecular weight. Note while there may be a very good
89  // calculated weight in the steam table class, using this weight may lead to
90  // codes exhibiting mass loss issues. We need to grab the elemental atomic
91  // weights used in the Element class and calculate a consistent H2O
92  // molecular weight based on that.
93  size_t nH = elementIndex("H");
94  if (nH == npos) {
95  throw CanteraError("WaterSSTP::initThermo",
96  "H not an element");
97  }
98  double mw_H = atomicWeight(nH);
99  size_t nO = elementIndex("O");
100  if (nO == npos) {
101  throw CanteraError("WaterSSTP::initThermo",
102  "O not an element");
103  }
104  double mw_O = atomicWeight(nO);
105  m_mw = 2.0 * mw_H + mw_O;
106  setMolecularWeight(0,m_mw);
107  double one = 1.0;
108  setMoleFractions(&one);
109 
110  // Set the baseline
111  doublereal T = 298.15;
112  Phase::setDensity(7.0E-8);
113  Phase::setTemperature(T);
114 
115  doublereal presLow = 1.0E-2;
116  doublereal oneBar = 1.0E5;
117  doublereal dd = m_sub.density(T, presLow, WATER_GAS, 7.0E-8);
118  setDensity(dd);
119  setTemperature(T);
120  SW_Offset = 0.0;
121  doublereal s = entropy_mole();
122  s -= GasConstant * log(oneBar/presLow);
123  if (s != 188.835E3) {
124  SW_Offset = 188.835E3 - s;
125  }
126  s = entropy_mole();
127  s -= GasConstant * log(oneBar/presLow);
128 
129  doublereal h = enthalpy_mole();
130  if (h != -241.826E6) {
131  EW_Offset = -241.826E6 - h;
132  }
133  h = enthalpy_mole();
134 
135  // Set the initial state of the system to 298.15 K and 1 bar.
136  setTemperature(298.15);
137  double rho0 = m_sub.density(298.15, OneAtm, WATER_LIQUID);
138  setDensity(rho0);
139 
140  m_waterProps.reset(new WaterProps(&m_sub));
141 
142  // We have to do something with the thermo function here.
143  delete m_spthermo;
144  m_spthermo = 0;
145 
146  // Set the flag to say we are ready to calculate stuff
147  m_ready = true;
148 }

196 void WaterSSTP::getEnthalpy_RT_ref(doublereal* hrt) const
197 {
198  doublereal p = pressure();
199  double T = temperature();
200  double dens = density();
201  int waterState = WATER_GAS;
202  double rc = m_sub.Rhocrit();
203  if (dens > rc) {
204  waterState = WATER_LIQUID;
205  }
206  doublereal dd = m_sub.density(T, OneAtm, waterState, dens);
207  if (dd <= 0.0) {
208  throw CanteraError("setPressure", "error");
209  }
210  doublereal h = m_sub.enthalpy();
211  *hrt = (h + EW_Offset) / RT();
212  dd = m_sub.density(T, p, waterState, dens);
213 }

334 doublereal WaterSSTP::dthermalExpansionCoeffdT() const
335 {
336  doublereal pres = pressure();
337  doublereal dens_save = density();
338  double T = temperature();
339  double tt = T - 0.04;
340  doublereal dd = m_sub.density(tt, pres, WATER_LIQUID, dens_save);
341  if (dd < 0.0) {
342  throw CanteraError("WaterSSTP::dthermalExpansionCoeffdT",
343  "Unable to solve for the density at T = {}, P = {}", tt, pres);
344  }
345  doublereal vald = m_sub.coeffThermExp();
346  m_sub.setState_TR(T, dens_save);
347  doublereal val2 = m_sub.coeffThermExp();
348  return (val2 - vald) / 0.04;
