213 void ThermoPhase::setState_HPorUV(doublereal Htarget, doublereal p,
214  doublereal dTtol, bool doUV)
215 {
216  doublereal dt;
217  doublereal v = 0.0;
218 
219  // Assign the specific volume or pressure and make sure it's positive
220  if (doUV) {
221  doublereal v = p;
222  if (v < 1.0E-300) {
223  throw CanteraError("setState_HPorUV (UV)",
224  "Input specific volume is too small or negative. v = " + fp2str(v));
225  }
226  setDensity(1.0/v);
227  } else {
228  if (p < 1.0E-300) {
229  throw CanteraError("setState_HPorUV (HP)",
230  "Input pressure is too small or negative. p = " + fp2str(p));
231  }
232  setPressure(p);
233  }
234  double Tmax = maxTemp() + 0.1;
235  double Tmin = minTemp() - 0.1;
236 
237  // Make sure we are within the temperature bounds at the start
238  // of the iteration
239  double Tnew = temperature();
240  double Tinit = Tnew;
241  if (Tnew > Tmax) {
242  Tnew = Tmax - 1.0;
243  } else if (Tnew < Tmin) {
244  Tnew = Tmin + 1.0;
245  }
246  if (Tnew != Tinit) {
247  setState_conditional_TP(Tnew, p, !doUV);
248  }
249 
250  double Hnew = (doUV) ? intEnergy_mass() : enthalpy_mass();
251  double Cpnew = (doUV) ? cv_mass() : cp_mass();
252  double Htop = Hnew;
253  double Ttop = Tnew;
254  double Hbot = Hnew;
255  double Tbot = Tnew;
256  double Told = Tnew;
257  double Hold = Hnew;
258 
259  bool ignoreBounds = false;
260  // Unstable phases are those for which
261  // cp < 0.0. These are possible for cases where
262  // we have passed the spinodal curve.
263  bool unstablePhase = false;
264  // Counter indicating the last temperature point where the
265  // phase was unstable
266  double Tunstable = -1.0;
267  bool unstablePhaseNew = false;
268 
269  // Newton iteration
270  for (int n = 0; n < 500; n++) {
271  Told = Tnew;
272  Hold = Hnew;
273  double cpd = Cpnew;
274  if (cpd < 0.0) {
275  unstablePhase = true;
276  Tunstable = Tnew;
277  }
278  // limit step size to 100 K
279  dt = clip((Htarget - Hold)/cpd, -100.0, 100.0);
280 
281  // Calculate the new T
282  Tnew = Told + dt;
283 
284  // Limit the step size so that we are convergent
285  // This is the step that makes it different from a
286  // Newton's algorithm
287  if ((dt > 0.0 && unstablePhase) || (dt <= 0.0 && !unstablePhase)) {
288  if (Hbot < Htarget && Tnew < (0.75 * Tbot + 0.25 * Told)) {
289  dt = 0.75 * (Tbot - Told);
290  Tnew = Told + dt;
291  }
292  } else if (Htop > Htarget && Tnew > (0.75 * Ttop + 0.25 * Told)) {
293  dt = 0.75 * (Ttop - Told);
294  Tnew = Told + dt;
295  }
296 
297  // Check Max and Min values
298  if (Tnew > Tmax && !ignoreBounds) {
299  setState_conditional_TP(Tmax, p, !doUV);
300  double Hmax = (doUV) ? intEnergy_mass() : enthalpy_mass();
301  if (Hmax >= Htarget) {
302  if (Htop < Htarget) {
303  Ttop = Tmax;
304  Htop = Hmax;
305  }
306  } else {
307  Tnew = Tmax + 1.0;
308  ignoreBounds = true;
309  }
310  }
311  if (Tnew < Tmin && !ignoreBounds) {
312  setState_conditional_TP(Tmin, p, !doUV);
313  double Hmin = (doUV) ? intEnergy_mass() : enthalpy_mass();
314  if (Hmin <= Htarget) {
315  if (Hbot > Htarget) {
316  Tbot = Tmin;
317  Hbot = Hmin;
318  }
319  } else {
320  Tnew = Tmin - 1.0;
321  ignoreBounds = true;
322  }
323  }
324 
325  // Try to keep phase within its region of stability
326  // -> Could do a lot better if I calculate the
327  // spinodal value of H.
328  for (int its = 0; its < 10; its++) {
329  Tnew = Told + dt;
330  if (Tnew < Told / 3.0) {
331  Tnew = Told / 3.0;
332  dt = -2.0 * Told / 3.0;
333  }
334  setState_conditional_TP(Tnew, p, !doUV);
335  if (doUV) {
336  Hnew = intEnergy_mass();
337  Cpnew = cv_mass();
338  } else {
339  Hnew = enthalpy_mass();
340  Cpnew = cp_mass();
341  }
342  if (Cpnew < 0.0) {
343  unstablePhaseNew = true;
344  Tunstable = Tnew;
345  } else {
346  unstablePhaseNew = false;
347  break;
348  }
349  if (unstablePhase == false && unstablePhaseNew == true) {
350  dt *= 0.25;
351  }
352  }
353 
354  if (Hnew == Htarget) {
355  return;
356  } else if (Hnew > Htarget && (Htop < Htarget || Hnew < Htop)) {
357  Htop = Hnew;
358  Ttop = Tnew;
359  } else if (Hnew < Htarget && (Hbot > Htarget || Hnew > Hbot)) {
360  Hbot = Hnew;
361  Tbot = Tnew;
362  }
363  // Convergence in H
364  double Herr = Htarget - Hnew;
365  double acpd = std::max(fabs(cpd), 1.0E-5);
366  double denom = std::max(fabs(Htarget), acpd * dTtol);
367  double HConvErr = fabs((Herr)/denom);
368  if (HConvErr < 0.00001 *dTtol || fabs(dt) < dTtol) {
369  return;
370  }
371  }
372  // We are here when there hasn't been convergence
373  /*
374  * Formulate a detailed error message, since questions seem to
375  * arise often about the lack of convergence.
376  */
377  string ErrString = "No convergence in 500 iterations\n";
378  if (doUV) {
379  ErrString += "\tTarget Internal Energy = " + fp2str(Htarget) + "\n";
380  ErrString += "\tCurrent Specific Volume = " + fp2str(v) + "\n";
381  ErrString += "\tStarting Temperature = " + fp2str(Tinit) + "\n";
382  ErrString += "\tCurrent Temperature = " + fp2str(Tnew) + "\n";
383  ErrString += "\tCurrent Internal Energy = " + fp2str(Hnew) + "\n";
384  ErrString += "\tCurrent Delta T = " + fp2str(dt) + "\n";
385  } else {
386  ErrString += "\tTarget Enthalpy = " + fp2str(Htarget) + "\n";
387  ErrString += "\tCurrent Pressure = " + fp2str(p) + "\n";
388  ErrString += "\tStarting Temperature = " + fp2str(Tinit) + "\n";
389  ErrString += "\tCurrent Temperature = " + fp2str(Tnew) + "\n";
390  ErrString += "\tCurrent Enthalpy = " + fp2str(Hnew) + "\n";
391  ErrString += "\tCurrent Delta T = " + fp2str(dt) + "\n";
392  }
393  if (unstablePhase) {
394  ErrString += "\t - The phase became unstable (Cp < 0) T_unstable_last = "
395  + fp2str(Tunstable) + "\n";
396  }
397  if (doUV) {
398  throw CanteraError("setState_HPorUV (UV)", ErrString);
399  } else {
400  throw CanteraError("setState_HPorUV (HP)", ErrString);
401  }
402 }

783 void ThermoPhase::getdlnActCoeffdlnN_numderiv(const size_t ld, doublereal* const dlnActCoeffdlnN)
784 {
785  double deltaMoles_j = 0.0;
786  double pres = pressure();
787 
788  /*
789  * Evaluate the current base activity coefficients if necessary
790  */
791  std::vector<double> ActCoeff_Base(m_kk);
792  getActivityCoefficients(DATA_PTR(ActCoeff_Base));
793  std::vector<double> Xmol_Base(m_kk);
794  getMoleFractions(DATA_PTR(Xmol_Base));
795 
796  // Make copies of ActCoeff and Xmol_ for use in taking differences
797  std::vector<double> ActCoeff(m_kk);
798  std::vector<double> Xmol(m_kk);
799  double v_totalMoles = 1.0;
800  double TMoles_base = v_totalMoles;
801 
802  /*
803  * Loop over the columns species to be deltad
804  */
805  for (size_t j = 0; j < m_kk; j++) {
806  /*
807  * Calculate a value for the delta moles of species j
808  * -> NOte Xmol_[] and Tmoles are always positive or zero
809  * quantities.
810  * -> experience has shown that you always need to make the deltas greater than needed to
811  * change the other mole fractions in order to capture some effects.
812  */
813  double moles_j_base = v_totalMoles * Xmol_Base[j];
814  deltaMoles_j = 1.0E-7 * moles_j_base + v_totalMoles * 1.0E-13 + 1.0E-150;
815  /*
816  * Now, update the total moles in the phase and all of the
817  * mole fractions based on this.
818  */
819  v_totalMoles = TMoles_base + deltaMoles_j;
820  for (size_t k = 0; k < m_kk; k++) {
821  Xmol[k] = Xmol_Base[k] * TMoles_base / v_totalMoles;
822  }
823  Xmol[j] = (moles_j_base + deltaMoles_j) / v_totalMoles;
824 
825  /*
826  * Go get new values for the activity coefficients.
827  * -> Note this calls setState_PX();
828  */
829  setState_PX(pres, DATA_PTR(Xmol));
830  getActivityCoefficients(DATA_PTR(ActCoeff));
831 
832  /*
833  * Calculate the column of the matrix
834  */
835  double* const lnActCoeffCol = dlnActCoeffdlnN + ld * j;
836  for (size_t k = 0; k < m_kk; k++) {
837  lnActCoeffCol[k] = (2*moles_j_base + deltaMoles_j) *(ActCoeff[k] - ActCoeff_Base[k]) /
838  ((ActCoeff[k] + ActCoeff_Base[k]) * deltaMoles_j);
839  }
840  /*
841  * Revert to the base case Xmol_, v_totalMoles
842  */
843  v_totalMoles = TMoles_base;
844  Xmol = Xmol_Base;
845  }
846  /*
847  * Go get base values for the activity coefficients.
848  * -> Note this calls setState_TPX() again;
849  * -> Just wanted to make sure that cantera is in sync
850  * with VolPhase after this call.
851  */
852  setState_PX(pres, DATA_PTR(Xmol_Base));
853 }

855 std::string ThermoPhase::report(bool show_thermo, doublereal threshold) const
856 {
857  char p[800];
858  string s = "";
859  try {
860  if (name() != "") {
861  sprintf(p, " \n %s:\n", name().c_str());
862  s += p;

865  s += p;
866  sprintf(p, " pressure %12.6g Pa\n", pressure());
867  s += p;

869  s += p;
870  sprintf(p, " mean mol. weight %12.6g amu\n", meanMolecularWeight());
871  s += p;
872 
873  doublereal phi = electricPotential();
874  if (phi != 0.0) {
875  sprintf(p, " potential %12.6g V\n", phi);
876  s += p;
877  }
878  if (show_thermo) {
879  sprintf(p, " \n");
880  s += p;
881  sprintf(p, " 1 kg 1 kmol\n");
882  s += p;
883  sprintf(p, " ----------- ------------\n");
884  s += p;
885  sprintf(p, " enthalpy %12.5g %12.4g J\n",
886  enthalpy_mass(), enthalpy_mole());
887  s += p;
888  sprintf(p, " internal energy %12.5g %12.4g J\n",
889  intEnergy_mass(), intEnergy_mole());
890  s += p;
891  sprintf(p, " entropy %12.5g %12.4g J/K\n",
892  entropy_mass(), entropy_mole());
893  s += p;
894  sprintf(p, " Gibbs function %12.5g %12.4g J\n",
895  gibbs_mass(), gibbs_mole());
896  s += p;
897  sprintf(p, " heat capacity c_p %12.5g %12.4g J/K\n",
898  cp_mass(), cp_mole());
899  s += p;
900  try {
901  sprintf(p, " heat capacity c_v %12.5g %12.4g J/K\n",
902  cv_mass(), cv_mole());
903  s += p;
904  } catch (CanteraError& err) {
905  err.save();
906  sprintf(p, " heat capacity c_v <not implemented> \n");
907  s += p;
908  }
909  }
910 
911  size_t kk = nSpecies();
912  vector_fp x(kk);
913  vector_fp y(kk);
914  vector_fp mu(kk);
915  getMoleFractions(&x[0]);
916  getMassFractions(&y[0]);
917  getChemPotentials(&mu[0]);
918  doublereal rt = GasConstant * temperature();
919 
920  int nMinor = 0;
921  doublereal xMinor = 0.0;
922  doublereal yMinor = 0.0;
923  if (show_thermo) {
924  sprintf(p, " \n X "
925  " Y Chem. Pot. / RT \n");
926  s += p;
927  sprintf(p, " ------------- "
928  "------------ ------------\n");
929  s += p;
930  for (size_t k = 0; k < kk; k++) {
931  if (x[k] >= threshold) {
932  if (x[k] > SmallNumber) {
933  sprintf(p, "%18s %12.6g %12.6g %12.6g\n",
934  speciesName(k).c_str(), x[k], y[k], mu[k]/rt);
935  } else {
936  sprintf(p, "%18s %12.6g %12.6g \n",
937  speciesName(k).c_str(), x[k], y[k]);
938  }
939  s += p;
940  } else {
941  nMinor++;
942  xMinor += x[k];
943  yMinor += y[k];
944  }
945  }
946  } else {
947  sprintf(p, " \n X "
948  " Y\n");
949  s += p;
950  sprintf(p, " -------------"
951  " ------------\n");
952  s += p;
953  for (size_t k = 0; k < kk; k++) {
954  if (x[k] >= threshold) {
955  sprintf(p, "%18s %12.6g %12.6g\n",
956  speciesName(k).c_str(), x[k], y[k]);
957  s += p;
958  } else {
959  nMinor++;
960  xMinor += x[k];
961  yMinor += y[k];
962  }
963  }
964  }
965  if (nMinor) {
966  sprintf(p, " [%+5i minor] %12.6g %12.6g\n",
967  nMinor, xMinor, yMinor);
968  s += p;
969  }
970  }
971  catch (CanteraError& err) {
972  err.save();
973  }
974  return s;
975 }
