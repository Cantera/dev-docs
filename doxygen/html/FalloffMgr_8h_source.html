74  /**
75  * Given a vector of reduced pressures for each falloff reaction,
76  * replace each entry by the value of the falloff function.
77  */
78  void pr_to_falloff(doublereal* values, const doublereal* work) {
79  for (size_t i = 0; i < m_rxn.size(); i++) {
80  double pr = values[m_rxn[i]];
81  if (m_reactionType[i] == FALLOFF_RXN) {
82  // Pr / (1 + Pr) * F
83  values[m_rxn[i]] *=
84  m_falloff[i]->F(pr, work + m_offset[i]) /(1.0 + pr);
85  } else {
86  // 1 / (1 + Pr) * F
87  values[m_rxn[i]] =
88  m_falloff[i]->F(pr, work + m_offset[i]) /(1.0 + pr);
89  }
90  }
91  }
