41 //! Factory to build instances of classes that manage the
42 //! standard-state thermodynamic properties of a set of species.
43 /*!
44  * This class is responsible for making the decision concerning which
45  * derivative of VPSSMgr object to use. The VPSSMgr object is used to calculate
46  * thermodynamic functions for the standard state. It queries the database of
47  * species to understand what the requirements are for the submodels for all of
48  * the species in the phase. Then, it picks the derived VPSSMgr object to use
49  * and passes it back to the calling routine. It doesn't load any data into the
50  * derived VPSSMgr object.
51  *
52  * Making the choice of VPSSMgr types is the only thing this class does.
53  *
54  * This class is implemented as a singleton -- one in which only one instance is
55  * needed. The recommended way to access the factory is to call this static
56  * method, which instantiates the class if it is the first call, but otherwise
57  * simply returns the pointer to the existing instance.
58  *
59  * @ingroup mgrpdssthermocalc
60  */
61 class VPSSMgrFactory : public FactoryBase
62 {
63 public:
64  //! Static method to return an instance of this class
65  /*!
66  * This class is implemented as a singleton -- one in which only one
67  * instance is needed. The recommended way to access the factory is to call
68  * this static method, which instantiates the class if it is the first call,
69  * but otherwise simply returns the pointer to the existing instance.
70  */
71  static VPSSMgrFactory* factory() {
72  std::unique_lock<std::mutex> lock(vpss_species_thermo_mutex);
73  if (!s_factory) {
74  s_factory = new VPSSMgrFactory;
75  }
76  return s_factory;
77  }
