957 void LiquidTransport::stefan_maxwell_solve()
958 {
959  doublereal tmp;
960  m_B.resize(m_nsp, m_nDim, 0.0);
961  m_A.resize(m_nsp, m_nsp, 0.0);
962 
963  //! grab a local copy of the molecular weights
964  const vector_fp& M = m_thermo->molecularWeights();
965  //! grad a local copy of the ion molar volume (inverse total ion concentration)
966  const doublereal vol = m_thermo->molarVolume();
967 
968  /*
969  * Update the temperature, concentrations and diffusion coefficients in the mixture.
970  */
971  update_T();
972  update_C();
973  if (!m_diff_temp_ok) {
974  updateDiff_T();
975  }
976 
977  double T = m_thermo->temperature();
978 
979  update_Grad_lnAC() ;
980 
981  m_thermo->getActivityCoefficients(DATA_PTR(m_actCoeff));
982 
983  /*
984  * Calculate the electrochemical potential gradient. This is the
985  * driving force for relative diffusional transport.
986  *
987  * Here we calculate
988  *
989  * X_i * (grad (mu_i) + S_i grad T - M_i / dens * grad P
990  *
991  * This is Eqn. 13-1 p. 318 Newman. The original equation is from
992  * Hershfeld, Curtis, and Bird.
993  *
994  * S_i is the partial molar entropy of species i. This term will cancel
995  * out a lot of the grad T terms in grad (mu_i), therefore simplifying
996  * the expression.
997  *
998  * Ok I think there may be many ways to do this. One way is to do it via basis
999  * functions, at the nodes, as a function of the variables in the problem.
1000  *
1001  * For calculation of molality based thermo systems, we current get
1002  * the molar based values. This may change.
1003  *
1004  * Note, we have broken the symmetry of the matrix here, due to
1005  * considerations involving species concentrations going to zero.
1006  *
1007  */
1008  for (size_t a = 0; a < m_nDim; a++) {
1009  for (size_t i = 0; i < m_nsp; i++) {
1010  m_Grad_mu[a*m_nsp + i] =
1011  m_chargeSpecies[i] * Faraday * m_Grad_V[a]
1012  + GasConstant * T * m_Grad_lnAC[a*m_nsp+i];
1013  }
1014  }
1015 
1016  if (m_thermo->activityConvention() == cAC_CONVENTION_MOLALITY) {
1017  int iSolvent = 0;
1018  double mwSolvent = m_thermo->molecularWeight(iSolvent);
1019  double mnaught = mwSolvent/ 1000.;
1020  double lnmnaught = log(mnaught);
1021  for (size_t a = 0; a < m_nDim; a++) {
1022  for (size_t i = 1; i < m_nsp; i++) {
1023  m_Grad_mu[a*m_nsp + i] -=
1024  m_molefracs[i] * GasConstant * m_Grad_T[a] * lnmnaught;
1025  }
1026  }
1027  }
1028 
1029  /*
1030  * Just for Note, m_A(i,j) refers to the ith row and jth column.
1031  * They are still fortran ordered, so that i varies fastest.
1032  */
1033 
1034  double condSum1;
1035  const doublereal invRT = 1.0 / (GasConstant * T);
1036  switch (m_nDim) {
1037  case 1: /* 1-D approximation */
1038 
1039  m_B(0,0) = 0.0;
1040  //equation for the reference velocity
1041  for (size_t j = 0; j < m_nsp; j++) {
1042  if (m_velocityBasis == VB_MOLEAVG) {
1043  m_A(0,j) = m_molefracs_tran[j];
1044  } else if (m_velocityBasis == VB_MASSAVG) {
1045  m_A(0,j) = m_massfracs_tran[j];
1046  } else if ((m_velocityBasis >= 0)
1047  && (m_velocityBasis < static_cast<int>(m_nsp)))
1048  // use species number m_velocityBasis as reference velocity
1049  if (m_velocityBasis == static_cast<int>(j)) {
1050  m_A(0,j) = 1.0;
1051  } else {
1052  m_A(0,j) = 0.0;
1053  }
1054  else
1055  throw CanteraError("LiquidTransport::stefan_maxwell_solve",
1056  "Unknown reference velocity provided.");
1057  }
1058  for (size_t i = 1; i < m_nsp; i++) {
1059  m_B(i,0) = m_Grad_mu[i] * invRT;
1060  m_A(i,i) = 0.0;
1061  for (size_t j = 0; j < m_nsp; j++) {
1062  if (j != i) {
1063  tmp = m_molefracs_tran[j] * m_bdiff(i,j);
1064  m_A(i,i) -= tmp;
1065  m_A(i,j) = tmp;
1066  }
1067  }
1068  }
1069 
1070  //! invert and solve the system Ax = b. Answer is in m_B
1071  solve(m_A, m_B);
1072 
1073  condSum1 = 0;
1074  for (size_t i = 0; i < m_nsp; i++) {
1075  condSum1 -= Faraday*m_chargeSpecies[i]*m_B(i,0)*m_molefracs_tran[i]/vol;
1076  }
1077  break;
1078  case 2: /* 2-D approximation */
1079  m_B(0,0) = 0.0;
1080  m_B(0,1) = 0.0;
1081  //equation for the reference velocity
1082  for (size_t j = 0; j < m_nsp; j++) {
1083  if (m_velocityBasis == VB_MOLEAVG) {
1084  m_A(0,j) = m_molefracs_tran[j];
1085  } else if (m_velocityBasis == VB_MASSAVG) {
1086  m_A(0,j) = m_massfracs_tran[j];
1087  } else if ((m_velocityBasis >= 0)
1088  && (m_velocityBasis < static_cast<int>(m_nsp)))
1089  // use species number m_velocityBasis as reference velocity
1090  if (m_velocityBasis == static_cast<int>(j)) {
1091  m_A(0,j) = 1.0;
1092  } else {
1093  m_A(0,j) = 0.0;
1094  }
1095  else
1096  throw CanteraError("LiquidTransport::stefan_maxwell_solve",
1097  "Unknown reference velocity provided.");
1098  }
1099  for (size_t i = 1; i < m_nsp; i++) {
1100  m_B(i,0) = m_Grad_mu[i] * invRT;
1101  m_B(i,1) = m_Grad_mu[m_nsp + i] * invRT;
1102  m_A(i,i) = 0.0;
1103  for (size_t j = 0; j < m_nsp; j++) {
1104  if (j != i) {
1105  tmp = m_molefracs_tran[j] * m_bdiff(i,j);
1106  m_A(i,i) -= tmp;
1107  m_A(i,j) = tmp;
1108  }
1109  }
1110  }
1111 
1112  //! invert and solve the system Ax = b. Answer is in m_B
1113  solve(m_A, m_B);
1114 
1115 
1116  break;
1117 
1118  case 3: /* 3-D approximation */
1119  m_B(0,0) = 0.0;
1120  m_B(0,1) = 0.0;
1121  m_B(0,2) = 0.0;
1122  //equation for the reference velocity
1123  for (size_t j = 0; j < m_nsp; j++) {
1124  if (m_velocityBasis == VB_MOLEAVG) {
1125  m_A(0,j) = m_molefracs_tran[j];
1126  } else if (m_velocityBasis == VB_MASSAVG) {
1127  m_A(0,j) = m_massfracs_tran[j];
1128  } else if ((m_velocityBasis >= 0)
1129  && (m_velocityBasis < static_cast<int>(m_nsp)))
1130  // use species number m_velocityBasis as reference velocity
1131  if (m_velocityBasis == static_cast<int>(j)) {
1132  m_A(0,j) = 1.0;
1133  } else {
1134  m_A(0,j) = 0.0;
1135  }
1136  else
1137  throw CanteraError("LiquidTransport::stefan_maxwell_solve",
1138  "Unknown reference velocity provided.");
1139  }
1140  for (size_t i = 1; i < m_nsp; i++) {
1141  m_B(i,0) = m_Grad_mu[i] * invRT;
1142  m_B(i,1) = m_Grad_mu[m_nsp + i] * invRT;
1143  m_B(i,2) = m_Grad_mu[2*m_nsp + i] * invRT;
1144  m_A(i,i) = 0.0;
1145  for (size_t j = 0; j < m_nsp; j++) {
