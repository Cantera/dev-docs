398 // Routine strips off white space from a c character string
399 /*
400  * This routine strips off blanks and tabs (only leading and trailing
401  * characters) in 'str'. On return, it returns the number of
402  * characters still included in the string (excluding the null character).
403  *
404  * Comments are excluded -> All instances of the comment character, '!',
405  * are replaced by '\0' thereby terminating
406  * the string
407  *
408  * Parameter list:
409  *
410  * @param str On output 'str' contains the same characters as on
411  * input except the leading and trailing white space and
412  * comments have been removed.
413  */
414 int stripLTWScstring(char str[])
415 {
416  int i = 0, j = 0;
417  char ch;
418  const char COM_CHAR='\0';
419  /*
420  * Quick Returns
421  */
422  if ((str == 0) || (str[0] == '\0')) {
423  return (0);
424  }
425 
426  /* Find first non-space character character */
427  while (((ch = str[i]) != '\0') && isspace(ch)) {
428  i++;
429  }
430 
431  /*
432  * Move real part of str to the front by copying the string
433  * - Comments are handled here, by terminating the copy at the
434  * first comment indicator, and inserting the null character at
435  * that point.
436  */
437 
438  while ((ch = str[j+i]) != '\0' &&
439  (ch != COM_CHAR)) {
440  str[j] = ch;
441  j++;
442  }
443  str[j] = '\0';
444  j--;
445  /* Remove trailing white space by inserting a null character */
446  while ((j != -1) && isspace(str[j])) {
447  j--;
448  }
449  j++;
450  str[j] = '\0';
451  return (j);
452 }

454 // Translate a char string into a single double
455 /*
456  * atofCheck is a wrapper around the C stdlib routine atof().
457  * It does quite a bit more error checking than atof() or
458  * strtod(), and is quite a bit more restrictive.
459  *
460  * First it interprets both E, e, d, and D as exponents.
461  * atof() only interprets e or E as an exponent character.
462  *
463  * It only accepts a string as well formed if it consists as a
464  * single token. Multiple words will produce an error message
465  *
466  * It will produce an error for NAN and inf entries as well,
467  * in contrast to atof() or strtod().
468  * The user needs to know that a serious numerical issue
469  * has occurred.
470  *
471  * It does not accept hexadecimal numbers.
472  *
473  * @param dptr pointer to the input c string
474  * @return Returns the double
475  *
476  * On any error, it will throw a CanteraError signal.
477  */
478 doublereal atofCheck(const char* const dptr)
479 {
480  if (!dptr) {
481  throw CanteraError("atofCheck", "null pointer to string");
482  }
483  char* eptr = (char*) malloc(strlen(dptr)+1);
484  strcpy(eptr, dptr);
485  int ll = stripLTWScstring(eptr);
486  if (ll == 0) {
487  throw CanteraError("atofCheck", "string has zero length");
488  }
489  int numDot = 0;
490  int numExp = 0;
491  char ch;
492  int istart = 0;
493  ch = eptr[0];
494  if (ch == '+' || ch == '-') {
495  istart = 1;
496  }
497  for (int i = istart; i < ll; i++) {
498  ch = eptr[i];
499  if (isdigit(ch)) {
500  } else if (ch == '.') {
501  numDot++;
502  if (numDot > 1) {
503  free(eptr);
504  throw CanteraError("atofCheck",
505  "string has more than one .");
506  }
507  } else if (ch == 'e' || ch == 'E' || ch == 'd' || ch == 'D') {
508  numExp++;
509  eptr[i] = 'E';
510  if (numExp > 1) {
511  free(eptr);
512  throw CanteraError("atofCheck",
513  "string has more than one exp char");
514  }
515  ch = eptr[i+1];
516  if (ch == '+' || ch == '-') {
517  i++;
518  }
519  } else {
520  std::string hh(dptr);
521  free(eptr);
522  throw CanteraError("atofCheck",
523  "Trouble processing string, " + hh);
524  }
525  }
526  doublereal rval = atof(eptr);
527  free(eptr);
528  return rval;
529 }
