46 VPStandardStateTP&
47 VPStandardStateTP::operator=(const VPStandardStateTP& b)
48 {
49  if (&b != this) {
50  /*
51  * Mostly, this is a passthrough to the underlying
52  * assignment operator for the ThermoPhase parent object.
53  */
54  ThermoPhase::operator=(b);
55  /*
56  * However, we have to handle data that we own.
57  */
58  m_Pcurrent = b.m_Pcurrent;
59  m_Tlast_ss = b.m_Tlast_ss;
60  m_Plast_ss = b.m_Plast_ss;
61  m_P0 = b.m_P0;
62 
63  /*
64  * Duplicate the pdss objects
65  */
66  if (m_PDSS_storage.size() > 0) {
67  for (int k = 0; k < (int) m_PDSS_storage.size(); k++) {
68  delete m_PDSS_storage[k];
69  }
70  }
71  m_PDSS_storage.resize(m_kk);
72  for (size_t k = 0; k < m_kk; k++) {
73  PDSS* ptmp = b.m_PDSS_storage[k];
74  m_PDSS_storage[k] = ptmp->duplMyselfAsPDSS();
75  }
76 
77  /*
78  * Duplicate the VPSS Manager object that conducts the calculations
79  */
80  delete m_VPSS_ptr;
81  m_VPSS_ptr = (b.m_VPSS_ptr)->duplMyselfAsVPSSMgr();
82 
83  /*
84  * The VPSSMgr object contains shallow pointers. Whenever you have shallow
85  * pointers, they have to be fixed up to point to the correct objects referring
86  * back to this ThermoPhase's properties.
87  */
88  m_VPSS_ptr->initAllPtrs(this, m_spthermo);
89  /*
90  * The PDSS objects contains shallow pointers. Whenever you have shallow
91  * pointers, they have to be fixed up to point to the correct objects referring
92  * back to this ThermoPhase's properties. This function also sets m_VPSS_ptr
93  * so it occurs after m_VPSS_ptr is set.
94  */
95  for (size_t k = 0; k < m_kk; k++) {
96  PDSS* ptmp = m_PDSS_storage[k];
97  ptmp->initAllPtrs(this, m_VPSS_ptr, m_spthermo);
98  }
99  /*
100  * Ok, the VPSSMgr object is ready for business.
101  * We need to resync the temperature and the pressure of the new standard states
102  * with what is stored in this object.
103  */
104  m_VPSS_ptr->setState_TP(m_Tlast_ss, m_Plast_ss);
105  }
106  return *this;
107 }

299 void VPStandardStateTP::setState_TP(doublereal t, doublereal pres)
300 {
301  /*
302  * A pretty tricky algorithm is needed here, due to problems involving
303  * standard states of real fluids. For those cases you need
304  * to combine the T and P specification for the standard state, or else
305  * you may venture into the forbidden zone, especially when nearing the
306  * triple point.
307  * Therefore, we need to do the standard state thermo calc with the
308  * (t, pres) combo.
309  */
310  Phase::setTemperature(t);
311  m_Pcurrent = pres;
312  updateStandardStateThermo();
313  /*
314  * Now, we still need to do the calculations for general ThermoPhase objects.
315  * So, we switch back to a virtual function call, setTemperature, and
316  * setPressure to recalculate stuff for child ThermoPhase objects of
317  * the VPStandardStateTP object. At this point,
318  * we haven't touched m_tlast or m_plast, so some calculations may still
319  * need to be done at the ThermoPhase object level.
320  */
321  //setTemperature(t);
322  //setPressure(pres);
323  calcDensity();
324 }
