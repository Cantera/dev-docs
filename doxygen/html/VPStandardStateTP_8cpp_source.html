48 VPStandardStateTP&
49 VPStandardStateTP::operator=(const VPStandardStateTP& b)
50 {
51  if (&b != this) {
52  /*
53  * Mostly, this is a passthrough to the underlying
54  * assignment operator for the ThermoPhase parent object.
55  */
56  ThermoPhase::operator=(b);
57  /*
58  * However, we have to handle data that we own.
59  */
60  m_Pcurrent = b.m_Pcurrent;
61  m_Tlast_ss = b.m_Tlast_ss;
62  m_Plast_ss = b.m_Plast_ss;
63  m_P0 = b.m_P0;
64 
65  /*
66  * Duplicate the pdss objects
67  */
68  if (m_PDSS_storage.size() > 0) {
69  for (int k = 0; k < (int) m_PDSS_storage.size(); k++) {
70  delete m_PDSS_storage[k];
71  }
72  }
73  m_PDSS_storage.resize(m_kk);
74  for (size_t k = 0; k < m_kk; k++) {
75  PDSS* ptmp = b.m_PDSS_storage[k];
76  m_PDSS_storage[k] = ptmp->duplMyselfAsPDSS();
77  }
78 
79  /*
80  * Duplicate the VPSS Manager object that conducts the calculations
81  */
82  delete m_VPSS_ptr;
83  m_VPSS_ptr = (b.m_VPSS_ptr)->duplMyselfAsVPSSMgr();
84 
85  /*
86  * The VPSSMgr object contains shallow pointers. Whenever you have shallow
87  * pointers, they have to be fixed up to point to the correct objects referring
88  * back to this ThermoPhase's properties.
89  */
90  m_VPSS_ptr->initAllPtrs(this, m_spthermo);
91  /*
92  * The PDSS objects contains shallow pointers. Whenever you have shallow
93  * pointers, they have to be fixed up to point to the correct objects referring
94  * back to this ThermoPhase's properties. This function also sets m_VPSS_ptr
95  * so it occurs after m_VPSS_ptr is set.
96  */
97  for (size_t k = 0; k < m_kk; k++) {
98  PDSS* ptmp = m_PDSS_storage[k];
99  ptmp->initAllPtrs(this, m_VPSS_ptr, m_spthermo);
100  }
101  /*
102  * Ok, the VPSSMgr object is ready for business.
103  * We need to resync the temperature and the pressure of the new standard states
104  * with what is stored in this object.
105  */
106  m_VPSS_ptr->setState_TP(m_Tlast_ss, m_Plast_ss);
107  }
108  return *this;
109 }

309 void VPStandardStateTP::setState_TP(doublereal t, doublereal pres)
310 {
311  /*
312  * A pretty tricky algorithm is needed here, due to problems involving
313  * standard states of real fluids. For those cases you need
314  * to combine the T and P specification for the standard state, or else
315  * you may venture into the forbidden zone, especially when nearing the
316  * triple point.
317  * Therefore, we need to do the standard state thermo calc with the
318  * (t, pres) combo.
319  */
320  Phase::setTemperature(t);
321  m_Pcurrent = pres;
322  updateStandardStateThermo();
323  /*
324  * Now, we still need to do the calculations for general ThermoPhase objects.
325  * So, we switch back to a virtual function call, setTemperature, and
326  * setPressure to recalculate stuff for child ThermoPhase objects of
327  * the VPStandardStateTP object. At this point,
328  * we haven't touched m_tlast or m_plast, so some calculations may still
329  * need to be done at the ThermoPhase object level.
330  */
331  //setTemperature(t);
332  //setPressure(pres);
333  calcDensity();
334 }
