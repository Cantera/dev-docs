326 void Phase::setMoleFractions(const doublereal* const x)
327 {
328  // Use m_y as a temporary work vector for the non-negative mole fractions
329  doublereal norm = 0.0;
330  /*
331  * sum is calculated below as the unnormalized molecular weight
332  */
333  doublereal sum = 0;
334  for (size_t k = 0; k < m_kk; k++) {
335  double xk = std::max(x[k], 0.0); // Ignore negative mole fractions
336  m_y[k] = xk;
337  norm += xk;
338  sum += m_molwts[k] * xk;
339  }
340  /*
341  * Set m_ym_ to the normalized mole fractions divided by the normalized mean molecular weight:
342  * m_ym_k = X_k / (sum_k X_k M_k)
343  */
344  const doublereal invSum = 1.0/sum;
345  for (size_t k=0; k < m_kk; k++) {
346  m_ym[k] = m_y[k]*invSum;
347  }
348  /*
349  * Now set m_y to the normalized mass fractions
350  * m_y = X_k M_k / (sum_k X_k M_k)
351  */
352  for (size_t k=0; k < m_kk; k++) {
353  m_y[k] = m_ym[k] * m_molwts[k];
354  }
355  /*
356  * Calculate the normalized molecular weight
357  */
358  m_mmw = sum/norm;
359  m_stateNum++;
360 }

390 void Phase::setMassFractions(const doublereal* const y)
391 {
392  for (size_t k = 0; k < m_kk; k++) {
393  m_y[k] = std::max(y[k], 0.0); // Ignore negative mass fractions
394  }
395  doublereal norm = accumulate(m_y.begin(), m_y.end(), 0.0);
396  scale(m_y.begin(), m_y.end(), m_y.begin(), 1.0/norm);

607 void Phase::setConcentrations(const doublereal* const conc)
608 {
609  // Use m_y as temporary storage for non-negative concentrations
610  doublereal sum = 0.0, norm = 0.0;
611  for (size_t k = 0; k != m_kk; ++k) {
612  double ck = std::max(conc[k], 0.0); // Ignore negative concentrations
613  m_y[k] = ck;
614  sum += ck * m_molwts[k];
615  norm += ck;
616  }
617  m_mmw = sum/norm;
618  setDensity(sum);
619  doublereal rsum = 1.0/sum;
620  for (size_t k = 0; k != m_kk; ++k) {
621  m_ym[k] = m_y[k] * rsum;
622  m_y[k] = m_ym[k] * m_molwts[k]; // m_y is now the mass fraction
623  }
624  m_stateNum++;
625 }

695 size_t Phase::addElement(const std::string& symbol, doublereal weight,
696  int atomic_number, doublereal entropy298,
697  int elem_type)
698 {
699  // Look up the atomic weight if not given
700  if (weight == -12345.0) {
701  weight = LookupWtElements(symbol);
702  if (weight < 0.0) {
703  throw CanteraError("Phase::addElement",
704  "No atomic weight found for element: " + symbol);
705  }
706  }
707 
708  // Check for duplicates
709  auto iter = find(m_elementNames.begin(), m_elementNames.end(), symbol);
710  if (iter != m_elementNames.end()) {
711  size_t m = iter - m_elementNames.begin();
712  if (m_atomicWeights[m] != weight) {
713  throw CanteraError("Phase::addElement",
714  "Duplicate elements ({}) have different weights", symbol);
715  } else {
716  // Ignore attempt to add duplicate element with the same weight
717  return m;
718  }
719  }
720 
721  // Add the new element
722  m_atomicWeights.push_back(weight);
723  m_elementNames.push_back(symbol);
724  m_atomicNumbers.push_back(atomic_number);
725  m_entropy298.push_back(entropy298);
726  if (symbol == "E") {
727  m_elem_type.push_back(CT_ELEM_TYPE_ELECTRONCHARGE);
728  } else {
729  m_elem_type.push_back(elem_type);
730  }
731  m_mm++;
732 
733  // Update species compositions
734  if (m_kk) {
735  vector_fp old(m_speciesComp);
736  m_speciesComp.resize(m_kk*m_mm, 0.0);
737  for (size_t k = 0; k < m_kk; k++) {
738  size_t m_old = m_mm - 1;
739  for (size_t m = 0; m < m_old; m++) {
740  m_speciesComp[k * m_mm + m] = old[k * (m_old) + m];
741  }
742  m_speciesComp[k * (m_mm) + (m_mm-1)] = 0.0;
743  }
744  }
745 
746  return m_mm-1;
747 }

749 bool Phase::addSpecies(shared_ptr<Species> spec) {
750  m_species[spec->name] = spec;
751  vector_fp comp(nElements());
752  for (const auto& elem : spec->composition) {
753  size_t m = elementIndex(elem.first);
754  if (m == npos) { // Element doesn't exist in this phase
755  switch (m_undefinedElementBehavior) {
756  case UndefElement::ignore:
757  return false;
758 
759  case UndefElement::add:
760  addElement(elem.first);
761  comp.resize(nElements());
762  m = elementIndex(elem.first);
763  break;
764 
765  case UndefElement::error:
766  default:
767  throw CanteraError("Phase::addSpecies",
768  "Species '{}' contains an undefined element '{}'.",
769  spec->name, elem.first);
770  }
771  }
772  comp[m] = elem.second;
773  }
774 
775  m_speciesNames.push_back(spec->name);
776  m_speciesIndices[spec->name] = m_kk;
777  m_speciesCharge.push_back(spec->charge);
778  m_speciesSize.push_back(spec->size);
779  size_t ne = nElements();
780 
781  double wt = 0.0;
782  const vector_fp& aw = atomicWeights();
783  if (spec->charge != 0.0) {
784  size_t eindex = elementIndex("E");
785  if (eindex != npos) {
786  doublereal ecomp = comp[eindex];
787  if (fabs(spec->charge + ecomp) > 0.001) {
788  if (ecomp != 0.0) {
789  throw CanteraError("Phase::addSpecies",
790  "Input charge and element E compositions differ "
791  "for species " + spec->name);
792  } else {
793  // Just fix up the element E composition based on the input
794  // species charge
795  comp[eindex] = -spec->charge;
796  }
797  }
798  } else {
799  addElement("E", 0.000545, 0, 0.0, CT_ELEM_TYPE_ELECTRONCHARGE);
800  ne = nElements();
801  eindex = elementIndex("E");
802  comp.resize(ne);
803  comp[ne - 1] = - spec->charge;
804  }
805  }
806  for (size_t m = 0; m < ne; m++) {
807  m_speciesComp.push_back(comp[m]);
808  wt += comp[m] * aw[m];
809  }
810 
811  // Some surface phases may define species representing empty sites
812  // that have zero molecular weight. Give them a very small molecular
813  // weight to avoid dividing by zero.
814  wt = std::max(wt, Tiny);
815  m_molwts.push_back(wt);
816  m_rmolwts.push_back(1.0/wt);
817  m_kk++;
818 
819  // Ensure that the Phase has a valid mass fraction vector that sums to
820  // one. We will assume that species 0 has a mass fraction of 1.0 and mass
821  // fraction of all other species is 0.0.
822  if (m_kk == 1) {
823  m_y.push_back(1.0);
824  m_ym.push_back(m_rmolwts[0]);
825  m_mmw = 1.0 / m_ym[0];
826  } else {
827  m_y.push_back(0.0);
828  m_ym.push_back(0.0);
829  }
830  return true;
831 }
