280  doublereal* clow, doublereal* chigh)

282  // heat capacity
283  doublereal cplow = cp_R(tmid, clow);

291  writelog("\tValue computed using low-temperature polynomial: "
292  +fp2str(cplow)+".\n");
293  writelog("\tValue computed using high-temperature polynomial: "
294  +fp2str(cphigh)+".\n");
295  }
296 
297  // enthalpy
298  doublereal hrtlow = enthalpy_RT(tmid, clow);
299  doublereal hrthigh = enthalpy_RT(tmid, chigh);
300  delta = hrtlow - hrthigh;
301  maxError = std::max(std::abs(delta), maxError);
302  if (fabs(delta/(fabs(hrtlow)+cplow*tmid)) > 0.001) {
303  writelog("\n\n**** WARNING ****\nFor species "+name+
304  ", discontinuity in h/RT detected at Tmid = "
305  +fp2str(tmid)+"\n");
306  writelog("\tValue computed using low-temperature polynomial: "
307  +fp2str(hrtlow)+".\n");
308  writelog("\tValue computed using high-temperature polynomial: "
309  +fp2str(hrthigh)+".\n");
310  }
311 
312  // entropy
313  doublereal srlow = entropy_R(tmid, clow);
314  doublereal srhigh = entropy_R(tmid, chigh);
315  delta = srlow - srhigh;
316  maxError = std::max(std::abs(delta), maxError);
317  if (fabs(delta/(fabs(srlow)+cplow)) > 0.001) {
318  writelog("\n\n**** WARNING ****\nFor species "+name+
319  ", discontinuity in s/R detected at Tmid = "
320  +fp2str(tmid)+"\n");
321  writelog("\tValue computed using low-temperature polynomial: "
322  +fp2str(srlow)+".\n");

330 void NasaThermo::fixDiscontinuities(doublereal Tlow, doublereal Tmid,
331  doublereal Thigh, doublereal* clow,
332  doublereal* chigh)
333 {
334  // The thermodynamic parameters can be written in terms nondimensionalized
335  // coefficients A[i] and the nondimensional temperature t = T/Tmid as:
336  //
337  // C_low(t) = A[0] + A[i] * t**i
338  // H_low(t) = A[0] + A[i] / (i+1) * t**i + A[5] / t
339  // S_low(t) = A[0]*ln(t) + A[i] / i * t**i + A[6]
340  //
341  // where the implicit sum is over the range 1 <= i <= 4 and the
342  // nondimensional coefficients are related to the dimensional coefficients
343  // a[i] by:
344  //
345  // A[0] = a[0]
346  // A[i] = Tmid**i * a[i], 1 <= i <= 4
347  // A[5] = a[5] / Tmid
348  // A[6] = a[6] + a[0] * ln(Tmid)
349  //
350  // and corresponding relationships hold for the high-temperature
351  // polynomial coefficients B[i]. This nondimensionalization is necessary
352  // in order for the resulting matrix to be well-conditioned.
353  //
354  // The requirement that C_low(1) = C_high(1) is satisfied by:
355  //
356  // B[0] = A[0] + (A[i] - B[i])
357  // C_high(t) = A[0] + (A[i] + B[i] * t**i - 1)
358  //
359  // The requirement that H_low(1) = H_high(1) is satisfied by:
360  //
361  // B[5] = A[5] + (i / (i+1) * (B[i] - A[i]))
362  // H_high(t) = A[0] + A[5] / t + (1 - i / (i+1) / t) * A[i] +
363  // (t**i / (i+1) - 1 + i / (i+1) / t) * B[i]
364  //
365  // The requirement that S_low(1) = S_high(1) is satisfied by:
366  //
367  // B[6] = A[6] + (A[i] - B[i]) / i
368  // S_high(t) = A[0] * ln(t) + A[6] + (ln(t) + 1 / i) * A[i] +
369  // (-ln(t) + t**i / i - 1 / i) * B[i]
370 
371  // Formulate a linear least squares problem for the nondimensionalized
372  // coefficients. In the system of equations M*x = b:
373  // - each row of M consists of the factors in one of the above equations
374  // for C_low, H_high, etc. evaluated at some temperature between Tlow
375  // and Thigh
376  // - x is a vector of the 11 independent coefficients (A[0] through A[6]
377  // and B[1] through B[4])
378  // - B is a vector of the corresponding value of C, H, or S computed using
379  // the original polynomial.
380 
381  const size_t nTemps = 12;
382  const size_t nCols = 11; // number of independent coefficients
383  const size_t nRows = 3*nTemps; // Evaluate C, H, and S at each temperature
384  DenseMatrix M(nRows, nCols, 0.0);
385  vector_fp b(nRows);
386  doublereal sqrtDeltaT = sqrt(Thigh) - sqrt(Tlow);
387  vector_fp tpow(5);
388  for (size_t j = 0; j < nTemps; j++) {
389  double T = pow(sqrt(Tlow) + sqrtDeltaT * j / (nTemps - 1.0), 2);
390  double t = T / Tmid; // non-dimensionalized temperature
391  double logt = std::log(t);
392  size_t n = 3 * j; // row index
393  for (int i = 1; i <= 4; i++) {
394  tpow[i] = pow(t, i);
395  }
396 
397  // row n: Cp/R
398  // row n+1: H/RT
399  // row n+2: S/R
400  // columns 0 through 6 are for the low-T coefficients
401  // columns 7 through 10 are for the independent high-T coefficients
402  M(n, 0) = 1.0;
403  M(n+1,0) = 1.0;
404  M(n+2,0) = logt;
405  M(n+1,5) = 1.0 / t;
406  M(n+2,6) = 1.0;
407  if (t <= 1.0) {
408  for (int i = 1; i <= 4; i++) {
409  M(n,i) = tpow[i];
410  M(n+1,i) = tpow[i] / (i+1);
411  M(n+2,i) = tpow[i] / i;
412  }
413  b[n] = cp_R(T, clow);
414  b[n+1] = enthalpy_RT(T, clow);
415  b[n+2] = entropy_R(T, clow);
416  } else {
417  for (int i = 1; i <= 4; i++) {
418  M(n,i) = 1.0;
419  M(n,i+6) = tpow[i] - 1.0;
420  M(n+1,i) = 1 - i / ((i + 1.0) * t);
421  M(n+1,i+6) = -1 + tpow[i] / (i+1) + i / ((i+1) * t);
422  M(n+2,i) = logt + 1.0 / i;
423  M(n+2,i+6) = -logt + (tpow[i] - 1.0) / i;
424  }
425  b[n] = cp_R(T, chigh);
426  b[n+1] = enthalpy_RT(T, chigh);
427  b[n+2] = entropy_R(T, chigh);
428  }
429  }
430 
431  // Solve the least squares problem
432  vector_fp sigma(nRows);
433  size_t rank;
434  int info;
435  vector_fp work(1);
436  int lwork = -1;
437  // First get the desired size of the work array
438  ct_dgelss(nRows, nCols, 1, &M(0,0), nRows, &b[0], nRows,
439  &sigma[0], -1, rank, &work[0], lwork, info);
440  work.resize(work[0]);
441  lwork = work[0];
442  ct_dgelss(nRows, nCols, 1, &M(0,0), nRows, &b[0], nRows,
443  &sigma[0], -1, rank, &work[0], lwork, info);
444 
445  AssertTrace(info == 0);
446  AssertTrace(rank == nCols);
447  AssertTrace(sigma[0] / sigma[10] < 1e20); // condition number
448 
449  // Compute the full set of nondimensionalized coefficients
450  // (dgelss returns the solution of M*x = b in b).
451 
452  // Note that clow and chigh store the coefficients in the order:
453  // clow = [a[5], a[6], a[0], a[1], a[2], a[3], a[4]]
454  clow[2] = chigh[2] = b[0];
455  clow[0] = chigh[0] = b[5];
456  clow[1] = chigh[1] = b[6];
457  for (int i = 1; i <= 4; i++) {
458  clow[2+i] = b[i];
459  chigh[2+i] = b[6+i];
460  chigh[2] += clow[2+i] - chigh[2+i];
461  chigh[0] += i / (i + 1.0) * (chigh[2+i] - clow[2+i]);
462  chigh[1] += (clow[2+i] - chigh[2+i]) / i;
463  }
464 
465  // redimensionalize
466  for (int i = 1; i <= 4; i++) {
467  clow[2+i] /= pow(Tmid, i);
468  chigh[2+i] /= pow(Tmid, i);
469  }
470  clow[0] *= Tmid;
471  chigh[0] *= Tmid;
472  clow[1] -= clow[2] * std::log(Tmid);
473  chigh[1] -= chigh[2] * std::log(Tmid);
474 }
