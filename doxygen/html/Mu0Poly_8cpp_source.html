35 void Mu0Poly::updateProperties(const doublereal* tt, doublereal* cp_R,
36  doublereal* h_RT, doublereal* s_R) const
37 {
38  size_t j = m_numIntervals;
39  double T = *tt;
40  for (size_t i = 0; i < m_numIntervals; i++) {
41  double T2 = m_t0_int[i+1];
42  if (T <=T2) {
43  j = i;
44  break;
45  }
46  }
47  double T1 = m_t0_int[j];
48  double cp_Rj = m_cp0_R_int[j];
49  *cp_R = cp_Rj;
50  *h_RT = (m_h0_R_int[j] + (T - T1) * cp_Rj)/T;
51  *s_R = m_s0_R_int[j] + cp_Rj * (log(T/T1));
52 }

151 void Mu0Poly::processCoeffs(const doublereal* coeffs)
152 {
153  size_t nPoints = (size_t) coeffs[0];
154  if (nPoints < 2) {
155  throw CanteraError("Mu0Poly",
156  "nPoints must be >= 2");
157  }
158  m_numIntervals = nPoints - 1;
159  m_H298 = coeffs[1] / GasConstant;
160  size_t iT298 = 0;
161  /*
162  * Resize according to the number of points
163  */
164  m_t0_int.resize(nPoints);
165  m_h0_R_int.resize(nPoints);
166  m_s0_R_int.resize(nPoints);
167  m_cp0_R_int.resize(nPoints);
168  m_mu0_R_int.resize(nPoints);
169  /*
170  * Calculate the T298 interval and make sure that
171  * the temperatures are strictly monotonic.
172  * Also distribute the data into the internal arrays.
173  */
174  bool ifound = false;
175  for (size_t i = 0, iindex = 2; i < nPoints; i++) {
176  double T1 = coeffs[iindex];
177  m_t0_int[i] = T1;
178  m_mu0_R_int[i] = coeffs[iindex+1] / GasConstant;
179  if (T1 == 298.15) {
180  iT298 = i;
181  ifound = true;
182  }
183  if (i < nPoints - 1 && coeffs[iindex+2] <= T1) {
184  throw CanteraError("Mu0Poly",
185  "Temperatures are not monotonic increasing");
186  }
187  iindex += 2;
188  }
189  if (!ifound) {
190  throw CanteraError("Mu0Poly",
191  "One temperature has to be 298.15");
192  }
193 
194  /*
195  * Starting from the interval with T298, we go up
196  */
197  m_h0_R_int[iT298] = m_H298;
198  m_s0_R_int[iT298] = - (m_mu0_R_int[iT298] - m_h0_R_int[iT298]) / m_t0_int[iT298];
199  for (size_t i = iT298; i < m_numIntervals; i++) {
200  double T1 = m_t0_int[i];
201  double s1 = m_s0_R_int[i];
202  double T2 = m_t0_int[i+1];
203  double deltaMu = m_mu0_R_int[i+1] - m_mu0_R_int[i];
204  double deltaT = T2 - T1;
205  double cpi = (deltaMu - T1 * s1 + T2 * s1) / (deltaT - T2 * log(T2/T1));
206  m_cp0_R_int[i] = cpi;
207  m_h0_R_int[i+1] = m_h0_R_int[i] + cpi * deltaT;
208  m_s0_R_int[i+1] = s1 + cpi * log(T2/T1);
209  m_cp0_R_int[i+1] = cpi;
210  }
211 
212  /*
213  * Starting from the interval with T298, we go down
214  */
215  if (iT298 != 0) {
216  m_h0_R_int[iT298] = m_H298;
217  m_s0_R_int[iT298] = - (m_mu0_R_int[iT298] - m_h0_R_int[iT298]) / m_t0_int[iT298];
218  for (size_t i = iT298 - 1; i != npos; i--) {
219  double T1 = m_t0_int[i];
220  double T2 = m_t0_int[i+1];
221  double s2 = m_s0_R_int[i+1];
222  double deltaMu = m_mu0_R_int[i+1] - m_mu0_R_int[i];
223  double deltaT = T2 - T1;
224  double cpi = (deltaMu - T1 * s2 + T2 * s2) / (deltaT - T1 * log(T2/T1));
225  m_cp0_R_int[i] = cpi;
226  m_h0_R_int[i] = m_h0_R_int[i+1] - cpi * deltaT;
227  m_s0_R_int[i] = s2 - cpi * log(T2/T1);
228  if (i == (m_numIntervals-1)) {
229  m_cp0_R_int[i+1] = cpi;
230  }
231  }
232  }
233 }
