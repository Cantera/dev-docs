66 void Mu0Poly::updateProperties(const doublereal* tt, doublereal* cp_R,
67  doublereal* h_RT, doublereal* s_R) const
68 {
69  size_t j = m_numIntervals;
70  double T = *tt;
71  for (size_t i = 0; i < m_numIntervals; i++) {
72  double T2 = m_t0_int[i+1];
73  if (T <=T2) {
74  j = i;
75  break;
76  }
77  }
78  double T1 = m_t0_int[j];
79  double cp_Rj = m_cp0_R_int[j];
80 
81  doublereal rt = 1.0/T;
82  cp_R[m_index] = cp_Rj;
83  h_RT[m_index] = rt*(m_h0_R_int[j] + (T - T1) * cp_Rj);
84  s_R[m_index] = m_s0_R_int[j] + cp_Rj * (log(T/T1));
85 }

203 void Mu0Poly::processCoeffs(const doublereal* coeffs)
204 {
205  size_t i, iindex;
206  double T1, T2;
207  size_t nPoints = (size_t) coeffs[0];
208  if (nPoints < 2) {
209  throw CanteraError("Mu0Poly",
210  "nPoints must be >= 2");
211  }
212  m_numIntervals = nPoints - 1;
213  m_H298 = coeffs[1] / GasConstant;
214  size_t iT298 = 0;
215  /*
216  * Resize according to the number of points
217  */
218  m_t0_int.resize(nPoints);
219  m_h0_R_int.resize(nPoints);
220  m_s0_R_int.resize(nPoints);
221  m_cp0_R_int.resize(nPoints);
222  m_mu0_R_int.resize(nPoints);
223  /*
224  * Calculate the T298 interval and make sure that
225  * the temperatures are strictly monotonic.
226  * Also distribute the data into the internal arrays.
227  */
228  bool ifound = false;
229  for (i = 0, iindex = 2; i < nPoints; i++) {
230  T1 = coeffs[iindex];
231  m_t0_int[i] = T1;
232  m_mu0_R_int[i] = coeffs[iindex+1] / GasConstant;
233  if (T1 == 298.15) {
234  iT298 = i;
235  ifound = true;
236  }
237  if (i < nPoints - 1) {
238  T2 = coeffs[iindex+2];
239  if (T2 <= T1) {
240  throw CanteraError("Mu0Poly",
241  "Temperatures are not monotonic increasing");
242  }
243  }
244  iindex += 2;
245  }
246  if (!ifound) {
247  throw CanteraError("Mu0Poly",
248  "One temperature has to be 298.15");
249  }
250 
251  /*
252  * Starting from the interval with T298, we go up
253  */
254  doublereal mu2, s1, s2, h1, h2, cpi, deltaMu, deltaT;
255  T1 = m_t0_int[iT298];
256  doublereal mu1 = m_mu0_R_int[iT298];
257  m_h0_R_int[iT298] = m_H298;
258  m_s0_R_int[iT298] = - (mu1 - m_h0_R_int[iT298]) / T1;
259  for (i = iT298; i < m_numIntervals; i++) {
260  T1 = m_t0_int[i];
261  s1 = m_s0_R_int[i];
262  h1 = m_h0_R_int[i];
263  mu1 = m_mu0_R_int[i];
264  T2 = m_t0_int[i+1];
265  mu2 = m_mu0_R_int[i+1];
266  deltaMu = mu2 - mu1;
267  deltaT = T2 - T1;
268  cpi = (deltaMu - T1 * s1 + T2 * s1) / (deltaT - T2 * log(T2/T1));
269  h2 = h1 + cpi * deltaT;
270  s2 = s1 + cpi * log(T2/T1);
271  m_cp0_R_int[i] = cpi;
272  m_h0_R_int[i+1] = h2;
273  m_s0_R_int[i+1] = s2;
274  m_cp0_R_int[i+1] = cpi;
275  }
276 
277  /*
278  * Starting from the interval with T298, we go down
279  */
280  if (iT298 != 0) {
281  T2 = m_t0_int[iT298];
282  mu2 = m_mu0_R_int[iT298];
283  m_h0_R_int[iT298] = m_H298;
284  m_s0_R_int[iT298] = - (mu2 - m_h0_R_int[iT298]) / T2;
285  for (i = iT298 - 1; i != npos; i--) {
286  T1 = m_t0_int[i];
287  mu1 = m_mu0_R_int[i];
288  T2 = m_t0_int[i+1];
289  mu2 = m_mu0_R_int[i+1];
290  s2 = m_s0_R_int[i+1];
291  h2 = m_h0_R_int[i+1];
292  deltaMu = mu2 - mu1;
293  deltaT = T2 - T1;
294  cpi = (deltaMu - T1 * s2 + T2 * s2) / (deltaT - T1 * log(T2/T1));
295  h1 = h2 - cpi * deltaT;
296  s1 = s2 - cpi * log(T2/T1);
297  m_cp0_R_int[i] = cpi;
298  m_h0_R_int[i] = h1;
299  m_s0_R_int[i] = s1;
300  if (i == (m_numIntervals-1)) {
301  m_cp0_R_int[i+1] = cpi;
302  }
303  }
304  }
305 #ifdef DEBUG_HKM_NOT
306  printf(" Temp mu0(J/kmol) cp0(J/kmol/K) "
307  " h0(J/kmol) s0(J/kmol/K) \n");
308  for (i = 0; i < nPoints; i++) {
309  printf("%12.3g %12.5g %12.5g %12.5g %12.5g\n",
310  m_t0_int[i], m_mu0_R_int[i] * GasConstant,
311  m_cp0_R_int[i]* GasConstant,
312  m_h0_R_int[i]* GasConstant,
313  m_s0_R_int[i]* GasConstant);
314  fflush(stdout);
315  }
316 #endif
317 }
