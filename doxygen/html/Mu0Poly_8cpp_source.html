96 updateProperties(const doublereal* tt, doublereal* cp_R,
97  doublereal* h_RT, doublereal* s_R) const
98 {
99  size_t j = m_numIntervals;
100  double T = *tt;
101  for (size_t i = 0; i < m_numIntervals; i++) {
102  double T2 = m_t0_int[i+1];
103  if (T <=T2) {
104  j = i;
105  break;
106  }
107  }
108  double T1 = m_t0_int[j];
109  double cp_Rj = m_cp0_R_int[j];
110 
111  doublereal rt = 1.0/T;
112  cp_R[m_index] = cp_Rj;
113  h_RT[m_index] = rt*(m_h0_R_int[j] + (T - T1) * cp_Rj);
114  s_R[m_index] = m_s0_R_int[j] + cp_Rj * (log(T/T1));
115 }

156  doublereal tmin, tmax;

158  const XML_Node& Mu0Node = *Mu0Node_ptr;
159 
160  tmin = fpValue(Mu0Node["Tmin"]);
161  tmax = fpValue(Mu0Node["Tmax"]);
162  doublereal pref = fpValue(Mu0Node["Pref"]);

164  doublereal h298 = 0.0;
165  if (Mu0Node.hasChild("H298")) {
166  h298 = getFloat(Mu0Node, "H298", "actEnergy");
167  }
168 
169  size_t numPoints = 1;
170  if (Mu0Node.hasChild("numPoints")) {
171  numPoints = getInteger(Mu0Node, "numPoints");
172  }
173 
174  vector_fp cValues(numPoints);
175  const XML_Node* valNode_ptr =
176  getByTitle(const_cast<XML_Node&>(Mu0Node), "Mu0Values");
177  if (!valNode_ptr) {
178  throw CanteraError("installMu0ThermoFromXML",

181  }
182  getFloatArray(*valNode_ptr, cValues, true, "actEnergy");
183  /*

186  * adjusted from the assumed T = 273.15
187  */
188  string uuu = (*valNode_ptr)["units"];

192  size_t ns = cValues.size();

195  "numPoints inconsistent while processing "
196  + speciesName);
197  }
198

200  const XML_Node* tempNode_ptr =
201  getByTitle(const_cast<XML_Node&>(Mu0Node), "Mu0Temperatures");
202  if (!tempNode_ptr) {
203  throw CanteraError("installMu0ThermoFromXML",
204  "missing required while processing + "
205  + speciesName);
206  }
207  getFloatArray(*tempNode_ptr, cTemperatures, false);
208  ns = cTemperatures.size();
209  if (ns != numPoints) {
210  throw CanteraError("installMu0ThermoFromXML",
211  "numPoints inconsistent while processing "
212  + speciesName);
213  }
214 
215  /*
216  * Fix up dimensionless Mu0 values if input
217  */
218  if (dimensionlessMu0Values) {
219  for (size_t i = 0; i < numPoints; i++) {
220  cValues[i] *= cTemperatures[i] / 273.15;
221  }

237 void Mu0Poly::processCoeffs(const doublereal* coeffs)
238 {
239  size_t i, iindex;
240  double T1, T2;
241  size_t nPoints = (size_t) coeffs[0];
242  if (nPoints < 2) {
243  throw CanteraError("Mu0Poly",
244  "nPoints must be >= 2");
245  }
246  m_numIntervals = nPoints - 1;
247  m_H298 = coeffs[1] / GasConstant;
248  size_t iT298 = 0;
249  /*
250  * Resize according to the number of points
251  */
252  m_t0_int.resize(nPoints);
253  m_h0_R_int.resize(nPoints);
254  m_s0_R_int.resize(nPoints);
255  m_cp0_R_int.resize(nPoints);
256  m_mu0_R_int.resize(nPoints);
257  /*
258  * Calculate the T298 interval and make sure that
259  * the temperatures are strictly monotonic.
260  * Also distribute the data into the internal arrays.
261  */
262  bool ifound = false;
263  for (i = 0, iindex = 2; i < nPoints; i++) {
264  T1 = coeffs[iindex];
265  m_t0_int[i] = T1;
266  m_mu0_R_int[i] = coeffs[iindex+1] / GasConstant;
267  if (T1 == 298.15) {
268  iT298 = i;
269  ifound = true;
270  }
271  if (i < nPoints - 1) {
272  T2 = coeffs[iindex+2];
273  if (T2 <= T1) {
274  throw CanteraError("Mu0Poly",
275  "Temperatures are not monotonic increasing");
276  }
277  }
278  iindex += 2;
279  }
280  if (!ifound) {
281  throw CanteraError("Mu0Poly",
282  "One temperature has to be 298.15");
283  }
284 
285  /*
286  * Starting from the interval with T298, we go up
287  */
288  doublereal mu2, s1, s2, h1, h2, cpi, deltaMu, deltaT;
289  T1 = m_t0_int[iT298];
290  doublereal mu1 = m_mu0_R_int[iT298];
291  m_h0_R_int[iT298] = m_H298;
292  m_s0_R_int[iT298] = - (mu1 - m_h0_R_int[iT298]) / T1;
293  for (i = iT298; i < m_numIntervals; i++) {
294  T1 = m_t0_int[i];
295  s1 = m_s0_R_int[i];
296  h1 = m_h0_R_int[i];
297  mu1 = m_mu0_R_int[i];
298  T2 = m_t0_int[i+1];
299  mu2 = m_mu0_R_int[i+1];
300  deltaMu = mu2 - mu1;
301  deltaT = T2 - T1;
302  cpi = (deltaMu - T1 * s1 + T2 * s1) / (deltaT - T2 * log(T2/T1));
303  h2 = h1 + cpi * deltaT;
304  s2 = s1 + cpi * log(T2/T1);
305  m_cp0_R_int[i] = cpi;
306  m_h0_R_int[i+1] = h2;
307  m_s0_R_int[i+1] = s2;
308  m_cp0_R_int[i+1] = cpi;
309  }
310 
311  /*
312  * Starting from the interval with T298, we go down
313  */
314  if (iT298 != 0) {
315  T2 = m_t0_int[iT298];
316  mu2 = m_mu0_R_int[iT298];
317  m_h0_R_int[iT298] = m_H298;
318  m_s0_R_int[iT298] = - (mu2 - m_h0_R_int[iT298]) / T2;
319  for (i = iT298 - 1; i != npos; i--) {
320  T1 = m_t0_int[i];
321  mu1 = m_mu0_R_int[i];
322  T2 = m_t0_int[i+1];
323  mu2 = m_mu0_R_int[i+1];
324  s2 = m_s0_R_int[i+1];
325  h2 = m_h0_R_int[i+1];
326  deltaMu = mu2 - mu1;
327  deltaT = T2 - T1;
328  cpi = (deltaMu - T1 * s2 + T2 * s2) / (deltaT - T1 * log(T2/T1));
329  h1 = h2 - cpi * deltaT;
330  s1 = s2 - cpi * log(T2/T1);
331  m_cp0_R_int[i] = cpi;
332  m_h0_R_int[i] = h1;
333  m_s0_R_int[i] = s1;
334  if (i == (m_numIntervals-1)) {
335  m_cp0_R_int[i+1] = cpi;
336  }
337  }
338  }
339 #ifdef DEBUG_HKM_NOT
340  printf(" Temp mu0(J/kmol) cp0(J/kmol/K) "
341  " h0(J/kmol) s0(J/kmol/K) \n");
342  for (i = 0; i < nPoints; i++) {
343  printf("%12.3g %12.5g %12.5g %12.5g %12.5g\n",
344  m_t0_int[i], m_mu0_R_int[i] * GasConstant,
345  m_cp0_R_int[i]* GasConstant,
346  m_h0_R_int[i]* GasConstant,
347  m_s0_R_int[i]* GasConstant);
348  fflush(stdout);
349  }
350 #endif
351 }
