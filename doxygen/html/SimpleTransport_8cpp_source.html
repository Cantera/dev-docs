55 bool SimpleTransport::initLiquid(LiquidTransportParams& tr)
56 {
57  // constant substance attributes
58  m_thermo = tr.thermo;
59  m_nsp = m_thermo->nSpecies();
60 
61  // Read the transport block in the phase XML Node
62  // It's not an error if this block doesn't exist. Just use the defaults
63  XML_Node& phaseNode = m_thermo->xml();
64  if (phaseNode.hasChild("transport")) {
65  XML_Node& transportNode = phaseNode.child("transport");
66  string transportModel = transportNode.attrib("model");
67  if (transportModel == "Simple") {
68  compositionDepType_ = tr.compositionDepTypeDefault_;
69  } else {
70  throw CanteraError("SimpleTransport::initLiquid()",
71  "transport model isn't the correct type: " + transportModel);
72  }
73  }
74 
75  // make a local copy of the molecular weights
76  m_mw = m_thermo->molecularWeights();
77 
78  // Get the input Viscosities
79  m_viscSpecies.resize(m_nsp);
80  m_coeffVisc_Ns.clear();
81  m_coeffVisc_Ns.resize(m_nsp);
82  for (size_t k = 0; k < m_nsp; k++) {
83  LiquidTransportData& ltd = tr.LTData[k];
84  m_coeffVisc_Ns[k] = ltd.viscosity;
85  ltd.viscosity = 0;
86  }
87 
88  // Get the input thermal conductivities
89  m_condSpecies.resize(m_nsp);
90  m_coeffLambda_Ns.clear();
91  m_coeffLambda_Ns.resize(m_nsp);
92  for (size_t k = 0; k < m_nsp; k++) {
93  LiquidTransportData& ltd = tr.LTData[k];
94  m_coeffLambda_Ns[k] = ltd.thermalCond;
95  ltd.thermalCond = 0;
96  }
97 
98  // Get the input species diffusivities
99  useHydroRadius_ = false;
100  m_diffSpecies.resize(m_nsp);
101  m_coeffDiff_Ns.clear();
102  m_coeffDiff_Ns.resize(m_nsp);
103  for (size_t k = 0; k < m_nsp; k++) {
104  string spName = m_thermo->speciesName(k);
105  LiquidTransportData& ltd = tr.LTData[k];
106  m_coeffDiff_Ns[k] = ltd.speciesDiffusivity;
107  ltd.speciesDiffusivity = 0;
108  if (!m_coeffDiff_Ns[k]) {
109  if (ltd.hydroRadius) {
110  m_coeffHydroRadius_Ns[k] = (ltd.hydroRadius)->duplMyselfAsLTPspecies();

112  if (!m_coeffHydroRadius_Ns[k]) {

139  m_diff_mix_ok = false;

144 {

146  update_C();
147

327 void SimpleTransport::getSpeciesFluxesExt(size_t ldf, doublereal* fluxes)
328 {
329  AssertThrow(ldf >= m_nsp ,"SimpleTransport::getSpeciesFluxesExt: Stride must be greater than m_nsp");
330  update_T();
331  update_C();
332 
333  getMixDiffCoeffs(m_spwork.data());
334 
335  const vector_fp& mw = m_thermo->molecularWeights();
336  const doublereal* y = m_thermo->massFractions();
337  doublereal concTotal = m_thermo->molarDensity();
338 
339  // Unroll wrt ndim
340  if (doMigration_) {
341  double FRT = ElectronCharge / (Boltzmann * m_temp);
342  for (size_t n = 0; n < m_nDim; n++) {
343  rhoVc[n] = 0.0;
344  for (size_t k = 0; k < m_nsp; k++) {
345  fluxes[n*ldf + k] = - concTotal * mw[k] * m_spwork[k] *
346  (m_Grad_X[n*m_nsp + k] + FRT * m_molefracs[k] * m_chargeSpecies[k] * m_Grad_V[n]);
347  rhoVc[n] += fluxes[n*ldf + k];
348  }
349  }
350  } else {
351  for (size_t n = 0; n < m_nDim; n++) {
352  rhoVc[n] = 0.0;
353  for (size_t k = 0; k < m_nsp; k++) {
354  fluxes[n*ldf + k] = - concTotal * mw[k] * m_spwork[k] * m_Grad_X[n*m_nsp + k];
355  rhoVc[n] += fluxes[n*ldf + k];
356  }
357  }
358  }
359 
360  if (m_velocityBasis == VB_MASSAVG) {
361  for (size_t n = 0; n < m_nDim; n++) {
362  rhoVc[n] = 0.0;
363  for (size_t k = 0; k < m_nsp; k++) {
364  rhoVc[n] += fluxes[n*ldf + k];
365  }
366  }
367  for (size_t n = 0; n < m_nDim; n++) {
368  for (size_t k = 0; k < m_nsp; k++) {
369  fluxes[n*ldf + k] -= y[k] * rhoVc[n];
370  }
371  }
372  } else if (m_velocityBasis == VB_MOLEAVG) {
373  for (size_t n = 0; n < m_nDim; n++) {
374  rhoVc[n] = 0.0;
375  for (size_t k = 0; k < m_nsp; k++) {
376  rhoVc[n] += fluxes[n*ldf + k] / mw[k];
377  }
378  }
379  for (size_t n = 0; n < m_nDim; n++) {
380  for (size_t k = 0; k < m_nsp; k++) {
381  fluxes[n*ldf + k] -= m_molefracs[k] * rhoVc[n] * mw[k];
382  }
383  }
384  } else if (m_velocityBasis >= 0) {
385  for (size_t n = 0; n < m_nDim; n++) {
386  rhoVc[n] = - fluxes[n*ldf + m_velocityBasis] / mw[m_velocityBasis];
387  for (size_t k = 0; k < m_nsp; k++) {
388  rhoVc[n] += fluxes[n*ldf + k] / mw[k];
389  }
390  }
391  for (size_t n = 0; n < m_nDim; n++) {
392  for (size_t k = 0; k < m_nsp; k++) {
393  fluxes[n*ldf + k] -= m_molefracs[k] * rhoVc[n] * mw[k];
394  }
395  fluxes[n*ldf + m_velocityBasis] = 0.0;
396  }
397  } else {
398  throw CanteraError("SimpleTransport::getSpeciesFluxesExt()",
399  "unknown velocity basis");
400  }
401 }

416 bool SimpleTransport::update_C()
417 {
418  // If the pressure has changed then the concentrations have changed.
419  doublereal pres = m_thermo->pressure();
420  bool qReturn = true;
421  if (pres != m_press) {
422  qReturn = false;
423  m_press = pres;
424  }
425  int iStateNew = m_thermo->stateMFNumber();
426  if (iStateNew != m_iStateMF) {
427  qReturn = false;
428  m_thermo->getMoleFractions(m_molefracs.data());
429  m_thermo->getConcentrations(m_concentrations.data());
430  concTot_ = 0.0;
431  for (size_t k = 0; k < m_nsp; k++) {
432  m_molefracs[k] = std::max(0.0, m_molefracs[k]);
433  concTot_ += m_concentrations[k];
434  }
435  dens_ = m_thermo->density();
436  meanMolecularWeight_ = m_thermo->meanMolecularWeight();
437  }
438  if (qReturn) {
439  return false;
440  }
441 
442  // Mixture stuff needs to be evaluated
443  m_visc_mix_ok = false;
444  m_diff_mix_ok = false;
445  m_cond_mix_ok = false;
446  return true;
447 }

494 {

496  if (t == m_temp) {
497  return false;

503 
504  // Compute various functions of temperature

506

509  m_visc_mix_ok = false;
