22 /**
23  * The NASA polynomial parameterization for one temperature range.
24  * This parameterization expresses the heat capacity as a
25  * fourth-order polynomial. Note that this is the form used in the
26  * 1971 NASA equilibrium program and by the Chemkin software
27  * package, but differs from the form used in the more recent NASA
28  * equilibrium program.
29  *
30  * Seven coefficients \f$(a_0,\dots,a_6)\f$ are used to represent
31  * \f$ c_p^0(T)\f$, \f$ h^0(T)\f$, and \f$ s^0(T) \f$ as
32  * polynomials in \f$ T \f$ :
33  * \f[
34  * \frac{c_p(T)}{R} = a_0 + a_1 T + a_2 T^2 + a_3 T^3 + a_4 T^4
35  * \f]
36  * \f[
37  * \frac{h^0(T)}{RT} = a_0 + \frac{a_1}{2} T + \frac{a_2}{3} T^2
38  * + \frac{a_3}{4} T^3 + \frac{a_4}{5} T^4 + \frac{a_5}{T}.
39  * \f]
40  * \f[
41  * \frac{s^0(T)}{R} = a_0\ln T + a_1 T + \frac{a_2}{2} T^2
42  + \frac{a_3}{3} T^3 + \frac{a_4}{4} T^4 + a_6.
43  * \f]
44  *
45  * This class is designed specifically for use by class NasaThermo.
46  * @ingroup spthermo
47  */

102  //! Update the properties for this species, given a temperature polynomial
103  /*!
104  * This method is called with a pointer to an array containing the
105  * functions of temperature needed by this parameterization, and three
106  * pointers to arrays where the computed property values should be
107  * written. This method updates only one value in each array.
108  *
109  * Temperature Polynomial:
110  * tt[0] = t;
111  * tt[1] = t*t;
112  * tt[2] = m_t[1]*t;
113  * tt[3] = m_t[2]*t;
114  * tt[4] = 1.0/t;
115  * tt[5] = std::log(t);
116  *
117  * @param tt vector of temperature polynomials
118  * @param cp_R Vector of Dimensionless heat capacities. (length m_kk).
119  * @param h_RT Vector of Dimensionless enthalpies. (length m_kk).
120  * @param s_R Vector of Dimensionless entropies. (length m_kk).
121  */
122  virtual void updateProperties(const doublereal* tt,
123  doublereal* cp_R, doublereal* h_RT, doublereal* s_R) const {
124  doublereal ct0 = m_coeff[2]; // a0
125  doublereal ct1 = m_coeff[3]*tt[0]; // a1 * T
126  doublereal ct2 = m_coeff[4]*tt[1]; // a2 * T^2
127  doublereal ct3 = m_coeff[5]*tt[2]; // a3 * T^3
128  doublereal ct4 = m_coeff[6]*tt[3]; // a4 * T^4
129 
130  doublereal cp, h, s;
131  cp = ct0 + ct1 + ct2 + ct3 + ct4;
132  h = ct0 + 0.5*ct1 + OneThird*ct2 + 0.25*ct3 + 0.2*ct4
133  + m_coeff[0]*tt[4]; // last term is a5/T
134  s = ct0*tt[5] + ct1 + 0.5*ct2 + OneThird*ct3
135  +0.25*ct4 + m_coeff[1]; // last term is a6
136 
137  // return the computed properties in the location in the output
138  // arrays for this species
139  cp_R[m_index] = cp;
140  h_RT[m_index] = h;
141  s_R[m_index] = s;
142  //writelog("NASA1: for species "+int2str(m_index)+", h_RT = "+
143  // fp2str(h)+"\n");
144  }

202
