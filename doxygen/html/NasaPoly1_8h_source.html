19 /**
20  * The NASA polynomial parameterization for one temperature range. This
21  * parameterization expresses the heat capacity as a fourth-order polynomial.
22  * Note that this is the form used in the 1971 NASA equilibrium program and by
23  * the Chemkin software package, but differs from the form used in the more
24  * recent NASA equilibrium program.
25  *
26  * Seven coefficients \f$(a_0,\dots,a_6)\f$ are used to represent
27  * \f$ c_p^0(T)\f$, \f$ h^0(T)\f$, and \f$ s^0(T) \f$ as
28  * polynomials in \f$ T \f$ :
29  * \f[
30  * \frac{c_p(T)}{R} = a_0 + a_1 T + a_2 T^2 + a_3 T^3 + a_4 T^4
31  * \f]
32  * \f[
33  * \frac{h^0(T)}{RT} = a_0 + \frac{a_1}{2} T + \frac{a_2}{3} T^2
34  * + \frac{a_3}{4} T^3 + \frac{a_4}{5} T^4 + \frac{a_5}{T}.
35  * \f]
36  * \f[
37  * \frac{s^0(T)}{R} = a_0\ln T + a_1 T + \frac{a_2}{2} T^2
38  * + \frac{a_3}{3} T^3 + \frac{a_4}{4} T^4 + a_6.
39  * \f]
40  *
41  * @ingroup spthermo
42  */

80  virtual void updateTemperaturePoly(double T, double* T_poly) const {
81  T_poly[0] = T;

83  T_poly[2] = T_poly[1] * T;

92  * Temperature Polynomial:

98  * tt[5] = std::log(t);
99  */
100  virtual void updateProperties(const doublereal* tt,
101  doublereal* cp_R, doublereal* h_RT, doublereal* s_R) const {
102  doublereal ct0 = m_coeff[0]; // a0
103  doublereal ct1 = m_coeff[1]*tt[0]; // a1 * T
104  doublereal ct2 = m_coeff[2]*tt[1]; // a2 * T^2
105  doublereal ct3 = m_coeff[3]*tt[2]; // a3 * T^3
106  doublereal ct4 = m_coeff[4]*tt[3]; // a4 * T^4
107 
108  doublereal cp, h, s;
109  cp = ct0 + ct1 + ct2 + ct3 + ct4;
110  h = ct0 + 0.5*ct1 + 1.0/3.0*ct2 + 0.25*ct3 + 0.2*ct4
111  + m_coeff[5]*tt[4]; // last term is a5/T
112  s = ct0*tt[5] + ct1 + 0.5*ct2 + 1.0/3.0*ct3
113  +0.25*ct4 + m_coeff[6]; // last term is a6
114 
115  // return the computed properties for this species
116  *cp_R = cp;
117  *h_RT = h;
118  *s_R = s;
119  }

149  virtual doublereal reportHf298(doublereal* const h298 = 0) const {
150  double tt[6];

152  updateTemperaturePoly(temp, tt);
153  doublereal ct0 = m_coeff[0]; // a0
154  doublereal ct1 = m_coeff[1]*tt[0]; // a1 * T
155  doublereal ct2 = m_coeff[2]*tt[1]; // a2 * T^2
156  doublereal ct3 = m_coeff[3]*tt[2]; // a3 * T^3
157  doublereal ct4 = m_coeff[4]*tt[3]; // a4 * T^4
158 
159  double h_RT = ct0 + 0.5*ct1 + 1.0/3.0*ct2 + 0.25*ct3 + 0.2*ct4
160  + m_coeff[5]*tt[4]; // last t
161 
162  double h = h_RT * GasConstant * temp;
163  if (h298) {
