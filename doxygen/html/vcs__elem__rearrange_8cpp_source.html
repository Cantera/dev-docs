20 int VCS_SOLVE::vcs_elem_rearrange(double* const aw, double* const sa,
21  double* const sm, double* const ss)
22 {
23  size_t ncomponents = m_numComponents;
24  if (DEBUG_MODE_ENABLED && m_debug_print_lvl >= 2) {
25  plogf(" ");
26  for (size_t i=0; i<77; i++) {
27  plogf("-");
28  }
29  plogf("\n");
30  plogf(" --- Subroutine elem_rearrange() called to ");
31  plogf("check stoich. coefficient matrix\n");
32  plogf(" --- and to rearrange the element ordering once");
33  plogendl();
34  }

36  /*
37  * Use a temporary work array for the element numbers
38  * Also make sure the value of test is unique.
39  */
40  bool lindep = true;
41  double test = -1.0E10;
42  while (lindep) {
43  lindep = false;
44  for (size_t i = 0; i < m_numElemConstraints; ++i) {
45  test -= 1.0;
46  aw[i] = m_elemAbundancesGoal[i];
47  if (test == aw[i]) {
48  lindep = true;
49  }
50  }
51  }
52 
53  /*
54  * Top of a loop of some sort based on the index JR. JR is the
55  * current number independent elements found.
56  */
57  size_t jr = 0;
58  while (jr < ncomponents) {
59  size_t k;
60  /*
61  * Top of another loop point based on finding a linearly
62  * independent species
63  */
64  while (true) {
65  /*
66  * Search the remaining part of the mole fraction vector, AW,
67  * for the largest remaining species. Return its identity in K.
68  */
69  k = m_numElemConstraints;
70  for (size_t ielem = jr; ielem < m_numElemConstraints; ielem++) {
71  if (m_elementActive[ielem] && aw[ielem] != test) {
72  k = ielem;
73  break;
74  }
75  }
76  if (k == m_numElemConstraints) {
77  throw CanteraError("vcs_elem_rearrange",
78  "Shouldn't be here. Algorithm misfired.");
79  }
80 
81  /*
82  * Assign a large negative number to the element that we have
83  * just found, in order to take it out of further consideration.
84  */
85  aw[k] = test;
86 
87  /* *********************************************************** */
88  /* **** CHECK LINEAR INDEPENDENCE OF CURRENT FORMULA MATRIX */
89  /* **** LINE WITH PREVIOUS LINES OF THE FORMULA MATRIX ****** */
90  /* *********************************************************** */
91  /*
92  * Modified Gram-Schmidt Method, p. 202 Dalquist
93  * QR factorization of a matrix without row pivoting.
94  */
95  size_t jl = jr;
96  /*
97  * Fill in the row for the current element, k, under consideration
98  * The row will contain the Formula matrix value for that element
99  * from the current component.
100  */
101  for (size_t j = 0; j < ncomponents; ++j) {
102  sm[j + jr*ncomponents] = m_formulaMatrix(j,k);
103  }
104  if (jl > 0) {
105  /*
106  * Compute the coefficients of JA column of the
107  * the upper triangular R matrix, SS(J) = R_J_JR
108  * (this is slightly different than Dalquist)
109  * R_JA_JA = 1
110  */
111  for (size_t j = 0; j < jl; ++j) {
112  ss[j] = 0.0;
113  for (size_t i = 0; i < ncomponents; ++i) {
114  ss[j] += sm[i + jr*ncomponents] * sm[i + j*ncomponents];
115  }
116  ss[j] /= sa[j];
117  }
118  /*
119  * Now make the new column, (*,JR), orthogonal to the
120  * previous columns
121  */
122  for (size_t j = 0; j < jl; ++j) {
123  for (size_t l = 0; l < ncomponents; ++l) {
124  sm[l + jr*ncomponents] -= ss[j] * sm[l + j*ncomponents];
125  }
126  }
127  }
128 
129  /*
130  * Find the new length of the new column in Q.
131  * It will be used in the denominator in future row calcs.
132  */
133  sa[jr] = 0.0;
134  for (size_t ml = 0; ml < ncomponents; ++ml) {
135  sa[jr] += pow(sm[ml + jr*ncomponents], 2);
136  }
137  /* **************************************************** */
138  /* **** IF NORM OF NEW ROW .LT. 1E-6 REJECT ********** */
139  /* **************************************************** */
140  if (sa[jr] > 1.0e-6) {
141  break;
142  }
143  }
144  /* ****************************************** */
145  /* **** REARRANGE THE DATA ****************** */
146  /* ****************************************** */
147  if (jr != k) {
148  if (DEBUG_MODE_ENABLED && m_debug_print_lvl >= 2) {
149  plogf(" --- ");
150  plogf("%-2.2s", m_elementName[k]);
151  plogf("(%9.2g) replaces ", m_elemAbundancesGoal[k]);
152  plogf("%-2.2s", m_elementName[jr]);
153  plogf("(%9.2g) as element %3d", m_elemAbundancesGoal[jr], jr);
154  plogendl();
155  }
156  vcs_switch_elem_pos(jr, k);
157  std::swap(aw[jr], aw[k]);
158  }
159 
160  // If we haven't found enough components, go back and find some more.
161  jr++;
162  }
163  return VCS_SUCCESS;

166 void VCS_SOLVE::vcs_switch_elem_pos(size_t ipos, size_t jpos)
167 {
168  if (ipos == jpos) {
169  return;
170  }
171  AssertThrowMsg(ipos < m_numElemConstraints && jpos < m_numElemConstraints,
172  "vcs_switch_elem_pos",
173  "inappropriate args: {} {}", ipos, jpos);
174  /*
175  * Change the element Global Index list in each vcs_VolPhase object
176  * to reflect the switch in the element positions.
177  */
178  for (size_t iph = 0; iph < m_numPhases; iph++) {
179  vcs_VolPhase* volPhase = m_VolPhaseList[iph];
180  for (size_t e = 0; e < volPhase->nElemConstraints(); e++) {
181  if (volPhase->elemGlobalIndex(e) == ipos) {
182  volPhase->setElemGlobalIndex(e, jpos);
183  }
184  if (volPhase->elemGlobalIndex(e) == jpos) {
185  volPhase->setElemGlobalIndex(e, ipos);
186  }
187  }
188  }
189  std::swap(m_elemAbundancesGoal[ipos], m_elemAbundancesGoal[jpos]);
190  std::swap(m_elemAbundances[ipos], m_elemAbundances[jpos]);
191  std::swap(m_elementMapIndex[ipos], m_elementMapIndex[jpos]);
192  std::swap(m_elType[ipos], m_elType[jpos]);
193  std::swap(m_elementActive[ipos], m_elementActive[jpos]);
194  for (size_t j = 0; j < m_numSpeciesTot; ++j) {
195  std::swap(m_formulaMatrix(j,ipos), m_formulaMatrix(j,jpos));
