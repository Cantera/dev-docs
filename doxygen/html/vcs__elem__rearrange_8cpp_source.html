22 int VCS_SOLVE::vcs_elem_rearrange(double* const aw, double* const sa,
23  double* const sm, double* const ss)
24 {
25  size_t ncomponents = m_numComponents;
26  if (DEBUG_MODE_ENABLED && m_debug_print_lvl >= 2) {
27  plogf(" ");
28  for (size_t i=0; i<77; i++) {
29  plogf("-");
30  }
31  plogf("\n");
32  plogf(" --- Subroutine elem_rearrange() called to ");
33  plogf("check stoich. coefficient matrix\n");
34  plogf(" --- and to rearrange the element ordering once");
35  plogendl();

38  /*
39  * Use a temporary work array for the element numbers
40  * Also make sure the value of test is unique.
41  */
42  bool lindep = false;
43  double test = -1.0E10;
44  do {
45  lindep = false;
46  for (size_t i = 0; i < m_numElemConstraints; ++i) {
47  test -= 1.0;
48  aw[i] = m_elemAbundancesGoal[i];
49  if (test == aw[i]) {
50  lindep = true;
51  }
52  }
53  } while (lindep);
54 
55  /*
56  * Top of a loop of some sort based on the index JR. JR is the
57  * current number independent elements found.
58  */
59  size_t jr = npos;
60  do {
61  ++jr;
62  size_t k;
63  /*
64  * Top of another loop point based on finding a linearly
65  * independent species
66  */
67  do {
68  /*
69  * Search the remaining part of the mole fraction vector, AW,
70  * for the largest remaining species. Return its identity in K.
71  */
72  k = m_numElemConstraints;
73  for (size_t ielem = jr; ielem < m_numElemConstraints; ielem++) {
74  if (m_elementActive[ielem]) {
75  if (aw[ielem] != test) {
76  k = ielem;
77  break;
78  }
79  }
80  }
81  if (k == m_numElemConstraints) {
82  throw CanteraError("vcs_elem_rearrange",
83  "Shouldn't be here. Algorithm misfired.");
84  }
85 
86  /*
87  * Assign a large negative number to the element that we have
88  * just found, in order to take it out of further consideration.
89  */
90  aw[k] = test;
91 
92  /* *********************************************************** */
93  /* **** CHECK LINEAR INDEPENDENCE OF CURRENT FORMULA MATRIX */
94  /* **** LINE WITH PREVIOUS LINES OF THE FORMULA MATRIX ****** */
95  /* *********************************************************** */
96  /*
97  * Modified Gram-Schmidt Method, p. 202 Dalquist
98  * QR factorization of a matrix without row pivoting.
99  */
100  size_t jl = jr;
101  /*
102  * Fill in the row for the current element, k, under consideration
103  * The row will contain the Formula matrix value for that element
104  * from the current component.
105  */
106  for (size_t j = 0; j < ncomponents; ++j) {
107  sm[j + jr*ncomponents] = m_formulaMatrix(j,k);
108  }
109  if (jl > 0) {
110  /*
111  * Compute the coefficients of JA column of the
112  * the upper triangular R matrix, SS(J) = R_J_JR
113  * (this is slightly different than Dalquist)
114  * R_JA_JA = 1
115  */
116  for (size_t j = 0; j < jl; ++j) {
117  ss[j] = 0.0;
118  for (size_t i = 0; i < ncomponents; ++i) {
119  ss[j] += sm[i + jr*ncomponents] * sm[i + j*ncomponents];
120  }
121  ss[j] /= sa[j];
122  }
123  /*
124  * Now make the new column, (*,JR), orthogonal to the
125  * previous columns
126  */
127  for (size_t j = 0; j < jl; ++j) {
128  for (size_t l = 0; l < ncomponents; ++l) {
129  sm[l + jr*ncomponents] -= ss[j] * sm[l + j*ncomponents];
130  }
131  }
132  }
133 
134  /*
135  * Find the new length of the new column in Q.
136  * It will be used in the denominator in future row calcs.
137  */
138  sa[jr] = 0.0;
139  for (size_t ml = 0; ml < ncomponents; ++ml) {
140  sa[jr] += SQUARE(sm[ml + jr*ncomponents]);
141  }
142  /* **************************************************** */
143  /* **** IF NORM OF NEW ROW .LT. 1E-6 REJECT ********** */
144  /* **************************************************** */
145  if (sa[jr] < 1.0e-6) {
146  lindep = true;
147  } else {
148  lindep = false;
149  }
150  } while (lindep);
151  /* ****************************************** */
152  /* **** REARRANGE THE DATA ****************** */
153  /* ****************************************** */
154  if (jr != k) {
155  if (DEBUG_MODE_ENABLED && m_debug_print_lvl >= 2) {
156  plogf(" --- ");
157  plogf("%-2.2s", (m_elementName[k]).c_str());
158  plogf("(%9.2g) replaces ", m_elemAbundancesGoal[k]);
159  plogf("%-2.2s", (m_elementName[jr]).c_str());
160  plogf("(%9.2g) as element %3d", m_elemAbundancesGoal[jr], jr);
161  plogendl();
162  }
163  vcs_switch_elem_pos(jr, k);
164  std::swap(aw[jr], aw[k]);
165  }
166 
167  /*
168  * If we haven't found enough components, go back
169  * and find some more. (nc -1 is used below, because
170  * jr is counted from 0, via the C convention.
171  */
172  } while (jr < (ncomponents-1));
173  return VCS_SUCCESS;

177 {
178  if (ipos == jpos) {

181  AssertThrowMsg(ipos < m_numElemConstraints && jpos < m_numElemConstraints,

183  "inappropriate args: " + int2str(ipos) + " " + int2str(jpos));
184  /*
185  * Change the element Global Index list in each vcs_VolPhase object

189  vcs_VolPhase* volPhase = m_VolPhaseList[iph];
190  for (size_t e = 0; e < volPhase->nElemConstraints(); e++) {

192  volPhase->setElemGlobalIndex(e, jpos);

195  volPhase->setElemGlobalIndex(e, ipos);
