41 static int get_modified_time(const std::string& path) {
42 #ifdef _WIN32
43  HANDLE hFile = CreateFile(path.c_str(), NULL, NULL,
44  NULL, OPEN_EXISTING, 0, NULL);
45  if (hFile == INVALID_HANDLE_VALUE) {
46  throw CanteraError("get_modified_time", "Couldn't open file:" + path);
47  }
48  FILETIME modified;
49  GetFileTime(hFile, NULL, NULL, &modified);
50  CloseHandle(hFile);
51  return static_cast<int>(modified.dwLowDateTime);
52 #else
53  struct stat attrib;
54  stat(path.c_str(), &attrib);
55  return static_cast<int>(attrib.st_mtime);
56 #endif
57 }

181 XML_Node* Application::get_XML_File(const std::string& file, int debug)
182 {
183  std::unique_lock<std::mutex> xmlLock(xml_mutex);
184  std::string path = findInputFile(file);
185  int mtime = get_modified_time(path);
186 
187  if (xmlfiles.find(path) != xmlfiles.end()) {
188  // Already have a parsed XML tree for this file cached. Check the
189  // last-modified time.
190  std::pair<XML_Node*, int> cache = xmlfiles[path];
191  if (cache.second == mtime) {
192  return cache.first;
193  }
194  }
195 
196  // Check whether or not the file is XML (based on the file extension). If
197  // not, it will be first processed with the preprocessor.
198  string::size_type idot = path.rfind('.');
199  string ext;
200  if (idot != string::npos) {
201  ext = path.substr(idot, path.size());
202  } else {
203  ext = "";
204  }
205  XML_Node* x = new XML_Node("doc");
206  if (ext != ".xml" && ext != ".ctml") {
207  // Assume that we are trying to open a cti file. Do the conversion to XML.
208  std::stringstream phase_xml(ct2ctml_string(path));
209  x->build(phase_xml, path);
210  } else {
211  x->build(path);
212  }
213  x->lock();
214  xmlfiles[path] = {x, mtime};
215  return x;
216 }

218 XML_Node* Application::get_XML_from_string(const std::string& text)
219 {
220  std::unique_lock<std::mutex> xmlLock(xml_mutex);
221  std::pair<XML_Node*, int>& entry = xmlfiles[text];
222  if (entry.first) {
223  // Return existing cached XML tree
224  return entry.first;
225  }
226  std::stringstream s;
227  size_t start = text.find_first_not_of(" \t\r\n");

309 void Application::setDefaultDirectories()
310 {
311  // always look in the local directory first
312  inputDirs.push_back(".");
313 
314  // if environment variable CANTERA_DATA is defined, then add it to the
315  // search path. CANTERA_DATA may include multiple directory, separated by
316  // the OS-dependent path separator (in the same manner as the PATH
317  // environment variable).
318 #ifdef _WIN32
319  std::string pathsep = ";";
320 #else
321  std::string pathsep = ":";
322 #endif
323 
324  if (getenv("CANTERA_DATA") != 0) {
325  string s = string(getenv("CANTERA_DATA"));
326  size_t start = 0;
327  size_t end = s.find(pathsep);
328  while(end != npos) {
329  inputDirs.push_back(s.substr(start, end-start));
330  start = end + 1;
331  end = s.find(pathsep, start);
332  }
333  inputDirs.push_back(s.substr(start,end));
334  }
335 
336 #ifdef _WIN32
337  // Under Windows, the Cantera setup utility records the installation
338  // directory in the registry. Data files are stored in the 'data'
339  // subdirectory of the main installation directory.
340  std::string installDir;
341  readStringRegistryKey("SOFTWARE\\Cantera\\Cantera " CANTERA_SHORT_VERSION,
342  "InstallDir", installDir, "");
343  if (installDir != "") {
344  inputDirs.push_back(installDir + "data");
345 
346  // Scripts for converting mechanisms to CTI and CMTL are installed in
347  // the 'bin' subdirectory. Add that directory to the PYTHONPATH.
348  const char* old_pythonpath = getenv("PYTHONPATH");
349  std::string pythonpath = "PYTHONPATH=" + installDir + "\\bin";
350  if (old_pythonpath) {
351  pythonpath += ";";
352  pythonpath.append(old_pythonpath);
353  }
354  _putenv(pythonpath.c_str());
355  }
356 
357 #endif
358 
359 #ifdef DARWIN
360  // add a default data location for Mac OS X
361  inputDirs.push_back("/Applications/Cantera/data");
362 #endif
363 
364  // CANTERA_DATA is defined in file config.h. This file is written during the
365  // build process (unix), and points to the directory specified by the
366  // 'prefix' option to 'configure', or else to /usr/local/cantera.
367 #ifdef CANTERA_DATA
368  string datadir = string(CANTERA_DATA);
369  inputDirs.push_back(datadir);
370 #endif
371 }

391 std::string Application::findInputFile(const std::string& name)
392 {
393  std::unique_lock<std::mutex> dirLock(dir_mutex);
394  string::size_type islash = name.find('/');
395  string::size_type ibslash = name.find('\\');
396  std::vector<string>& dirs = inputDirs;
397 
398  // Expand "~/" to user's home directory, if possible
399  if (name.find("~/") == 0) {
400  char* home = getenv("HOME"); // POSIX systems
401  if (!home) {
402  home = getenv("USERPROFILE"); // Windows systems
403  }
404  if (home) {
405  return home + name.substr(1, npos);
406  }
407  }
408 
409  if (islash == string::npos && ibslash == string::npos) {
410  size_t nd = dirs.size();
411  for (size_t i = 0; i < nd; i++) {
412  string inname = dirs[i] + "/" + name;
413  std::ifstream fin(inname);
414  if (fin) {
415  return inname;
416  }
417  }
418  string msg = "\nInput file " + name + " not found in director";
419  msg += (nd == 1 ? "y " : "ies ");
420  for (size_t i = 0; i < nd; i++) {
421  msg += "\n'" + dirs[i] + "'";
422  if (i+1 < nd) {
423  msg += ", ";
424  }
425  }
426  msg += "\n\n";
427  msg += "To fix this problem, either:\n";
428  msg += " a) move the missing files into the local directory;\n";
429  msg += " b) define environment variable CANTERA_DATA to\n";
430  msg += " point to the directory containing the file.";
431  throw CanteraError("findInputFile", msg);
432  }
433 
434  return name;
435 }
