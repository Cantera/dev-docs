73 void TransportFactory::getBinDiffCorrection(doublereal t,
74  const GasTransportParams& tr, MMCollisionInt& integrals,
75  size_t k, size_t j, doublereal xk, doublereal xj,
76  doublereal& fkj, doublereal& fjk)
77 {
78  doublereal w1, w2, wsum, sig1, sig2, sig12, sigratio, sigratio2,
79  sigratio3, tstar1, tstar2, tstar12,
80  om22_1, om22_2, om11_12, astar_12, bstar_12, cstar_12,
81  cnst, wmwp, sqw12, p1, p2, p12, q1, q2, q12;
82 
83  w1 = tr.mw[k];
84  w2 = tr.mw[j];
85  wsum = w1 + w2;
86  wmwp = (w1 - w2)/wsum;
87  sqw12 = sqrt(w1*w2);
88 
89  sig1 = tr.sigma[k];
90  sig2 = tr.sigma[j];
91  sig12 = 0.5*(tr.sigma[k] + tr.sigma[j]);
92  sigratio = sig1*sig1/(sig2*sig2);
93  sigratio2 = sig1*sig1/(sig12*sig12);
94  sigratio3 = sig2*sig2/(sig12*sig12);
95 
96  tstar1 = Boltzmann * t / tr.eps[k];
97  tstar2 = Boltzmann * t / tr.eps[j];
98  tstar12 = Boltzmann * t / sqrt(tr.eps[k] * tr.eps[j]);
99 
100  om22_1 = integrals.omega22(tstar1, tr.delta(k,k));
101  om22_2 = integrals.omega22(tstar2, tr.delta(j,j));
102  om11_12 = integrals.omega11(tstar12, tr.delta(k,j));
103  astar_12 = integrals.astar(tstar12, tr.delta(k,j));
104  bstar_12 = integrals.bstar(tstar12, tr.delta(k,j));
105  cstar_12 = integrals.cstar(tstar12, tr.delta(k,j));
106 
107  cnst = sigratio * sqrt(2.0*w2/wsum) * 2.0 *
108  w1*w1/(wsum * w2);
109  p1 = cnst * om22_1 / om11_12;
110 
111  cnst = (1.0/sigratio) * sqrt(2.0*w1/wsum) * 2.0*w2*w2/(wsum*w1);
112  p2 = cnst * om22_2 / om11_12;
113 
114  p12 = 15.0 * wmwp*wmwp + 8.0*w1*w2*astar_12/(wsum*wsum);
115 
116  cnst = (2.0/(w2*wsum))*sqrt(2.0*w2/wsum)*sigratio2;
117  q1 = cnst*((2.5 - 1.2*bstar_12)*w1*w1 + 3.0*w2*w2
118  + 1.6*w1*w2*astar_12);
119 
120  cnst = (2.0/(w1*wsum))*sqrt(2.0*w1/wsum)*sigratio3;
121  q2 = cnst*((2.5 - 1.2*bstar_12)*w2*w2 + 3.0*w1*w1
122  + 1.6*w1*w2*astar_12);
123 
124  q12 = wmwp*wmwp*15.0*(2.5 - 1.2*bstar_12)
125  + 4.0*w1*w2*astar_12*(11.0 - 2.4*bstar_12)/(wsum*wsum)
126  + 1.6*wsum*om22_1*om22_2/(om11_12*om11_12*sqw12)
127  * sigratio2 * sigratio3;
128 
129  cnst = 6.0*cstar_12 - 5.0;
130  fkj = 1.0 + 0.1*cnst*cnst *
131  (p1*xk*xk + p2*xj*xj + p12*xk*xj)/
132  (q1*xk*xk + q2*xj*xj + q12*xk*xj);
133  fjk = 1.0 + 0.1*cnst*cnst *
134  (p2*xk*xk + p1*xj*xj + p12*xk*xj)/
135  (q2*xk*xk + q1*xj*xj + q12*xk*xj);

138 void TransportFactory::makePolarCorrections(size_t i, size_t j,
139  const GasTransportParams& tr, doublereal& f_eps, doublereal& f_sigma)
140 {
141  // no correction if both are nonpolar, or both are polar
142  if (tr.polar[i] == tr.polar[j]) {
143  f_eps = 1.0;
144  f_sigma = 1.0;
145  return;
146  }
147 
148  // corrections to the effective diameter and well depth
149  // if one is polar and one is non-polar
150 
151  size_t kp = (tr.polar[i] ? i : j); // the polar one
152  size_t knp = (i == kp ? j : i); // the nonpolar one
153 
154  doublereal d3np, d3p, alpha_star, mu_p_star, xi;
155  d3np = pow(tr.sigma[knp],3);
156  d3p = pow(tr.sigma[kp],3);
157  alpha_star = tr.alpha[knp]/d3np;
158  mu_p_star = tr.dipole(kp,kp)/sqrt(d3p * tr.eps[kp]);
159  xi = 1.0 + 0.25 * alpha_star * mu_p_star * mu_p_star *
160  sqrt(tr.eps[kp]/tr.eps[knp]);
161  f_sigma = pow(xi, -1.0/6.0);
162  f_eps = xi*xi;
163 }

412 void TransportFactory::setupMM(const std::vector<const XML_Node*> &transport_database,
413  thermo_t* thermo, int mode, int log_level, GasTransportParams& tr)
414 {
415 
416  // constant mixture attributes
417  tr.thermo = thermo;
418  tr.nsp_ = tr.thermo->nSpecies();
419  size_t nsp = tr.nsp_;
420 
421  tr.tmin = thermo->minTemp();
422  tr.tmax = thermo->maxTemp();
423  tr.mw.resize(nsp);
424  tr.log_level = log_level;
425 
426  copy(tr.thermo->molecularWeights().begin(), tr.thermo->molecularWeights().end(), tr.mw.begin());
427 
428  tr.mode_ = mode;
429  tr.epsilon.resize(nsp, nsp, 0.0);
430  tr.delta.resize(nsp, nsp, 0.0);
431  tr.reducedMass.resize(nsp, nsp, 0.0);
432  tr.dipole.resize(nsp, nsp, 0.0);
433  tr.diam.resize(nsp, nsp, 0.0);
434  tr.crot.resize(nsp);
435  tr.zrot.resize(nsp);
436  tr.polar.resize(nsp, false);
437  tr.alpha.resize(nsp, 0.0);
438  tr.poly.resize(nsp);
439  tr.sigma.resize(nsp);
440  tr.eps.resize(nsp);
441 
442  XML_Node root, log;
443  getTransportData(transport_database, log, tr.thermo->speciesNames(), tr);
444 
445  for (size_t i = 0; i < nsp; i++) {
446  tr.poly[i].resize(nsp);
447  }
448 
449  doublereal tstar_min = 1.e8, tstar_max = 0.0;
450  doublereal f_eps, f_sigma;
451 
452  DenseMatrix& diam = tr.diam;
453  DenseMatrix& epsilon = tr.epsilon;
454 
455  for (size_t i = 0; i < nsp; i++) {
456  for (size_t j = i; j < nsp; j++) {
457  // the reduced mass
458  tr.reducedMass(i,j) = tr.mw[i] * tr.mw[j] / (Avogadro * (tr.mw[i] + tr.mw[j]));
459 
460  // hard-sphere diameter for (i,j) collisions
461  diam(i,j) = 0.5*(tr.sigma[i] + tr.sigma[j]);
462 
463  // the effective well depth for (i,j) collisions
464  epsilon(i,j) = sqrt(tr.eps[i]*tr.eps[j]);
465 
466  // The polynomial fits of collision integrals vs. T*
467  // will be done for the T* from tstar_min to tstar_max
468  tstar_min = std::min(tstar_min, Boltzmann * tr.tmin/epsilon(i,j));
469  tstar_max = std::max(tstar_max, Boltzmann * tr.tmax/epsilon(i,j));
470 
471  // the effective dipole moment for (i,j) collisions
472  tr.dipole(i,j) = sqrt(tr.dipole(i,i)*tr.dipole(j,j));
473 
474  // reduced dipole moment delta* (nondimensional)
475  doublereal d = diam(i,j);
476  tr.delta(i,j) = 0.5 * tr.dipole(i,j)*tr.dipole(i,j)
477  / (epsilon(i,j) * d * d * d);
478 
479  makePolarCorrections(i, j, tr, f_eps, f_sigma);
480  tr.diam(i,j) *= f_sigma;
481  epsilon(i,j) *= f_eps;
482 
483  // properties are symmetric
484  tr.reducedMass(j,i) = tr.reducedMass(i,j);
485  diam(j,i) = diam(i,j);
486  epsilon(j,i) = epsilon(i,j);
487  tr.dipole(j,i) = tr.dipole(i,j);
488  tr.delta(j,i) = tr.delta(i,j);
489  }
490  }
491 
492  // Chemkin fits the entire T* range in the Monchick and Mason tables,
493  // so modify tstar_min and tstar_max if in Chemkin compatibility mode
494 
495  if (mode == CK_Mode) {
496  tstar_min = 0.101;
497  tstar_max = 99.9;
498  }
499 
500 
501  // initialize the collision integral calculator for the desired
502  // T* range
503  if (DEBUG_MODE_ENABLED && m_verbose) {
504  writelog("*** collision_integrals ***\n");
505  }
506  MMCollisionInt integrals;
507  integrals.init(tstar_min, tstar_max, log_level);
508  fitCollisionIntegrals(tr, integrals);
509  if (DEBUG_MODE_ENABLED && m_verbose) {
510  writelog("*** end of collision_integrals ***\n");
511  }
512  // make polynomial fits
513  if (DEBUG_MODE_ENABLED && m_verbose) {
514  writelog("*** property fits ***\n");
515  }
516  fitProperties(tr, integrals);
517  if (DEBUG_MODE_ENABLED && m_verbose) {
518  writelog("*** end of property fits ***\n");
519  }
520 }

633 void TransportFactory::fitCollisionIntegrals(GasTransportParams& tr,
634  MMCollisionInt& integrals)
635 {
636  vector_fp::iterator dptr;
637  doublereal dstar;
638  size_t nsp = tr.nsp_;
639  int mode = tr.mode_;
640  size_t i, j;
641 
642  // Chemkin fits to sixth order polynomials
643  int degree = (mode == CK_Mode ? 6 : COLL_INT_POLY_DEGREE);
644  if (DEBUG_MODE_ENABLED && m_verbose) {
645  writelog("tstar_fits\n"
646  "fits to A*, B*, and C* vs. log(T*).\n"
647  "These are done only for the required dstar(j,k) values.\n\n");
648  if (tr.log_level < 3) {
649  writelog("*** polynomial coefficients not printed (log_level < 3) ***\n");
650  }
651  }
652  for (i = 0; i < nsp; i++) {
653  for (j = i; j < nsp; j++) {
654  // Chemkin fits only delta* = 0
655  if (mode != CK_Mode) {
656  dstar = tr.delta(i,j);
657  } else {
658  dstar = 0.0;
659  }
660 
661  // if a fit has already been generated for
662  // delta* = tr.delta(i,j), then use it. Otherwise,
663  // make a new fit, and add tr.delta(i,j) to the list
664  // of delta* values for which fits have been done.
665 
666  // 'find' returns a pointer to end() if not found
667  dptr = find(tr.fitlist.begin(), tr.fitlist.end(), dstar);
668  if (dptr == tr.fitlist.end()) {
669  vector_fp ca(degree+1), cb(degree+1), cc(degree+1);
670  vector_fp co22(degree+1);
671  integrals.fit(degree, dstar,
672  DATA_PTR(ca), DATA_PTR(cb), DATA_PTR(cc));
673  integrals.fit_omega22(degree, dstar,
674  DATA_PTR(co22));
675  tr.omega22_poly.push_back(co22);
676  tr.astar_poly.push_back(ca);
677  tr.bstar_poly.push_back(cb);
678  tr.cstar_poly.push_back(cc);
679  tr.poly[i][j] = static_cast<int>(tr.astar_poly.size()) - 1;
680  tr.fitlist.push_back(dstar);
681  }
682 
683  // delta* found in fitlist, so just point to this
684  // polynomial
685  else {
686  tr.poly[i][j] = static_cast<int>((dptr - tr.fitlist.begin()));
687  }
688  tr.poly[j][i] = tr.poly[i][j];
689  }
690  }

693 void TransportFactory::getTransportData(const std::vector<const XML_Node*> &xspecies,
694  XML_Node& log, const std::vector<std::string> &names, GasTransportParams& tr)
695 {
696  std::map<std::string, size_t> speciesIndices;
697  for (size_t i = 0; i < names.size(); i++) {
698  speciesIndices[names[i]] = i;
699  }
700 
701  for (size_t i = 0; i < xspecies.size(); i++) {
702  const XML_Node& sp = *xspecies[i];
703 
704  // Find the index for this species in 'names'
705  std::map<std::string, size_t>::const_iterator iter =
706  speciesIndices.find(sp["name"]);
707  size_t j;
708  if (iter != speciesIndices.end()) {
709  j = iter->second;
710  } else {
711  // Don't need transport data for this species
712  continue;
713  }
714 
715  XML_Node& node = sp.child("transport");
716 
717  // parameters are converted to SI units before storing
718 
719  // Molecular geometry; rotational heat capacity / R
720  XML_Node* geomNode = ctml::getByTitle(node, "geometry");
721  std::string geom = (geomNode) ? geomNode->value() : "";
722  if (geom == "atom") {
723  tr.crot[j] = 0.0;
724  } else if (geom == "linear") {
725  tr.crot[j] = 1.0;
726  } else if (geom == "nonlinear") {
727  tr.crot[j] = 1.5;
728  } else {
729  throw TransportDBError(i, "invalid geometry");
730  }
731 
732  // Well-depth parameter in Kelvin (converted to Joules)
733  double welldepth = ctml::getFloat(node, "LJ_welldepth");
734  if (welldepth >= 0.0) {
735  tr.eps[j] = Boltzmann * welldepth;
736  } else {
737  throw TransportDBError(i, "negative well depth");
738  }
739 
740  // Lennard-Jones diameter of the molecule, given in Angstroms.
741  double diam = ctml::getFloat(node, "LJ_diameter");
742  if (diam > 0.0) {
743  tr.sigma[j] = 1.e-10 * diam; // A -> m
744  } else {
745  throw TransportDBError(i, "negative or zero diameter");
746  }
747 
748  // Dipole moment of the molecule.
749  // Given in Debye (a debye is 10-18 cm3/2 erg1/2)
750  double dipole = ctml::getFloat(node, "dipoleMoment");
751  if (dipole >= 0.0) {
752  tr.dipole(j,j) = 1.e-25 * SqrtTen * dipole;
753  tr.polar[j] = (dipole > 0.0);
754  } else {
755  throw TransportDBError(i, "negative dipole moment");
756  }
757 
758  // Polarizability of the molecule, given in cubic Angstroms.
759  double polar = ctml::getFloat(node, "polarizability");
760  if (polar >= 0.0) {
761  tr.alpha[j] = 1.e-30 * polar; // A^3 -> m^3
762  } else {
763  throw TransportDBError(i, "negative polarizability");
764  }
765 
766  // Rotational relaxation number. (Number of collisions it takes to
767  // equilibrate the rotational dofs with the temperature)
768  double rot = ctml::getFloat(node, "rotRelax");
769  if (rot >= 0.0) {
770  tr.zrot[j] = std::max(1.0, rot);
771  } else {
772  throw TransportDBError(i, "negative rotation relaxation number");
773  }
774  }
775 }

1093 void TransportFactory::fitProperties(GasTransportParams& tr,
1094  MMCollisionInt& integrals)
1095 {
1096  doublereal tstar;
1097  int ndeg = 0;
1098  // number of points to use in generating fit data
1099  const size_t np = 50;
1100 
1101  int mode = tr.mode_;
1102  int degree = (mode == CK_Mode ? 3 : 4);
1103 
1104  doublereal t, om22;
1105  doublereal dt = (tr.tmax - tr.tmin)/(np-1);
1106  vector_fp tlog(np), spvisc(np), spcond(np);
1107  doublereal val, fit;
1108 
1109  vector_fp w(np), w2(np);
1110 
1111  // generate array of log(t) values
1112  for (size_t n = 0; n < np; n++) {
1113  t = tr.tmin + dt*n;
1114  tlog[n] = log(t);
1115  }
1116 
1117  // vector of polynomial coefficients
1118  vector_fp c(degree + 1), c2(degree + 1);
1119 
1120 
1121  // fit the pure-species viscosity and thermal conductivity for
1122  // each species
1123  if (DEBUG_MODE_ENABLED && tr.log_level < 2 && m_verbose) {
1124  writelog("*** polynomial coefficients not printed (log_level < 2) ***\n");
1125  }
1126  doublereal sqrt_T, visc, err, relerr,
1127  mxerr = 0.0, mxrelerr = 0.0, mxerr_cond = 0.0, mxrelerr_cond = 0.0;
1128 
1129  if (DEBUG_MODE_ENABLED && m_verbose) {
1130  writelog("Polynomial fits for viscosity:\n");
1131  if (mode == CK_Mode) {
1132  writelog("log(viscosity) fit to cubic polynomial in log(T)\n");
1133  } else {
1134  writelogf("viscosity/sqrt(T) fit to polynomial of degree "
1135  "%d in log(T)", degree);
1136  }
1137  }
1138 
1139  doublereal cp_R, cond, w_RT, f_int, A_factor, B_factor,
1140  c1, cv_rot, cv_int, f_rot, f_trans, om11;
1141  doublereal diffcoeff;
1142 
1143  for (size_t k = 0; k < tr.nsp_; k++) {
1144  for (size_t n = 0; n < np; n++) {
1145  t = tr.tmin + dt*n;
1146 
1147  tr.thermo->setTemperature(t);
1148  cp_R = ((IdealGasPhase*)tr.thermo)->cp_R_ref()[k];
1149 
1150  tstar = Boltzmann * t/ tr.eps[k];
1151  sqrt_T = sqrt(t);
1152  om22 = integrals.omega22(tstar, tr.delta(k,k));
1153  om11 = integrals.omega11(tstar, tr.delta(k,k));
1154 
1155  // self-diffusion coefficient, without polar
1156  // corrections
1157  diffcoeff = ThreeSixteenths *
1158  sqrt(2.0 * Pi/tr.reducedMass(k,k)) *
1159  pow((Boltzmann * t), 1.5)/
1160  (Pi * tr.sigma[k] * tr.sigma[k] * om11);
1161 
1162  // viscosity
1163  visc = FiveSixteenths
1164  * sqrt(Pi * tr.mw[k] * Boltzmann * t / Avogadro) /
1165  (om22 * Pi * tr.sigma[k]*tr.sigma[k]);
1166 
1167  // thermal conductivity
1168  w_RT = tr.mw[k]/(GasConstant * t);
1169  f_int = w_RT * diffcoeff/visc;
1170  cv_rot = tr.crot[k];
1171
