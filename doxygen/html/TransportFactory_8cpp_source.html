221  break;

226  break;

351 void TransportFactory::getLiquidSpeciesTransportData(const std::vector<const XML_Node*> &xspecies,
352  XML_Node& log,
353  const std::vector<std::string> &names,
354  LiquidTransportParams& trParam)
355 {
356  std::string name;
357  /*
358  Create a map of species names versus liquid transport data parameters
359  */
360  std::map<std::string, LiquidTransportData> datatable;
361 
362  // Store the number of species in the phase
363  size_t nsp = trParam.nsp_;
364 
365  // Store the number of off-diagonal symmetric interactions between species in the phase
366  size_t nBinInt = nsp*(nsp-1)/2;
367 
368  // read all entries in database into 'datatable' and check for
369  // errors. Note that this procedure validates all entries, not
370  // only those for the species listed in 'names'.
371  for (size_t i = 0; i < nsp; i++) {
372  const XML_Node& sp = *xspecies[i];
373  name = sp["name"];
374  vector_fp vCoeff;
375 
376  // Species with no 'transport' child are skipped. However, if that species is in the list,
377  // it will throw an exception below.
378  try {
379  if (sp.hasChild("transport")) {
380  XML_Node& trNode = sp.child("transport");
381 
382  // Fill datatable with LiquidTransportData objects for error checking
383  // and then insertion into LiquidTransportData objects below.
384  LiquidTransportData data;
385  data.speciesName = name;
386  data.mobilityRatio.resize(nsp*nsp,0);
387  data.selfDiffusion.resize(nsp,0);
388  ThermoPhase* temp_thermo = trParam.thermo;
389  size_t num = trNode.nChildren();
390  for (size_t iChild = 0; iChild < num; iChild++) {
391  XML_Node& xmlChild = trNode.child(iChild);
392  std::string nodeName = xmlChild.name();
393 
394  switch (m_tranPropMap[nodeName]) {
395  case TP_VISCOSITY:
396  data.viscosity = newLTP(xmlChild, name, m_tranPropMap[nodeName], temp_thermo);
397  break;
398  case TP_IONCONDUCTIVITY:
399  data.ionConductivity = newLTP(xmlChild, name, m_tranPropMap[nodeName], temp_thermo);
400  break;
401  case TP_MOBILITYRATIO: {
402  for (size_t iSpec = 0; iSpec< nBinInt; iSpec++) {
403  XML_Node& propSpecNode = xmlChild.child(iSpec);
404  std::string specName = propSpecNode.name();
405  size_t loc = specName.find(":");
406  std::string firstSpec = specName.substr(0,loc);
407  std::string secondSpec = specName.substr(loc+1);
408  size_t index = temp_thermo->speciesIndex(firstSpec)+nsp*temp_thermo->speciesIndex(secondSpec);
409  data.mobilityRatio[index] = newLTP(propSpecNode, name, m_tranPropMap[nodeName], temp_thermo);
410  };
411  };
412  break;
413  case TP_SELFDIFFUSION: {
414  for (size_t iSpec = 0; iSpec< nsp; iSpec++) {
415  XML_Node& propSpecNode = xmlChild.child(iSpec);
416  std::string specName = propSpecNode.name();
417  size_t index = temp_thermo->speciesIndex(specName);
418  data.selfDiffusion[index] = newLTP(propSpecNode, name, m_tranPropMap[nodeName], temp_thermo);
419  };
420  };
421  break;
422  case TP_THERMALCOND:
423  data.thermalCond = newLTP(xmlChild,
424  name,
425  m_tranPropMap[nodeName],
426  temp_thermo);
427  break;
428  case TP_DIFFUSIVITY:
429  data.speciesDiffusivity = newLTP(xmlChild,
430  name,
431  m_tranPropMap[nodeName],
432  temp_thermo);
433  break;
434  case TP_HYDRORADIUS:
435  data.hydroRadius = newLTP(xmlChild,
436  name,
437  m_tranPropMap[nodeName],
438  temp_thermo);
439  break;
440  case TP_ELECTCOND:
441  data.electCond = newLTP(xmlChild,
442  name,
443  m_tranPropMap[nodeName],
444  temp_thermo);
445  break;
446  default:
447  throw CanteraError("getLiquidSpeciesTransportData","unknown transport property: " + nodeName);
448  }
449  }
450  datatable[name] = data;
451  }
452  } catch (CanteraError& err) {
453  err.save();
454  throw err;
455  }
456  }
457 
458  trParam.LTData.clear();
459  for (size_t i = 0; i < trParam.nsp_; i++) {
460  /*
461  Check to see that we have a LiquidTransportData object for all of the
462  species in the phase. If not, throw an error.
463  */
464  auto it = datatable.find(names[i]);
465  if (it == datatable.end()) {
466  throw TransportDBError(0,"No transport data found for species " + names[i]);
467  }
468  LiquidTransportData& trdat = it->second;
469 
470  /*
471  Now, transfer these objects into LTData in the correct phase index order by
472  calling the default copy constructor for LiquidTransportData.
473  */
474  trParam.LTData.push_back(trdat);
475  }

573  }
574  }
575  /* Allow a switch between mass-averaged, mole-averaged
576  * and solvent specified reference velocities.
577  * XML code within the transportProperty node
578  * (i.e. within <viscosity>) should read as follows
579  * <velocityBasis basis="mass"> <!-- mass averaged -->
580  * <velocityBasis basis="mole"> <!-- mole averaged -->

592  } else {

595  }
596  }
597  }

605 void TransportFactory::getSolidTransportData(const XML_Node& transportNode,
606  XML_Node& log,
607  const std::string phaseName,
608  SolidTransportData& trParam)
609 {
610  try {
611  size_t num = transportNode.nChildren();
612  for (size_t iChild = 0; iChild < num; iChild++) {
613  //tranTypeNode is a type of transport property like viscosity
614  XML_Node& tranTypeNode = transportNode.child(iChild);
615  std::string nodeName = tranTypeNode.name();
616  ThermoPhase* temp_thermo = trParam.thermo;
617 
618  //tranTypeNode contains the interaction model
619  switch (m_tranPropMap[nodeName]) {
620  case TP_IONCONDUCTIVITY:
621  trParam.ionConductivity = newLTP(tranTypeNode, phaseName,
622  m_tranPropMap[nodeName],
623  temp_thermo);
624  break;
625  case TP_THERMALCOND:
626  trParam.thermalConductivity = newLTP(tranTypeNode, phaseName,
627  m_tranPropMap[nodeName],
628  temp_thermo);
629  break;
630  case TP_DEFECTDIFF:
631  trParam.defectDiffusivity = newLTP(tranTypeNode, phaseName,
632  m_tranPropMap[nodeName],
633  temp_thermo);
634  break;
635  case TP_DEFECTCONC:
636  trParam.defectActivity = newLTP(tranTypeNode, phaseName,
637  m_tranPropMap[nodeName],
638  temp_thermo);
639  break;
640  case TP_ELECTCOND:
641  trParam.electConductivity = newLTP(tranTypeNode, phaseName,
642  m_tranPropMap[nodeName],
643  temp_thermo);
644  break;
645  default:
646  throw CanteraError("getSolidTransportData","unknown transport property: " + nodeName);
647  }
648  }
649  } catch (CanteraError) {
650  showErrors(std::cout);
651  }
652  return;
