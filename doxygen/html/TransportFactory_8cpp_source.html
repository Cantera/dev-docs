176 Transport* TransportFactory::newTransport(const std::string& transportModel,
177  thermo_t* phase, int log_level, int ndim)
178 {
179  vector_fp state;
180  Transport* tr = 0;
181  phase->saveState(state);
182 
183  if (transportModel == "Solid") {
184  tr = new SolidTransport;
185  initSolidTransport(tr, phase, log_level);
186  tr->setThermo(*phase);
187  } else if (transportModel == "DustyGas") {
188  tr = new DustyGasTransport;
189  Transport* gastr = new MultiTransport;
190  gastr->init(phase, 0, log_level);
191  DustyGasTransport* dtr = (DustyGasTransport*)tr;
192  dtr->initialize(phase, gastr);
193  } else if (transportModel == "Simple") {
194  tr = new SimpleTransport();
195  initLiquidTransport(tr, phase, log_level);
196  tr->setThermo(*phase);
197  } else if (transportModel == "Liquid") {
198  tr = new LiquidTransport(phase, ndim);
199  initLiquidTransport(tr, phase, log_level);
200  tr->setThermo(*phase);
201  } else {
202  tr = create(transportModel);
203  tr->init(phase, m_CK_mode[transportModel], log_level);
204  }
205  phase->restoreState(state);
206  return tr;
207 }

219 void TransportFactory::setupLiquidTransport(thermo_t* thermo, int log_level,
220  LiquidTransportParams& trParam)
221 {
222  const std::vector<const XML_Node*> & species_database = thermo->speciesData();
223  const XML_Node* phase_database = &thermo->xml();
224 
225  // constant mixture attributes
226  trParam.thermo = thermo;
227  trParam.nsp_ = trParam.thermo->nSpecies();
228  size_t nsp = trParam.nsp_;
229  trParam.tmin = thermo->minTemp();
230  trParam.tmax = thermo->maxTemp();
231  trParam.log_level = log_level;
232 
233  // Get the molecular weights and load them into trParam
234  trParam.mw = trParam.thermo->molecularWeights();
235 
236  // Resize all other vectors in trParam
237  trParam.LTData.resize(nsp);
238 
239  // Need to identify a method to obtain interaction matrices.
240  // This will fill LiquidTransportParams members visc_Eij, visc_Sij
241  trParam.thermalCond_Aij.resize(nsp,nsp);
242  trParam.diff_Dij.resize(nsp,nsp);
243  trParam.radius_Aij.resize(nsp,nsp);
244 
245  XML_Node log;
246  // Note that getLiquidSpeciesTransportData just populates the pure species transport data.
247  getLiquidSpeciesTransportData(species_database, log, trParam.thermo->speciesNames(), trParam);
248 
249  // getLiquidInteractionsTransportData() populates the species-species
250  // interaction models parameters like visc_Eij
251  if (phase_database->hasChild("transport")) {
252  XML_Node& transportNode = phase_database->child("transport");
253  getLiquidInteractionsTransportData(transportNode, log, trParam.thermo->speciesNames(), trParam);
254  }

301 void TransportFactory::getLiquidSpeciesTransportData(const std::vector<const XML_Node*> &xspecies,
302  XML_Node& log,
303  const std::vector<std::string> &names,
304  LiquidTransportParams& trParam)
305 {
306  // Create a map of species names versus liquid transport data parameters
307  std::map<std::string, LiquidTransportData> datatable;
308 
309  // Store the number of species in the phase
310  size_t nsp = trParam.nsp_;
311 
312  // Store the number of off-diagonal symmetric interactions between species in the phase
313  size_t nBinInt = nsp*(nsp-1)/2;
314 
315  // read all entries in database into 'datatable' and check for errors. Note
316  // that this procedure validates all entries, not only those for the species
317  // listed in 'names'.
318  for (size_t i = 0; i < nsp; i++) {
319  const XML_Node& sp = *xspecies[i];
320  string name = sp["name"];
321 
322  // Species with no 'transport' child are skipped. However, if that
323  // species is in the list, it will throw an exception below.
324  if (sp.hasChild("transport")) {
325  XML_Node& trNode = sp.child("transport");
326 
327  // Fill datatable with LiquidTransportData objects for error checking
328  // and then insertion into LiquidTransportData objects below.
329  LiquidTransportData data;
330  data.speciesName = name;
331  data.mobilityRatio.resize(nsp*nsp,0);
332  data.selfDiffusion.resize(nsp,0);
333  size_t num = trNode.nChildren();
334  for (size_t iChild = 0; iChild < num; iChild++) {
335  XML_Node& xmlChild = trNode.child(iChild);
336  std::string nodeName = xmlChild.name();
337 
338  switch (m_tranPropMap[nodeName]) {
339  case TP_VISCOSITY:
340  data.viscosity = newLTP(xmlChild, name, m_tranPropMap[nodeName], trParam.thermo);
341  break;
342  case TP_IONCONDUCTIVITY:
343  data.ionConductivity = newLTP(xmlChild, name, m_tranPropMap[nodeName], trParam.thermo);
344  break;
345  case TP_MOBILITYRATIO: {
346  for (size_t iSpec = 0; iSpec< nBinInt; iSpec++) {
347  XML_Node& propSpecNode = xmlChild.child(iSpec);
348  std::string specName = propSpecNode.name();
349  size_t loc = specName.find(":");
350  std::string firstSpec = specName.substr(0,loc);
351  std::string secondSpec = specName.substr(loc+1);
352  size_t index = trParam.thermo->speciesIndex(firstSpec)+nsp*trParam.thermo->speciesIndex(secondSpec);
353  data.mobilityRatio[index] = newLTP(propSpecNode, name, m_tranPropMap[nodeName], trParam.thermo);
354  };
355  };
356  break;
357  case TP_SELFDIFFUSION: {
358  for (size_t iSpec = 0; iSpec< nsp; iSpec++) {
359  XML_Node& propSpecNode = xmlChild.child(iSpec);
360  std::string specName = propSpecNode.name();
361  size_t index = trParam.thermo->speciesIndex(specName);
362  data.selfDiffusion[index] = newLTP(propSpecNode, name, m_tranPropMap[nodeName], trParam.thermo);
363  };
364  };
365  break;
366  case TP_THERMALCOND:
367  data.thermalCond = newLTP(xmlChild,
368  name,
369  m_tranPropMap[nodeName],
370  trParam.thermo);
371  break;
372  case TP_DIFFUSIVITY:
373  data.speciesDiffusivity = newLTP(xmlChild,
374  name,
375  m_tranPropMap[nodeName],
376  trParam.thermo);
377  break;
378  case TP_HYDRORADIUS:
379  data.hydroRadius = newLTP(xmlChild,
380  name,
381  m_tranPropMap[nodeName],
382  trParam.thermo);
383  break;
384  case TP_ELECTCOND:
385  data.electCond = newLTP(xmlChild,
386  name,
387  m_tranPropMap[nodeName],
388  trParam.thermo);
389  break;
390  default:
391  throw CanteraError("getLiquidSpeciesTransportData","unknown transport property: " + nodeName);
392  }
393  }
394  datatable[name] = data;
395  }
396  }
397 
398  trParam.LTData.clear();
399  for (size_t i = 0; i < trParam.nsp_; i++) {
400  // Check to see that we have a LiquidTransportData object for all of the
401  // species in the phase. If not, throw an error.
402  auto it = datatable.find(names[i]);
403  if (it == datatable.end()) {
404  throw TransportDBError(0,"No transport data found for species " + names[i]);
405  }
406 
407  // Now, transfer these objects into LTData in the correct phase index
408  // order by calling the default copy constructor for
409  // LiquidTransportData.
410  trParam.LTData.push_back(it->second);
411  }
412 }

414 /*
415  * Read transport property data from a file for interactions between species in
416  * a liquid. Given the name of a file containing transport property parameters
417  * and a list of species names, this method returns an instance of
418  * TransportParams containing the transport data for these species read from the
419  * file.
420  */
421 void TransportFactory::getLiquidInteractionsTransportData(const XML_Node& transportNode,
422  XML_Node& log,
423  const std::vector<std::string> &names,
424  LiquidTransportParams& trParam)
425 {
426  try {
427  size_t nsp = trParam.nsp_;
428  size_t nBinInt = nsp*(nsp-1)/2;
429  for (size_t iChild = 0; iChild < transportNode.nChildren(); iChild++) {
430  //tranTypeNode is a type of transport property like viscosity
431  XML_Node& tranTypeNode = transportNode.child(iChild);
432  std::string nodeName = tranTypeNode.name();
433  trParam.mobilityRatio.resize(nsp*nsp,0);
434  trParam.selfDiffusion.resize(nsp,0);
435 
436  if (tranTypeNode.name() == "compositionDependence") {
437  std::string modelName = tranTypeNode.attrib("model");
438  auto it = m_LTImodelMap.find(modelName);
439  if (it == m_LTImodelMap.end()) {
440  throw CanteraError("TransportFactory::getLiquidInteractionsTransportData",
441  "Unknown compositionDependence string: " + modelName);
442  } else {
443  trParam.compositionDepTypeDefault_ = it->second;
444  }
445  } else {
446  if (tranTypeNode.hasChild("compositionDependence")) {
447  //compDepNode contains the interaction model
448  XML_Node& compDepNode = tranTypeNode.child("compositionDependence");
449  switch (m_tranPropMap[nodeName]) {
450  break;
451  case TP_VISCOSITY:
452  trParam.viscosity = newLTI(compDepNode, m_tranPropMap[nodeName], trParam);
453  break;
454  case TP_IONCONDUCTIVITY:
455  trParam.ionConductivity = newLTI(compDepNode,
456  m_tranPropMap[nodeName],
457  trParam);
458  break;
459  case TP_MOBILITYRATIO: {
460  for (size_t iSpec = 0; iSpec< nBinInt; iSpec++) {
461  XML_Node& propSpecNode = compDepNode.child(iSpec);
462  string specName = propSpecNode.name();
463  size_t loc = specName.find(":");
464  string firstSpec = specName.substr(0,loc);
465  string secondSpec = specName.substr(loc+1);
466  size_t index = trParam.thermo->speciesIndex(firstSpec)+nsp*trParam.thermo->speciesIndex(secondSpec);
467  trParam.mobilityRatio[index] = newLTI(propSpecNode,
468  m_tranPropMap[nodeName],
469  trParam);
470  };
471  };
472  break;
473  case TP_SELFDIFFUSION: {
474  for (size_t iSpec = 0; iSpec< nsp; iSpec++) {
475  XML_Node& propSpecNode = compDepNode.child(iSpec);
476  string specName = propSpecNode.name();
477  size_t index = trParam.thermo->speciesIndex(specName);
478  trParam.selfDiffusion[index] = newLTI(propSpecNode,
479  m_tranPropMap[nodeName],
480  trParam);
481  };
482  };
483  break;
484  case TP_THERMALCOND:
485  trParam.thermalCond = newLTI(compDepNode,
486  m_tranPropMap[nodeName],
487  trParam);
488  break;
489  case TP_DIFFUSIVITY:
490  trParam.speciesDiffusivity = newLTI(compDepNode,
491  m_tranPropMap[nodeName],
492  trParam);
493  break;
494  case TP_HYDRORADIUS:
495  trParam.hydroRadius = newLTI(compDepNode,
496  m_tranPropMap[nodeName],
497  trParam);
498  break;
499  case TP_ELECTCOND:
500  trParam.electCond = newLTI(compDepNode,
501  m_tranPropMap[nodeName],
502  trParam);
503  break;
504  default:
505  throw CanteraError("getLiquidInteractionsTransportData","unknown transport property: " + nodeName);
506  }
507  }
508  /* Allow a switch between mass-averaged, mole-averaged
509  * and solvent specified reference velocities.
510  * XML code within the transportProperty node
511  * (i.e. within <viscosity>) should read as follows
512  * <velocityBasis basis="mass"> <!-- mass averaged -->
513  * <velocityBasis basis="mole"> <!-- mole averaged -->
514  * <velocityBasis basis="H2O"> <!-- H2O solvent -->
515  */
516  if (tranTypeNode.hasChild("velocityBasis")) {
517  std::string velocityBasis =
518  tranTypeNode.child("velocityBasis").attrib("basis");
519  if (velocityBasis == "mass") {
520  trParam.velocityBasis_ = VB_MASSAVG;
521  } else if (velocityBasis == "mole") {
522  trParam.velocityBasis_ = VB_MOLEAVG;
523  } else if (trParam.thermo->speciesIndex(velocityBasis) > 0) {
524  trParam.velocityBasis_ = static_cast<int>(trParam.thermo->speciesIndex(velocityBasis));
525  } else {
526  int linenum = __LINE__;
527  throw TransportDBError(linenum, "Unknown attribute \"" + velocityBasis + "\" for <velocityBasis> node. ");
528  }
529  }
530  }
531  }
532  } catch (CanteraError& err) {
533  std::cout << err.what() << std::endl;
534  }
535  return;
536 }
