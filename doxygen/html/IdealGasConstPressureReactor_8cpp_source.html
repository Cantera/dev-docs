63 void IdealGasConstPressureReactor::initialize(doublereal t0)
64 {

67  m_wdot.resize(m_nsp, 0.0);
68  m_hk.resize(m_nsp, 0.0);

76  m_pressure = m_thermo->pressure();

78 
79  size_t nt = 0, maxnt = 0;
80  for (size_t m = 0; m < m_nwalls; m++) {
81  if (m_wall[m]->kinetics(m_lr[m])) {
82  nt = m_wall[m]->kinetics(m_lr[m])->nTotalSpecies();
83  if (nt > maxnt) {
84  maxnt = nt;
85  }
86  if (m_wall[m]->kinetics(m_lr[m])) {
87  if (&m_kin->thermo(0) !=
88  &m_wall[m]->kinetics(m_lr[m])->thermo(0)) {
89  throw CanteraError("IdealGasConstPressureReactor::initialize",
90  "First phase of all kinetics managers must be"
91  " the gas.");
92  }
93  }
94  }
95  }
96  m_work.resize(maxnt);
97  std::sort(m_pnum.begin(), m_pnum.end());

101 void IdealGasConstPressureReactor::updateState(doublereal* y)
102 {
103  // The components of y are [0] the total mass, [1] the temperature,
104  // [2...K+2) are the mass fractions of each species, and [K+2...] are the
105  // coverages of surface species on each wall.
106  m_mass = y[0];
107  m_thermo->setMassFractions_NoNorm(y+2);
108  m_thermo->setState_TP(y[1], m_pressure);
109  m_vol = m_mass / m_thermo->density();
110 
111  size_t loc = m_nsp + 2;
112  SurfPhase* surf;
113  for (size_t m = 0; m < m_nwalls; m++) {
114  surf = m_wall[m]->surface(m_lr[m]);
115  if (surf) {
116  m_wall[m]->setCoverages(m_lr[m], y+loc);
117  loc += surf->nSpecies();
118  }
119  }
120 
121  // save parameters needed by other connected reactors
122  m_enthalpy = m_thermo->enthalpy_mass();
123  m_intEnergy = m_thermo->intEnergy_mass();
124  m_thermo->saveState(m_state);

127 void IdealGasConstPressureReactor::evalEqs(doublereal time, doublereal* y,
128  doublereal* ydot, doublereal* params)
129 {
130  size_t nk;
131  m_thermo->restoreState(m_state);
132  applySensitivity(params);
133 
134  Kinetics* kin;
135  m_Q = 0.0;
136 
137  // compute wall terms
138  doublereal rs0, sum, wallarea;
139  double mcpdTdt = 0.0; // m * c_p * dT/dt
140  double dmdt = 0.0; // dm/dt (gas phase)
141  double* dYdt = ydot + 2;
142  m_thermo->getPartialMolarEnthalpies(&m_hk[0]);
143 
144  SurfPhase* surf;
145  size_t lr, ns, loc = m_nsp+2, surfloc;
146  fill(m_sdot.begin(), m_sdot.end(), 0.0);
147  for (size_t i = 0; i < m_nwalls; i++) {
148  lr = 1 - 2*m_lr[i];
149  m_Q += lr*m_wall[i]->Q(time);
150  kin = m_wall[i]->kinetics(m_lr[i]);
151  surf = m_wall[i]->surface(m_lr[i]);
152  if (surf && kin) {
153  rs0 = 1.0/surf->siteDensity();
154  nk = surf->nSpecies();
155  sum = 0.0;
156  surf->setTemperature(m_state[0]);
157  m_wall[i]->syncCoverages(m_lr[i]);
158  kin->getNetProductionRates(DATA_PTR(m_work));
159  ns = kin->surfacePhaseIndex();
160  surfloc = kin->kineticsSpeciesIndex(0,ns);
161  for (size_t k = 1; k < nk; k++) {
162  ydot[loc + k] = m_work[surfloc+k]*rs0*surf->size(k);
163  sum -= ydot[loc + k];
164  }
165  ydot[loc] = sum;
166  loc += nk;
167 
168  wallarea = m_wall[i]->area();
169  for (size_t k = 0; k < m_nsp; k++) {
170  m_sdot[k] += m_work[k]*wallarea;
171  }
172  }
173  }
174 
175  const vector_fp& mw = m_thermo->molecularWeights();
176  const doublereal* Y = m_thermo->massFractions();
177 
178  if (m_chem) {
179  m_kin->getNetProductionRates(&m_wdot[0]); // "omega dot"
180  }
181 
182  double mdot_surf = 0.0; // net mass flux from surface
183  for (size_t k = 0; k < m_nsp; k++) {
184  // production in gas phase and from surfaces
185  dYdt[k] = (m_wdot[k] * m_vol + m_sdot[k]) * mw[k] / m_mass;
186  mdot_surf += m_sdot[k] * mw[k];
187  }
188  dmdt += mdot_surf;
189 
190  // external heat transfer
191  mcpdTdt -= m_Q;
192 
193  for (size_t n = 0; n < m_nsp; n++) {
194  // heat release from gas phase and surface reations
195  mcpdTdt -= m_wdot[n] * m_hk[n] * m_vol;
196  mcpdTdt -= m_sdot[n] * m_hk[n];
197  // dilution by net surface mass flux
198  dYdt[n] -= Y[n] * mdot_surf / m_mass;
199  }
200 
201  // add terms for open system
202  if (m_open) {
203  // outlets
204  for (size_t i = 0; i < m_nOutlets; i++) {
205  dmdt -= m_outlet[i]->massFlowRate(time); // mass flow out of system
206  }
207 
208  // inlets
209  for (size_t i = 0; i < m_nInlets; i++) {
210  double mdot_in = m_inlet[i]->massFlowRate(time);
211  dmdt += mdot_in; // mass flow into system
212  mcpdTdt += m_inlet[i]->enthalpy_mass() * mdot_in;
213  for (size_t n = 0; n < m_nsp; n++) {
214  double mdot_spec = m_inlet[i]->outletSpeciesMassFlowRate(n);
215  // flow of species into system and dilution by other species
216  dYdt[n] += (mdot_spec - mdot_in * Y[n]) / m_mass;
217  mcpdTdt -= m_hk[n] / mw[n] * mdot_spec;
218  }
219  }
220  }
221 
222  ydot[0] = dmdt;
223  if (m_energy) {
224  ydot[1] = mcpdTdt / (m_mass * m_thermo->cp_mass());
225  } else {
226  ydot[1] = 0.0;
227  }
228 
229  resetSensitivity(params);
230 }
