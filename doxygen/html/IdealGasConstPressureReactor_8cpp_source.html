61 void IdealGasConstPressureReactor::updateState(doublereal* y)
62 {
63  // The components of y are [0] the total mass, [1] the temperature,
64  // [2...K+2) are the mass fractions of each species, and [K+2...] are the
65  // coverages of surface species on each wall.
66  m_mass = y[0];
67  m_thermo->setMassFractions_NoNorm(y+2);
68  m_thermo->setState_TP(y[1], m_pressure);
69  m_vol = m_mass / m_thermo->density();
70  updateSurfaceState(y + m_nsp + 2);
71 
72  // save parameters needed by other connected reactors
73  m_enthalpy = m_thermo->enthalpy_mass();
74  m_intEnergy = m_thermo->intEnergy_mass();

78 void IdealGasConstPressureReactor::evalEqs(doublereal time, doublereal* y,
79  doublereal* ydot, doublereal* params)
80 {
81  double dmdt = 0.0; // dm/dt (gas phase)
82  double mcpdTdt = 0.0; // m * c_p * dT/dt
83  double* dYdt = ydot + 2;
84 
85  m_thermo->restoreState(m_state);
86  applySensitivity(params);
87  evalWalls(time);
88  double mdot_surf = evalSurfaces(time, ydot + m_nsp + 2);
89  dmdt += mdot_surf;
90 
91  m_thermo->getPartialMolarEnthalpies(&m_hk[0]);
92  const vector_fp& mw = m_thermo->molecularWeights();
93  const doublereal* Y = m_thermo->massFractions();
94 
95  if (m_chem) {
96  m_kin->getNetProductionRates(&m_wdot[0]); // "omega dot"
97  }
98 
99  // external heat transfer
100  mcpdTdt -= m_Q;
101 
102  for (size_t n = 0; n < m_nsp; n++) {
103  // heat release from gas phase and surface reations
104  mcpdTdt -= m_wdot[n] * m_hk[n] * m_vol;
105  mcpdTdt -= m_sdot[n] * m_hk[n];
106  // production in gas phase and from surfaces
107  dYdt[n] = (m_wdot[n] * m_vol + m_sdot[n]) * mw[n] / m_mass;
108  // dilution by net surface mass flux
109  dYdt[n] -= Y[n] * mdot_surf / m_mass;
110  }
111 
112  // add terms for outlets
113  for (size_t i = 0; i < m_outlet.size(); i++) {
114  dmdt -= m_outlet[i]->massFlowRate(time); // mass flow out of system
115  }
116 
117  // add terms for inlets
118  for (size_t i = 0; i < m_inlet.size(); i++) {
119  double mdot_in = m_inlet[i]->massFlowRate(time);
120  dmdt += mdot_in; // mass flow into system
121  mcpdTdt += m_inlet[i]->enthalpy_mass() * mdot_in;
122  for (size_t n = 0; n < m_nsp; n++) {
123  double mdot_spec = m_inlet[i]->outletSpeciesMassFlowRate(n);
124  // flow of species into system and dilution by other species
125  dYdt[n] += (mdot_spec - mdot_in * Y[n]) / m_mass;
126  mcpdTdt -= m_hk[n] / mw[n] * mdot_spec;
127  }
128  }
129 
130  ydot[0] = dmdt;
131  if (m_energy) {
132  ydot[1] = mcpdTdt / (m_mass * m_thermo->cp_mass());
133  } else {
134  ydot[1] = 0.0;
135  }
136 
137  resetSensitivity(params);
138 }
