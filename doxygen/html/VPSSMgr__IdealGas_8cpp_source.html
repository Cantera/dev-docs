69 void VPSSMgr_IdealGas::_updateStandardStateThermo()
70 {
71 
72  doublereal pp = log(m_plast / m_p0);
73  doublereal v = temperature() *GasConstant /m_plast;
74 
75  for (size_t k = 0; k < m_kk; k++) {
76  m_hss_RT[k] = m_h0_RT[k];
77  m_cpss_R[k] = m_cp0_R[k];
78  m_sss_R[k] = m_s0_R[k] - pp;
79  m_gss_RT[k] = m_hss_RT[k] - m_sss_R[k];
80  m_Vss[k] = v;
81  }
82 }

91 VPSSMgr_IdealGas::createInstallPDSS(size_t k, const XML_Node& speciesNode,
92  const XML_Node* const phaseNode_ptr)

95  if (ss) {

98  "standardState model for species isn't "

100  }
101  }
102  if (m_Vss.size() < k+1) {
103  m_Vss.resize(k+1, 0.0);
104  }
105 
106  shared_ptr<SpeciesThermoInterpType> stit(newSpeciesThermoInterpType(speciesNode));
107  stit->validate(speciesNode["name"]);
108  m_spthermo->install_STIT(k, stit);
109 
110  PDSS* kPDSS = new PDSS_IdealGas(m_vptp_ptr, k, speciesNode,
111  *phaseNode_ptr, true);
112 
113  m_p0 = m_spthermo->refPressure(k);
