170  m_p0 = refPressure_;
171  if (type == m_thermo1.ID) {
172  m_thermo1.install(name, sp, 0, c, minTemp_, maxTemp_,
173  refPressure_);
174  speciesToType[sp] = m_thermo1.ID;
175  } else if (type == m_thermo2.ID) {
176  m_thermo2.install(name, sp, 0, c, minTemp_, maxTemp_,
177  refPressure_);
178  speciesToType[sp] = m_thermo2.ID;
179  } else {
180  throw UnknownSpeciesThermo("SpeciesThermoDuo:install",type);
181  }
182 }
