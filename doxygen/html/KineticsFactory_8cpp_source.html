30  * Look for a child of the xml element phase called

34  string kintype = phaseData.child("kinetics")["model"];
35  /*
36  * look up the string kintype in the list of known
37  * kinetics managers (list is kept at the top of this file).
38  * Translate it to an integer value, ikin.

40  int ikin=-1;
41  int n;
42  for (n = 0; n < ntypes; n++) {

44  ikin = _itypes[n];
45  }
46  }
47  /*
48  * Assign the kinetics manager based on the value of ikin.
49  * Kinetics managers are classes derived from the base
50  * Kinetics class. Unknown kinetics managers will throw a
51  * CanteraError here.

53  Kinetics* k=0;
54  switch (ikin) {

66  break;
67
