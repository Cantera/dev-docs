179 void MixedSolventElectrolyte::getPartialMolarEnthalpies(doublereal* hbar) const
180 {
181  /*
182  * Get the nondimensional standard state enthalpies
183  */
184  getEnthalpy_RT(hbar);
185  /*
186  * dimensionalize it.
187  */
188  double T = temperature();
189  double RT = GasConstant * T;
190  for (size_t k = 0; k < m_kk; k++) {
191  hbar[k] *= RT;
192  }
193  /*
194  * Update the activity coefficients, This also update the
195  * internally stored molalities.
196  */
197  s_update_lnActCoeff();
198  s_update_dlnActCoeff_dT();
199  double RTT = RT * T;
200  for (size_t k = 0; k < m_kk; k++) {
201  hbar[k] -= RTT * dlnActCoeffdT_Scaled_[k];
202  }
203 }

355 void MixedSolventElectrolyte::s_update_lnActCoeff() const
356 {
357  double T = temperature();
358  double RT = GasConstant*T;
359  lnActCoeff_Scaled_.assign(m_kk, 0.0);
360  for (size_t iK = 0; iK < m_kk; iK++) {
361  for (size_t i = 0; i < numBinaryInteractions_; i++) {
362  size_t iA = m_pSpecies_A_ij[i];
363  size_t iB = m_pSpecies_B_ij[i];
364  int delAK = 0;
365  int delBK = 0;
366  if (iA==iK) {
367  delAK = 1;
368  } else if (iB==iK) {
369  delBK = 1;
370  }
371  double XA = moleFractions_[iA];
372  double XB = moleFractions_[iB];
373  double g0 = (m_HE_b_ij[i] - T * m_SE_b_ij[i]) / RT;
374  double g1 = (m_HE_c_ij[i] - T * m_SE_c_ij[i]) / RT;
375  lnActCoeff_Scaled_[iK] += (delAK * XB + XA * delBK - XA * XB) * (g0 + g1 * XB) + XA * XB * (delBK - XB) * g1;
376  }
377  }
378 }

380 void MixedSolventElectrolyte::s_update_dlnActCoeff_dT() const
381 {

384  dlnActCoeffdT_Scaled_.assign(m_kk, 0.0);
385  d2lnActCoeffdT2_Scaled_.assign(m_kk, 0.0);
386  for (size_t iK = 0; iK < m_kk; iK++) {
387  for (size_t i = 0; i < numBinaryInteractions_; i++) {
388  size_t iA = m_pSpecies_A_ij[i];
389  size_t iB = m_pSpecies_B_ij[i];
390  int delAK = 0;
391  int delBK = 0;
392  if (iA==iK) {
393  delAK = 1;
394  } else if (iB==iK) {
395  delBK = 1;
396  }
397  double XA = moleFractions_[iA];
398  double XB = moleFractions_[iB];
399  double g0 = -m_HE_b_ij[i] / RTT;
400  double g1 = -m_HE_c_ij[i] / RTT;
401  double temp = (delAK * XB + XA * delBK - XA * XB) * (g0 + g1 * XB) + XA * XB * (delBK - XB) * g1;
402  dlnActCoeffdT_Scaled_[iK] += temp;

616 void MixedSolventElectrolyte::readXMLBinarySpecies(XML_Node& xmLBinarySpecies)
617 {
618  string xname = xmLBinarySpecies.name();
619  if (xname != "binaryNeutralSpeciesParameters") {
620  throw CanteraError("MixedSolventElectrolyte::readXMLBinarySpecies",
621  "Incorrect name for processing this routine: " + xname);
622  }
623  vector_fp vParams;
624  string iName = xmLBinarySpecies.attrib("speciesA");
625  if (iName == "") {
626  throw CanteraError("MixedSolventElectrolyte::readXMLBinarySpecies", "no speciesA attrib");
627  }
628  string jName = xmLBinarySpecies.attrib("speciesB");
629  if (jName == "") {
630  throw CanteraError("MixedSolventElectrolyte::readXMLBinarySpecies", "no speciesB attrib");
631  }
632  /*
633  * Find the index of the species in the current phase. It's not
634  * an error to not find the species
635  */
636  size_t iSpecies = speciesIndex(iName);
637  if (iSpecies == npos) {
638  return;
639  }
640  string ispName = speciesName(iSpecies);
641  if (charge(iSpecies) != 0) {
642  throw CanteraError("MixedSolventElectrolyte::readXMLBinarySpecies", "speciesA charge problem");
643  }
644  size_t jSpecies = speciesIndex(jName);
645  if (jSpecies == npos) {
646  return;
647  }
648  string jspName = speciesName(jSpecies);
649  if (charge(jSpecies) != 0) {
650  throw CanteraError("MixedSolventElectrolyte::readXMLBinarySpecies", "speciesB charge problem");
651  }
652 
653  resizeNumInteractions(numBinaryInteractions_ + 1);
654  size_t iSpot = numBinaryInteractions_ - 1;
655  m_pSpecies_A_ij[iSpot] = iSpecies;
656  m_pSpecies_B_ij[iSpot] = jSpecies;
657 
658  for (size_t iChild = 0; iChild < xmLBinarySpecies.nChildren(); iChild++) {
659  XML_Node& xmlChild = xmLBinarySpecies.child(iChild);
660  string nodeName = lowercase(xmlChild.name());
661  /*
662  * Process the binary species interaction child elements
663  */
664  if (nodeName == "excessenthalpy") {
665  /*
666  * Get the string containing all of the values
667  */
668  getFloatArray(xmlChild, vParams, true, "toSI", "excessEnthalpy");
669  if (vParams.size() != 2) {
670  throw CanteraError("MixedSolventElectrolyte::readXMLBinarySpecies::excessEnthalpy for " + ispName
671  + "::" + jspName,
672  "wrong number of params found");
673  }
674  m_HE_b_ij[iSpot] = vParams[0];
675  m_HE_c_ij[iSpot] = vParams[1];
676  }
677 
678  if (nodeName == "excessentropy") {
679  /*
680  * Get the string containing all of the values
681  */

687  }

690  }
