142 int SquareMatrix::factorQR()
143 {
144  if (tau.size() < m_nrows) {
145  tau.resize(m_nrows, 0.0);
146  work.resize(8 * m_nrows, 0.0);
147  }
148  a1norm_ = ct_dlange('1', m_nrows, m_nrows, &*begin(), m_nrows, work.data());
149  int info = 0;

151  size_t lwork = work.size();

153  if (info != 0) {

162  if (lworkOpt > lwork) {

168 int SquareMatrix::solveQR(doublereal* b)
169 {
170  int info=0;
171  /*
172  * Check to see whether the matrix has been factored.
173  */
174  if (!m_factored) {
175  int retn = factorQR();
176  if (retn) {
177  return retn;
178  }
179  }
180 
181  size_t lwork = work.size();
182  if (lwork < m_nrows) {
183  work.resize(8 * m_nrows, 0.0);
184  lwork = 8 * m_nrows;
185  }
186 
187  /*
188  * Solve the factored system
189  */
190  ct_dormqr(ctlapack::Left, ctlapack::Transpose, m_nrows, 1, m_nrows, &*begin(), m_nrows, tau.data(), b, m_nrows,
191  work.data(), lwork, info);
192  if (info != 0) {
193  if (m_printLevel) {
194  writelogf("SquareMatrix::solveQR(): DORMQR returned INFO = %d\n", info);
195  }
196  if (! m_useReturnErrorCode) {
197  throw CELapackError("SquareMatrix::solveQR()", "DORMQR returned INFO = " + int2str(info));
198  }
199  }
200  size_t lworkOpt = static_cast<size_t>(work[0]);
201  if (lworkOpt > lwork) {
202  work.resize(lworkOpt);
203  }
204 
205  char dd = 'N';
206  ct_dtrtrs(ctlapack::UpperTriangular, ctlapack::NoTranspose, &dd, m_nrows, 1, &*begin(), m_nrows, b,
207  m_nrows, info);
208  if (info != 0) {
209  if (m_printLevel) {
210  writelogf("SquareMatrix::solveQR(): DTRTRS returned INFO = %d\n", info);
211  }
212  if (! m_useReturnErrorCode) {

214  }
