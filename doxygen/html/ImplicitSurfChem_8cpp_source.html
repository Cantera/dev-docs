32 {
33  m_nsurf = k.size();
34  size_t ns, nsp;
35  size_t nt, ntmax = 0;
36  size_t kinSpIndex = 0;
37  // Loop over the number of surface kinetics objects
38  for (size_t n = 0; n < m_nsurf; n++) {
39  InterfaceKinetics* kinPtr = k[n];
40  m_vecKinPtrs.push_back(kinPtr);
41  ns = k[n]->surfacePhaseIndex();
42  if (ns == npos) {
43  throw CanteraError("ImplicitSurfChem",
44  "kinetics manager contains no surface phase");
45  }
46  m_surfindex.push_back(ns);
47  m_surf.push_back((SurfPhase*)&k[n]->thermo(ns));
48  nsp = m_surf.back()->nSpecies();
49  m_nsp.push_back(nsp);
50  m_nv += m_nsp.back();

52  ntmax = std::max(nt, ntmax);
53  m_specStartIndex.push_back(kinSpIndex);
54  kinSpIndex += nsp;
55  size_t nPhases = kinPtr->nPhases();

57  size_t imatch = npos;

59  if (ip != ns) {
60  ThermoPhase* thPtr = & kinPtr->thermo(ip);
61  if ((imatch = checkMatch(m_bulkPhases, thPtr)) == npos) {
62  m_bulkPhases.push_back(thPtr);
63  m_numBulkPhases++;
64  nsp = thPtr->nSpecies();
65  m_nspBulkPhases.push_back(nsp);
66  m_numTotalBulkSpecies += nsp;
67  imatch = m_bulkPhases.size() - 1;
68  }
69  pLocTmp[ip] = int(imatch);
70  } else {
71  pLocTmp[ip] = -int(n);
72  }
73  }
74  pLocVec.push_back(pLocTmp);
75  }
76  m_numTotalSpecies = m_nv + m_numTotalBulkSpecies;
77  m_concSpecies.resize(m_numTotalSpecies, 0.0);
78  m_concSpeciesSave.resize(m_numTotalSpecies, 0.0);
79 
80  m_integ.reset(newIntegrator("CVODE"));
81 
82  // use backward differencing, with a full Jacobian computed
83  // numerically, and use a Newton linear iterator
84  m_integ->setMethod(BDF_Method);
85  m_integ->setProblemType(DENSE + NOJAC);
86  m_integ->setIterator(Newton_Iter);
87  m_work.resize(ntmax);
88 }

141 void ImplicitSurfChem::eval(doublereal time, doublereal* y,
142  doublereal* ydot, doublereal* p)
143 {
144  updateState(y); // synchronize the surface state(s) with y
145  doublereal rs0, sum;
146  size_t loc, kstart;
147  for (size_t n = 0; n < m_nsurf; n++) {
148  rs0 = 1.0/m_surf[n]->siteDensity();
149  m_vecKinPtrs[n]->getNetProductionRates(m_work.data());
150  kstart = m_vecKinPtrs[n]->kineticsSpeciesIndex(0,m_surfindex[n]);
151  sum = 0.0;
152  loc = 0;
153  for (size_t k = 1; k < m_nsp[n]; k++) {
154  ydot[k + loc] = m_work[kstart + k] * rs0 * m_surf[n]->size(k);
155  sum -= ydot[k];
156  }
157  ydot[loc] = sum;
158  loc += m_nsp[n];
159  }

162 void ImplicitSurfChem::solvePseudoSteadyStateProblem(int ifuncOverride,
163  doublereal timeScaleOverride)
164 {
165  int ifunc;
166  /*
167  * set bulkFunc
168  * -> We assume that the bulk concentrations are constant.
169  */
170  int bulkFunc = BULK_ETCH;
171  /*
172  * time scale - time over which to integrate equations
173  */
174  doublereal time_scale = timeScaleOverride;
175  if (!m_surfSolver) {
176  m_surfSolver.reset(new solveSP(this, bulkFunc));
177  /*
178  * set ifunc, which sets the algorithm.
179  */
180  ifunc = SFLUX_INITIALIZE;
181  } else {
182  ifunc = SFLUX_RESIDUAL;
183  }
184 
185  // Possibly override the ifunc value
186  if (ifuncOverride >= 0) {
187  ifunc = ifuncOverride;
188  }
189 
190  /*
191  * Get the specifications for the problem from the values
192  * in the ThermoPhase objects for all phases.
193  *
194  * 1) concentrations of all species in all phases, m_concSpecies[]
195  * 2) Temperature and pressure
196  */
197  getConcSpecies(m_concSpecies.data());
198  InterfaceKinetics* ik = m_vecKinPtrs[0];
199  ThermoPhase& tp = ik->thermo(0);
200  doublereal TKelvin = tp.temperature();
201  doublereal PGas = tp.pressure();
202  /*
203  * Make sure that there is a common temperature and
204  * pressure for all ThermoPhase objects belonging to the
205  * interfacial kinetics object, if it is required by
206  * the problem statement.
207  */
208  if (m_commonTempPressForPhases) {
209  setCommonState_TP(TKelvin, PGas);
210  }
211 
212  doublereal reltol = 1.0E-6;
213  doublereal atol = 1.0E-20;
214 
215  /*
216  * Install a filter for negative concentrations. One of the
217  * few ways solveSS can fail is if concentrations on input
218  * are below zero.
219  */
220  bool rset = false;
221  for (size_t k = 0; k < m_nv; k++) {
222  if (m_concSpecies[k] < 0.0) {
223  rset = true;
224  m_concSpecies[k] = 0.0;
225  }
226  }
227  if (rset) {
228  setConcSpecies(m_concSpecies.data());
229  }
230 
231  m_surfSolver->m_ioflag = m_ioFlag;
232 
233  // Save the current solution
234  copy(m_concSpecies.begin(), m_concSpecies.end(), m_concSpeciesSave.begin());
235 
236  int retn = m_surfSolver->solveSurfProb(ifunc, time_scale, TKelvin, PGas,
237  reltol, atol);
238  if (retn != 1) {
239  // reset the concentrations
240  copy(m_concSpeciesSave.begin(), m_concSpeciesSave.end(), m_concSpecies.begin());
241  setConcSpecies(m_concSpeciesSave.data());
242  ifunc = SFLUX_INITIALIZE;
243  retn = m_surfSolver->solveSurfProb(ifunc, time_scale, TKelvin, PGas,
244  reltol, atol);
245  if (retn != 1) {
246  throw CanteraError("ImplicitSurfChem::solvePseudoSteadyStateProblem",
247  "solveSP return an error condition!");
248  }
249  }
