123 double VCS_SPECIES_THERMO::GStar_R_calc(size_t kglob, double TKelvin,
124  double pres)
125 {
126  double fe = G0_R_calc(kglob, TKelvin);
127  double T = TKelvin;
128  if (UseCanteraCalls) {
129  if (m_VCS_UnitsFormat != VCS_UNITS_MKS) {
130  throw Cantera::CanteraError("VCS_SPECIES_THERMO::GStar_R_calc",
131  "Possible inconsistency");
132  }
133  size_t kspec = IndexSpeciesPhase;
134  OwningPhase->setState_TP(TKelvin, pres);
135  fe = OwningPhase->GStar_calc_one(kspec);
136  double R = vcsUtil_gasConstant(m_VCS_UnitsFormat);
137  fe /= R;
138  } else {
139  double pref = SS0_Pref;
140  switch (SSStar_Model) {
141  case VCS_SSSTAR_CONSTANT:
142  break;
143  case VCS_SSSTAR_IDEAL_GAS:
144  fe += T * log(pres/ pref);
145  break;
146  default:
147  throw CanteraError("VCS_SPECIES_THERMO::GStar_R_calc",
148  "unknown SSStar model");
149  }
150  }
151  return fe;
152 }

184 double VCS_SPECIES_THERMO::G0_R_calc(size_t kglob, double TKelvin)
185 {
186  double fe, H, S;
187  if (SS0_Model == VCS_SS0_CONSTANT) {
188  return SS0_feSave;
189  }
190  if (TKelvin == SS0_TSave) {
191  return SS0_feSave;
192  }
193  if (UseCanteraCalls) {
194  if (m_VCS_UnitsFormat != VCS_UNITS_MKS) {
195  throw Cantera::CanteraError("VCS_SPECIES_THERMO::G0_R_calc",
196  "Possible inconsistency");
197  }
198  size_t kspec = IndexSpeciesPhase;
199  OwningPhase->setState_T(TKelvin);
200  fe = OwningPhase->G0_calc_one(kspec);
201  double R = vcsUtil_gasConstant(m_VCS_UnitsFormat);
202  fe /= R;
203  } else {
204  switch (SS0_Model) {
205  case VCS_SS0_CONSTANT:
206  fe = SS0_feSave;
207  break;
208  case VCS_SS0_CONSTANT_CP:
209  H = SS0_H0 + (TKelvin - SS0_T0) * SS0_Cp0;
210  S = SS0_Cp0 + SS0_Cp0 * log((TKelvin / SS0_T0));
211  fe = H - TKelvin * S;
212  break;
213  default:
214  throw CanteraError("VCS_SPECIES_THERMO::G0_R_calc",
215  "unknown model");
216  }
217  }
218  SS0_feSave = fe;
219  SS0_TSave = TKelvin;
220  return fe;
221 }

223 double VCS_SPECIES_THERMO::eval_ac(size_t kglob)
224 {
225  double ac;
226  /*
227  * Activity coefficients are frequently evaluated on a per phase
228  * basis. If they are, then the currPhAC[] boolean may be used
229  * to reduce repeated work. Just set currPhAC[iph], when the
230  * activity coefficients for all species in the phase are reevaluated.
231  */
232  if (UseCanteraCalls) {
233  size_t kspec = IndexSpeciesPhase;
234  ac = OwningPhase->AC_calc_one(kspec);
235  } else {
236  switch (Activity_Coeff_Model) {
237  case VCS_AC_CONSTANT:
238  ac = 1.0;
239  break;
240  default:
241  throw CanteraError("VCS_SPECIES_THERMO::eval_ac" ,"unknown model");
242  }
243  }
244  return ac;
245 }
