119  const XML_Node& speciesNode,
120  const XML_Node& phaseNode, bool spInstalled)
121 {
122  PDSS::initThermo();
123  SpeciesThermo& sp = m_tp->speciesThermo();
124  m_p0 = sp.refPressure(m_spindex);
125 
126  if (!spInstalled) {
127  throw CanteraError("PDSS_SSVol::constructPDSSXML", "spInstalled false not handled");
128  }
129 
130  const XML_Node* ss = speciesNode.findByName("standardState");
131  if (!ss) {
132  throw CanteraError("PDSS_SSVol::constructPDSSXML",
133  "no standardState Node for species " + speciesNode.name());
134  }
135  std::string model = (*ss)["model"];
136  if (model == "constant_incompressible" || model == "constant") {
137  volumeModel_ = cSSVOLUME_CONSTANT;
138  m_constMolarVolume = ctml::getFloat(*ss, "molarVolume", "toSI");
139  } else if (model == "temperature_polynomial") {
140  volumeModel_ = cSSVOLUME_TPOLY;
141  size_t num = ctml::getFloatArray(*ss, TCoeff_, true, "toSI", "volumeTemperaturePolynomial");
142  if (num != 4) {
143  throw CanteraError("PDSS_SSVol::constructPDSSXML",
144  " Didn't get 4 density polynomial numbers for species " + speciesNode.name());
145  }
146  } else if (model == "density_temperature_polynomial") {
147  volumeModel_ = cSSVOLUME_DENSITY_TPOLY;
148  size_t num = ctml::getFloatArray(*ss, TCoeff_, true, "toSI", "densityTemperaturePolynomial");
149  if (num != 4) {
150  throw CanteraError("PDSS_SSVol::constructPDSSXML",
151  " Didn't get 4 density polynomial numbers for species " + speciesNode.name());
152  }
153  } else {
154  throw CanteraError("PDSS_SSVol::constructPDSSXML",
155  "standardState model for species isn't constant_incompressible: " + speciesNode.name());
156  }
157  std::string id = "";

358 void PDSS_SSVol::calcMolarVolume() const
359 {
360  if (volumeModel_ == cSSVOLUME_CONSTANT) {
361  m_Vss_ptr[m_spindex] = m_constMolarVolume;
362  } else if (volumeModel_ == cSSVOLUME_TPOLY) {
363  m_Vss_ptr[m_spindex] = TCoeff_[0] + m_temp * (TCoeff_[1] + m_temp * (TCoeff_[2] + m_temp * TCoeff_[3]));
364  dVdT_ = TCoeff_[1] + 2.0 * m_temp * TCoeff_[2] + 3.0 * m_temp * m_temp * TCoeff_[3];
365  d2VdT2_ = 2.0 * TCoeff_[2] + 6.0 * m_temp * TCoeff_[3];
366  } else if (volumeModel_ == cSSVOLUME_DENSITY_TPOLY) {
367  doublereal dens = TCoeff_[0] + m_temp * (TCoeff_[1] + m_temp * (TCoeff_[2] + m_temp * TCoeff_[3]));
368  m_Vss_ptr[m_spindex] = m_mw / dens;
369  doublereal dens2 = dens * dens;
370  doublereal ddensdT = TCoeff_[1] + 2.0 * m_temp * TCoeff_[2] + 3.0 * m_temp * m_temp * TCoeff_[3];
371  doublereal d2densdT2 = 2.0 * TCoeff_[2] + 6.0 * m_temp * TCoeff_[3];
372  dVdT_ = - m_mw / (dens2) * (ddensdT);
373  d2VdT2_ = 2.0 * m_mw / (dens2 * dens) * ddensdT * ddensdT - m_mw / dens2 * d2densdT2;
374  } else {
375  throw CanteraError("PDSS_SSVol::calcMolarVolume", "unimplemented");
376  }
377 }

403 void PDSS_SSVol::setPressure(doublereal p)
404 {
405  m_pres = p;
406  doublereal deltaP = m_pres - m_p0;
407  if (fabs(deltaP) < 1.0E-10) {
408  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex];
409  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex];
410  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
411  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex];
412  } else {
413  doublereal del_pRT = deltaP / (GasConstant * m_temp);
414  doublereal sV_term = - deltaP / (GasConstant) * dVdT_;
415  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] + sV_term + del_pRT * (m_Vss_ptr[m_spindex]);
416  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex] + sV_term;
417  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
418  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex] - m_temp * deltaP * d2VdT2_;
419  }
420 }

422 void PDSS_SSVol::setTemperature(doublereal temp)
423 {
424  m_temp = temp;
425  m_spthermo->update_one(m_spindex, temp, m_cp0_R_ptr, m_h0_RT_ptr, m_s0_R_ptr);
426  calcMolarVolume();
427  m_g0_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] - m_s0_R_ptr[m_spindex];
428  doublereal deltaP = m_pres - m_p0;
429  if (fabs(deltaP) < 1.0E-10) {
430  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex];
431  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex];
432  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
433  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex];
434  } else {
435  doublereal del_pRT = deltaP / (GasConstant * m_temp);
436  doublereal sV_term = - deltaP / (GasConstant) * dVdT_;
437  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] + sV_term + del_pRT * (m_Vss_ptr[m_spindex]);
438  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex] + sV_term;
439  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
440  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex] - m_temp * deltaP * d2VdT2_;
441  }
