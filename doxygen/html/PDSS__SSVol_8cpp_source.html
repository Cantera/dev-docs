84  const XML_Node& speciesNode,
85  const XML_Node& phaseNode, bool spInstalled)
86 {
87  PDSS::initThermo();
88  m_p0 = m_tp->speciesThermo().refPressure(m_spindex);
89 
90  if (!spInstalled) {
91  throw CanteraError("PDSS_SSVol::constructPDSSXML", "spInstalled false not handled");
92  }
93 
94  const XML_Node* ss = speciesNode.findByName("standardState");
95  if (!ss) {
96  throw CanteraError("PDSS_SSVol::constructPDSSXML",
97  "no standardState Node for species " + speciesNode.name());
98  }
99  std::string model = ss->attrib("model");
100  if (model == "constant_incompressible" || model == "constant") {
101  volumeModel_ = cSSVOLUME_CONSTANT;
102  m_constMolarVolume = getFloat(*ss, "molarVolume", "toSI");
103  } else if (model == "temperature_polynomial") {
104  volumeModel_ = cSSVOLUME_TPOLY;
105  size_t num = getFloatArray(*ss, TCoeff_, true, "toSI", "volumeTemperaturePolynomial");
106  if (num != 4) {
107  throw CanteraError("PDSS_SSVol::constructPDSSXML",
108  " Didn't get 4 density polynomial numbers for species " + speciesNode.name());
109  }
110  } else if (model == "density_temperature_polynomial") {
111  volumeModel_ = cSSVOLUME_DENSITY_TPOLY;
112  size_t num = getFloatArray(*ss, TCoeff_, true, "toSI", "densityTemperaturePolynomial");
113  if (num != 4) {
114  throw CanteraError("PDSS_SSVol::constructPDSSXML",
115  " Didn't get 4 density polynomial numbers for species " + speciesNode.name());
116  }
117  } else {
118  throw CanteraError("PDSS_SSVol::constructPDSSXML",
119  "standardState model for species isn't constant_incompressible: " + speciesNode.name());
120  }

240 void PDSS_SSVol::calcMolarVolume() const
241 {
242  if (volumeModel_ == cSSVOLUME_CONSTANT) {
243  m_Vss_ptr[m_spindex] = m_constMolarVolume;
244  } else if (volumeModel_ == cSSVOLUME_TPOLY) {
245  m_Vss_ptr[m_spindex] = TCoeff_[0] + m_temp * (TCoeff_[1] + m_temp * (TCoeff_[2] + m_temp * TCoeff_[3]));
246  dVdT_ = TCoeff_[1] + 2.0 * m_temp * TCoeff_[2] + 3.0 * m_temp * m_temp * TCoeff_[3];
247  d2VdT2_ = 2.0 * TCoeff_[2] + 6.0 * m_temp * TCoeff_[3];
248  } else if (volumeModel_ == cSSVOLUME_DENSITY_TPOLY) {
249  doublereal dens = TCoeff_[0] + m_temp * (TCoeff_[1] + m_temp * (TCoeff_[2] + m_temp * TCoeff_[3]));
250  m_Vss_ptr[m_spindex] = m_mw / dens;
251  doublereal dens2 = dens * dens;
252  doublereal ddensdT = TCoeff_[1] + 2.0 * m_temp * TCoeff_[2] + 3.0 * m_temp * m_temp * TCoeff_[3];
253  doublereal d2densdT2 = 2.0 * TCoeff_[2] + 6.0 * m_temp * TCoeff_[3];
254  dVdT_ = - m_mw / dens2 * ddensdT;
255  d2VdT2_ = 2.0 * m_mw / (dens2 * dens) * ddensdT * ddensdT - m_mw / dens2 * d2densdT2;
256  } else {
257  throw CanteraError("PDSS_SSVol::calcMolarVolume", "unimplemented");
258  }
259 }

261 void PDSS_SSVol::setPressure(doublereal p)
262 {
263  m_pres = p;
264  doublereal deltaP = m_pres - m_p0;
265  if (fabs(deltaP) < 1.0E-10) {
266  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex];
267  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex];
268  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
269  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex];
270  } else {
271  doublereal del_pRT = deltaP / (GasConstant * m_temp);
272  doublereal sV_term = - deltaP / GasConstant * dVdT_;
273  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] + sV_term + del_pRT * m_Vss_ptr[m_spindex];
274  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex] + sV_term;
275  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
276  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex] - m_temp * deltaP * d2VdT2_;
277  }
278 }

280 void PDSS_SSVol::setTemperature(doublereal temp)
281 {
282  m_temp = temp;
283  m_spthermo->update_one(m_spindex, temp, m_cp0_R_ptr, m_h0_RT_ptr, m_s0_R_ptr);
284  calcMolarVolume();
285  m_g0_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] - m_s0_R_ptr[m_spindex];
286  doublereal deltaP = m_pres - m_p0;
287  if (fabs(deltaP) < 1.0E-10) {
288  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex];
289  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex];
290  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
291  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex];
292  } else {
293  doublereal del_pRT = deltaP / (GasConstant * m_temp);
294  doublereal sV_term = - deltaP / GasConstant * dVdT_;
295  m_hss_RT_ptr[m_spindex] = m_h0_RT_ptr[m_spindex] + sV_term + del_pRT * m_Vss_ptr[m_spindex];
296  m_sss_R_ptr[m_spindex] = m_s0_R_ptr[m_spindex] + sV_term;
297  m_gss_RT_ptr[m_spindex] = m_hss_RT_ptr[m_spindex] - m_sss_R_ptr[m_spindex];
298  m_cpss_R_ptr[m_spindex] = m_cp0_R_ptr[m_spindex] - m_temp * deltaP * d2VdT2_;
299  }
