309 void IdealMolalSoln::getActivities(doublereal* ac) const
310 {
311  _updateStandardStateThermo();
312  /*
313  * Update the molality array, m_molalities()
314  * This requires an update due to mole fractions
315  */
316  if (IMS_typeCutoff_ == 0) {
317  calcMolalities();
318  for (size_t k = 0; k < m_kk; k++) {
319  ac[k] = m_molalities[k];
320  }
321  double xmolSolvent = moleFraction(m_indexSolvent);
322  // Limit the activity coefficient to be finite as the solvent mole
323  // fraction goes to zero.
324  xmolSolvent = std::max(m_xmolSolventMIN, xmolSolvent);
325  ac[m_indexSolvent] =
326  exp((xmolSolvent - 1.0)/xmolSolvent);
327  } else {
328 
329  s_updateIMS_lnMolalityActCoeff();
330  /*
331  * Now calculate the array of activities.
332  */
333  for (size_t k = 1; k < m_kk; k++) {
334  ac[k] = m_molalities[k] * exp(IMS_lnActCoeffMolal_[k]);
335  }
336  double xmolSolvent = moleFraction(m_indexSolvent);
337  ac[m_indexSolvent] =
338  exp(IMS_lnActCoeffMolal_[m_indexSolvent]) * xmolSolvent;
339 
340  }

343 void IdealMolalSoln::
344 getMolalityActivityCoefficients(doublereal* acMolality) const
345 {
346  if (IMS_typeCutoff_ == 0) {
347  for (size_t k = 0; k < m_kk; k++) {
348  acMolality[k] = 1.0;
349  }
350  double xmolSolvent = moleFraction(m_indexSolvent);
351  // Limit the activity coefficient to be finite as the solvent mole
352  // fraction goes to zero.
353  xmolSolvent = std::max(m_xmolSolventMIN, xmolSolvent);
354  acMolality[m_indexSolvent] =
355  exp((xmolSolvent - 1.0)/xmolSolvent) / xmolSolvent;
356  } else {
357  s_updateIMS_lnMolalityActCoeff();
358  std::copy(IMS_lnActCoeffMolal_.begin(), IMS_lnActCoeffMolal_.end(), acMolality);
359  for (size_t k = 0; k < m_kk; k++) {

369 void IdealMolalSoln::getChemPotentials(doublereal* mu) const
370 {
371  double xx;
372 
373  // Assertion is made for speed
374  AssertThrow(m_indexSolvent == 0, "solvent not the first species");
375 
376  /*
377  * First get the standard chemical potentials
378  * -> this requires updates of standard state as a function
379  * of T and P
380  * These are defined at unit molality.
381  */
382  getStandardChemPotentials(mu);
383  /*
384  * Update the molality array, m_molalities()
385  * This requires an update due to mole fractions
386  */
387  calcMolalities();
388  /*
389  * get the solvent mole fraction
390  */
391  double xmolSolvent = moleFraction(m_indexSolvent);
392  doublereal RT = GasConstant * temperature();
393 
394  if (IMS_typeCutoff_ == 0 || xmolSolvent > 3.* IMS_X_o_cutoff_/2.0) {
395 
396  for (size_t k = 1; k < m_kk; k++) {
397  xx = std::max(m_molalities[k], SmallNumber);
398  mu[k] += RT * log(xx);
399  }
400  /*
401  * Do the solvent
402  * -> see my notes
403  */
404 
405  xx = std::max(xmolSolvent, SmallNumber);
406  mu[m_indexSolvent] +=
407  (RT * (xmolSolvent - 1.0) / xx);
408  } else {
409  /*
410  * Update the activity coefficients
411  * This also updates the internal molality array.
412  */
413  s_updateIMS_lnMolalityActCoeff();
414 
415 
416  for (size_t k = 1; k < m_kk; k++) {
417  xx = std::max(m_molalities[k], SmallNumber);
418  mu[k] += RT * (log(xx) + IMS_lnActCoeffMolal_[k]);
419  }
420  xx = std::max(xmolSolvent, SmallNumber);
421  mu[m_indexSolvent] +=
422  RT * (log(xx) + IMS_lnActCoeffMolal_[m_indexSolvent]);
423  }
424 
425 }

436 void IdealMolalSoln::getPartialMolarEntropies(doublereal* sbar) const
437 {
438  getEntropy_R(sbar);
439  doublereal R = GasConstant;
440  calcMolalities();
441  if (IMS_typeCutoff_ == 0) {
442  for (size_t k = 0; k < m_kk; k++) {
443  if (k != m_indexSolvent) {
444  doublereal mm = std::max(SmallNumber, m_molalities[k]);
445  sbar[k] -= R * log(mm);
446  }
447  }
448  double xmolSolvent = moleFraction(m_indexSolvent);
449  sbar[m_indexSolvent] -= (R * (xmolSolvent - 1.0) / xmolSolvent);
450  } else {
451  /*
452  * Update the activity coefficients, This also update the
453  * internally stored molalities.
454  */
455  s_updateIMS_lnMolalityActCoeff();
456  /*
457  * First we will add in the obvious dependence on the T
458  * term out front of the log activity term
459  */
460  doublereal mm;
461  for (size_t k = 0; k < m_kk; k++) {
462  if (k != m_indexSolvent) {
463  mm = std::max(SmallNumber, m_molalities[k]);
464  sbar[k] -= R * (log(mm) + IMS_lnActCoeffMolal_[k]);
465  }
466  }
467  double xmolSolvent = moleFraction(m_indexSolvent);
468  mm = std::max(SmallNumber, xmolSolvent);
469  sbar[m_indexSolvent] -= R *(log(mm) + IMS_lnActCoeffMolal_[m_indexSolvent]);
470 
471  }
472 }

701 void IdealMolalSoln::s_updateIMS_lnMolalityActCoeff() const
702 {
703  double tmp;
704  /*
705  * Calculate the molalities. Currently, the molalities
706  * may not be current with respect to the contents of the
707  * State objects' data.
708  */
709  calcMolalities();
710 
711  double xmolSolvent = moleFraction(m_indexSolvent);
712  double xx = std::max(m_xmolSolventMIN, xmolSolvent);
713 
714  if (IMS_typeCutoff_ == 0) {
715  for (size_t k = 1; k < m_kk; k++) {
716  IMS_lnActCoeffMolal_[k]= 0.0;
717  }
718  IMS_lnActCoeffMolal_[m_indexSolvent] = - log(xx) + (xx - 1.0)/xx;
719  return;
720  } else if (IMS_typeCutoff_ == 1) {
721  if (xmolSolvent > 3.0 * IMS_X_o_cutoff_/2.0) {
722  for (size_t k = 1; k < m_kk; k++) {
723  IMS_lnActCoeffMolal_[k]= 0.0;
724  }
725  IMS_lnActCoeffMolal_[m_indexSolvent] = - log(xx) + (xx - 1.0)/xx;
726  return;
727  } else if (xmolSolvent < IMS_X_o_cutoff_/2.0) {
728  tmp = log(xx * IMS_gamma_k_min_);
729  for (size_t k = 1; k < m_kk; k++) {
730  IMS_lnActCoeffMolal_[k]= tmp;
731  }
732  IMS_lnActCoeffMolal_[m_indexSolvent] = log(IMS_gamma_o_min_);
733  return;
734  } else {
735 
736  /*
737  * If we are in the middle region, calculate the connecting polynomials
738  */
739  double xminus = xmolSolvent - IMS_X_o_cutoff_/2.0;
740  double xminus2 = xminus * xminus;
741  double xminus3 = xminus2 * xminus;
742  double x_o_cut2 = IMS_X_o_cutoff_ * IMS_X_o_cutoff_;
743  double x_o_cut3 = x_o_cut2 * IMS_X_o_cutoff_;
744 
745  double h2 = 3.5 * xminus2 / IMS_X_o_cutoff_ - 2.0 * xminus3 / x_o_cut2;
746  double h2_prime = 7.0 * xminus / IMS_X_o_cutoff_ - 6.0 * xminus2 / x_o_cut2;
747 
748  double h1 = (1.0 - 3.0 * xminus2 / x_o_cut2 + 2.0 * xminus3/ x_o_cut3);
749  double h1_prime = (- 6.0 * xminus / x_o_cut2 + 6.0 * xminus2/ x_o_cut3);
750 
751  double h1_g = h1 / IMS_gamma_o_min_;
752  double h1_g_prime = h1_prime / IMS_gamma_o_min_;
753 
754  double alpha = 1.0 / (exp(1.0) * IMS_gamma_k_min_);
755  double h1_f = h1 * alpha;
756  double h1_f_prime = h1_prime * alpha;
757 
758  double f = h2 + h1_f;
759  double f_prime = h2_prime + h1_f_prime;
760 
761  double g = h2 + h1_g;
762  double g_prime = h2_prime + h1_g_prime;
763 
764  tmp = (xmolSolvent/ g * g_prime + (1.0-xmolSolvent) / f * f_prime);
765  double lngammak = -1.0 - log(f) + tmp * xmolSolvent;
766  double lngammao =-log(g) - tmp * (1.0-xmolSolvent);
767 
768  tmp = log(xmolSolvent) + lngammak;
769  for (size_t k = 1; k < m_kk; k++) {
770  IMS_lnActCoeffMolal_[k]= tmp;
771  }
772  IMS_lnActCoeffMolal_[m_indexSolvent] = lngammao;
773  }
774  }
775 
776  // Exponentials - trial 2
777  else if (IMS_typeCutoff_ == 2) {
778  if (xmolSolvent > IMS_X_o_cutoff_) {
779  for (size_t k = 1; k < m_kk; k++) {
780  IMS_lnActCoeffMolal_[k]= 0.0;
781  }
782  IMS_lnActCoeffMolal_[m_indexSolvent] = - log(xx) + (xx - 1.0)/xx;
783  return;
784  } else {
785 
786  double xoverc = xmolSolvent/IMS_cCut_;
787  double eterm = std::exp(-xoverc);
788 
789  double fptmp = IMS_bfCut_ - IMS_afCut_ / IMS_cCut_ - IMS_bfCut_*xoverc
790  + 2.0*IMS_dfCut_*xmolSolvent - IMS_dfCut_*xmolSolvent*xoverc;
791  double f_prime = 1.0 + eterm*fptmp;
792  double f = xmolSolvent + IMS_efCut_ + eterm * (IMS_afCut_ + xmolSolvent * (IMS_bfCut_ + IMS_dfCut_*xmolSolvent));
793 
794  double gptmp = IMS_bgCut_ - IMS_agCut_ / IMS_cCut_ - IMS_bgCut_*xoverc
795  + 2.0*IMS_dgCut_*xmolSolvent - IMS_dgCut_*xmolSolvent*xoverc;
796  double g_prime = 1.0 + eterm*gptmp;
797  double g = xmolSolvent + IMS_egCut_ + eterm * (IMS_agCut_ + xmolSolvent * (IMS_bgCut_ + IMS_dgCut_*xmolSolvent));
798 
799  tmp = (xmolSolvent / g * g_prime + (1.0 - xmolSolvent) / f * f_prime);
800  double lngammak = -1.0 - log(f) + tmp * xmolSolvent;
801  double lngammao =-log(g) - tmp * (1.0-xmolSolvent);
802 
803  tmp = log(xx) + lngammak;
804  for (size_t k = 1; k < m_kk; k++) {
805  IMS_lnActCoeffMolal_[k]= tmp;
806  }
807  IMS_lnActCoeffMolal_[m_indexSolvent] = lngammao;
808  }
809  }
810  return;
811 }

826 void IdealMolalSoln::calcIMSCutoffParams_()
827 {
828  IMS_afCut_ = 1.0 / (std::exp(1.0) * IMS_gamma_k_min_);
829  IMS_efCut_ = 0.0;
830  bool converged = false;
831  double oldV = 0.0;
832  int its;
833  for (its = 0; its < 100 && !converged; its++) {
834  oldV = IMS_efCut_;
835  IMS_afCut_ = 1.0 / (std::exp(1.0) * IMS_gamma_k_min_) - IMS_efCut_;
836  IMS_bfCut_ = IMS_afCut_ / IMS_cCut_ + IMS_slopefCut_ - 1.0;
837  IMS_dfCut_ = ((- IMS_afCut_/IMS_cCut_ + IMS_bfCut_ - IMS_bfCut_*IMS_X_o_cutoff_/IMS_cCut_)
838  /
839  (IMS_X_o_cutoff_*IMS_X_o_cutoff_/IMS_cCut_ - 2.0 * IMS_X_o_cutoff_));
840  double tmp = IMS_afCut_ + IMS_X_o_cutoff_*(IMS_bfCut_ + IMS_dfCut_ * IMS_X_o_cutoff_);
841  double eterm = std::exp(-IMS_X_o_cutoff_/IMS_cCut_);
842  IMS_efCut_ = - eterm * (tmp);
843  if (fabs(IMS_efCut_ - oldV) < 1.0E-14) {
844  converged = true;
845  }
846  }
847  if (!converged) {
848  throw CanteraError(" IdealMolalSoln::calcCutoffParams_()",
849  " failed to converge on the f polynomial");
850  }
851  converged = false;
852  double f_0 = IMS_afCut_ + IMS_efCut_;
853  double f_prime_0 = 1.0 - IMS_afCut_ / IMS_cCut_ + IMS_bfCut_;
854  IMS_egCut_ = 0.0;
855  for (its = 0; its < 100 && !converged; its++) {
856  oldV = IMS_egCut_;
857  double lng_0 = -log(IMS_gamma_o_min_) - f_prime_0 / f_0;
858  IMS_agCut_ = exp(lng_0) - IMS_egCut_;
859  IMS_bgCut_ = IMS_agCut_ / IMS_cCut_ + IMS_slopegCut_ - 1.0;
860  IMS_dgCut_ = ((- IMS_agCut_/IMS_cCut_ + IMS_bgCut_ - IMS_bgCut_*IMS_X_o_cutoff_/IMS_cCut_)
861  /
862  (IMS_X_o_cutoff_*IMS_X_o_cutoff_/IMS_cCut_ - 2.0 * IMS_X_o_cutoff_));
863  double tmp = IMS_agCut_ + IMS_X_o_cutoff_*(IMS_bgCut_ + IMS_dgCut_ *IMS_X_o_cutoff_);
864  double eterm = std::exp(-IMS_X_o_cutoff_/IMS_cCut_);
865  IMS_egCut_ = - eterm * (tmp);
866  if (fabs(IMS_egCut_ - oldV) < 1.0E-14) {
867  converged = true;
868  }
869  }
870  if (!converged) {
871  throw CanteraError(" IdealMolalSoln::calcCutoffParams_()",
872  " failed to converge on the g polynomial");
873  }
874 }
