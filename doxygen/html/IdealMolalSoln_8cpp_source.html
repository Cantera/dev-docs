302 void IdealMolalSoln::getActivities(doublereal* ac) const
303 {
304  _updateStandardStateThermo();
305  /*
306  * Update the molality array, m_molalities()
307  * This requires an update due to mole fractions
308  */
309  if (IMS_typeCutoff_ == 0) {
310  calcMolalities();
311  for (size_t k = 0; k < m_kk; k++) {
312  ac[k] = m_molalities[k];
313  }
314  double xmolSolvent = moleFraction(m_indexSolvent);
315  // Limit the activity coefficient to be finite as the solvent mole
316  // fraction goes to zero.
317  xmolSolvent = std::max(m_xmolSolventMIN, xmolSolvent);
318  ac[m_indexSolvent] =
319  exp((xmolSolvent - 1.0)/xmolSolvent);
320  } else {
321 
322  s_updateIMS_lnMolalityActCoeff();
323  /*
324  * Now calculate the array of activities.
325  */
326  for (size_t k = 1; k < m_kk; k++) {
327  ac[k] = m_molalities[k] * exp(IMS_lnActCoeffMolal_[k]);
328  }
329  double xmolSolvent = moleFraction(m_indexSolvent);
330  ac[m_indexSolvent] =
331  exp(IMS_lnActCoeffMolal_[m_indexSolvent]) * xmolSolvent;
332 
333  }

336 void IdealMolalSoln::getMolalityActivityCoefficients(doublereal* acMolality) const
337 {
338  if (IMS_typeCutoff_ == 0) {
339  for (size_t k = 0; k < m_kk; k++) {
340  acMolality[k] = 1.0;
341  }
342  double xmolSolvent = moleFraction(m_indexSolvent);
343  // Limit the activity coefficient to be finite as the solvent mole
344  // fraction goes to zero.
345  xmolSolvent = std::max(m_xmolSolventMIN, xmolSolvent);
346  acMolality[m_indexSolvent] =

348  } else {

361 void IdealMolalSoln::getChemPotentials(doublereal* mu) const
362 {
363  // Assertion is made for speed
364  AssertThrow(m_indexSolvent == 0, "solvent not the first species");
365 
366  /*
367  * First get the standard chemical potentials
368  * -> this requires updates of standard state as a function
369  * of T and P
370  * These are defined at unit molality.
371  */
372  getStandardChemPotentials(mu);
373  /*
374  * Update the molality array, m_molalities()
375  * This requires an update due to mole fractions
376  */
377  calcMolalities();
378  /*
379  * get the solvent mole fraction
380  */
381  double xmolSolvent = moleFraction(m_indexSolvent);
382  doublereal RT = GasConstant * temperature();
383 
384  if (IMS_typeCutoff_ == 0 || xmolSolvent > 3.* IMS_X_o_cutoff_/2.0) {
385 
386  for (size_t k = 1; k < m_kk; k++) {
387  double xx = std::max(m_molalities[k], SmallNumber);
388  mu[k] += RT * log(xx);
389  }
390  /*
391  * Do the solvent
392  * -> see my notes
393  */
394 
395  double xx = std::max(xmolSolvent, SmallNumber);
396  mu[m_indexSolvent] +=
397  (RT * (xmolSolvent - 1.0) / xx);
398  } else {
399  /*
400  * Update the activity coefficients
401  * This also updates the internal molality array.
402  */
403  s_updateIMS_lnMolalityActCoeff();
404 
405 
406  for (size_t k = 1; k < m_kk; k++) {
407  double xx = std::max(m_molalities[k], SmallNumber);
408  mu[k] += RT * (log(xx) + IMS_lnActCoeffMolal_[k]);
409  }
410  double xx = std::max(xmolSolvent, SmallNumber);
411  mu[m_indexSolvent] +=
412  RT * (log(xx) + IMS_lnActCoeffMolal_[m_indexSolvent]);
413  }
414 
415 }

426 void IdealMolalSoln::getPartialMolarEntropies(doublereal* sbar) const
427 {
428  getEntropy_R(sbar);
429  calcMolalities();
430  if (IMS_typeCutoff_ == 0) {
431  for (size_t k = 0; k < m_kk; k++) {
432  if (k != m_indexSolvent) {
433  doublereal mm = std::max(SmallNumber, m_molalities[k]);
434  sbar[k] -= GasConstant * log(mm);
435  }
436  }
437  double xmolSolvent = moleFraction(m_indexSolvent);
438  sbar[m_indexSolvent] -= (GasConstant * (xmolSolvent - 1.0) / xmolSolvent);
439  } else {
440  /*
441  * Update the activity coefficients, This also update the
442  * internally stored molalities.
443  */
444  s_updateIMS_lnMolalityActCoeff();
445  /*
446  * First we will add in the obvious dependence on the T
447  * term out front of the log activity term
448  */
449  doublereal mm;
450  for (size_t k = 0; k < m_kk; k++) {
451  if (k != m_indexSolvent) {
452  mm = std::max(SmallNumber, m_molalities[k]);
453  sbar[k] -= GasConstant * (log(mm) + IMS_lnActCoeffMolal_[k]);
454  }
455  }
456  double xmolSolvent = moleFraction(m_indexSolvent);
457  mm = std::max(SmallNumber, xmolSolvent);
458  sbar[m_indexSolvent] -= GasConstant *(log(mm) + IMS_lnActCoeffMolal_[m_indexSolvent]);
459 
460  }
461 }

663 void IdealMolalSoln::s_updateIMS_lnMolalityActCoeff() const
664 {
665  /*
666  * Calculate the molalities. Currently, the molalities
667  * may not be current with respect to the contents of the
668  * State objects' data.
669  */
670  calcMolalities();
671 
672  double xmolSolvent = moleFraction(m_indexSolvent);
673  double xx = std::max(m_xmolSolventMIN, xmolSolvent);
674 
675  if (IMS_typeCutoff_ == 0) {
676  for (size_t k = 1; k < m_kk; k++) {
677  IMS_lnActCoeffMolal_[k]= 0.0;
678  }
679  IMS_lnActCoeffMolal_[m_indexSolvent] = - log(xx) + (xx - 1.0)/xx;
680  return;
681  } else if (IMS_typeCutoff_ == 1) {
682  if (xmolSolvent > 3.0 * IMS_X_o_cutoff_/2.0) {
683  for (size_t k = 1; k < m_kk; k++) {
684  IMS_lnActCoeffMolal_[k]= 0.0;
685  }
686  IMS_lnActCoeffMolal_[m_indexSolvent] = - log(xx) + (xx - 1.0)/xx;
687  return;
688  } else if (xmolSolvent < IMS_X_o_cutoff_/2.0) {
689  double tmp = log(xx * IMS_gamma_k_min_);
690  for (size_t k = 1; k < m_kk; k++) {
691  IMS_lnActCoeffMolal_[k]= tmp;
692  }
693  IMS_lnActCoeffMolal_[m_indexSolvent] = log(IMS_gamma_o_min_);
694  return;
695  } else {
696 
697  /*
698  * If we are in the middle region, calculate the connecting polynomials
699  */
700  double xminus = xmolSolvent - IMS_X_o_cutoff_/2.0;
701  double xminus2 = xminus * xminus;
702  double xminus3 = xminus2 * xminus;
703  double x_o_cut2 = IMS_X_o_cutoff_ * IMS_X_o_cutoff_;
704  double x_o_cut3 = x_o_cut2 * IMS_X_o_cutoff_;
705 
706  double h2 = 3.5 * xminus2 / IMS_X_o_cutoff_ - 2.0 * xminus3 / x_o_cut2;
707  double h2_prime = 7.0 * xminus / IMS_X_o_cutoff_ - 6.0 * xminus2 / x_o_cut2;
708 
709  double h1 = (1.0 - 3.0 * xminus2 / x_o_cut2 + 2.0 * xminus3/ x_o_cut3);
710  double h1_prime = (- 6.0 * xminus / x_o_cut2 + 6.0 * xminus2/ x_o_cut3);
711 
712  double h1_g = h1 / IMS_gamma_o_min_;
713  double h1_g_prime = h1_prime / IMS_gamma_o_min_;
714 
715  double alpha = 1.0 / (exp(1.0) * IMS_gamma_k_min_);
716  double h1_f = h1 * alpha;
717  double h1_f_prime = h1_prime * alpha;
718 
719  double f = h2 + h1_f;
720  double f_prime = h2_prime + h1_f_prime;
721 
722  double g = h2 + h1_g;
723  double g_prime = h2_prime + h1_g_prime;
724 
725  double tmp = (xmolSolvent/ g * g_prime + (1.0-xmolSolvent) / f * f_prime);
726  double lngammak = -1.0 - log(f) + tmp * xmolSolvent;
727  double lngammao =-log(g) - tmp * (1.0-xmolSolvent);
728 
729  tmp = log(xmolSolvent) + lngammak;
730  for (size_t k = 1; k < m_kk; k++) {
731  IMS_lnActCoeffMolal_[k]= tmp;
732  }
733  IMS_lnActCoeffMolal_[m_indexSolvent] = lngammao;
734  }
735  }
736 
737  // Exponentials - trial 2
738  else if (IMS_typeCutoff_ == 2) {
739  if (xmolSolvent > IMS_X_o_cutoff_) {
740  for (size_t k = 1; k < m_kk; k++) {
741  IMS_lnActCoeffMolal_[k]= 0.0;
742  }
743  IMS_lnActCoeffMolal_[m_indexSolvent] = - log(xx) + (xx - 1.0)/xx;
744  return;
745  } else {
746 
747  double xoverc = xmolSolvent/IMS_cCut_;
748  double eterm = std::exp(-xoverc);
749 
750  double fptmp = IMS_bfCut_ - IMS_afCut_ / IMS_cCut_ - IMS_bfCut_*xoverc
751  + 2.0*IMS_dfCut_*xmolSolvent - IMS_dfCut_*xmolSolvent*xoverc;
752  double f_prime = 1.0 + eterm*fptmp;
753  double f = xmolSolvent + IMS_efCut_ + eterm * (IMS_afCut_ + xmolSolvent * (IMS_bfCut_ + IMS_dfCut_*xmolSolvent));
754 
755  double gptmp = IMS_bgCut_ - IMS_agCut_ / IMS_cCut_ - IMS_bgCut_*xoverc
756  + 2.0*IMS_dgCut_*xmolSolvent - IMS_dgCut_*xmolSolvent*xoverc;
757  double g_prime = 1.0 + eterm*gptmp;
758  double g = xmolSolvent + IMS_egCut_ + eterm * (IMS_agCut_ + xmolSolvent * (IMS_bgCut_ + IMS_dgCut_*xmolSolvent));
759 
760  double tmp = (xmolSolvent / g * g_prime + (1.0 - xmolSolvent) / f * f_prime);
761  double lngammak = -1.0 - log(f) + tmp * xmolSolvent;
762  double lngammao =-log(g) - tmp * (1.0-xmolSolvent);
763 
764  tmp = log(xx) + lngammak;
765  for (size_t k = 1; k < m_kk; k++) {
766  IMS_lnActCoeffMolal_[k]= tmp;
767  }
768  IMS_lnActCoeffMolal_[m_indexSolvent] = lngammao;
769  }
770  }
771 }

785 void IdealMolalSoln::calcIMSCutoffParams_()
786 {
787  IMS_afCut_ = 1.0 / (std::exp(1.0) * IMS_gamma_k_min_);
788  IMS_efCut_ = 0.0;
789  bool converged = false;
790  for (int its = 0; its < 100 && !converged; its++) {
791  double oldV = IMS_efCut_;
792  IMS_afCut_ = 1.0 / (std::exp(1.0) * IMS_gamma_k_min_) - IMS_efCut_;
793  IMS_bfCut_ = IMS_afCut_ / IMS_cCut_ + IMS_slopefCut_ - 1.0;
794  IMS_dfCut_ = ((- IMS_afCut_/IMS_cCut_ + IMS_bfCut_ - IMS_bfCut_*IMS_X_o_cutoff_/IMS_cCut_)
795  /
796  (IMS_X_o_cutoff_*IMS_X_o_cutoff_/IMS_cCut_ - 2.0 * IMS_X_o_cutoff_));
797  double tmp = IMS_afCut_ + IMS_X_o_cutoff_*(IMS_bfCut_ + IMS_dfCut_ * IMS_X_o_cutoff_);
798  double eterm = std::exp(-IMS_X_o_cutoff_/IMS_cCut_);
799  IMS_efCut_ = - eterm * (tmp);
800  if (fabs(IMS_efCut_ - oldV) < 1.0E-14) {
801  converged = true;
802  }
803  }
804  if (!converged) {
805  throw CanteraError(" IdealMolalSoln::calcCutoffParams_()",
806  " failed to converge on the f polynomial");
807  }
808  converged = false;
809  double f_0 = IMS_afCut_ + IMS_efCut_;
810  double f_prime_0 = 1.0 - IMS_afCut_ / IMS_cCut_ + IMS_bfCut_;
811  IMS_egCut_ = 0.0;
812  for (int its = 0; its < 100 && !converged; its++) {
813  double oldV = IMS_egCut_;
814  double lng_0 = -log(IMS_gamma_o_min_) - f_prime_0 / f_0;
815  IMS_agCut_ = exp(lng_0) - IMS_egCut_;
816  IMS_bgCut_ = IMS_agCut_ / IMS_cCut_ + IMS_slopegCut_ - 1.0;
817  IMS_dgCut_ = ((- IMS_agCut_/IMS_cCut_ + IMS_bgCut_ - IMS_bgCut_*IMS_X_o_cutoff_/IMS_cCut_)
818  /
819  (IMS_X_o_cutoff_*IMS_X_o_cutoff_/IMS_cCut_ - 2.0 * IMS_X_o_cutoff_));
820  double tmp = IMS_agCut_ + IMS_X_o_cutoff_*(IMS_bgCut_ + IMS_dgCut_ *IMS_X_o_cutoff_);
821  double eterm = std::exp(-IMS_X_o_cutoff_/IMS_cCut_);
822  IMS_egCut_ = - eterm * (tmp);
823  if (fabs(IMS_egCut_ - oldV) < 1.0E-14) {
824  converged = true;
825  }
826  }
827  if (!converged) {
828  throw CanteraError(" IdealMolalSoln::calcCutoffParams_()",
829  " failed to converge on the g polynomial");
830  }
831 }
