67  int& has_nasa_idealGas,

77 {
78  string ssModel = "idealGas";
79  for (size_t n = 0; n < spDataNodeList.size(); n++) {
80  bool ifound = false;
81  XML_Node* spNode = spDataNodeList[n];
82  if (spNode->hasChild("standardState")) {
83  string mm = spNode->child("standardState")["model"];
84  if (mm == "waterIAPWS" || mm == "waterPDSS") {
85  has_water++;
86  ifound = true;
87  }
88  if (mm == "HKFT") {
89  has_hptx++;
90  ifound = true;
91  }
92  }
93  if (!ifound) {
94  if (spNode->hasChild("thermo")) {
95  const XML_Node& th = spNode->child("thermo");
96  if (spNode->hasChild("standardState")) {
97  ssModel = spNode->child("standardState")["model"];
98  }
99  if (th.hasChild("NASA")) {
100  if (ssModel == "idealGas") {
101  has_nasa_idealGas++;
102  } else if (ssModel == "constant_incompressible" ||
103  ssModel == "constantVolume") {
104  has_nasa_constVol++;
105  } else if (ssModel == "temperature_polynomial" ||
106  ssModel == "density_temperature_polynomial" ||
107  ssModel == "constant") {
108  has_other++;
109  } else {
110  throw CanteraError("getVPSSMgrTypes",
111  "Specified VPSSMgr model {} does not match any known type.",
112  spNode->attrib("name"));
113  }
114  ifound = true;
115  }
116  if (th.hasChild("Shomate")) {
117  if (ssModel == "idealGas") {
118  has_shomate_idealGas++;
119  } else if (ssModel == "constant_incompressible" ||
120  ssModel == "constantVolume") {
121  has_shomate_constVol++;
122  } else if (ssModel == "temperature_polynomial" ||
123  ssModel == "density_temperature_polynomial" ||
124  ssModel == "constant") {
125  has_other++;
126  } else {
127  throw CanteraError("getVPSSMgrTypes",
128  "Specified VPSSMgr model {} does not match any known type.",
129  spNode->attrib("name"));
130  }
131  ifound = true;
132  }
133  if (th.hasChild("const_cp")) {
134  if (ssModel == "idealGas") {
135  has_simple_idealGas++;
136  } else if (ssModel == "constant_incompressible" ||
137  ssModel == "constantVolume") {
138  has_simple_constVol++;
139  } else if (ssModel == "temperature_polynomial" ||
140  ssModel == "density_temperature_polynomial" ||
141  ssModel == "constant") {
142  has_other++;
143  } else {
144  throw CanteraError("getVPSSMgrTypes",
145  "Specified VPSSMgr model {} does not match any known type.",
146  spNode->attrib("name"));
147  }
148  ifound = true;
149  }
150  if (th.hasChild("poly")) {
151  if (th.child("poly")["order"] == "1") {
152  has_simple_constVol = 1;
153  ifound = true;
154  } else throw CanteraError("newSpeciesThermo",
155  "poly with order > 1 not yet supported");
156  }
157  if (th.hasChild("Mu0")) {
158  has_other++;
159  ifound = true;
160  }
161  if (th.hasChild("NASA9")) {
162  has_other++;
163  ifound = true;
164  }
165  if (th.hasChild("NASA9MULTITEMP")) {
166  has_other++;
167  ifound = true;
168  }
169  if (th.hasChild("adsorbate")) {
170  has_other++;
171  ifound = true;
172  }
173  if (th.hasChild("HKFT")) {
174  has_hptx++;
175  ifound = true;
176  }
177  } else {
178  throw CanteraError("getVPSSMgrTypes",
179  "Specified VPSSMgr model {} does not match any known type.",
180  spNode->attrib("name"));
181  }
182  }

195 {
196  std::string lssModel = lowercase(ssModel);
197  VPSSMgr_enumType type;
198  if (lssModel == "idealgas") {
199  type = cVPSSMGR_IDEALGAS;
200  } else if (lssModel == "constvol") {
201  type = cVPSSMGR_CONSTVOL;
202  } else if (lssModel == "purefuild") {
203  type = cVPSSMGR_PUREFLUID;
204  } else if (lssModel == "water_constvol") {
205  type = cVPSSMGR_WATER_CONSTVOL;
206  } else if (lssModel == "water_hkft") {
207  type = cVPSSMGR_WATER_HKFT;
208  } else if (lssModel == "general") {
209  type = cVPSSMGR_GENERAL;
210  } else {
211  type = cVPSSMGR_UNDEF;
212  }
213  return type;

216 VPSSMgr* VPSSMgrFactory::newVPSSMgr(VPStandardStateTP* vp_ptr,
217  XML_Node* phaseNode_ptr,
218  std::vector<XML_Node*> & spDataNodeList)
219 {
220  std::string ssManager;
221  std::string vpssManager;
222 
223  // First look for any explicit instructions within the XML Database for the
224  // standard state manager and the variable pressure standard state manager
225  if (phaseNode_ptr && phaseNode_ptr->hasChild("thermo")) {
226  const XML_Node& thermoNode = phaseNode_ptr->child("thermo");
227  if (thermoNode.hasChild("standardStateManager")) {
228  const XML_Node& ssNode = thermoNode.child("standardStateManager");
229  ssManager = ssNode["model"];
230  }
231  if (thermoNode.hasChild("variablePressureStandardStateManager")) {
232  const XML_Node& vpssNode = thermoNode.child("variablePressureStandardStateManager");
233  vpssManager = vpssNode["model"];
234  }
235  }
236 
237  // first get the reference state handler.
238  SpeciesThermo* spth = new GeneralSpeciesThermo();
239  vp_ptr->setSpeciesThermo(spth);
240 
241  // Next, if we have specific directions, use them to get the VPSSSMgr object
242  // and return immediately
243  if (vpssManager != "") {
244  VPSSMgr_enumType type = VPSSMgr_StringConversion(vpssManager);
245  return newVPSSMgr(type, vp_ptr);
246  }
247 
248  // Handle special cases based on the VPStandardState types
249  if (vp_ptr->eosType() == cVPSS_IdealGas) {
250  return new VPSSMgr_IdealGas(vp_ptr, spth);
251  } else if (vp_ptr->eosType() == cVPSS_ConstVol) {
252  return new VPSSMgr_ConstVol(vp_ptr, spth);
253  }
254 
255  int inasaIG = 0, inasaCV = 0, ishomateIG = 0, ishomateCV = 0,
256  isimpleIG = 0, isimpleCV = 0, iwater = 0, itpx = 0, iother = 0;
257  int ihptx = 0;
258 
259  try {
260  getVPSSMgrTypes(spDataNodeList, inasaIG, inasaCV, ishomateIG, ishomateCV,
261  isimpleIG, isimpleCV, iwater, itpx, ihptx, iother);
262  } catch (CanteraError) {
263  iother = 1;
264  }
265 
266  if (iwater == 1) {
267  if (ihptx == 0) {
268  if (inasaIG || ishomateIG || isimpleIG) {
269  throw CanteraError("newVPSSMgr", "Ideal gas with liquid water");
270  } else {
271  return new VPSSMgr_Water_ConstVol(vp_ptr, spth);
272  }
273  } else {
274  if (inasaIG || ishomateIG || isimpleIG) {
275  throw CanteraError("newVPSSMgr", "Ideal gas with liquid water");
276  } else if (inasaCV || ishomateCV || isimpleCV) {
277  return new VPSSMgr_General(vp_ptr, spth);
278  } else {
279  return new VPSSMgr_Water_HKFT(vp_ptr, spth);
280  }
281  }
282  }
283  if ((inasaCV || ishomateCV || isimpleCV) &&
284  !inasaIG && !ishomateIG && !isimpleIG && !itpx && !ihptx && !iother) {
285  return new VPSSMgr_ConstVol(vp_ptr, spth);
286  }
287  return new VPSSMgr_General(vp_ptr, spth);
288 }

290 // I don't think this is currently used. However, this is a virtual
291 // function where additional capabilities may be added.
292 VPSSMgr* VPSSMgrFactory::newVPSSMgr(VPSSMgr_enumType type,
293  VPStandardStateTP* vp_ptr)
294 {
295  SpeciesThermo& spthermoRef = vp_ptr->speciesThermo();
296  switch (type) {
297  case cVPSSMGR_IDEALGAS:
298  return new VPSSMgr_IdealGas(vp_ptr, &spthermoRef);
299  case cVPSSMGR_CONSTVOL:
300  return new VPSSMgr_ConstVol(vp_ptr, &spthermoRef);
301  case cVPSSMGR_PUREFLUID:
302  throw CanteraError("VPSSMgrFactory::newVPSSMgr",
303  "unimplemented");
304  case cVPSSMGR_WATER_CONSTVOL:
305  return new VPSSMgr_Water_ConstVol(vp_ptr, &spthermoRef);
306  case cVPSSMGR_WATER_HKFT:
307  return new VPSSMgr_Water_HKFT(vp_ptr, &spthermoRef);
308  case cVPSSMGR_GENERAL:
309  return new VPSSMgr_General(vp_ptr, &spthermoRef);
310  case cVPSSMGR_UNDEF:
311  default:
312  throw CanteraError("VPSSMgrFactory::newVPSSMgr",
313  "Specified VPSSMgr model {} does not match any known type.", type);
314  return 0;
315  }
