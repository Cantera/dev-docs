120 void MetalSHEelectrons::initThermoXML(XML_Node& phaseNode, const std::string& id_)
121 {
122  // Find the Thermo XML node

124  throw CanteraError("MetalSHEelectrons::initThermoXML",
125  "no thermo XML node");

127  XML_Node& tnode = phaseNode.child("thermo");
128  doublereal dens = 2.65E3;

130  dens = getFloat(tnode, "density", "toSI");
131  }
132  setDensity(dens);

134 }
