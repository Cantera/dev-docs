103 doublereal SingleSpeciesTP::cv_mole() const
104 {
105  // For single species, we go directory to the general Cp - Cv relation
106  //
107  // Cp = Cv + alpha**2 * V * T / beta
108  //
109  // where
110  // alpha = volume thermal expansion coefficient
111  // beta = isothermal compressibility
112  doublereal cvbar = cp_mole();
113  doublereal alpha = thermalExpansionCoeff();
114  doublereal beta = isothermalCompressibility();
115  doublereal V = molecularWeight(0)/density();
116  doublereal T = temperature();
117  if (beta != 0.0) {
118  cvbar -= alpha * alpha * V * T / beta;
119  }
120  return cvbar;
121 }

212 void SingleSpeciesTP::setState_HP(doublereal h, doublereal p,
213  doublereal tol)
214 {
215  doublereal dt;
216  setPressure(p);
217  for (int n = 0; n < 50; n++) {
218  dt = clip((h - enthalpy_mass())/cp_mass(), -100.0, 100.0);
219  setState_TP(temperature() + dt, p);
220  if (fabs(dt) < tol) {
221  return;
222  }
223  }
224  throw CanteraError("setState_HP","no convergence. dt = {}", dt);
225 }
