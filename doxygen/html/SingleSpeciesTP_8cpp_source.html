101 doublereal SingleSpeciesTP::cv_mole() const
102 {
103  // For single species, we go directory to the general Cp - Cv relation
104  //
105  // Cp = Cv + alpha**2 * V * T / beta
106  //
107  // where
108  // alpha = volume thermal expansion coefficient
109  // beta = isothermal compressibility
110  doublereal cvbar = cp_mole();
111  doublereal alpha = thermalExpansionCoeff();
112  doublereal beta = isothermalCompressibility();
113  doublereal V = molecularWeight(0)/density();
114  doublereal T = temperature();
115  if (beta != 0.0) {
116  cvbar -= alpha * alpha * V * T / beta;
117  }
118  return cvbar;
119 }

215 void SingleSpeciesTP::setState_HP(doublereal h, doublereal p,
216  doublereal tol)
217 {
218  doublereal dt;
219  setPressure(p);
220  for (int n = 0; n < 50; n++) {
221  dt = clip((h - enthalpy_mass())/cp_mass(), -100.0, 100.0);
222  setState_TP(temperature() + dt, p);
223  if (fabs(dt) < tol) {
224  return;
225  }
226  }
227  throw CanteraError("setState_HP","no convergence. dt = {}", dt);
228 }

230 void SingleSpeciesTP::setState_UV(doublereal u, doublereal v,
231  doublereal tol)

284 void SingleSpeciesTP::initThermo()
285 {
286  // Make sure there is one and only one species in this phase.
287  if (nSpecies() != 1) {
288  throw CanteraError("initThermo",
289  "stoichiometric substances may only contain one species.");
290  }
291 
292  // Resize temporary arrays.

294  m_cp0_R.resize(1);
295  m_s0_R.resize(1);
296 
297  // Make sure the species mole fraction is equal to 1.0;

300
