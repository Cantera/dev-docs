117 doublereal SingleSpeciesTP::cv_mole() const
118 {
119  /*
120  * For single species, we go directory to the general Cp - Cv relation
121  *
122  * Cp = Cv + alpha**2 * V * T / beta
123  *
124  * where
125  * alpha = volume thermal expansion coefficient
126  * beta = isothermal compressibility
127  */
128  doublereal cvbar = cp_mole();
129  doublereal alpha = thermalExpansionCoeff();
130  doublereal beta = isothermalCompressibility();
131  doublereal molecW = molecularWeight(0);
132  doublereal V = molecW/density();
133  doublereal T = temperature();
134  if (beta != 0.0) {
135  cvbar -= alpha * alpha * V * T / beta;
136  }
137  return cvbar;
138 }

310 void SingleSpeciesTP::setState_HP(doublereal h, doublereal p,
311  doublereal tol)
312 {
313  doublereal dt;
314  setPressure(p);
315  for (int n = 0; n < 50; n++) {
316  dt = (h - enthalpy_mass())/cp_mass();
317  if (dt > 100.0) {
318  dt = 100.0;
319  } else if (dt < -100.0) {
320  dt = -100.0;
321  }
322  setState_TP(temperature() + dt, p);
323  if (fabs(dt) < tol) {
324  return;
325  }
326  }
327  throw CanteraError("setState_HP","no convergence. dt = " + fp2str(dt));
328 }

418  /*

426  /*
