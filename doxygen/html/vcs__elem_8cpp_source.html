25 bool VCS_SOLVE::vcs_elabcheck(int ibound)
26 {
27  size_t top = m_numComponents;
28  if (ibound) {
29  top = m_numElemConstraints;
30  }
31 
32  for (size_t i = 0; i < top; ++i) {
33  // Require 12 digits of accuracy on non-zero constraints.
34  if (m_elementActive[i] && fabs(m_elemAbundances[i] - m_elemAbundancesGoal[i]) > fabs(m_elemAbundancesGoal[i]) * 1.0e-12) {
35  // This logic is for charge neutrality condition
36  if (m_elType[i] == VCS_ELEM_TYPE_CHARGENEUTRALITY &&
37  m_elemAbundancesGoal[i] != 0.0) {
38  throw CanteraError("VCS_SOLVE::vcs_elabcheck",
39  "Problem with charge neutrality condition");
40  }
41  if (m_elemAbundancesGoal[i] == 0.0 || (m_elType[i] == VCS_ELEM_TYPE_ELECTRONCHARGE)) {
42  double scale = VCS_DELETE_MINORSPECIES_CUTOFF;
43 
44  // Find out if the constraint is a multisign constraint. If it
45  // is, then we have to worry about roundoff error in the
46  // addition of terms. We are limited to 13 digits of finite
47  // arithmetic accuracy.
48  bool multisign = false;
49  for (size_t kspec = 0; kspec < m_numSpeciesTot; kspec++) {
50  double eval = m_formulaMatrix(kspec,i);
51  if (eval < 0.0) {
52  multisign = true;
53  }
54  if (eval != 0.0) {
55  scale = std::max(scale, fabs(eval * m_molNumSpecies_old[kspec]));
56  }
57  }
58  if (multisign) {
59  if (fabs(m_elemAbundances[i] - m_elemAbundancesGoal[i]) > 1e-11 * scale) {
60  return false;
61  }
62  } else {
63  if (fabs(m_elemAbundances[i] - m_elemAbundancesGoal[i]) > VCS_DELETE_MINORSPECIES_CUTOFF) {
64  return false;
65  }
66  }
67  } else {
68  // For normal element balances, we require absolute compliance
69  // even for ridiculously small numbers.
70  if (m_elType[i] == VCS_ELEM_TYPE_ABSPOS) {
71  return false;
72  } else {
73  return false;
74  }
75  }
76  }
77  }
78  return true;
79 }
