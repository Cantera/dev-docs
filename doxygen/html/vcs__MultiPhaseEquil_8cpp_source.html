329  Herr, cpb, HConvErr);

332  }

334  if (Tnew < 0.0) {
335  Tnew = 0.5*Tnow;
336  }
337  m_mix->setTemperature(Tnew);
338 
339  } catch (CanteraError err) {
340  if (!estimateEquil) {
341  addLogEntry("no convergence",
342  "try estimating composition at the start");

360  "No convergence for T");
361 done:
362  ;
363  return iSuccess;

943  vprob->Vol = mphase->volume();

945 
946  Cantera::ThermoPhase* tPhase = 0;
947 
948  bool gasPhase;
949  int printLvl = vprob->m_printLvl;
950 
951  /*
952  * Loop over the phases, transferring pertinent information
953  */
954  int kT = 0;
955  for (size_t iphase = 0; iphase < totNumPhases; iphase++) {
956 
957  /*
958  * Get the thermophase object - assume volume phase
959  */
960  tPhase = &(mphase->phase(iphase));
961  size_t nelem = tPhase->nElements();
962 
963  /*
964  * Query Cantera for the equation of state type of the
965  * current phase.
966  */
967  int eos = tPhase->eosType();
968  if (eos == cIdealGas) {
969  gasPhase = true;
970  } else {
971  gasPhase = false;
972  }
973 
974  /*
975  * Find out the number of species in the phase
976  */
977  size_t nSpPhase = tPhase->nSpecies();
978  /*
979  * Find out the name of the phase
980  */
981  string phaseName = tPhase->name();
982 
983  /*
984  * Call the basic vcs_VolPhase creation routine.
985  * Properties set here:
986  * ->PhaseNum = phase number in the thermo problem
987  * ->GasPhase = Boolean indicating whether it is a gas phase
988  * ->NumSpecies = number of species in the phase
989  * ->TMolesInert = Inerts in the phase = 0.0 for cantera
990  * ->PhaseName = Name of the phase
991  */
992 
993  vcs_VolPhase* VolPhase = vprob->VPhaseList[iphase];
994  VolPhase->resize(iphase, nSpPhase, nelem, phaseName.c_str(), 0.0);
995  VolPhase->m_gasPhase = gasPhase;
996  /*
997  * Tell the vcs_VolPhase pointer about cantera
998  */
999  VolPhase->p_VCS_UnitsFormat = vprob->m_VCS_UnitsFormat;
1000  VolPhase->setPtrThermoPhase(tPhase);
1001  VolPhase->setTotalMoles(0.0);
1002  /*
1003  * Set the electric potential of the volume phase from the
1004  * ThermoPhase object's value.
1005  */
1006  VolPhase->setElectricPotential(tPhase->electricPotential());
1007  /*
1008  * Query the ThermoPhase object to find out what convention
1009  * it uses for the specification of activity and Standard State.
1010  */
1011  VolPhase->p_activityConvention = tPhase->activityConvention();
1012  /*
1013  * Assign the value of eqn of state
1014  * -> Handle conflicts here.
1015  */
1016  switch (eos) {
1017  case cIdealGas:
1018  VolPhase->m_eqnState = VCS_EOS_IDEAL_GAS;
1019  break;
1020  case cIncompressible:
1021  VolPhase->m_eqnState = VCS_EOS_CONSTANT;
1022  break;
1023  case cSurf:
1024  plogf("cSurf not handled yet\n");
1025  exit(EXIT_FAILURE);
1026  case cStoichSubstance:
1027  VolPhase->m_eqnState = VCS_EOS_STOICH_SUB;
1028  break;
1029  case cPureFluid:
1030  if (printLvl > 1) {
1031  plogf("cPureFluid not recognized yet by VCSnonideal\n");
1032  }
1033  break;
1034  case cEdge:
1035  plogf("cEdge not handled yet\n");
1036  exit(EXIT_FAILURE);
1037  case cIdealSolidSolnPhase0:
1038  case cIdealSolidSolnPhase1:
1039  case cIdealSolidSolnPhase2:
1040  VolPhase->m_eqnState = VCS_EOS_IDEAL_SOLN;
1041  break;
1042  default:
1043  if (printLvl > 1) {
1044  plogf("Unknown Cantera EOS to VCSnonideal: %d\n", eos);
1045  }
1046  VolPhase->m_eqnState = VCS_EOS_UNK_CANTERA;
1047  if (!VolPhase->usingCanteraCalls()) {
1048  plogf("vcs functions asked for, but unimplemented\n");
1049  exit(EXIT_FAILURE);
1050  }
1051  break;
1052  }
1053

1055  * Transfer all of the element information from the
1056  * ThermoPhase object to the vcs_VolPhase object.
1057  * Also decide whether we need a new charge neutrality
1058  * element in the phase to enforce a charge neutrality
1059  * constraint.
1060  * We also decide whether this is a single species phase
1061  * with the voltage being the independent variable setting
1062  * the chemical potential of the electrons.
1063  */
1064  VolPhase->transferElementsFM(tPhase);
1065

1067  * Combine the element information in the vcs_VolPhase
1068  * object into the vprob object.
1069  */
1070  vprob->addPhaseElements(VolPhase);

1077  * Loop through each species in the current phase
1078  */

1082  * ThermoPhase object
1083  */
1084  vprob->WtSpecies[kT] = tPhase->molecularWeight(k);
1085 
1086  /*
1087  * Obtain the charges of the species from the

1089  */

1094  */

1107  */

1109  tMoles += vprob->w[kT];
1110  vprob->mf[kT] = mphase->moleFraction(kT);
1111 
1112  /*
1113  * transfer chemical potential vector
1114  */
1115  vprob->m_gibbsSpecies[kT] = muPhase[k];
1116  /*
1117  * Transfer the type of unknown
1118  */
1119  vprob->SpeciesUnknownType[kT] = VolPhase->speciesUnknownType(k);
1120  /*
1121  * Transfer the species information from the
1122  * volPhase structure to the VPROB structure
1123  * This includes:
1124  * FormulaMatrix[][]
1125  * VolPhase->IndSpecies[]
1126  */
1127  vprob->addOnePhaseSpecies(VolPhase, k, kT);
1128 
1129  /*
1130  * Get a pointer to the thermo object
1131  */
1132  ts_ptr = vprob->SpeciesThermo[kT];
1133  /*
1134  * Fill in the vcs_SpeciesProperty structure
1135  */

1137  sProp->NumElements = vprob->ne;
1138  sProp->SpName = vprob->SpName[kT];
1139  sProp->SpeciesThermo = ts_ptr;

1143  sProp->FormulaMatrixCol[e] = vprob->FormulaMatrix[e][kT];

1146  sProp->SurfaceSpecies = false;
1147  sProp->VolPM = 0.0;
1148

1150  * Transfer the thermo specification of the species
1151  * vprob->SpeciesThermo[]
1152  */

1155  /*
1156  * Add lookback connectivity into the thermo object first
1157  */
1158  ts_ptr->IndexPhase = iphase;

1160  ts_ptr->OwningPhase = VolPhase;

1175  ts_ptr->SS0_S0 = c[2];
1176  ts_ptr->SS0_Cp0 = c[3];

1185  ts_ptr->Activity_Coeff_Params = NULL;

1220  * estimate of the total number of moles is zero.
