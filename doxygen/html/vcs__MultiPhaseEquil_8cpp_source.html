98  P2 = Pnow;

100  V1 = Vnow;

104  V2 = V1;
105  P1 = Pnow;

139  }

143  "No convergence for V");

151  int XY, double Tlow, double Thigh,
152  int estimateEquil,
153  int printLvl, doublereal err,
154  int maxsteps, int loglevel)
155 {
156  int maxiter = 100;
157  int iSuccess;
158  if (XY != HP && XY != UP) {
159  throw CanteraError("vcs_MultiPhaseEquil::equilibrate_HP",
160  "Wrong XP", XY);
161  }
162  int strt = estimateEquil;
163 
164  // Lower bound on T. This will change as we progress in the calculation
165  if (Tlow <= 0.0) {
166  Tlow = 0.5 * m_mix->minTemp();
167  }
168  // Upper bound on T. This will change as we progress in the calculation
169  if (Thigh <= 0.0 || Thigh > 1.0E6) {
170  Thigh = 2.0 * m_mix->maxTemp();
171  }
172 
173  doublereal cpb = 1.0;
174  doublereal Hlow = Undef;
175  doublereal Hhigh = Undef;
176  doublereal Tnow = m_mix->temperature();
177  int printLvlSub = std::max(printLvl - 1, 0);
178 
179  for (int n = 0; n < maxiter; n++) {
180  // start with a loose error tolerance, but tighten it as we get
181  // close to the final temperature
182  try {
183  Tnow = m_mix->temperature();
184  iSuccess = equilibrate_TP(strt, printLvlSub, err, maxsteps, loglevel);
185  strt = 0;
186  double Hnow = (XY == UP) ? m_mix->IntEnergy() : m_mix->enthalpy();
187  double pmoles[10];
188  pmoles[0] = m_mix->phaseMoles(0);
189  double Tmoles = pmoles[0];
190  double HperMole = Hnow/Tmoles;
191  if (printLvl > 0) {
192  plogf("T = %g, Hnow = %g ,Tmoles = %g, HperMole = %g",
193  Tnow, Hnow, Tmoles, HperMole);
194  plogendl();
195  }
196 
197  // the equilibrium enthalpy monotonically increases with T;
198  // if the current value is below the target, then we know the
199  // current temperature is too low. Set the lower bounds.
200  if (Hnow < Htarget) {
201  if (Tnow > Tlow) {
202  Tlow = Tnow;
203  Hlow = Hnow;
204  }
205  } else {
206  // the current enthalpy is greater than the target; therefore
207  // the current temperature is too high. Set the high bounds.
208  if (Tnow < Thigh) {
209  Thigh = Tnow;
210  Hhigh = Hnow;
211  }
212  }
213  double dT;
214  if (Hlow != Undef && Hhigh != Undef) {
215  cpb = (Hhigh - Hlow)/(Thigh - Tlow);
216  dT = (Htarget - Hnow)/cpb;
217  double dTa = fabs(dT);
218  double dTmax = 0.5*fabs(Thigh - Tlow);
219  if (dTa > dTmax) {
220  dT *= dTmax/dTa;
221  }
222  } else {
223  double Tnew = sqrt(Tlow*Thigh);
224  dT = clip(Tnew - Tnow, -200.0, 200.0);
225  }
226  double acpb = std::max(fabs(cpb), 1.0E-6);
227  double denom = std::max(fabs(Htarget), acpb);
228  double Herr = Htarget - Hnow;
229  double HConvErr = fabs((Herr)/denom);
230  if (printLvl > 0) {
231  plogf(" equilibrate_HP: It = %d, Tcurr = %g Hcurr = %g, Htarget = %g\n",
232  n, Tnow, Hnow, Htarget);
233  plogf(" H rel error = %g, cp = %g, HConvErr = %g\n",
234  Herr, cpb, HConvErr);
235  }
236 
237  if (HConvErr < err) { // || dTa < 1.0e-4) {
238  if (printLvl > 0) {
239  plogf(" equilibrate_HP: CONVERGENCE: Hfinal = %g Tfinal = %g, Its = %d \n",
240  Hnow, Tnow, n);
241  plogf(" H rel error = %g, cp = %g, HConvErr = %g\n",
242  Herr, cpb, HConvErr);
243  }
244  goto done;
245  }
246  double Tnew = Tnow + dT;
247  if (Tnew < 0.0) {
248  Tnew = 0.5*Tnow;
249  }
250  m_mix->setTemperature(Tnew);
251  } catch (CanteraError err) {
252  if (!estimateEquil) {
253  strt = -1;
254  } else {
255  double Tnew = 0.5*(Tnow + Thigh);
256  if (fabs(Tnew - Tnow) < 1.0) {
257  Tnew = Tnow + 1.0;
258  }
259  m_mix->setTemperature(Tnew);
260  }
261  }
262  }
263  throw CanteraError("MultiPhase::equilibrate_HP",
264  "No convergence for T");
265 done:
266  ;
267  return iSuccess;
268 }

753  VolPhase->m_eqnState = VCS_EOS_CONSTANT;
754  } else if (eos == "StoichSubstance") {
755  VolPhase->m_eqnState = VCS_EOS_STOICH_SUB;
756  } else if (eos == "IdealSolidSoln") {

823  // VolPhase->IndSpecies[]
