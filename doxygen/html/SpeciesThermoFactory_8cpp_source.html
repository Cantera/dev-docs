39 //! Examine the types of species thermo parameterizations,
40 //! and return a flag indicating the type of reference state thermo manager
41 //! that will be needed in order to evaluate them all.
42 /*!
43  *
44  * @param spDataNodeList This vector contains a list
45  * of species XML nodes that will be in the phase
46  * @param has_nasa Return int that indicates whether the phase has a NASA polynomial form for one of its species
47  * @param has_shomate Return int that indicates whether the phase has a SHOMATE polynomial form for one of its species
48  * @param has_simple Return int that indicates whether the phase has a SIMPLE polynomial form for one of its species
49  * @param has_other Return int that indicates whether the phase has a form for one of its species that is not one of the ones listed above.

52  * @deprecated

54 static void getSpeciesThermoTypes(std::vector<XML_Node*> & spDataNodeList,
55  int& has_nasa, int& has_shomate, int& has_simple,
56  int& has_other)
57 {
58  for (size_t n = 0; n < spDataNodeList.size(); n++) {
59  XML_Node* spNode = spDataNodeList[n];
60  if (spNode->hasChild("standardState")) {
61  string mname = spNode->child("standardState")["model"];
62  if (mname == "water" || mname == "waterIAPWS") {
63  has_other = 1;
64  continue;
65  }
66  }
67  if (spNode->hasChild("thermo")) {
68  const XML_Node& th = spNode->child("thermo");
69  if (th.hasChild("NASA")) {
70  has_nasa = 1;
71  } else if (th.hasChild("Shomate")) {
72  has_shomate = 1;
73  } else if (th.hasChild("MinEQ3")) {
74  has_shomate = 1;
75  } else if (th.hasChild("const_cp")) {
76  has_simple = 1;
77  } else if (th.hasChild("poly")) {
78  if (th.child("poly")["order"] == "1") {
79  has_simple = 1;
80  } else throw CanteraError("newSpeciesThermo",
81  "poly with order > 1 not yet supported");
82  } else if (th.hasChild("Mu0")) {
83  has_other = 1;
84  } else if (th.hasChild("NASA9")) {
85  has_other = 1;
86  } else if (th.hasChild("NASA9MULTITEMP")) {
87  has_other = 1;
88  } else if (th.hasChild("adsorbate")) {
89  has_other = 1;
90  } else {
91  has_other = 1;
92  }
93  } else {
94  throw CanteraError("getSpeciesThermoTypes:",
95  spNode->attrib("name") + " is missing the thermo XML node");
96  }
97  }

244 //! species from a set ! of XML nodes
245 /*!
246  * This is called if a 'NASA' node is found in the XML input.
247  *
248  * @param nodes vector of 1 or 2 'NASA' XML_Nodes, each defining the
249  * coefficients for a temperature range
250  */
251 static SpeciesThermoInterpType* newNasaThermoFromXML(vector<XML_Node*> nodes)
252 {
253  const XML_Node& f0 = *nodes[0];
254  bool dualRange = (nodes.size() > 1);
255 
256  double tmin0 = fpValue(f0["Tmin"]);
257  double tmax0 = fpValue(f0["Tmax"]);
258 
259  doublereal p0 = OneAtm;
260  if (f0.hasAttrib("P0")) {
261  p0 = fpValue(f0["P0"]);
262  }
263  if (f0.hasAttrib("Pref")) {
264  p0 = fpValue(f0["Pref"]);
265  }
266  p0 = OneAtm;
267 
268  double tmin1 = tmax0;
269  double tmax1 = tmin1 + 0.0001;
270  if (dualRange) {
271  tmin1 = fpValue(nodes[1]->attrib("Tmin"));
272  tmax1 = fpValue(nodes[1]->attrib("Tmax"));
273  }
274 
275  vector_fp c0, c1;
276  doublereal tmin, tmid, tmax;
277  if (fabs(tmax0 - tmin1) < 0.01) {
278  // f0 has the lower T data, and f1 the higher T data
279  tmin = tmin0;
280  tmid = tmax0;
281  tmax = tmax1;
282  getFloatArray(f0.child("floatArray"), c0, false);
283  if (dualRange) {
284  getFloatArray(nodes[1]->child("floatArray"), c1, false);
285  } else {
286  // if there is no higher range data, then copy c0 to c1.
287  c1.resize(7,0.0);
288  copy(c0.begin(), c0.end(), c1.begin());
289  }
290  } else if (fabs(tmax1 - tmin0) < 0.01) {
291  // f1 has the lower T data, and f0 the higher T data
292  tmin = tmin1;
293  tmid = tmax1;
294  tmax = tmax0;
295  getFloatArray(nodes[1]->child("floatArray"), c0, false);
296  getFloatArray(f0.child("floatArray"), c1, false);
297  } else {
298  throw CanteraError("installNasaThermo",
299  "non-continuous temperature ranges.");
300  }
301 
302  vector_fp c(15);
303  c[0] = tmid;
304  copy(c1.begin(), c1.begin()+7, c.begin() + 1); // high-T coefficients
305  copy(c0.begin(), c0.begin()+7, c.begin() + 8); // low-T coefficients
306  return newSpeciesThermoInterpType(NASA, tmin, tmax, p0, &c[0]);
307 }

309 //! Create a Shomate polynomial from an XML node giving the 'EQ3' coefficients
310 /*!
311  * This is called if a 'MinEQ3' node is found in the XML input.
312  * @param MinEQ3node The XML_Node containing the MinEQ3 parameterization
313  */
314 SpeciesThermoInterpType* newShomateForMineralEQ3(const XML_Node& MinEQ3node)
315 {
316  doublereal tmin0 = strSItoDbl(MinEQ3node["Tmin"]);
317  doublereal tmax0 = strSItoDbl(MinEQ3node["Tmax"]);
318  doublereal p0 = strSItoDbl(MinEQ3node["Pref"]);
319 
320  doublereal deltaG_formation_pr_tr =
321  getFloatDefaultUnits(MinEQ3node, "DG0_f_Pr_Tr", "cal/gmol", "actEnergy");
322  doublereal deltaH_formation_pr_tr =
323  getFloatDefaultUnits(MinEQ3node, "DH0_f_Pr_Tr", "cal/gmol", "actEnergy");
324  doublereal Entrop_pr_tr = getFloatDefaultUnits(MinEQ3node, "S0_Pr_Tr", "cal/gmol/K");
325  doublereal a = getFloatDefaultUnits(MinEQ3node, "a", "cal/gmol/K");
326  doublereal b = getFloatDefaultUnits(MinEQ3node, "b", "cal/gmol/K2");
327  doublereal c = getFloatDefaultUnits(MinEQ3node, "c", "cal-K/gmol");
328  doublereal dg = deltaG_formation_pr_tr * 4.184 * 1.0E3;
329  doublereal DHjmol = deltaH_formation_pr_tr * 1.0E3 * 4.184;
330  doublereal fac = DHjmol - dg - 298.15 * Entrop_pr_tr * 1.0E3 * 4.184;
331  doublereal Mu0_tr_pr = fac + dg;
332  doublereal e = Entrop_pr_tr * 1.0E3 * 4.184;
333  doublereal Hcalc = Mu0_tr_pr + 298.15 * e;
334 
335  /*
336  * Now calculate the shomate polynomials
337  *
338  * Cp first
339  *
340  * Shomate: (Joules / gmol / K)
341  * Cp = As + Bs * t + Cs * t*t + Ds * t*t*t + Es / (t*t)
342  * where
343  * t = temperature(Kelvin) / 1000
344  */
345  double As = a * 4.184;
346  double Bs = b * 4.184 * 1000.;
347  double Cs = 0.0;
348  double Ds = 0.0;
349  double Es = c * 4.184 / (1.0E6);
350 
351  double t = 298.15 / 1000.;
352  double H298smFs = As * t + Bs * t * t / 2.0 - Es / t;
353 
354  double HcalcS = Hcalc / 1.0E6;
355  double Fs = HcalcS - H298smFs;
356 
357  double S298smGs = As * log(t) + Bs * t - Es/(2.0*t*t);
358  double ScalcS = e / 1.0E3;
359  double Gs = ScalcS - S298smGs;
360 
361  double c0[7] = {As, Bs, Cs, Ds, Es, Fs, Gs};
362 
363  return newSpeciesThermoInterpType(SHOMATE1, tmin0, tmax0, p0, c0);
364 }

599 SpeciesThermoInterpType* newSpeciesThermoInterpType(const XML_Node& thermo)
600 {
601  // Get the children of the thermo XML node. In the next bit of code we take out the comments that
602  // may have been children of the thermo XML node by doing a selective copy.
603  // These shouldn't interfere with the algorithm at any point.
604  const std::vector<XML_Node*>& tpWC = thermo.children();
605  std::vector<XML_Node*> tp;
606  for (size_t i = 0; i < tpWC.size(); i++) {
607  if (!(tpWC[i])->isComment()) {
608  tp.push_back(tpWC[i]);
609  }
610  }
611 
612  std::string thermoType = lowercase(tp[0]->name());
613 
614  for (size_t i = 1; i < tp.size(); i++) {
615  if (lowercase(tp[i]->name()) != thermoType) {
616  throw CanteraError("newSpeciesThermoInterpType",
617  "Encounterd unsupported mixed species thermo parameterizations");
618  }
619  }
620  if ((tp.size() > 2 && thermoType != "nasa9") ||
621  (tp.size() > 1 && (thermoType == "const_cp" ||
622  thermoType == "mu0" ||
623  thermoType == "adsorbate"))) {
624  throw CanteraError("newSpeciesThermoInterpType",
625  "Too many regions in thermo parameterization.");
626  }
627 
628  std::string model = lowercase(thermo["model"]);
629  if (model == "mineraleq3") {
630  if (thermoType != "mineq3") {
631  throw CanteraError("SpeciesThermoFactory::installThermoForSpecies",
632  "confused: expected MinEQ3");
633  }
634  return newShomateForMineralEQ3(*tp[0]);
635  } else if (thermoType == "shomate") {
636  return newShomateThermoFromXML(tp);
637  } else if (thermoType == "const_cp") {
638  return newConstCpThermoFromXML(*tp[0]);
639  } else if (thermoType == "nasa") {
640  return newNasaThermoFromXML(tp);
641  } else if (thermoType == "mu0") {
642  return newMu0ThermoFromXML(*tp[0]);
643  } else if (thermoType == "nasa9") {
644  return newNasa9ThermoFromXML(tp);
645  } else if (thermoType == "adsorbate") {
646  return newAdsorbateThermoFromXML(*tp[0]);
647  } else if (thermoType == "statmech") {
648  return newStatMechThermoFromXML(*tp[0]);
649  } else if (model == "hkft" || model == "ionfromneutral") {
650  // Some PDSS species use the 'thermo' node, but don't specify a
651  // SpeciesThermoInterpType parameterization. This function needs to just
652  // ignore this data.
653  return 0;
654  } else {
655  throw CanteraError("newSpeciesThermoInterpType",
656  "Unknown species thermo model '" + thermoType + "'.");
657  }
658 }
