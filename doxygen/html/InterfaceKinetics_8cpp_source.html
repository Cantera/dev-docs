302 /**
303  * Update the equilibrium constants in molar units for all
304  * reversible reactions. Irreversible reactions have their
305  * equilibrium constant set to zero.
306  * For reactions involving charged species the equilibrium
307  * constant is adjusted according to the electrostatic potential.
308  */
309 void InterfaceKinetics::updateKc()
310 {
311  fill(m_rkcn.begin(), m_rkcn.end(), 0.0);
312 
313  //static vector_fp mu(nTotalSpecies());
314  if (m_nrev > 0) {
315  /*
316  * Get the vector of standard state electrochemical potentials for species in the Interfacial
317  * kinetics object and store it in m_mu0[]
318  */
319  size_t nsp, ik = 0;
320  doublereal rt = GasConstant*thermo(0).temperature();
321  doublereal rrt = 1.0 / rt;
322  size_t np = nPhases();
323  for (size_t n = 0; n < np; n++) {
324  thermo(n).getStandardChemPotentials(DATA_PTR(m_mu0) + m_start[n]);
325  nsp = thermo(n).nSpecies();
326  for (size_t k = 0; k < nsp; k++) {
327  m_mu0[ik] -= rt * thermo(n).logStandardConc(k);
328  m_mu0[ik] += Faraday * m_phi[n] * thermo(n).charge(k);
329  ik++;
330  }
331  }
332 
333  // compute Delta mu^0 for all reversible reactions
334  m_rxnstoich.getRevReactionDelta(m_ii, DATA_PTR(m_mu0), DATA_PTR(m_rkcn));
335 
336  for (size_t i = 0; i < m_nrev; i++) {
337  size_t irxn = m_revindex[i];
338  if (irxn == npos || irxn >= nReactions()) {
339  throw CanteraError("InterfaceKinetics",
340  "illegal value: irxn = "+int2str(irxn));
341  }
342  // WARNING this may overflow HKM
343  m_rkcn[irxn] = exp(m_rkcn[irxn]*rrt);
344  }
345  for (size_t i = 0; i != m_nirrev; ++i) {
346  m_rkcn[ m_irrev[i] ] = 0.0;
347  }
348  }
349 }

416  doublereal rrt = 1.0/rt;
417  for (size_t n = 0; n < nPhases(); n++) {

419  size_t nsp = thermo(n).nSpecies();

422  m_mu0[ik] += Faraday * m_phi[n] * thermo(n).charge(k);

512 // Apply corrections for interfacial charge transfer reactions
513 /*
514  * For reactions that transfer charge across a potential difference,
515  * the activation energies are modified by the potential difference.
516  * (see, for example, ...). This method applies this correction.
517  *
518  * @param kf Vector of forward reaction rate constants on which to have
519  * the correction applied
520  */
521 void InterfaceKinetics::applyButlerVolmerCorrection(doublereal* const kf)
522 {
523  // compute the electrical potential energy of each species
524  size_t ik = 0;
525  for (size_t n = 0; n < nPhases(); n++) {
526  size_t nsp = thermo(n).nSpecies();
527  for (size_t k = 0; k < nsp; k++) {
528  m_pot[ik] = Faraday*thermo(n).charge(k)*m_phi[n];
529  ik++;
530  }
531  }
532 
533  // Compute the change in electrical potential energy for each
534  // reaction. This will only be non-zero if a potential
535  // difference is present.
536  m_rxnstoich.getReactionDelta(m_ii, DATA_PTR(m_pot), DATA_PTR(m_rwork));
537 
538  // Modify the reaction rates. Only modify those with a
539  // non-zero activation energy. Below we decrease the
540  // activation energy below zero but in some debug modes
541  // we print out a warning message about this.
542  /*
543  * NOTE, there is some discussion about this point.
544  * Should we decrease the activation energy below zero?
545  * I don't think this has been decided in any definitive way.
546  * The treatment below is numerically more stable, however.
547  */
548  doublereal eamod;
549 #ifdef DEBUG_KIN_MODE
550  doublereal ea;
551 #endif
552  for (size_t i = 0; i < m_beta.size(); i++) {
553  size_t irxn = m_ctrxn[i];
554  eamod = m_beta[i]*m_rwork[irxn];
555  // if (eamod != 0.0 && m_E[irxn] != 0.0) {
556  if (eamod != 0.0) {
557 #ifdef DEBUG_KIN_MODE
558  ea = GasConstant * m_E[irxn];
559  if (eamod + ea < 0.0) {
560  writelog("Warning: act energy mod too large!\n");
561  writelog(" Delta phi = "+fp2str(m_rwork[irxn]/Faraday)+"\n");
562  writelog(" Delta Ea = "+fp2str(eamod)+"\n");
563  writelog(" Ea = "+fp2str(ea)+"\n");
564  for (n = 0; n < np; n++) {
565  writelog("Phase "+int2str(n)+": phi = "
566  +fp2str(m_phi[n])+"\n");
567  }
568  }
569 #endif
570  doublereal rt = GasConstant*thermo(0).temperature();
571  doublereal rrt = 1.0/rt;
572  kf[irxn] *= exp(-eamod*rrt);
573  }
574  }

577 void InterfaceKinetics::applyExchangeCurrentDensityFormulation(doublereal* const kfwd)
578 {
579  getExchangeCurrentQuantities();
580  doublereal rt = GasConstant*thermo(0).temperature();
581  doublereal rrt = 1.0/rt;
582  for (size_t i = 0; i < m_ctrxn.size(); i++) {
583  size_t irxn = m_ctrxn[i];
584  int iECDFormulation = m_ctrxn_ecdf[i];
585  if (iECDFormulation) {
586  double tmp = exp(- m_beta[i] * m_deltaG0[irxn] * rrt);
587  double tmp2 = m_ProdStanConcReac[irxn];
588  tmp *= 1.0 / tmp2 / Faraday;
589  kfwd[irxn] *= tmp;
590  }
591  }

1130 void InterfaceKinetics::installReagents(const ReactionData& r)
1131 {
1132 
1133  size_t n, ns, m;
1134  doublereal nsFlt;
1135  /*
1136  * extend temporary storage by one for this rxn.
1137  */
1138  m_ropf.push_back(0.0);
1139  m_ropr.push_back(0.0);
1140  m_ropnet.push_back(0.0);
1141  m_rkcn.push_back(0.0);
1142 
1143  /*
1144  * Obtain the current reaction index for the reaction that we
1145  * are adding. The first reaction is labeled 0.
1146  */
1147  size_t rnum = reactionNumber();
1148 
1149  // vectors rk and pk are lists of species numbers, with
1150  // repeated entries for species with stoichiometric
1151  // coefficients > 1. This allows the reaction to be defined
1152  // with unity reaction order for each reactant, and so the
1153  // faster method 'multiply' can be used to compute the rate of
1154  // progress instead of 'power'.
1155 
1156  std::vector<size_t> rk;
1157  size_t nr = r.reactants.size();
1158  for (n = 0; n < nr; n++) {
1159  nsFlt = r.rstoich[n];
1160  ns = (size_t) nsFlt;
1161  if ((doublereal) ns != nsFlt) {
1162  if (ns < 1) {
1163  ns = 1;
1164  }
1165  }
1166  /*
1167  * Add to m_rrxn. m_rrxn is a vector of maps. m_rrxn has a length
1168  * equal to the total number of species for each species, there
1169  * exists a map, with the reaction number being the key, and the
1170  * reactant stoichiometric coefficient being the value.
1171  */
1172  m_rrxn[r.reactants[n]][rnum] = nsFlt;
1173  for (m = 0; m < ns; m++) {
1174  rk.push_back(r.reactants[n]);
1175  }
1176  }
1177  /*
1178  * Now that we have rk[], we add it into the vector<vector_int> m_reactants
1179  * in the rnum index spot. Thus m_reactants[rnum] yields a vector
1180  * of reactants for the rnum'th reaction
1181  */
1182  m_reactants.push_back(rk);
1183  std::vector<size_t> pk;
1184  size_t np = r.products.size();
1185  for (n = 0; n < np; n++) {
1186  nsFlt = r.pstoich[n];
1187  ns = (size_t) nsFlt;
1188  if ((doublereal) ns != nsFlt) {
1189  if (ns < 1) {
1190  ns = 1;
1191  }
1192  }
1193  /*
1194  * Add to m_prxn. m_prxn is a vector of maps. m_prxn has a length
1195  * equal to the total number of species for each species, there
1196  * exists a map, with the reaction number being the key, and the
1197  * product stoichiometric coefficient being the value.
1198  */
1199  m_prxn[r.products[n]][rnum] = nsFlt;
1200  for (m = 0; m < ns; m++) {
1201  pk.push_back(r.products[n]);
1202  }
1203  }
1204  /*
1205  * Now that we have pk[], we add it into the vector<vector_int> m_products
1206  * in the rnum index spot. Thus m_products[rnum] yields a vector
1207  * of products for the rnum'th reaction
1208  */
1209  m_products.push_back(pk);
1210  /*
1211  * Add this reaction to the stoichiometric coefficient manager. This
1212  * calculates rates of species production from reaction rates of
1213  * progress.
1214  */
1215  m_rxnstoich.add(reactionNumber(), r);
1216  /*
1217  * register reaction in lists of reversible and irreversible rxns.
1218  */
1219  if (r.reversible) {
1220  m_revindex.push_back(reactionNumber());
1221  m_nrev++;
1222  } else {
1223  m_irrev.push_back(reactionNumber());
1224  m_nirrev++;
1225  }
1226 }
