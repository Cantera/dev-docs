658  }

660  }

716  }
717  }
718  }
719  }
720  }
721  if (m_rowScaling) {

732

735  jptr = &(*(jac.begin()));
736  for (jcol = 0; jcol < (int) neq_; jcol++) {

738  *jptr *= m_rowScales[irow];
739  jptr++;

903 int NonlinearSolver::doAffineNewtonSolve(const doublereal* const y_curr, const doublereal* const ydot_curr,
904  doublereal* const delta_y, GeneralMatrix& jac)
905 {
906  /*
907  * Notes on banded Hessian solve: The matrix for jT j has a larger band
908  * width. Both the top and bottom band widths are doubled, going from KU
909  * to KU+KL and KL to KU+KL in size. This is not an impossible increase
910  * in cost, but has to be considered.
911  */
912  bool newtonGood = true;
913  // We can default to QR here ( or not )
914  jac.useFactorAlgorithm(1);
915  int useQR = jac.factorAlgorithm();
916  // multiply the residual by -1
917  // Scale the residual if there is row scaling. Note, the matrix has already been scaled
918  if (m_rowScaling && !m_resid_scaled) {
919  for (size_t n = 0; n < neq_; n++) {
920  delta_y[n] = -m_rowScales[n] * m_resid[n];
921  }
922  m_resid_scaled = true;
923  } else {
924  for (size_t n = 0; n < neq_; n++) {
925  delta_y[n] = -m_resid[n];
926  }
927  }
928 
929  // Factor the matrix using a standard Newton solve
930  m_conditionNumber = 1.0E300;
931  int info = 0;
932  if (!jac.factored()) {
933  if (useQR) {
934  info = jac.factorQR();
935  } else {
936  info = jac.factor();
937  }
938  }
939  /*
940  * Find the condition number of the matrix
941  * If we have failed to factor, we will fall back to calculating and factoring a modified Hessian
942  */
943  if (info == 0) {
944  doublereal rcond = 0.0;
945  if (useQR) {
946  rcond = jac.rcondQR();
947  } else {
948  doublereal a1norm = jac.oneNorm();
949  rcond = jac.rcond(a1norm);
950  }
951  if (rcond > 0.0) {
952  m_conditionNumber = 1.0 / rcond;
953  }
954  } else {
955  m_conditionNumber = 1.0E300;
956  newtonGood = false;
957  if (m_print_flag >= 1) {
958  printf("\t\t doAffineNewtonSolve: ");
959  if (useQR) {
960  printf("factorQR()");
961  } else {
962  printf("factor()");
963  }
964  printf(" returned with info = %d, indicating a zero row or column\n", info);
965  }
966  }
967  bool doHessian = false;
968  if (s_doBothSolvesAndCompare) {
969  doHessian = true;
970  }
971  if (m_conditionNumber < 1.0E7) {
972  if (m_print_flag >= 4) {
973  printf("\t\t doAffineNewtonSolve: Condition number = %g during regular solve\n", m_conditionNumber);
974  }
975 
976  /*
977  * Solve the system -> This also involves inverting the matrix
978  */
979  int info = jac.solve(DATA_PTR(delta_y));
980  if (info) {
981  if (m_print_flag >= 2) {
982  printf("\t\t doAffineNewtonSolve() ERROR: QRSolve returned INFO = %d. Switching to Hessian solve\n", info);
983  }
984  doHessian = true;
985  newtonGood = false;
986  }
987  /*
988  * reverse the column scaling if there was any on a successful solve
989  */
990  if (m_colScaling) {
991  for (size_t irow = 0; irow < neq_; irow++) {
992  delta_y[irow] = delta_y[irow] * m_colScales[irow];
993  }
994  }
995 
996  } else {
997  if (jac.matrixType_ == 1) {
998  newtonGood = true;
999  if (m_print_flag >= 3) {
1000  printf("\t\t doAffineNewtonSolve() WARNING: Condition number too large, %g, But Banded Hessian solve "
1001  "not implemented yet \n", m_conditionNumber);
1002  }
1003  } else {
1004  doHessian = true;
1005  newtonGood = false;
1006  if (m_print_flag >= 3) {
1007  printf("\t\t doAffineNewtonSolve() WARNING: Condition number too large, %g. Doing a Hessian solve \n", m_conditionNumber);
1008  }
1009  }
1010  }
1011 
1012  if (doHessian) {
1013  // Store the old value for later comparison
1014  vector_fp delyNewton(neq_);
1015  for (size_t irow = 0; irow < neq_; irow++) {
1016  delyNewton[irow] = delta_y[irow];
1017  }
1018 
1019  // Get memory if not done before
1020  if (HessianPtr_ == 0) {
1021  HessianPtr_ = jac.duplMyselfAsGeneralMatrix();
1022  }
1023 
1024  /*
1025  * Calculate the symmetric Hessian
1026  */
1027  GeneralMatrix& hessian = *HessianPtr_;
1028  GeneralMatrix& jacCopy = *jacCopyPtr_;
1029  hessian.zero();
1030  if (m_rowScaling) {
1031  for (size_t i = 0; i < neq_; i++) {
1032  for (size_t j = i; j < neq_; j++) {
1033  for (size_t k = 0; k < neq_; k++) {
1034  hessian(i,j) += jacCopy(k,i) * jacCopy(k,j) * m_rowScales[k] * m_rowScales[k];

1046  }

1065  hnorm = hcol;
1066  }
1067  }

1071  hcol += fabs(hessian(j,i));
1072  }

1074  hcol += fabs(hessian(i,j));
1075  }

1094  }
1095  } else {
1096  for (size_t i = 0; i < neq_; i++) {
1097  hessian(i,i) += hcol;
1098  }

1109  }

1116  for (size_t n = 0; n < neq_; n++) {
1117  delyH[n] = -m_rowScales[n] * m_resid[n];
1118  }
1119  } else {
1120  for (size_t n = 0; n < neq_; n++) {
1121  delyH[n] = -m_resid[n];
1122  }
1123  }
1124 
1125  if (m_rowScaling) {
1126  for (size_t j = 0; j < neq_; j++) {
1127  delta_y[j] = 0.0;
1128  for (size_t i = 0; i < neq_; i++) {
1129  delta_y[j] += delyH[i] * jacCopy(i,j) * m_rowScales[i];
1130  }
1131  }
1132  } else {
1133  for (size_t j = 0; j < neq_; j++) {
1134  delta_y[j] = 0.0;
1135  for (size_t i = 0; i < neq_; i++) {
1136  delta_y[j] += delyH[i] * jacCopy(i,j);
1137  }
1138  }
1139  }

1144  */

1146  if (info) {
1147  if (m_print_flag >= 2) {
1148  printf("\t\t NonlinearSolver::doAffineNewtonSolve() ERROR: DPOTRS returned INFO = %d\n", info);
1149  }

1154  */

1252  doublereal normSoln;

1254  /*
1255  * For confirmation of the scaling factors, see Dennis and Schnabel p, 152, p, 156 and my notes
1256  *
1257  * The colFac and rowFac values are used to eliminate the scaling of the matrix from the
1258  * actual equation
1259  *
1260  * Here we calculate the steepest descent direction. This is equation (11) in the notes. It is
1261  * stored in deltaX_CP_[].The value corresponds to d_descent[].
1262  */
1263  for (size_t j = 0; j < neq_; j++) {
1264  deltaX_CP_[j] = 0.0;
1265  if (m_colScaling) {
1266  colFac = 1.0 / m_colScales[j];
1267  }
1268  for (size_t i = 0; i < neq_; i++) {
1269  if (m_rowScaling) {
1270  rowFac = 1.0 / m_rowScales[i];
1271  }
1272  deltaX_CP_[j] -= m_resid[i] * jac(i,j) * colFac * rowFac * m_ewt[j] * m_ewt[j]
1273  / (m_residWts[i] * m_residWts[i]);
1274  if (DEBUG_MODE_ENABLED) {
1275  checkFinite(deltaX_CP_[j]);
1276  }
1277  }
1278  }
1279

1281  * Calculate J_hat d_y_descent. This is formula 18 in the notes.
1282  */
1283  for (size_t i = 0; i < neq_; i++) {
1284  Jd_[i] = 0.0;
1285  if (m_rowScaling) {
1286  rowFac = 1.0 / m_rowScales[i];
1287  } else {
1288  rowFac = 1.0;
