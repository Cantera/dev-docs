32 void PureFluidPhase::initThermo()
33 {
34  if (m_tpx_name != "") {
35  m_sub.reset(tpx::newSubstance(m_tpx_name));
36  } else {
37  m_sub.reset(tpx::GetSub(m_subflag));
38  }
39  if (!m_sub) {
40  throw CanteraError("PureFluidPhase::initThermo",
41  "could not create new substance object.");
42  }
43  m_mw = m_sub->MolWt();
44  setMolecularWeight(0,m_mw);
45  double one = 1.0;
46  setMoleFractions(&one);
47  double cp0_R, h0_RT, s0_R, p;
48  double T0 = 298.15;
49  if (T0 < m_sub->Tcrit()) {
50  m_sub->Set(tpx::PropertyPair::TX, T0, 1.0);
51  p = 0.01*m_sub->P();
52  } else {
53  p = 0.001*m_sub->Pcrit();
54  }
55  p = 0.001 * p;
56  m_sub->Set(tpx::PropertyPair::TP, T0, p);
57 
58  m_spthermo.update_single(0, T0, &cp0_R, &h0_RT, &s0_R);
59  double s_R = s0_R - log(p/refPressure());
60  m_sub->setStdState(h0_RT*GasConstant*298.15/m_mw,
61  s_R*GasConstant/m_mw, T0, p);
62  debuglog("PureFluidPhase::initThermo: initialized phase "
63  +id()+"\n", m_verbose);
64 }

379 std::string PureFluidPhase::report(bool show_thermo, doublereal threshold) const
380 {
381  fmt::MemoryWriter b;
382  if (name() != "") {
383  b.write("\n {}:\n", name());
384  }
385  b.write("\n");
386  b.write(" temperature {:12.6g} K\n", temperature());
387  b.write(" pressure {:12.6g} Pa\n", pressure());
388  b.write(" density {:12.6g} kg/m^3\n", density());
389  b.write(" mean mol. weight {:12.6g} amu\n", meanMolecularWeight());
390  b.write(" vapor fraction {:12.6g}\n", vaporFraction());
391 
392  doublereal phi = electricPotential();
393  if (phi != 0.0) {
394  b.write(" potential {:12.6g} V\n", phi);
395  }
396  if (show_thermo) {
397  b.write("\n");
398  b.write(" 1 kg 1 kmol\n");
399  b.write(" ----------- ------------\n");
400  b.write(" enthalpy {:12.6g} {:12.4g} J\n",
401  enthalpy_mass(), enthalpy_mole());
402  b.write(" internal energy {:12.6g} {:12.4g} J\n",
403  intEnergy_mass(), intEnergy_mole());
404  b.write(" entropy {:12.6g} {:12.4g} J/K\n",
405  entropy_mass(), entropy_mole());
406  b.write(" Gibbs function {:12.6g} {:12.4g} J\n",
407  gibbs_mass(), gibbs_mole());
408  b.write(" heat capacity c_p {:12.6g} {:12.4g} J/K\n",
409  cp_mass(), cp_mole());
410  try {
411  b.write(" heat capacity c_v {:12.6g} {:12.4g} J/K\n",
412  cv_mass(), cv_mole());
413  } catch (NotImplementedError&) {
414  b.write(" heat capacity c_v <not implemented>\n");
415  }
416  }
417  return b.str();
418 }
