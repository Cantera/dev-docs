54 void PureFluidPhase::initThermo()
55 {
56  m_sub.reset(tpx::GetSub(m_subflag));
57  if (!m_sub) {
58  throw CanteraError("PureFluidPhase::initThermo",
59  "could not create new substance object.");
60  }
61  m_mw = m_sub->MolWt();
62  setMolecularWeight(0,m_mw);
63  double one = 1.0;
64  setMoleFractions(&one);
65  double cp0_R, h0_RT, s0_R, p;
66  double T0 = 298.15;
67  if (T0 < m_sub->Tcrit()) {
68  m_sub->Set(tpx::PropertyPair::TX, T0, 1.0);
69  p = 0.01*m_sub->P();
70  } else {
71  p = 0.001*m_sub->Pcrit();
72  }
73  p = 0.001 * p;
74  m_sub->Set(tpx::PropertyPair::TP, T0, p);
75 
76  m_spthermo->update_one(0, T0, &cp0_R, &h0_RT, &s0_R);
77  double s_R = s0_R - log(p/refPressure());
78  m_sub->setStdState(h0_RT*GasConstant*298.15/m_mw,
79  s_R*GasConstant/m_mw, T0, p);
80  debuglog("PureFluidPhase::initThermo: initialized phase "
81  +id()+"\n", m_verbose);
82 }

344 std::string PureFluidPhase::report(bool show_thermo, doublereal threshold) const
345 {
346  fmt::MemoryWriter b;
347  if (name() != "") {
348  b.write("\n {}:\n", name());
349  }
350  b.write("\n");
351  b.write(" temperature {:12.6g} K\n", temperature());
352  b.write(" pressure {:12.6g} Pa\n", pressure());
353  b.write(" density {:12.6g} kg/m^3\n", density());
354  b.write(" mean mol. weight {:12.6g} amu\n", meanMolecularWeight());
355  b.write(" vapor fraction {:12.6g}\n", vaporFraction());
356 
357  doublereal phi = electricPotential();
358  if (phi != 0.0) {
359  b.write(" potential {:12.6g} V\n", phi);
360  }
361  if (show_thermo) {
362  b.write("\n");
363  b.write(" 1 kg 1 kmol\n");
364  b.write(" ----------- ------------\n");
365  b.write(" enthalpy {:12.6g} {:12.4g} J\n",
366  enthalpy_mass(), enthalpy_mole());
367  b.write(" internal energy {:12.6g} {:12.4g} J\n",
368  intEnergy_mass(), intEnergy_mole());
369  b.write(" entropy {:12.6g} {:12.4g} J/K\n",
370  entropy_mass(), entropy_mole());
371  b.write(" Gibbs function {:12.6g} {:12.4g} J\n",
372  gibbs_mass(), gibbs_mole());
373  b.write(" heat capacity c_p {:12.6g} {:12.4g} J/K\n",
374  cp_mass(), cp_mole());
375  try {
376  b.write(" heat capacity c_v {:12.6g} {:12.4g} J/K\n",
377  cv_mass(), cv_mole());
378  } catch (CanteraError& e) {
379  e.save();
380  b.write(" heat capacity c_v <not implemented>\n");
381  }
382  }
383  return b.str();
384 }
