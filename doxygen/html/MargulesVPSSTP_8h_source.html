18 //! MargulesVPSSTP is a derived class of GibbsExcessVPSSTP that employs the
19 //! Margules approximation for the excess Gibbs free energy
20 /*!
21  * MargulesVPSSTP derives from class GibbsExcessVPSSTP which is derived from
22  * VPStandardStateTP, and overloads the virtual methods defined there with ones
23  * that use expressions appropriate for the Margules Excess Gibbs free energy
24  * approximation.
25  *
26  * The independent unknowns are pressure, temperature, and mass fraction.
27  *
28  * <HR>
29  * <H2> Specification of Species Standard State Properties </H2>
30  * <HR>
31  *
32  * All species are defined to have standard states that depend upon both the
33  * temperature and the pressure. The Margules approximation assumes symmetric
34  * standard states, where all of the standard state assume that the species are
35  * in pure component states at the temperature and pressure of the solution. I
36  * don't think it prevents, however, some species from being dilute in the
37  * solution.
38  *
39  * <HR>
40  * <H2> Specification of Solution Thermodynamic Properties </H2>
41  * <HR>
42  *
43  * The molar excess Gibbs free energy is given by the following formula which is
44  * a sum over interactions <I>i</I>. Each of the interactions are binary
45  * interactions involving two of the species in the phase, denoted, <I>Ai</I>
46  * and <I>Bi</I>. This is the generalization of the Margules formulation for a
47  * phase that has more than 2 species.
48  *
49  * \f[
50  * G^E = \sum_i \left( H_{Ei} - T S_{Ei} \right)
51  * \f]
52  * \f[
53  * H^E_i = n X_{Ai} X_{Bi} \left( h_{o,i} + h_{1,i} X_{Bi} \right)
54  * \f]
55  * \f[
56  * S^E_i = n X_{Ai} X_{Bi} \left( s_{o,i} + s_{1,i} X_{Bi} \right)
57  * \f]
58  *
59  * where n is the total moles in the solution.
60  *
61  * The activity of a species defined in the phase is given by an excess Gibbs
62  * free energy formulation.
63  *
64  * \f[
65  * a_k = \gamma_k X_k
66  * \f]
67  *
68  * where
69  *
70  * \f[
71  * R T \ln( \gamma_k )= \frac{d(n G^E)}{d(n_k)}\Bigg|_{n_i}
72  * \f]
73  *
74  * Taking the derivatives results in the following expression
75  *
76  * \f[
77  * R T \ln( \gamma_k )= \sum_i \left( \left( \delta_{Ai,k} X_{Bi} + \delta_{Bi,k} X_{Ai} - X_{Ai} X_{Bi} \right)
78  * \left( g^E_{o,i} + g^E_{1,i} X_{Bi} \right) +
79  * \left( \delta_{Bi,k} - X_{Bi} \right) X_{Ai} X_{Bi} g^E_{1,i} \right)
80  * \f]
81  * where
82  * \f$ g^E_{o,i} = h_{o,i} - T s_{o,i} \f$ and
83  * \f$ g^E_{1,i} = h_{1,i} - T s_{1,i} \f$ and where
84  * \f$ X_k \f$ is the mole fraction of species <I>k</I>.
85  *
86  * This object inherits from the class VPStandardStateTP. Therefore, the
87  * specification and calculation of all standard state and reference state
88  * values are handled at that level. Various functional forms for the standard
89  * state are permissible. The chemical potential for species <I>k</I> is equal
90  * to
91  *
92  * \f[
93  * \mu_k(T,P) = \mu^o_k(T, P) + R T \ln(\gamma_k X_k)
94  * \f]
95  *
96  * The partial molar entropy for species <I>k</I> is given by
97  *
98  * \f[
99  * \tilde{s}_k(T,P) = s^o_k(T,P) - R \ln( \gamma_k X_k )
100  * - R T \frac{d \ln(\gamma_k) }{dT}
101  * \f]
102  *
103  * The partial molar enthalpy for species <I>k</I> is given by
104  *
105  * \f[
106  * \tilde{h}_k(T,P) = h^o_k(T,P) - R T^2 \frac{d \ln(\gamma_k)}{dT}
107  * \f]
108  *
109  * The partial molar volume for species <I>k</I> is
110  *
111  * \f[
112  * \tilde V_k(T,P) = V^o_k(T,P) + R T \frac{d \ln(\gamma_k) }{dP}
113  * \f]
114  *
115  * The partial molar Heat Capacity for species <I>k</I> is
116  *
117  * \f[
118  * \tilde{C}_{p,k}(T,P) = C^o_{p,k}(T,P) - 2 R T \frac{d \ln( \gamma_k )}{dT}
119  * - R T^2 \frac{d^2 \ln(\gamma_k) }{{dT}^2}
120  * \f]
121  *
122  * <HR>
123  * <H2> %Application within Kinetics Managers </H2>
124  * <HR>
125  *
126  * \f$ C^a_k\f$ are defined such that \f$ a_k = C^a_k / C^s_k, \f$ where
127  * \f$ C^s_k \f$ is a standard concentration defined below and \f$ a_k \f$ are
128  * activities used in the thermodynamic functions. These activity (or
129  * generalized) concentrations are used by kinetics manager classes to compute
130  * the forward and reverse rates of elementary reactions. The activity
131  * concentration,\f$ C^a_k \f$,is given by the following expression.
132  *
133  * \f[
134  * C^a_k = C^s_k X_k = \frac{P}{R T} X_k
135  * \f]
136  *
137  * The standard concentration for species <I>k</I> is independent of <I>k</I> and equal to
138  *
139  * \f[
140  * C^s_k = C^s = \frac{P}{R T}
141  * \f]
142  *
143  * For example, a bulk-phase binary gas reaction between species j and k,
144  * producing a new gas species l would have the following equation for its rate
145  * of progress variable, \f$ R^1 \f$, which has units of kmol m-3 s-1.
146  *
147  * \f[
148  * R^1 = k^1 C_j^a C_k^a = k^1 (C^s a_j) (C^s a_k)
149  * \f]
150  * where
151  * \f[
152  * C_j^a = C^s a_j \mbox{\quad and \quad} C_k^a = C^s a_k
153  * \f]
154  *
155  * \f$ C_j^a \f$ is the activity concentration of species j, and \f$ C_k^a \f$
156  * is the activity concentration of species k. \f$ C^s \f$ is the standard
157  * concentration. \f$ a_j \f$ is the activity of species j which is equal to the
158  * mole fraction of j.
159  *
160  * The reverse rate constant can then be obtained from the law of microscopic
161  * reversibility and the equilibrium expression for the system.
162  *
163  * \f[
164  * \frac{a_j a_k}{ a_l} = K_a^{o,1} = \exp(\frac{\mu^o_l - \mu^o_j - \mu^o_k}{R T} )
165  * \f]
166  *
167  * \f$ K_a^{o,1} \f$ is the dimensionless form of the equilibrium constant,
168  * associated with the pressure dependent standard states \f$ \mu^o_l(T,P) \f$
169  * and their associated activities, \f$ a_l \f$, repeated here:
170  *
171  * \f[
172  * \mu_l(T,P) = \mu^o_l(T, P) + R T \log(a_l)
173  * \f]
174  *
175  * We can switch over to expressing the equilibrium constant in terms of the
176  * reference state chemical potentials
177  *
178  * \f[
179  * K_a^{o,1} = \exp(\frac{\mu^{ref}_l - \mu^{ref}_j - \mu^{ref}_k}{R T} ) * \frac{P_{ref}}{P}
180  * \f]
181  *
182  * The concentration equilibrium constant, \f$ K_c \f$, may be obtained by
183  * changing over to activity concentrations. When this is done:
184  *
185  * \f[
186  * \frac{C^a_j C^a_k}{ C^a_l} = C^o K_a^{o,1} = K_c^1 =
187  * \exp(\frac{\mu^{ref}_l - \mu^{ref}_j - \mu^{ref}_k}{R T} ) * \frac{P_{ref}}{RT}
188  * \f]
189  *
190  * %Kinetics managers will calculate the concentration equilibrium constant, \f$
191  * K_c \f$, using the second and third part of the above expression as a
192  * definition for the concentration equilibrium constant.
193  *
194  * For completeness, the pressure equilibrium constant may be obtained as well
195  *
196  * \f[
197  * \frac{P_j P_k}{ P_l P_{ref}} = K_p^1 = \exp(\frac{\mu^{ref}_l - \mu^{ref}_j - \mu^{ref}_k}{R T} )
198  * \f]
199  *
200  * \f$ K_p \f$ is the simplest form of the equilibrium constant for ideal gases.
201  * However, it isn't necessarily the simplest form of the equilibrium constant
202  * for other types of phases; \f$ K_c \f$ is used instead because it is
203  * completely general.
204  *
205  * The reverse rate of progress may be written down as
206  * \f[
207  * R^{-1} = k^{-1} C_l^a = k^{-1} (C^o a_l)
208  * \f]
209  *
210  * where we can use the concept of microscopic reversibility to write the
211  * reverse rate constant in terms of the forward reate constant and the
212  * concentration equilibrium constant, \f$ K_c \f$.
213  *
214  * \f[
215  * k^{-1} = k^1 K^1_c
216  * \f]
217  *
218  * \f$k^{-1} \f$ has units of s-1.
219  *
220  * @ingroup thermoprops
221  */
