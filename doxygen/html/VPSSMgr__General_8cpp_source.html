101 void VPSSMgr_General::_updateRefStateThermo() const
102 {
103  if (m_useTmpRefStateStorage) {
104  for (size_t k = 0; k < m_kk; k++) {
105  PDSS* kPDSS = m_PDSS_ptrs[k];
106  kPDSS->setState_TP(m_tlast, m_plast);
107  m_h0_RT[k] = kPDSS->enthalpy_RT_ref();
108  m_s0_R[k] = kPDSS->entropy_R_ref();
109  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
110  m_cp0_R[k] = kPDSS->cp_R_ref();
111  m_V0[k] = kPDSS->molarVolume_ref();
112  }
113  }

135 /*!
136  * Returns the vector of the
137  * gibbs function of the reference state at the current temperature
138  * of the solution and the reference pressure for the species.
139  * units = J/kmol
140  *
141  * @param g Output vector contain the Gibbs free energies
142  * of the reference state of the species
143  * length = m_kk, units = J/kmol.
144  */
145 void VPSSMgr_General::getGibbs_ref(doublereal* g) const
146 {
147  doublereal _rt = GasConstant * m_tlast;
148  if (m_useTmpRefStateStorage) {
149  std::copy(m_g0_RT.begin(), m_g0_RT.end(), g);
150  scale(g, g+m_kk, g, _rt);
151  } else {
152  for (size_t k = 0; k < m_kk; k++) {
153  PDSS* kPDSS = m_PDSS_ptrs[k];
154  kPDSS->setState_TP(m_tlast, m_plast);
155  double h0_RT = kPDSS->enthalpy_RT_ref();
156  double s0_R = kPDSS->entropy_R_ref();
157  g[k] = _rt * (h0_RT - s0_R);
158  }
159  }
160 }

168 PDSS*
169 VPSSMgr_General::returnPDSS_ptr(size_t k, const XML_Node& speciesNode,
170  const XML_Node* const phaseNode_ptr, bool& doST)
171 {
172  PDSS* kPDSS = 0;
173  doST = true;
174  GeneralSpeciesThermo* genSpthermo = dynamic_cast<GeneralSpeciesThermo*>(m_spthermo);
175 
176 
177  const XML_Node* const ss = speciesNode.findByName("standardState");
178  if (!ss) {
179  VPSSMgr::installSTSpecies(k, speciesNode, phaseNode_ptr);
180  kPDSS = new PDSS_IdealGas(m_vptp_ptr, k, speciesNode, *phaseNode_ptr, true);
181  return kPDSS;
182  }
183  std::string model = (*ss)["model"];
184  if (model == "constant_incompressible") {
185  VPSSMgr::installSTSpecies(k, speciesNode, phaseNode_ptr);
186  kPDSS = new PDSS_ConstVol(m_vptp_ptr, k, speciesNode, *phaseNode_ptr, true);
187  if (!kPDSS) {
188  throw CanteraError("VPSSMgr_General::returnPDSS_ptr", "new PDSS_ConstVol failed");
189  }
190  } else if (model == "waterIAPWS" || model == "waterPDSS") {
191  // VPSSMgr::installSTSpecies(k, speciesNode, phaseNode_ptr);
192  kPDSS = new PDSS_Water(m_vptp_ptr, 0);
193  if (!genSpthermo) {
194  throw CanteraError("VPSSMgr_General::returnPDSS_ptr",
195  "failed dynamic cast");
196  }
197  genSpthermo->installPDSShandler(k, kPDSS, this);
198  m_useTmpRefStateStorage = false;
199  } else if (model == "HKFT") {
200  doST = false;
201  kPDSS = new PDSS_HKFT(m_vptp_ptr, k, speciesNode, *phaseNode_ptr, true);
202  if (!genSpthermo) {
203  throw CanteraError("VPSSMgr_General::returnPDSS_ptr",
204  "failed dynamic cast");
205  }
206  genSpthermo->installPDSShandler(k, kPDSS, this);
207 
208  } else if (model == "IonFromNeutral") {
209  if (!genSpthermo) {
210  throw CanteraError("VPSSMgr_General::returnPDSS_ptr",
211  "failed dynamic cast");
212  }
213  doST = false;
214  kPDSS = new PDSS_IonsFromNeutral(m_vptp_ptr, k, speciesNode, *phaseNode_ptr, true);
215  if (!kPDSS) {
216  throw CanteraError("VPSSMgr_General::returnPDSS_ptr",
217  "new PDSS_IonsFromNeutral failed");
218  }
219  genSpthermo->installPDSShandler(k, kPDSS, this);
220 
221  } else if (model == "constant" || model == "temperature_polynomial" || model == "density_temperature_polynomial") {
222  VPSSMgr::installSTSpecies(k, speciesNode, phaseNode_ptr);
223  kPDSS = new PDSS_SSVol(m_vptp_ptr, k, speciesNode, *phaseNode_ptr, true);
224  if (!kPDSS) {
225  throw CanteraError("VPSSMgr_General::returnPDSS_ptr", "new PDSS_SSVol failed");
226  }
227  } else {
228  throw CanteraError("VPSSMgr_General::returnPDSS_ptr",
229  "unknown standard state formulation: " + model);
230  }
231  return kPDSS;
232 }

234 PDSS*
235 VPSSMgr_General::createInstallPDSS(size_t k, const XML_Node& speciesNode,
236  const XML_Node* const phaseNode_ptr)
237 {
238  bool doST;
239  PDSS* kPDSS = returnPDSS_ptr(k, speciesNode, phaseNode_ptr, doST);
240  // VPSSMgr::installSTSpecies(k, speciesNode, phaseNode_ptr);
241  if (m_PDSS_ptrs.size() < k+1) {
242  m_PDSS_ptrs.resize(k+1, 0);
243  }
244  m_PDSS_ptrs[k] = kPDSS;
245  if ((k+1) >= m_kk) {
246  m_kk = k+1;
247  }
248 
249  doublereal minTemp = kPDSS->minTemp();
250  if (minTemp > m_minTemp) {
251  m_minTemp = minTemp;
252  }
253 
254  doublereal maxTemp = kPDSS->maxTemp();
255  if (maxTemp < m_maxTemp) {
256  m_maxTemp = maxTemp;
257  }
258 
259  doublereal p0 = kPDSS->refPressure();
260  if (k == 0) {
261  m_p0 = p0;
262  }
263  return kPDSS;
264 }
