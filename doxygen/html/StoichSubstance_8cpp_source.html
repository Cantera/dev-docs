150 void StoichSubstance::initThermo()
151 {
152  // Make sure there is one and only one species in this phase.
153  if (m_kk != 1) {
154  throw CanteraError("initThermo",
155  "stoichiometric substances may only contain one species.");

165 void StoichSubstance::initThermoXML(XML_Node& phaseNode, const std::string& id_)
166 {
167  // Find the Thermo XML node
168  if (!phaseNode.hasChild("thermo")) {
169  throw CanteraError("StoichSubstance::initThermoXML",
170  "no thermo XML node");
171  }
172  XML_Node& tnode = phaseNode.child("thermo");
173  std::string model = tnode["model"];
174  if (model != "StoichSubstance" && model != "StoichSubstanceSSTP") {
175  throw CanteraError("StoichSubstance::initThermoXML",
176  "thermo model attribute must be StoichSubstance");
177  }
178  double dens = getFloat(tnode, "density", "toSI");
179  setDensity(dens);
180  SingleSpeciesTP::initThermoXML(phaseNode, id_);
181 }
