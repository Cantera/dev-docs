182 void GibbsExcessVPSSTP::getActivityCoefficients(doublereal* const ac) const
183 {
184 
185  getLnActivityCoefficients(ac);
186 
187  // Protect against roundoff when taking exponentials
188  for (size_t k = 0; k < m_kk; k++) {
189  if (ac[k] > 700.) {
190  ac[k] = exp(700.0);
191  } else if (ac[k] < -700.) {
192  ac[k] = exp(-700.0);
193  } else {
194  ac[k] = exp(ac[k]);
