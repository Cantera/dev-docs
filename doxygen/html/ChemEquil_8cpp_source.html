605  adjustEloc(s, elMolesGoal);
606  }
607 
608  // Save the calculated and converged element potentials to the
609  // original ThermoPhase object.
610  s.setElementPotentials(m_lambda);
611  if (s.temperature() > s.maxTemp() + 1.0 ||
612  s.temperature() < s.minTemp() - 1.0) {
613  writelog("Warning: Temperature ({} K) outside valid range of "
614  "{} K to {} K\n",
615  s.temperature(), s.minTemp(), s.maxTemp());
616  }
617  return 0;
618  }
619  // compute the residual and the Jacobian using the current
620  // solution vector
621  equilResidual(s, x, elMolesGoal, res_trial, xval, yval);
622  f = 0.5*dot(res_trial.begin(), res_trial.end(), res_trial.begin());
623 
624  // Compute the Jacobian matrix
625  equilJacobian(s, x, elMolesGoal, jac, xval, yval);
626 
627  if (ChemEquil_print_lvl > 0) {
628  writelogf("Jacobian matrix %d:\n", iter);
629  for (size_t m = 0; m <= m_mm; m++) {
630  writelog(" [ ");
631  for (size_t n = 0; n <= m_mm; n++) {
632  writelog("{:10.5g} ", jac(m,n));
633  }
634  writelog(" ]");
635  if (m < m_mm) {
636  writelog("x_{:10s}", s.elementName(m));
637  } else if (m_eloc == m) {
638  writelog("x_ELOC");
639  } else if (m == m_skip) {
640  writelog("x_YY");
641  } else {
642  writelog("x_XX");
643  }
644  writelog(" = - ({:10.5g})\n", res_trial[m]);
645  }
646  }
647 
648  oldx = x;
649  double oldf = f;
650  scale(res_trial.begin(), res_trial.end(), res_trial.begin(), -1.0);
651 
652  // Solve the system
653  try {
654  info = solve(jac, res_trial.data());
655  } catch (CanteraError& err) {
656  s.restoreState(state);
657  throw CanteraError("equilibrate",
658  "Jacobian is singular. \nTry adding more species, "
659  "changing the elemental composition slightly, \nor removing "
660  "unused elements.\n\n" + err.getMessage());
661  }
662 
663  // find the factor by which the Newton step can be multiplied
664  // to keep the solution within bounds.
665  double fctr = 1.0;
666  for (size_t m = 0; m < nvar; m++) {
667  double newval = x[m] + res_trial[m];
668  if (newval > above[m]) {
669  fctr = std::max(0.0,
670  std::min(fctr,0.8*(above[m] - x[m])/(newval - x[m])));
671  } else if (newval < below[m]) {
672  if (m < m_mm && (m != m_skip)) {
673  res_trial[m] = -50;
674  if (x[m] < below[m] + 50.) {
675  res_trial[m] = below[m] - x[m];
676  }
677  } else {
678  fctr = std::min(fctr, 0.8*(x[m] - below[m])/(x[m] - newval));
679  }
680  }
681  // Delta Damping
682  if (m == mm && fabs(res_trial[mm]) > 0.2) {
683  fctr = std::min(fctr, 0.2/fabs(res_trial[mm]));
684  }
685  }
686  if (fctr != 1.0 && ChemEquil_print_lvl > 0) {
687  writelogf("WARNING Soln Damping because of bounds: %g\n", fctr);
688  }
689 
690  // multiply the step by the scaling factor
691  scale(res_trial.begin(), res_trial.end(), res_trial.begin(), fctr);
692 
693  if (!dampStep(s, oldx, oldf, grad, res_trial,
694  x, f, elMolesGoal , xval, yval)) {
695  fail++;
696  if (fail > 3) {
697  s.restoreState(state);
698  throw CanteraError("equilibrate",
699  "Cannot find an acceptable Newton damping coefficient.");
700  }
701  } else {
702  fail = 0;
703  }
704  }
705 
706  // no convergence
707  s.restoreState(state);
708  throw CanteraError("ChemEquil::equilibrate",
709  "no convergence in {} iterations.", options.maxIterations);
710 }

713 int ChemEquil::dampStep(thermo_t& mix, vector_fp& oldx,
714  double oldf, vector_fp& grad, vector_fp& step, vector_fp& x,
715  double& f, vector_fp& elmols, double xval, double yval)
716 {
717  // Carry out a delta damping approach on the dimensionless element
718  // potentials.
719  double damp = 1.0;
720  for (size_t m = 0; m < m_mm; m++) {
721  if (m == m_eloc) {
722  if (step[m] > 1.25) {
723  damp = std::min(damp, 1.25 /step[m]);
724  }
725  if (step[m] < -1.25) {
726  damp = std::min(damp, -1.25 / step[m]);
727  }
728  } else {
729  if (step[m] > 0.75) {
730  damp = std::min(damp, 0.75 /step[m]);
731  }
732  if (step[m] < -0.75) {
733  damp = std::min(damp, -0.75 / step[m]);
734  }
735  }
736  }
737 
738  // Update the solution unknown
739  for (size_t m = 0; m < x.size(); m++) {
740  x[m] = oldx[m] + damp * step[m];
741  }
742  if (ChemEquil_print_lvl > 0) {
743  writelogf("Solution Unknowns: damp = %g\n", damp);
744  writelog(" X_new X_old Step\n");
745  for (size_t m = 0; m < m_mm; m++) {
746  writelogf(" % -10.5g % -10.5g % -10.5g\n", x[m], oldx[m], step[m]);
747  }
748  }
749  return 1;

752 void ChemEquil::equilResidual(thermo_t& s, const vector_fp& x,
753  const vector_fp& elmFracGoal, vector_fp& resid,
754  doublereal xval, doublereal yval, int loglevel)
755 {
756  setToEquilState(s, x, exp(x[m_mm]));
757 
758  // residuals are the total element moles
759  vector_fp& elmFrac = m_elementmolefracs;
760  for (size_t n = 0; n < m_mm; n++) {
761  size_t m = m_orderVectorElements[n];
762  // drive element potential for absent elements to -1000
763  if (elmFracGoal[m] < m_elemFracCutoff && m != m_eloc) {
764  resid[m] = x[m] + 1000.0;
765  } else if (n >= m_nComponents) {
766  resid[m] = x[m];
767  } else {
768  // Change the calculation for small element number, using
769  // L'Hopital's rule. The log formulation is unstable.
770  if (elmFracGoal[m] < 1.0E-10 || elmFrac[m] < 1.0E-10 || m == m_eloc) {
771  resid[m] = elmFracGoal[m] - elmFrac[m];
772  } else {
773  resid[m] = log((1.0 + elmFracGoal[m]) / (1.0 + elmFrac[m]));
774  }
775  }
776  }
777 
778  if (ChemEquil_print_lvl > 0 && !m_doResPerturb) {
779  writelog("Residual: ElFracGoal ElFracCurrent Resid\n");
780  for (size_t n = 0; n < m_mm; n++) {
781  writelogf(" % -14.7E % -14.7E % -10.5E\n",
782  elmFracGoal[n], elmFrac[n], resid[n]);
783  }
784  }
785 
786  double xx = m_p1(s);
787  double yy = m_p2(s);
788  resid[m_mm] = xx/xval - 1.0;
789  resid[m_skip] = yy/yval - 1.0;
790 
791  if (ChemEquil_print_lvl > 0 && !m_doResPerturb) {
792  writelog(" Goal Xvalue Resid\n");
793  writelogf(" XX : % -14.7E % -14.7E % -10.5E\n", xval, xx, resid[m_mm]);
794  writelogf(" YY(%1d): % -14.7E % -14.7E % -10.5E\n", m_skip, yval, yy, resid[m_skip]);
795  }
796 }

798 void ChemEquil::equilJacobian(thermo_t& s, vector_fp& x,
799  const vector_fp& elmols, DenseMatrix& jac,
800  doublereal xval, doublereal yval, int loglevel)
801 {
802  vector_fp& r0 = m_jwork1;
803  vector_fp& r1 = m_jwork2;
804  size_t len = x.size();
805  r0.resize(len);
806  r1.resize(len);
807  doublereal atol = 1.e-10;
808 
809  equilResidual(s, x, elmols, r0, xval, yval, loglevel-1);
810 
811  m_doResPerturb = false;
812  for (size_t n = 0; n < len; n++) {
813  double xsave = x[n];
814  double dx = std::max(atol, fabs(xsave) * 1.0E-7);
815  x[n] = xsave + dx;
816  dx = x[n] - xsave;
817  double rdx = 1.0/dx;
818 
819  // calculate perturbed residual
820  equilResidual(s, x, elmols, r1, xval, yval, loglevel-1);
821 
822  // compute nth column of Jacobian
823  for (size_t m = 0; m < x.size(); m++) {
824  jac(m, n) = (r1[m] - r0[m])*rdx;
825  }
826  x[n] = xsave;
827  }
828  m_doResPerturb = false;
829 }

831 double ChemEquil::calcEmoles(thermo_t& s, vector_fp& x, const double& n_t,
832  const vector_fp& Xmol_i_calc,

867 int ChemEquil::estimateEP_Brinkley(thermo_t& s, vector_fp& x,
868  vector_fp& elMoles)
869 {
870  // Before we do anything, we will save the state of the solution. Then, if
871  // things go drastically wrong, we will restore the saved state.
872  vector_fp state;
873  s.saveState(state);
874  bool modifiedMatrix = false;
875  size_t neq = m_mm+1;
876  int retn = 1;
877  DenseMatrix a1(neq, neq, 0.0);
878  vector_fp b(neq, 0.0);
879  vector_fp n_i(m_kk,0.0);
880  vector_fp n_i_calc(m_kk,0.0);
881  vector_fp actCoeff(m_kk, 1.0);
882  double beta = 1.0;
883 
884  s.getMoleFractions(n_i.data());
885  double pressureConst = s.pressure();
886  vector_fp Xmol_i_calc = n_i;
887 
888  vector_fp x_old(m_mm+1, 0.0);
889  vector_fp resid(m_mm+1, 0.0);
890  vector_int lumpSum(m_mm+1, 0);
891 
892  // Get the nondimensional Gibbs functions for the species at their standard
893  // states of solution at the current T and P of the solution.
894  s.getGibbs_RT(m_muSS_RT.data());
895 
896  vector_fp eMolesCalc(m_mm, 0.0);
897  vector_fp eMolesFix(m_mm, 0.0);
898  double elMolesTotal = 0.0;
899  for (size_t m = 0; m < m_mm; m++) {
900  elMolesTotal += elMoles[m];
901  for (size_t k = 0; k < m_kk; k++) {
902  eMolesFix[m] += nAtoms(k,m) * n_i[k];
903  }
904  }
905 
906  for (size_t m = 0; m < m_mm; m++) {
907  if (elMoles[m] > 1.0E-70) {
908  x[m] = clip(x[m], -100.0, 50.0);
909  } else {
910  x[m] = clip(x[m], -1000.0, 50.0);
911  }
912  }
913 
914  double n_t = 0.0;
915  double nAtomsMax = 1.0;
916  s.setMoleFractions(Xmol_i_calc.data());
917  s.setPressure(pressureConst);
918  s.getActivityCoefficients(actCoeff.data());
919  for (size_t k = 0; k < m_kk; k++) {
920  double tmp = - (m_muSS_RT[k] + log(actCoeff[k]));
921  double sum2 = 0.0;
922  for (size_t m = 0; m < m_mm; m++) {
923  double sum = nAtoms(k,m);
924  tmp += sum * x[m];
925  sum2 += sum;
926  nAtomsMax = std::max(nAtomsMax, sum2);
927  }
928  if (tmp > 100.) {
929  n_t += 2.8E43;
930  } else {
931  n_t += exp(tmp);
932  }
933  }
934 
935  if (ChemEquil_print_lvl > 0) {
936  writelog("estimateEP_Brinkley::\n\n");
937  writelogf("temp = %g\n", s.temperature());
938  writelogf("pres = %g\n", s.pressure());
939  writelog("Initial mole numbers and mu_SS:\n");
940  writelog(" Name MoleNum mu_SS actCoeff\n");
941  for (size_t k = 0; k < m_kk; k++) {
942  writelogf("%15s %13.5g %13.5g %13.5g\n",
943  s.speciesName(k), n_i[k], m_muSS_RT[k], actCoeff[k]);
944  }
945  writelogf("Initial n_t = %10.5g\n", n_t);
946  writelog("Comparison of Goal Element Abundance with Initial Guess:\n");
947  writelog(" eName eCurrent eGoal\n");
948  for (size_t m = 0; m < m_mm; m++) {
949  writelogf("%5s %13.5g %13.5g\n",
950  s.elementName(m), eMolesFix[m], elMoles[m]);
951  }
952  }
953  for (size_t m = 0; m < m_mm; m++) {
954  if (m != m_eloc && elMoles[m] <= options.absElemTol) {
955  x[m] = -200.;
956  }
957  }
958 
959  // Main Loop.
960  for (int iter = 0; iter < 20* options.maxIterations; iter++) {
961  // Save the old solution
962  for (size_t m = 0; m < m_mm; m++) {
963  x_old[m] = x[m];
964  }
965  x_old[m_mm] = n_t;
966  // Calculate the mole numbers of species
967  if (ChemEquil_print_lvl > 0) {
968  writelogf("START ITERATION %d:\n", iter);
969  }
970  // Calculate the mole numbers of species and elements.
971  double n_t_calc = calcEmoles(s, x, n_t, Xmol_i_calc, eMolesCalc, n_i_calc,
972  pressureConst);
973 
974  for (size_t k = 0; k < m_kk; k++) {
975  Xmol_i_calc[k] = n_i_calc[k]/n_t_calc;
976  }
977 
978  if (ChemEquil_print_lvl > 0) {
979  writelog(" Species: Calculated_Moles Calculated_Mole_Fraction\n");
980  for (size_t k = 0; k < m_kk; k++) {
981  writelogf("%15s: %10.5g %10.5g\n",
982  s.speciesName(k), n_i_calc[k], Xmol_i_calc[k]);
983  }
984  writelogf("%15s: %10.5g\n", "Total Molar Sum", n_t_calc);
985  writelogf("(iter %d) element moles bal: Goal Calculated\n", iter);
986  for (size_t m = 0; m < m_mm; m++) {
987  writelogf(" %8s: %10.5g %10.5g \n",
988  s.elementName(m), elMoles[m], eMolesCalc[m]);
989  }
990  }
991 
992  bool normalStep = true;
993  // Decide if we are to do a normal step or a modified step
994  size_t iM = npos;
995  for (size_t m = 0; m < m_mm; m++) {
996  if (elMoles[m] > 0.001 * elMolesTotal) {
997  if (eMolesCalc[m] > 1000. * elMoles[m]) {
998  normalStep = false;
999  iM = m;
1000  }
1001  if (1000 * eMolesCalc[m] < elMoles[m]) {
1002  normalStep = false;
1003  iM = m;
1004  }
1005  }
1006  }
1007  if (ChemEquil_print_lvl > 0 && !normalStep) {
1008  writelogf(" NOTE: iter(%d) Doing an abnormal step due to row %d\n", iter, iM);
1009  }
1010  if (!normalStep) {
1011  beta = 1.0;
1012  resid[m_mm] = 0.0;
1013  for (size_t im = 0; im < m_mm; im++) {
1014  size_t m = m_orderVectorElements[im];
1015  resid[m] = 0.0;
1016  if (im < m_nComponents && elMoles[m] > 0.001 * elMolesTotal) {
1017  if (eMolesCalc[m] > 1000. * elMoles[m]) {
1018  resid[m] = -0.5;
1019  resid[m_mm] -= 0.5;
1020  }
1021  if (1000 * eMolesCalc[m] < elMoles[m]) {
1022  resid[m] = 0.5;
1023  resid[m_mm] += 0.5;
1024  }
1025  }
1026  }
1027  if (n_t < (elMolesTotal / nAtomsMax)) {
1028  if (resid[m_mm] < 0.0) {
1029  resid[m_mm] = 0.1;
1030  }
1031  } else if (n_t > elMolesTotal) {
1032  resid[m_mm] = std::min(resid[m_mm], 0.0);
1033  }
1034  } else {
1035  // Determine whether the matrix should be dumbed down because the
1036  // coefficient matrix of species (with significant concentrations)
1037  // is rank deficient.
1038  //
1039  // The basic idea is that at any time during the calculation only a
1040  // small subset of species with sufficient concentration matters. If
1041  // the rank of the element coefficient matrix for that subset of
1042  // species is less than the number of elements, then the matrix
1043  // created by the Brinkley method below may become singular.
1044  //
1045  // The logic below looks for obvious cases where the current element
1046  // coefficient matrix is rank deficient.
1047  //
1048  // The way around rank-deficiency is to lump-sum the corresponding
1049  // row of the matrix. Note, lump-summing seems to work very well in
1050  // terms of its stability properties, i.e., it heads in the right
1051  // direction, albeit with lousy convergence rates.
1052  //
1053  // NOTE: This probably should be extended to a full blown Gauss-
1054  // Jordan factorization scheme in the future. For Example the scheme
1055  // below would fail for the set: HCl NH4Cl, NH3. Hopefully, it's
1056  // caught by the equal rows logic below.
1057  for (size_t m = 0; m < m_mm; m++) {
1058  lumpSum[m] = 1;
1059  }
1060 
1061  double nCutoff = 1.0E-9 * n_t_calc;
1062  if (ChemEquil_print_lvl > 0) {
1063  writelog(" Lump Sum Elements Calculation: \n");
1064  }
1065  for (size_t m = 0; m < m_mm; m++) {
1066  size_t kMSp = npos;
1067  size_t kMSp2 = npos;
1068  int nSpeciesWithElem = 0;
1069  for (size_t k = 0; k < m_kk; k++) {
1070  if (n_i_calc[k] > nCutoff && fabs(nAtoms(k,m)) > 0.001) {
1071  nSpeciesWithElem++;
1072  if (kMSp != npos) {
1073  kMSp2 = k;
1074  double factor = fabs(nAtoms(kMSp,m) / nAtoms(kMSp2,m));
1075  for (size_t n = 0; n < m_mm; n++) {
1076  if (fabs(factor * nAtoms(kMSp2,n) - nAtoms(kMSp,n)) > 1.0E-8) {
1077  lumpSum[m] = 0;
1078  break;
1079  }
1080  }
1081  } else {
1082  kMSp = k;
1083  }
1084  }
1085  }
1086  if (ChemEquil_print_lvl > 0) {
1087  writelogf(" %5s %3d : %5d %5d\n",
1088  s.elementName(m), lumpSum[m], kMSp, kMSp2);
1089  }
1090  }
1091 
1092  // Formulate the matrix.
1093  for (size_t im = 0; im < m_mm; im++) {
1094  size_t m = m_orderVectorElements[im];
1095  if (im < m_nComponents) {
1096  for (size_t n = 0; n < m_mm; n++) {
1097  a1(m,n) = 0.0;
1098  for (size_t k = 0; k < m_kk; k++) {
1099  a1(m,n) += nAtoms(k,m) * nAtoms(k,n) * n_i_calc[k];
1100  }
1101  }
1102  a1(m,m_mm) = eMolesCalc[m];
1103  a1(m_mm, m) = eMolesCalc[m];
1104  } else {
1105  for (size_t n = 0; n <= m_mm; n++) {
1106  a1(m,n) = 0.0;
1107  }
1108  a1(m,m) = 1.0;
1109  }
1110  }
1111  a1(m_mm, m_mm) = 0.0;
1112 
1113  // Formulate the residual, resid, and the estimate for the
1114  // convergence criteria, sum
1115  double sum = 0.0;
1116  for (size_t im = 0; im < m_mm; im++) {
1117  size_t m = m_orderVectorElements[im];
1118  if (im < m_nComponents) {
1119  resid[m] = elMoles[m] - eMolesCalc[m];
1120  } else {
1121  resid[m] = 0.0;
1122  }
1123 
1124  // For equations with positive and negative coefficients,
1125  // (electronic charge), we must mitigate the convergence
1126  // criteria by a condition limited by finite precision of
1127  // inverting a matrix. Other equations with just positive
1128  // coefficients aren't limited by this.
1129  double tmp;
1130  if (m == m_eloc) {
1131  tmp = resid[m] / (elMoles[m] + elMolesTotal*1.0E-6 + options.absElemTol);
1132  } else {
1133  tmp = resid[m] / (elMoles[m] + options.absElemTol);
1134  }
1135  sum += tmp * tmp;
1136  }
1137 
1138  for (size_t m = 0; m < m_mm; m++) {
1139  if (a1(m,m) < 1.0E-50) {
1140  if (ChemEquil_print_lvl > 0) {
1141  writelogf(" NOTE: Diagonalizing the analytical Jac row %d\n", m);
1142  }
1143  for (size_t n = 0; n < m_mm; n++) {
1144  a1(m,n) = 0.0;
1145  }
1146  a1(m,m) = 1.0;
