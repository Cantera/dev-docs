213 /*
214  * calcMolalities():
215  * We calculate the vector of molalities of the species
216  * in the phase and store the result internally:
217  * \f[
218  * m_i = (n_i) / (1000 * M_o * n_{o,p})
219  * \f]
220  * where
221  * - \f$ M_o \f$ is the molecular weight of the solvent
222  * - \f$ n_o \f$ is the mole fraction of the solvent
223  * - \f$ n_i \f$ is the mole fraction of the solute.
224  * - \f$ n_{o,p} = max (n_{o, min}, n_o) \f$
225  * - \f$ n_{o,min} \f$ = minimum mole fraction of solvent allowed
226  * in the denominator.
227  */
228 void MolalityVPSSTP::calcMolalities() const
229 {
230  getMoleFractions(DATA_PTR(m_molalities));
231  double xmolSolvent = m_molalities[m_indexSolvent];
232  if (xmolSolvent < m_xmolSolventMIN) {
233  xmolSolvent = m_xmolSolventMIN;
234  }
235  double denomInv = 1.0/ (m_Mnaught * xmolSolvent);
236  for (size_t k = 0; k < m_kk; k++) {
237  m_molalities[k] *= denomInv;
238  }
239 }

264 /*
265  * setMolalities():
266  * We are supplied with the molalities of all of the
267  * solute species. We then calculate the mole fractions of all
268  * species and update the ThermoPhase object.
269  *
270  * m_i = (n_i) / (W_o/1000 * n_o_p)
271  *
272  * where M_o is the molecular weight of the solvent
273  * n_o is the mole fraction of the solvent
274  * n_i is the mole fraction of the solute.
275  * n_o_p = max (n_o_min, n_o)
276  * n_o_min = minimum mole fraction of solvent allowed
277  * in the denominator.
278  */
279 void MolalityVPSSTP::setMolalities(const doublereal* const molal)
280 {
281 
282  double Lsum = 1.0 / m_Mnaught;
283  for (size_t k = 1; k < m_kk; k++) {
284  m_molalities[k] = molal[k];
285  Lsum += molal[k];
286  }
287  double tmp = 1.0 / Lsum;
288  m_molalities[m_indexSolvent] = tmp / m_Mnaught;
289  double sum = m_molalities[m_indexSolvent];
290  for (size_t k = 1; k < m_kk; k++) {
291  m_molalities[k] = tmp * molal[k];
292  sum += m_molalities[k];
293  }
294  if (sum != 1.0) {
295  tmp = 1.0 / sum;
296  for (size_t k = 0; k < m_kk; k++) {
297  m_molalities[k] *= tmp;
298  }
299  }
300  setMoleFractions(DATA_PTR(m_molalities));
301  /*
302  * Essentially we don't trust the input: We calculate
303  * the molalities from the mole fractions that we
304  * just obtained.
305  */
306  calcMolalities();
307 }

309 /*
310  * setMolalitiesByName()
311  *
312  * This routine sets the molalities by name
313  * HKM -> Might need to be more complicated here, setting
314  * neutrals so that the existing mole fractions are
315  * preserved.
316  */
317 void MolalityVPSSTP::setMolalitiesByName(compositionMap& mMap)
318 {
319  size_t kk = nSpecies();
320  doublereal x;
321  /*
322  * Get a vector of mole fractions
323  */
324  vector_fp mf(kk, 0.0);
325  getMoleFractions(DATA_PTR(mf));
326  double xmolS = mf[m_indexSolvent];
327  double xmolSmin = std::max(xmolS, m_xmolSolventMIN);
328  compositionMap::iterator p;
329  for (size_t k = 0; k < kk; k++) {
330  p = mMap.find(speciesName(k));
331  if (p != mMap.end()) {
332  x = mMap[speciesName(k)];
333  if (x > 0.0) {
334  mf[k] = x * m_Mnaught * xmolSmin;
335  }
336  }
337  }
338  /*
339  * check charge neutrality
340  */
341  size_t largePos = npos;
342  double cPos = 0.0;
343  size_t largeNeg = npos;
344  double cNeg = 0.0;
345  double sum = 0.0;
346  for (size_t k = 0; k < kk; k++) {
347  double ch = charge(k);
348  if (mf[k] > 0.0) {
349  if (ch > 0.0) {
350  if (ch * mf[k] > cPos) {
351  largePos = k;
352  cPos = ch * mf[k];
353  }
354  }
355  if (ch < 0.0) {
356  if (fabs(ch) * mf[k] > cNeg) {
357  largeNeg = k;
358  cNeg = fabs(ch) * mf[k];
359  }
360  }
361  }
362  sum += mf[k] * ch;
363  }
364  if (sum != 0.0) {
365  if (sum > 0.0) {
366  if (cPos > sum) {
367  mf[largePos] -= sum / charge(largePos);
368  } else {
369  throw CanteraError("MolalityVPSSTP:setMolalitiesbyName",
370  "unbalanced charges");
371  }
372  } else {
373  if (cNeg > (-sum)) {
374  mf[largeNeg] -= (-sum) / fabs(charge(largeNeg));
375  } else {
376  throw CanteraError("MolalityVPSSTP:setMolalitiesbyName",
377  "unbalanced charges");
378  }
379  }
380 
381  }
382  sum = 0.0;
383  for (size_t k = 0; k < kk; k++) {
384  sum += mf[k];
385  }
386  sum = 1.0/sum;
387  for (size_t k = 0; k < kk; k++) {
388  mf[k] *= sum;
389  }
390  setMoleFractions(DATA_PTR(mf));
391  /*
392  * After we formally set the mole fractions, we
393  * calculate the molalities again and store it in
394  * this object.
395  */
396  calcMolalities();
397 }

512 /*
513  * osmotic coefficient:
514  *
515  * Calculate the osmotic coefficient of the solvent. Note there
516  * are lots of definitions of the osmotic coefficient floating
517  * around. We use the one defined in the Pitzer's book:
518  * (Activity Coeff in Electrolyte Solutions, K. S. Pitzer
519  * CRC Press, Boca Raton, 1991, p. 85, Eqn. 28).
520  *
521  * Definition:
522  * - sum(m_i) * Mnaught * oc = ln(activity_solvent)
523  */
524 doublereal MolalityVPSSTP::osmoticCoefficient() const
525 {
526  /*
527  * First, we calculate the activities all over again
528  */
529  vector_fp act(m_kk);
530  getActivities(DATA_PTR(act));
531  /*
532  * Then, we calculate the sum of the solvent molalities
533  */
534  double sum = 0;
535  for (size_t k = 1; k < m_kk; k++) {
536  sum += std::max(m_molalities[k], 0.0);
537  }
538  double oc = 1.0;
539  double lac = log(act[m_indexSolvent]);
540  if (sum > 1.0E-200) {
541  oc = - lac / (m_Mnaught * sum);
542  }
543  return oc;
544 }

723 // Returns the index of the Cl- species.
724 /*
725  * The Cl- species is special in the sense that its single ion
726  * molality-based activity coefficient is used in the specification
727  * of the pH scale for single ions. Therefore, we need to know
728  * what species index Cl- is. If the species isn't in the species
729  * list then this routine returns -1, and we can't use the NBS
730  * pH scale.
731  *
732  * Right now we use a restrictive interpretation. The species
733  * must be named "Cl-". It must consist of exactly one Cl and one E
734  * atom.
735  */
736 size_t MolalityVPSSTP::findCLMIndex() const
737 {
738  size_t indexCLM = npos;
739  size_t eCl = npos;
740  size_t eE = npos;
741  size_t ne = nElements();
742  string sn;
743  for (size_t e = 0; e < ne; e++) {
744  sn = elementName(e);
745  if (sn == "Cl" || sn == "CL") {
746  eCl = e;
747  break;
748  }
749  }
750  // We have failed if we can't find the Cl element index
751  if (eCl == npos) {
752  return npos;
753  }
754  for (size_t e = 0; e < ne; e++) {
755  sn = elementName(e);
756  if (sn == "E" || sn == "e") {
757  eE = e;
758  break;
759  }
760  }
761  // We have failed if we can't find the E element index
762  if (eE == npos) {
763  return npos;
764  }
765  for (size_t k = 1; k < m_kk; k++) {
766  doublereal nCl = nAtoms(k, eCl);
767  if (nCl != 1.0) {
768  continue;
769  }
770  doublereal nE = nAtoms(k, eE);
771  if (nE != 1.0) {
772  continue;
773  }
774  for (size_t e = 0; e < ne; e++) {
775  if (e != eE && e != eCl) {
776  doublereal nA = nAtoms(k, e);
777  if (nA != 0.0) {
778  continue;
779  }
780  }
781  }
782  sn = speciesName(k);
783  if (sn != "Cl-" && sn != "CL-") {
784  continue;
785  }
786 
787  indexCLM = k;
788  break;
789  }
790  return indexCLM;
791 }
