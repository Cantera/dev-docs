96 void ConstDensityThermo::getChemPotentials(doublereal* mu) const
97 {
98  doublereal vdp = (pressure() - m_spthermo->refPressure())/
99  molarDensity();
100  const vector_fp& g_RT = gibbs_RT();
101  for (size_t k = 0; k < m_kk; k++) {

129 void ConstDensityThermo::_updateThermo() const
130 {
131  doublereal tnow = temperature();
132  if (m_tlast != tnow) {
133  m_spthermo->update(tnow, &m_cp0_R[0], &m_h0_RT[0],
134  &m_s0_R[0]);
135  m_tlast = tnow;
136  for (size_t k = 0; k < m_kk; k++) {
137  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
138  }
139  m_tlast = tnow;
