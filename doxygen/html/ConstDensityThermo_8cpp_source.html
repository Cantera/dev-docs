98 void ConstDensityThermo::getChemPotentials(doublereal* mu) const
99 {
100  doublereal vdp = (pressure() - m_spthermo->refPressure())/
101  molarDensity();
102  doublereal rt = temperature() * GasConstant;
103  const vector_fp& g_RT = gibbs_RT();
104  for (size_t k = 0; k < m_kk; k++) {
105  double xx = std::max(SmallNumber, moleFraction(k));
106  mu[k] = rt*(g_RT[k] + log(xx)) + vdp;
107  }
