134 void Reactor::updateState(doublereal* y)
135 {
136  // The components of y are [0] the total mass, [1] the total volume,
137  // [2] the total internal energy, [3...K+3] are the mass fractions of each
138  // species, and [K+3...] are the coverages of surface species on each wall.
139  m_mass = y[0];
140  m_vol = y[1];
141  m_thermo->setMassFractions_NoNorm(y+3);
142 
143  if (m_energy) {
144  // Use a damped Newton's method to determine the mixture temperature.
145  // Tight tolerances are required both for Jacobian evaluation and for
146  // sensitivity analysis to work correctly.
147  doublereal U = y[2];
148  doublereal T = temperature();
149  double dT = 100;
150  double dUprev = 1e10;
151  double dU = 1e10;
152  int i = 0;
153  double damp = 1.0;
154  while (abs(dT / T) > 10 * DBL_EPSILON) {
155  dUprev = dU;
156  m_thermo->setState_TR(T, m_mass / m_vol);
157  double dUdT = m_thermo->cv_mass() * m_mass;
158  dU = m_thermo->intEnergy_mass() * m_mass - U;
159  dT = dU / dUdT;
160  // Reduce the damping coefficient if the magnitude of the error
161  // isn't decreasing
162  if (std::abs(dU) < std::abs(dUprev)) {
163  damp = 1.0;
164  } else {
165  damp *= 0.8;
166  }
167  dT = std::min(dT, 0.5 * T) * damp;
168  T -= dT;
169  i++;
170  if (i > 100) {
171  throw CanteraError("Reactor::updateState",
172  "no convergence\nU/m = {}\nT = {}\nrho = {}\n",
173  U / m_mass, T, m_mass / m_vol);
174  }
175  }
176  } else {
177  m_thermo->setDensity(m_mass/m_vol);
178  }
179 
180  updateSurfaceState(y + m_nsp + 3);
181 
182  // save parameters needed by other connected reactors
183  m_enthalpy = m_thermo->enthalpy_mass();
184  m_pressure = m_thermo->pressure();
185  m_intEnergy = m_thermo->intEnergy_mass();
186  m_thermo->saveState(m_state);
187 }

201 void Reactor::evalEqs(doublereal time, doublereal* y,
202  doublereal* ydot, doublereal* params)
203 {
204  double dmdt = 0.0; // dm/dt (gas phase)
205  double* dYdt = ydot + 3;
206 
207  m_thermo->restoreState(m_state);
208  applySensitivity(params);
209  evalWalls(time);
210  double mdot_surf = evalSurfaces(time, ydot + m_nsp + 3);
211  dmdt += mdot_surf; // mass added to gas phase from surface reations
212 
213  // volume equation
214  ydot[1] = m_vdot;
215 
216  const vector_fp& mw = m_thermo->molecularWeights();
217  const doublereal* Y = m_thermo->massFractions();
218 
219  if (m_chem) {
220  m_kin->getNetProductionRates(&m_wdot[0]); // "omega dot"
221  }
222 
223  for (size_t k = 0; k < m_nsp; k++) {
224  // production in gas phase and from surfaces
225  dYdt[k] = (m_wdot[k] * m_vol + m_sdot[k]) * mw[k] / m_mass;
226  // dilution by net surface mass flux
227  dYdt[k] -= Y[k] * mdot_surf / m_mass;
228  }
229 
230  // Energy equation.
231  // \f[
232  // \dot U = -P\dot V + A \dot q + \dot m_{in} h_{in} - \dot m_{out} h.
233  // \f]
234  if (m_energy) {
235  ydot[2] = - m_thermo->pressure() * m_vdot - m_Q;
236  } else {
237  ydot[2] = 0.0;
238  }
239 
240  // add terms for outlets
241  for (size_t i = 0; i < m_outlet.size(); i++) {
242  double mdot_out = m_outlet[i]->massFlowRate(time);
243  dmdt -= mdot_out; // mass flow out of system
244  if (m_energy) {
245  ydot[2] -= mdot_out * m_enthalpy;
246  }
247  }
248 
249  // add terms for inlets
250  for (size_t i = 0; i < m_inlet.size(); i++) {
251  double mdot_in = m_inlet[i]->massFlowRate(time);
252  dmdt += mdot_in; // mass flow into system
253  for (size_t n = 0; n < m_nsp; n++) {
254  double mdot_spec = m_inlet[i]->outletSpeciesMassFlowRate(n);
255  // flow of species into system and dilution by other species
256  dYdt[n] += (mdot_spec - mdot_in * Y[n]) / m_mass;
257  }
258  if (m_energy) {
259  ydot[2] += mdot_in * m_inlet[i]->enthalpy_mass();
260  }
261  }
262 
263  ydot[0] = dmdt;
264  resetSensitivity(params);
265 }

278 double Reactor::evalSurfaces(double t, double* ydot)
279 {
280  const vector_fp& mw = m_thermo->molecularWeights();
281  fill(m_sdot.begin(), m_sdot.end(), 0.0);
282  size_t loc = 0; // offset into ydot
283  double mdot_surf = 0.0; // net mass flux from surface
284 
285  for (size_t i = 0; i < m_wall.size(); i++) {
286  Kinetics* kin = m_wall[i]->kinetics(m_lr[i]);
287  SurfPhase* surf = m_wall[i]->surface(m_lr[i]);
288  if (surf && kin) {
289  double rs0 = 1.0/surf->siteDensity();
290  size_t nk = surf->nSpecies();
291  double sum = 0.0;
292  surf->setTemperature(m_state[0]);
293  m_wall[i]->syncCoverages(m_lr[i]);
294  kin->getNetProductionRates(&m_work[0]);
295  size_t ns = kin->surfacePhaseIndex();
296  size_t surfloc = kin->kineticsSpeciesIndex(0,ns);
297  for (size_t k = 1; k < nk; k++) {
298  ydot[loc + k] = m_work[surfloc+k]*rs0*surf->size(k);
299  sum -= ydot[loc + k];
300  }
301  ydot[loc] = sum;
302  loc += nk;
303 
304  double wallarea = m_wall[i]->area();
305  for (size_t k = 0; k < m_nsp; k++) {
306  m_sdot[k] += m_work[k]*wallarea;
307  mdot_surf += m_sdot[k] * mw[k];
308  }
309  }
310  }
311  return mdot_surf;
312 }
