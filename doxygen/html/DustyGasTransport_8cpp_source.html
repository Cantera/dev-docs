130 {

146 void DustyGasTransport::updateKnudsenDiffCoeffs()
147 {
148  if (m_knudsen_ok) {
149  return;
150  }
151  doublereal K_g = m_pore_radius * m_porosity / m_tortuosity;
152  const doublereal TwoThirds = 2.0/3.0;
153  for (size_t k = 0; k < m_nsp; k++) {
154  m_dk[k] = TwoThirds * K_g * sqrt((8.0 * GasConstant * m_temp)/
155  (Pi * m_mw[k]));
156  }
157  m_knudsen_ok = true;
158 }

160 void DustyGasTransport::eval_H_matrix()
161 {
162  updateBinaryDiffCoeffs();
163  updateKnudsenDiffCoeffs();
164  doublereal sum;
165  for (size_t k = 0; k < m_nsp; k++) {
166 
167  // evaluate off-diagonal terms
168  for (size_t l = 0; l < m_nsp; l++) {
169  m_multidiff(k,l) = -m_x[k]/m_d(k,l);
170  }
171 
172  // evaluate diagonal term
173  sum = 0.0;
174  for (size_t j = 0; j < m_nsp; j++) {
175  if (j != k) {
176  sum += m_x[j]/m_d(k,j);
177  }
178  }
179  m_multidiff(k,k) = 1.0/m_dk[k] + sum;
180  }
181 }

183 void DustyGasTransport::getMolarFluxes(const doublereal* const state1,
184  const doublereal* const state2,
185  const doublereal delta,
186  doublereal* const fluxes)
187 {
188  doublereal conc1, conc2;
189 
190  // cbar will be the average concentration between the two points
191  doublereal* const cbar = DATA_PTR(m_spwork);
192  doublereal* const gradc = DATA_PTR(m_spwork2);
193  const doublereal t1 = state1[0];
194  const doublereal t2 = state2[0];
195  const doublereal rho1 = state1[1];
196  const doublereal rho2 = state2[1];
197  const doublereal* const y1 = state1 + 2;
198  const doublereal* const y2 = state2 + 2;
199  doublereal c1sum = 0.0, c2sum = 0.0;
200 
201  for (size_t k = 0; k < m_nsp; k++) {
202  conc1 = rho1 * y1[k] / m_mw[k];
203  conc2 = rho2 * y2[k] / m_mw[k];
204  cbar[k] = 0.5*(conc1 + conc2);
205  gradc[k] = (conc2 - conc1) / delta;
206  c1sum += conc1;
207  c2sum += conc2;
208  }
209 
210  // Calculate the pressures at p1 p2 and pbar
211  doublereal p1 = c1sum * GasConstant * t1;
212  doublereal p2 = c2sum * GasConstant * t2;
213  doublereal pbar = 0.5*(p1 + p2);
214  doublereal gradp = (p2 - p1)/delta;
215  doublereal tbar = 0.5*(t1 + t2);
216 
217  m_thermo->setState_TPX(tbar, pbar, cbar);
218 
219  updateMultiDiffCoeffs();
220 
221  // Multiply m_multidiff and gradc together and store the result in fluxes[]
222  multiply(m_multidiff, gradc, fluxes);
223 
224  divide_each(cbar, cbar + m_nsp, m_dk.begin());
225 
226  // if no permeability has been specified, use result for
227  // close-packed spheres
228  double b = 0.0;
229  if (m_perm < 0.0) {
230  double p = m_porosity;
231  double d = m_diam;
232  double t = m_tortuosity;
233  b = p*p*p*d*d/(72.0*t*(1.0-p)*(1.0-p));
234  } else {
235  b = m_perm;
236  }
237  b *= gradp / m_gastran->viscosity();
238  scale(cbar, cbar + m_nsp, cbar, b);
239 
240  // Multiply m_multidiff with cbar and add it to fluxes
241  increment(m_multidiff, cbar, fluxes);
242  scale(fluxes, fluxes + m_nsp, fluxes, -1.0);
243 }
