220 // Private routine to update the dusty gas binary diffusion coefficients
221 /*
222  * The dusty gas binary diffusion coefficients \f$ D^{dg}_{i,j} \f$ are evaluated from the binary
223  * gas-phase diffusion coefficients \f$ D^{bin}_{i,j} \f$ using the following formula
224  *
225  * \f[
226  * D^{dg}_{i,j} = \frac{\phi}{\tau} D^{bin}_{i,j}
227  * \f]
228  *
229  * where \f$ \phi \f$ is the porosity of the media and \f$ \tau \f$ is the tortuosity of the media.
230  *
231  */
232 void DustyGasTransport::updateBinaryDiffCoeffs()
233 {
234  if (m_bulk_ok) {
235  return;
236  }
237 
238  // get the gaseous binary diffusion coefficients
239  m_gastran->getBinaryDiffCoeffs(m_nsp, m_d.ptrColumn(0));
240  doublereal por2tort = m_porosity / m_tortuosity;
241  for (size_t n = 0; n < m_nsp; n++) {
242  for (size_t m = 0; m < m_nsp; m++) {
243  m_d(n,m) *= por2tort;
244  }
245  }
246  m_bulk_ok = true;
247 }
248 //====================================================================================================================
249 // Private routine to update the Knudsen diffusion coefficients
250 /*
251  * The Knudsen diffusion coefficients are given by the following form
252  *
253  * \f[
254  * \mathcal{D}^{knud}_k = \frac{2}{3} \frac{r_{pore} \phi}{\tau} \left( \frac{8 R T}{\pi W_k} \right)^{1/2}
255  * \f]
256  *
257  */
258 void DustyGasTransport::updateKnudsenDiffCoeffs()
259 {
260  if (m_knudsen_ok) {
261  return;
262  }
263  doublereal K_g = m_pore_radius * m_porosity / m_tortuosity;
264  const doublereal TwoThirds = 2.0/3.0;
265  for (size_t k = 0; k < m_nsp; k++) {
266  m_dk[k] = TwoThirds * K_g * sqrt((8.0 * GasConstant * m_temp)/
267  (Pi * m_mw[k]));
268  }
269  m_knudsen_ok = true;
270 }

272 //====================================================================================================================
273 // Private routine to calculate the H matrix
274 /*
275  * The H matrix is the term we have given to the matrix of coefficients in the equation for the molar
276  * fluxes. The matrix must be inverted in order to calculate the molar fluxes.
277  *
278  * The multicomponent diffusion H matrix \f$ H_{k,l} \f$ is given by the following formulas
279  *
280  * \f[
281  * H_{k,l} = - \frac{X_k}{D^e_{k,l}}
282  * \f]
283  * \f[
284  * H_{k,k} = \frac{1}{\mathcal(D)^{e}_{k, knud}} + \sum_{j \ne k}^N{ \frac{X_j}{D^e_{k,j}} }
285  * \f]
286  */
287 void DustyGasTransport::eval_H_matrix()
288 {
289  updateBinaryDiffCoeffs();
290  updateKnudsenDiffCoeffs();
291  doublereal sum;
292  for (size_t k = 0; k < m_nsp; k++) {
293 
294  // evaluate off-diagonal terms
295  for (size_t l = 0; l < m_nsp; l++) {
296  m_multidiff(k,l) = -m_x[k]/m_d(k,l);
297  }
298 
299  // evaluate diagonal term
300  sum = 0.0;
301  for (size_t j = 0; j < m_nsp; j++) {
302  if (j != k) {
303  sum += m_x[j]/m_d(k,j);
304  }
305  }
306  m_multidiff(k,k) = 1.0/m_dk[k] + sum;
307  }
308 }

310 void DustyGasTransport::getMolarFluxes(const doublereal* const state1,
311  const doublereal* const state2,
312  const doublereal delta,
313  doublereal* const fluxes)
314 {
315 
316  doublereal conc1, conc2;
317 
318  // cbar will be the average concentration between the two points
319  doublereal* const cbar = DATA_PTR(m_spwork);
320  doublereal* const gradc = DATA_PTR(m_spwork2);
321  const doublereal t1 = state1[0];
322  const doublereal t2 = state2[0];
323  const doublereal rho1 = state1[1];
324  const doublereal rho2 = state2[1];
325  const doublereal* const y1 = state1 + 2;
326  const doublereal* const y2 = state2 + 2;
327  doublereal c1sum = 0.0, c2sum = 0.0;
328 
329  for (size_t k = 0; k < m_nsp; k++) {
330  conc1 = rho1 * y1[k] / m_mw[k];
331  conc2 = rho2 * y2[k] / m_mw[k];
332  cbar[k] = 0.5*(conc1 + conc2);
333  gradc[k] = (conc2 - conc1) / delta;
334  c1sum += conc1;
335  c2sum += conc2;
336  }
337 
338  // Calculate the pressures at p1 p2 and pbar
339  doublereal p1 = c1sum * GasConstant * t1;
340  doublereal p2 = c2sum * GasConstant * t2;
341  doublereal pbar = 0.5*(p1 + p2);
342  doublereal gradp = (p2 - p1)/delta;
343  doublereal tbar = 0.5*(t1 + t2);
344 
345  m_thermo->setState_TPX(tbar, pbar, cbar);
346 
347  updateMultiDiffCoeffs();
348 
349  // Multiply m_multidiff and gradc together and store the result in fluxes[]
350  multiply(m_multidiff, gradc, fluxes);
351 
352  divide_each(cbar, cbar + m_nsp, m_dk.begin());
353 
354  // if no permeability has been specified, use result for
355  // close-packed spheres
356  double b = 0.0;
357  if (m_perm < 0.0) {
358  double p = m_porosity;
359  double d = m_diam;
360  double t = m_tortuosity;
361  b = p*p*p*d*d/(72.0*t*(1.0-p)*(1.0-p));
362  } else {
363  b = m_perm;
364  }
365  b *= gradp / m_gastran->viscosity();
366  scale(cbar, cbar + m_nsp, cbar, b);
367 
368  // Multiply m_multidiff with cbar and add it to fluxes
369  increment(m_multidiff, cbar, fluxes);
370  scale(fluxes, fluxes + m_nsp, fluxes, -1.0);
371 }

377 void DustyGasTransport::updateMultiDiffCoeffs()

396 // Return the Multicomponent diffusion coefficients. Units: [m^2/s].
397 /*
398  * Returns the array of multicomponent diffusion coefficients.
399  *
400  * @param ld The dimension of the inner loop of d (usually equal to m_nsp)
401  * @param d flat vector of diffusion coefficients, fortran ordering.
402  * d[ld*j+i] is the D_ij diffusion coefficient (the diffusion

405 void DustyGasTransport::getMultiDiffCoeffs(const size_t ld, doublereal* const d)

408  for (size_t i = 0; i < m_nsp; i++) {
409  for (size_t j = 0; j < m_nsp; j++) {
410  d[ld*j + i] = m_multidiff(i,j);
411  }
412  }
