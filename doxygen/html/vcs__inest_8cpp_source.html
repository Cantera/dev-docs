33  pprefix);

46  if (m_elementActive[j]) {
47  double tmp = 0.0;
48  for (size_t kspec = 0; kspec < nspecies; ++kspec) {

56  plogendl();

59  // Make sure all species have positive definite mole numbers Set voltages to
60  // zero for now, until we figure out what to do
61  m_deltaMolNumSpecies.assign(m_deltaMolNumSpecies.size(), 0.0);
62  for (size_t kspec = 0; kspec < nspecies; ++kspec) {
63  if (m_speciesUnknownType[kspec] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
64  if (m_molNumSpecies_old[kspec] <= 0.0) {
65  // HKM Should eventually include logic here for non SS phases
66  if (!m_SSPhase[kspec]) {
67  m_molNumSpecies_old[kspec] = 1.0e-30;
68  }
69  }
70  } else {
71  m_molNumSpecies_old[kspec] = 0.0;
72  }
73  }
74 
75  // Now find the optimized basis that spans the stoichiometric coefficient
76  // matrix
77  bool conv;
78  vcs_basopt(false, aw, sa, sm, ss, test, &conv);
79 
80  // CALCULATE TOTAL MOLES, CHEMICAL POTENTIALS OF BASIS
81 
82  // Calculate TMoles and m_tPhaseMoles_old[]
83  vcs_tmoles();
84 
85  // m_tPhaseMoles_new[] will consist of just the component moles
86  for (size_t iph = 0; iph < m_numPhases; iph++) {
87  m_tPhaseMoles_new[iph] = TPhInertMoles[iph] + 1.0E-20;
88  }
89  for (size_t kspec = 0; kspec < m_numComponents; ++kspec) {
90  if (m_speciesUnknownType[kspec] == VCS_SPECIES_TYPE_MOLNUM) {
91  m_tPhaseMoles_new[m_phaseID[kspec]] += m_molNumSpecies_old[kspec];
92  }
93  }
94  double TMolesMultiphase = 0.0;
95  for (size_t iph = 0; iph < m_numPhases; iph++) {
96  if (! m_VolPhaseList[iph]->m_singleSpecies) {
97  TMolesMultiphase += m_tPhaseMoles_new[iph];
98  }
99  }
100  m_molNumSpecies_new = m_molNumSpecies_old;
101  for (size_t kspec = 0; kspec < m_numComponents; ++kspec) {
102  if (m_speciesUnknownType[kspec] != VCS_SPECIES_TYPE_MOLNUM) {
103  m_molNumSpecies_new[kspec] = 0.0;
104  }
105  }
106  m_feSpecies_new = m_SSfeSpecies;
107 
108  for (size_t kspec = 0; kspec < m_numComponents; ++kspec) {
109  if (m_speciesUnknownType[kspec] == VCS_SPECIES_TYPE_MOLNUM) {
110  if (! m_SSPhase[kspec]) {
111  size_t iph = m_phaseID[kspec];
112  m_feSpecies_new[kspec] += log(m_molNumSpecies_new[kspec] / m_tPhaseMoles_old[iph]);
113  }
114  } else {
115  m_molNumSpecies_new[kspec] = 0.0;
116  }
117  }
118  vcs_deltag(0, true, VCS_STATECALC_NEW);
119  if (m_debug_print_lvl >= 2) {
120  for (size_t kspec = 0; kspec < nspecies; ++kspec) {
121  plogf("%s", pprefix);
122  plogf("%-12.12s", m_speciesName[kspec]);
123  if (kspec < m_numComponents) {
124  plogf("fe* = %15.5g ff = %15.5g\n", m_feSpecies_new[kspec],
125  m_SSfeSpecies[kspec]);
126  } else {
127  plogf("fe* = %15.5g ff = %15.5g dg* = %15.5g\n",
128  m_feSpecies_new[kspec], m_SSfeSpecies[kspec], m_deltaGRxn_new[kspec-m_numComponents]);
129  }
130  }
131  }
132 
133  // ESTIMATE REACTION ADJUSTMENTS
134  vector_fp& xtphMax = m_TmpPhase;
135  vector_fp& xtphMin = m_TmpPhase2;
136  m_deltaPhaseMoles.assign(m_deltaPhaseMoles.size(), 0.0);
137  for (size_t iph = 0; iph < m_numPhases; iph++) {
138  xtphMax[iph] = log(m_tPhaseMoles_new[iph] * 1.0E32);
139  xtphMin[iph] = log(m_tPhaseMoles_new[iph] * 1.0E-32);
140  }
141  for (size_t irxn = 0; irxn < nrxn; ++irxn) {
142  size_t kspec = m_indexRxnToSpecies[irxn];
143 
144  // For single species phases, we will not estimate the mole numbers. If
145  // the phase exists, it stays. If it doesn't exist in the estimate, it
146  // doesn't come into existence here.
147  if (! m_SSPhase[kspec]) {
148  size_t iph = m_phaseID[kspec];
149  if (m_deltaGRxn_new[irxn] > xtphMax[iph]) {
150  m_deltaGRxn_new[irxn] = 0.8 * xtphMax[iph];
151  }
152  if (m_deltaGRxn_new[irxn] < xtphMin[iph]) {
153  m_deltaGRxn_new[irxn] = 0.8 * xtphMin[iph];
154  }
155 
156  // HKM -> The TMolesMultiphase is a change of mine. It more evenly
157  // distributes the initial moles amongst multiple multispecies
158  // phases according to the relative values of the standard state
159  // free energies. There is no change for problems with one
160  // multispecies phase. It cut diamond4.vin iterations down from 62
161  // to 14.
162  m_deltaMolNumSpecies[kspec] = 0.5 * (m_tPhaseMoles_new[iph] + TMolesMultiphase)
163  * exp(-m_deltaGRxn_new[irxn]);
164 
165  for (size_t k = 0; k < m_numComponents; ++k) {
166  m_deltaMolNumSpecies[k] += m_stoichCoeffRxnMatrix(k,irxn) * m_deltaMolNumSpecies[kspec];
167  }
168 
169  for (iph = 0; iph < m_numPhases; iph++) {
170  m_deltaPhaseMoles[iph] += m_deltaMolNumPhase(iph,irxn) * m_deltaMolNumSpecies[kspec];
171  }
172  }
173  }
174  if (m_debug_print_lvl >= 2) {
175  for (size_t kspec = 0; kspec < nspecies; ++kspec) {
176  if (m_speciesUnknownType[kspec] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
177  plogf("%sdirection (", pprefix);
178  plogf("%-12.12s", m_speciesName[kspec]);
179  plogf(") = %g", m_deltaMolNumSpecies[kspec]);
180  if (m_SSPhase[kspec]) {
181  if (m_molNumSpecies_old[kspec] > 0.0) {

183  } else {
184  plogf(" (ssPhase doesn't exist -> stability not checked)");
185  }
186  }
187  plogendl();
188  }
189  }
190  }
191 
192  // KEEP COMPONENT SPECIES POSITIVE
193  double par = 0.5;
194  for (size_t kspec = 0; kspec < m_numComponents; ++kspec) {
195  if (m_speciesUnknownType[kspec] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE &&
196  par < -m_deltaMolNumSpecies[kspec] / m_molNumSpecies_new[kspec]) {
197  par = -m_deltaMolNumSpecies[kspec] / m_molNumSpecies_new[kspec];
198  }
199  }
200  par = 1. / par;
201  if (par <= 1.0 && par > 0.0) {
202  par *= 0.8;
203  } else {
204  par = 1.0;
205  }
206 
207  // CALCULATE NEW MOLE NUMBERS
208  size_t lt = 0;
209  size_t ikl = 0;
210  double s1 = 0.0;
211  while (true) {
212  for (size_t kspec = 0; kspec < m_numComponents; ++kspec) {
213  if (m_speciesUnknownType[kspec] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE) {
214  m_molNumSpecies_old[kspec] = m_molNumSpecies_new[kspec] + par * m_deltaMolNumSpecies[kspec];
215  } else {
216  m_deltaMolNumSpecies[kspec] = 0.0;
217  }
218  }
219  for (size_t kspec = m_numComponents; kspec < nspecies; ++kspec) {
220  if (m_speciesUnknownType[kspec] != VCS_SPECIES_TYPE_INTERFACIALVOLTAGE &&
221  m_deltaMolNumSpecies[kspec] != 0.0) {
222  m_molNumSpecies_old[kspec] = m_deltaMolNumSpecies[kspec] * par;
223  }
224  }
225 
226  // We have a new w[] estimate, go get the TMoles and m_tPhaseMoles_old[]
227  // values
228  vcs_tmoles();
229  if (lt > 0) {
230  break;
231  }
232 
233  // CONVERGENCE FORCING SECTION
234  vcs_setFlagsVolPhases(false, VCS_STATECALC_OLD);
235  vcs_dfe(VCS_STATECALC_OLD, 0, 0, nspecies);
236  double s = 0.0;
237  for (size_t kspec = 0; kspec < nspecies; ++kspec) {
238  s += m_deltaMolNumSpecies[kspec] * m_feSpecies_old[kspec];
239  }
240  if (s == 0.0) {
241  break;
242  }
243  if (s < 0.0 && ikl == 0) {
244  break;
245  }
246 
247  // TRY HALF STEP SIZE
248  if (ikl == 0) {
249  s1 = s;
250  par *= 0.5;
251  ikl = 1;
252  continue;
253  }
254 
255  // FIT PARABOLA THROUGH HALF AND FULL STEPS
256  double xl = (1.0 - s / (s1 - s)) * 0.5;
257  if (xl < 0.0) {
258  // POOR DIRECTION, REDUCE STEP SIZE TO 0.2
259  par *= 0.2;
260  } else {
261  if (xl > 1.0) {
262  // TOO BIG A STEP, TAKE ORIGINAL FULL STEP
263  par *= 2.0;
264  } else {
265  // ACCEPT RESULTS OF FORCER
266  par = par * 2.0 * xl;
267  }
268  }
269  lt = 1;
270  }
271 
272  if (m_debug_print_lvl >= 2) {
273  plogf("%s Final Mole Numbers produced by inest:\n",
274  pprefix);
275  plogf("%s SPECIES MOLE_NUMBER\n", pprefix);
276  for (size_t kspec = 0; kspec < nspecies; ++kspec) {
277  plogf("%s ", pprefix);
278  plogf("%-12.12s", m_speciesName[kspec]);
279  plogf(" %g", m_molNumSpecies_old[kspec]);
280  plogendl();
281  }
282  }

285 int VCS_SOLVE::vcs_inest_TP()
286 {
287  int retn = 0;
288  clockWC tickTock;
289  if (m_doEstimateEquil > 0) {
290  // Calculate the elemental abundances
291  vcs_elab();
292  if (vcs_elabcheck(0)) {
293  if (m_debug_print_lvl >= 2) {
294  plogf("%s Initial guess passed element abundances on input\n", pprefix);
295  plogf("%s m_doEstimateEquil = 1 so will use the input mole "
296  "numbers as estimates", pprefix);
297  plogendl();
298  }
299  return retn;
300  } else if (m_debug_print_lvl >= 2) {
301  plogf("%s Initial guess failed element abundances on input\n", pprefix);
302  plogf("%s m_doEstimateEquil = 1 so will discard input "
303  "mole numbers and find our own estimate", pprefix);
304  plogendl();
305  }
306  }
307 
308  // Malloc temporary space for usage in this routine and in subroutines
309  vector_fp sm(m_numElemConstraints*m_numElemConstraints, 0.0);
310  vector_fp ss(m_numElemConstraints, 0.0);
311  vector_fp sa(m_numElemConstraints, 0.0);
312  vector_fp aw(m_numSpeciesTot+ m_numElemConstraints, 0.0);
313 
314  // Go get the estimate of the solution
315  if (m_debug_print_lvl >= 2) {
316  plogf("%sGo find an initial estimate for the equilibrium problem",
317  pprefix);
318  plogendl();
319  }
320  double test = -1.0E20;
321  vcs_inest(&aw[0], &sa[0], &sm[0], &ss[0], test);
322 
323  // Calculate the elemental abundances
324  vcs_elab();
325 
326  // If we still fail to achieve the correct elemental abundances, try to fix
327  // the problem again by calling the main elemental abundances fixer routine,
328  // used in the main program. This attempts to tweak the mole numbers of the
329  // component species to satisfy the element abundance constraints.
330  //
331  // Note: We won't do this unless we have to since it involves inverting a
332  // matrix.
333  bool rangeCheck = vcs_elabcheck(1);
334  if (!vcs_elabcheck(0)) {
335  if (m_debug_print_lvl >= 2) {
336  plogf("%sInitial guess failed element abundances\n", pprefix);
337  plogf("%sCall vcs_elcorr to attempt fix", pprefix);
338  plogendl();
339  }
340  vcs_elcorr(&sm[0], &aw[0]);
341  rangeCheck = vcs_elabcheck(1);
342  if (!vcs_elabcheck(0)) {
343  plogf("%sInitial guess still fails element abundance equations\n",
344  pprefix);
345  plogf("%s - Inability to ever satisfy element abundance "
346  "constraints is probable", pprefix);
347  plogendl();
348  retn = -1;
349  } else {
350  if (m_debug_print_lvl >= 2) {
351  if (rangeCheck) {
352  plogf("%sInitial guess now satisfies element abundances", pprefix);
353  plogendl();
354  } else {
355  plogf("%sElement Abundances RANGE ERROR\n", pprefix);
356  plogf("%s - Initial guess satisfies NC=%d element abundances, "
357  "BUT not NE=%d element abundances", pprefix,
358  m_numComponents, m_numElemConstraints);
359  plogendl();
360  }
361  }
362  }
363  } else {
364  if (m_debug_print_lvl >= 2) {
365  if (rangeCheck) {
366  plogf("%sInitial guess satisfies element abundances", pprefix);
367  plogendl();
368  } else {
369  plogf("%sElement Abundances RANGE ERROR\n", pprefix);
370  plogf("%s - Initial guess satisfies NC=%d element abundances, "
371  "BUT not NE=%d element abundances", pprefix,
372  m_numComponents, m_numElemConstraints);
373  plogendl();
374  }
375  }
376  }
377 
378  if (m_debug_print_lvl >= 2) {
379  plogf("%sTotal Dimensionless Gibbs Free Energy = %15.7E", pprefix,
380  vcs_Total_Gibbs(&m_molNumSpecies_old[0], &m_feSpecies_new[0],
381  &m_tPhaseMoles_old[0]));
382  plogendl();
383  }
384 
385  // Record time
386  m_VCount->T_Time_inest += tickTock.secondsWC();
387  m_VCount->T_Calls_Inest++;
388  return retn;
389 }
