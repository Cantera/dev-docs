21 doublereal quadInterp(doublereal x0, doublereal* x, doublereal* y)
22 {
23  doublereal dx21, dx32, dx31, dy32, dy21, a;
24  dx21 = x[1] - x[0];
25  dx32 = x[2] - x[1];
26  dx31 = dx21 + dx32;
27  dy32 = y[2] - y[1];
28  dy21 = y[1] - y[0];
29  a = (dx21*dy32 - dy21*dx32)/(dx21*dx31*dx32);
30  return a*(x0 - x[0])*(x0 - x[1]) + (dy21/dx21)*(x0 - x[1]) + y[1];
31 }
