86 {

135 void MaskellSolidSolnPhase::getActivityCoefficients(doublereal* ac) const
136 {
137  _updateThermo();
138  static const int cacheId = m_cache.getId();
139  CachedArray cached = m_cache.getArray(cacheId);
140  if (!cached.validate(temperature(), pressure(), stateMFNumber())) {
141  cached.value.resize(2);
142 
143  const doublereal r = moleFraction(product_species_index);
144  const doublereal pval = p(r);
145  const doublereal rfm = r * fm(r);
146  const doublereal A = (std::pow(1 - rfm, pval) * std::pow(rfm, pval) * std::pow(r - rfm, 1 - pval)) /
147  (std::pow(1 - r - rfm, 1 + pval) * (1 - r));
148  const doublereal B = pval * h_mixing / RT();
149  cached.value[product_species_index] = A * std::exp(B);
150  cached.value[reactant_species_index] = 1 / (A * r * (1-r) ) * std::exp(-B);
151  }
152  std::copy(cached.value.begin(), cached.value.end(), ac);
153 }

155 void MaskellSolidSolnPhase::getChemPotentials(doublereal* mu) const
156 {
157  _updateThermo();
158  const doublereal r = moleFraction(product_species_index);
159  const doublereal pval = p(r);
160  const doublereal rfm = r * fm(r);
161  const doublereal DgbarDr = pval * h_mixing +
162  RT() *
163  std::log( (std::pow(1 - rfm, pval) * std::pow(rfm, pval) * std::pow(r - rfm, 1 - pval) * r) /
164  (std::pow(1 - r - rfm, 1 + pval) * (1 - r)) );
165  mu[product_species_index] = RT() * m_g0_RT[product_species_index] + DgbarDr;
166  mu[reactant_species_index] = RT() * m_g0_RT[reactant_species_index] - DgbarDr;
167 }

216 void MaskellSolidSolnPhase::initThermoXML(XML_Node& phaseNode, const std::string& id_)
217 {
218  if (id_.size() > 0 && phaseNode.id() != id_) {
219  throw CanteraError("MaskellSolidSolnPhase::initThermoXML",
220  "phasenode and Id are incompatible");
221  }
222 
223  // Check on the thermo field. Must have:
224  // <thermo model="MaskellSolidSolution" />
225  if (phaseNode.hasChild("thermo")) {
226  XML_Node& thNode = phaseNode.child("thermo");
227  std::string mString = thNode.attrib("model");
228  if (lowercase(mString) != "maskellsolidsolnphase") {
229  throw CanteraError("MaskellSolidSolnPhase::initThermoXML",
230  "Unknown thermo model: " + mString);
231  }
232 
233  // Parse the enthalpy of mixing constant
234  if (thNode.hasChild("h_mix")) {
235  set_h_mix(fpValue(thNode.child("h_mix").value()));
236  } else {
237  throw CanteraError("MaskellSolidSolnPhase::initThermoXML",
238  "Mixing enthalpy parameter not specified.");
239  }
240 
241  if (thNode.hasChild("product_species")) {
242  std::string product_species_name = thNode.child("product_species").value();
243  product_species_index = static_cast<int>(speciesIndex(product_species_name));
244  if (product_species_index == -1) {
245  throw CanteraError("MaskellSolidSolnPhase::initThermoXML",
246  "Species " + product_species_name + " not found.");
247  }
248  if (product_species_index == 0) {
249  reactant_species_index = 1;
250  } else {
251  reactant_species_index = 0;
252  }
253  }
254  } else {
255  throw CanteraError("MaskellSolidSolnPhase::initThermoXML",
256  "Unspecified thermo model");
257  }
258 
259  // Confirm that the phase only contains 2 species
260  if (m_kk != 2) {
261  throw CanteraError("MaskellSolidSolnPhase::initThermoXML",
262  "MaskellSolidSolution model requires exactly 2 species.");
263  }
264 
265  // Call the base initThermo, which handles setting the initial state.
266  VPStandardStateTP::initThermoXML(phaseNode, id_);
267 }

269 void MaskellSolidSolnPhase::_updateThermo() const
270 {
271  assert(m_kk == 2);
272  static const int cacheId = m_cache.getId();
273  CachedScalar cached = m_cache.getScalar(cacheId);
274 
275  // Update the thermodynamic functions of the reference state.
276  doublereal tnow = temperature();
277  if (!cached.validate(tnow)) {
278  m_spthermo->update(tnow, m_cp0_R.data(), m_h0_RT.data(), m_s0_R.data());
279  for (size_t k = 0; k < m_kk; k++) {
280  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
281  }
282  }
283 }

290 doublereal MaskellSolidSolnPhase::fm(const doublereal r) const
291 {
292  return (1 - std::sqrt(1 - 4*r*(1-r)/s())) / (2*r);

295 doublereal MaskellSolidSolnPhase::p(const doublereal r) const
296 {
297  const doublereal sval = s();
298  return (1 - 2*r) / std::sqrt(sval*sval - 4 * sval * r + 4 * sval * r * r);
