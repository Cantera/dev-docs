149 void MaskellSolidSolnPhase::getActivityCoefficients(doublereal* ac) const
150 {
151  _updateThermo();
152  static const int cacheId = m_cache.getId();
153  CachedArray cached = m_cache.getArray(cacheId);
154  if( !cached.validate(temperature(), pressure(), stateMFNumber()) ) {
155  cached.value.resize(2);
156 
157  const doublereal r = moleFraction(product_species_index);
158  const doublereal pval = p(r);
159  const doublereal fmval = fm(r);
160  const doublereal rfm = r * fmval;
161  const doublereal RT = GasConstant * temperature();
162  const doublereal A = (std::pow(1 - rfm, pval) * std::pow(rfm, pval) * std::pow(r - rfm, 1 - pval)) /
163  (std::pow(1 - r - rfm, 1 + pval) * (1 - r));
164  const doublereal B = pval * h_mixing / RT;
165  cached.value[product_species_index] = A * std::exp(B);
166  cached.value[reactant_species_index] = 1 / (A * r * (1-r) ) * std::exp(-B);
167  }
168  std::copy(cached.value.begin(), cached.value.end(), ac);
169 }

171 void MaskellSolidSolnPhase::getChemPotentials(doublereal* mu) const
172 {
173  _updateThermo();
174  const doublereal r = moleFraction(product_species_index);
175  const doublereal pval = p(r);
176  const doublereal fmval = fm(r);
177  const doublereal rfm = r * fmval;
178  const doublereal RT = GasConstant * temperature();
179  const doublereal DgbarDr = pval * h_mixing +
180  GasConstant * temperature() *
181  std::log( (std::pow(1 - rfm, pval) * std::pow(rfm, pval) * std::pow(r - rfm, 1 - pval) * r) /
182  (std::pow(1 - r - rfm, 1 + pval) * (1 - r)) );
183  mu[product_species_index] = RT * m_g0_RT[product_species_index] + DgbarDr;
184  mu[reactant_species_index] = RT * m_g0_RT[reactant_species_index] - DgbarDr;
185 }

240 void MaskellSolidSolnPhase::initThermoXML(XML_Node& phaseNode, const std::string& id_)
241 {
242  std::string subname = "MaskellSolidSolnPhase::initThermoXML";
243  if (id_.size() > 0) {
244  std::string idp = phaseNode.id();
245  if (idp != id_) {
246  throw CanteraError(subname.c_str(),
247  "phasenode and Id are incompatible");
248  }
249  }
250 
251  /*
252  * Check on the thermo field. Must have:
253  * <thermo model="MaskellSolidSolution" />
254  */
255  if (phaseNode.hasChild("thermo")) {
256  XML_Node& thNode = phaseNode.child("thermo");
257  std::string mStringa = thNode.attrib("model");
258  std::string mString = lowercase(mStringa);
259  if (mString != "maskellsolidsolnphase") {
260  throw CanteraError(subname.c_str(),
261  "Unknown thermo model: " + mStringa);
262  }
263 
264 
265  /*
266  * Parse the enthalpy of mixing constant
267  */
268  if (thNode.hasChild("h_mix")) {
269  XML_Node& scNode = thNode.child("h_mix");
270  set_h_mix(fpValue(scNode.value()));
271  } else {
272  throw CanteraError(subname.c_str(),
273  "Mixing enthalpy parameter not specified.");
274  }
275 
276  if (thNode.hasChild("product_species")) {
277  XML_Node& scNode = thNode.child("product_species");
278  std::string product_species_name = scNode.value();
279  product_species_index = speciesIndex(product_species_name);
280  if( product_species_index == static_cast<int>(npos) )
281  {
282  throw CanteraError(subname.c_str(),
283  "Species " + product_species_name + " not found.");
284  }
285  if( product_species_index == 0 )
286  {
287  reactant_species_index = 1;
288  } else {
289  reactant_species_index = 0;
290  }
291  }
292 
293  } else {
294  throw CanteraError(subname.c_str(),
295  "Unspecified thermo model");
296  }
297 
298 
299  // Confirm that the phase only contains 2 species
300  if( m_kk != 2 )
301  {
302  throw CanteraError( subname.c_str(), "MaskellSolidSolution model requires exactly 2 species.");
303  }
304 
305  /*
306  * Call the base initThermo, which handles setting the initial
307  * state.
308  */
309  VPStandardStateTP::initThermoXML(phaseNode, id_);
310 }

312 void MaskellSolidSolnPhase::_updateThermo() const
313 {
314  assert(m_kk == 2);
315  static const int cacheId = m_cache.getId();
316  CachedScalar cached = m_cache.getScalar(cacheId);
317  /*
318  * Update the thermodynamic functions of the reference state.
319  */
320  doublereal tnow = temperature();
321  if( !cached.validate(tnow) )
322  {
323  m_spthermo->update(tnow, DATA_PTR(m_cp0_R), DATA_PTR(m_h0_RT),
324  DATA_PTR(m_s0_R));
325  for (size_t k = 0; k < m_kk; k++) {
326  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
327  }
328  }
329 }

336 doublereal MaskellSolidSolnPhase::fm(const doublereal r) const
337 {
338  const doublereal sval = s();

342 doublereal MaskellSolidSolnPhase::p(const doublereal r) const
343 {
344  const doublereal sval = s();
345  return (1 - 2*r) / std::sqrt(sval*sval - 4 * sval * r + 4 * sval * r * r);
