19 //! Specific Class to handle tortuosity corrections for diffusive transport
20 //! in porous media using the Bruggeman exponent
21 /*!
22  * Class to compute the increase in diffusive path length associated with
23  * tortuous path diffusion through, for example, porous media. This base class
24  * implementation relates tortuosity to volume fraction through a power-law
25  * relationship that goes back to Bruggeman. The exponent is referred to as the
26  * Bruggeman exponent.
27  *
28  * Note that the total diffusional flux is generally written as
29  *
30  * \f[
31  * \frac{ \phi C_T D_i \nabla X_i }{ \tau^2 }
32  * \f]
33  *
34  * where \f$ \phi \f$ is the volume fraction of the transported phase,
35  * \f$ \tau \f$ is referred to as the tortuosity. (Other variables are
36  * \f$ C_T \f$, the total concentration, \f$ D_i \f$, the diffusion coefficient,
37  * and \f$ X_i \f$, the mole fraction with Fickian transport assumed.)
38  */
39 class Tortuosity
40 {
41 public:
42  //! Default constructor uses Bruggeman exponent of 1.5
43  Tortuosity(double setPower = 1.5) : expBrug_(setPower) {
44  }
45 
46  //! The tortuosity factor models the effective increase in the
47  //! diffusive transport length.
48  /**
49  * This method returns \f$ 1/\tau^2 \f$ in the description of the
50  * flux \f$ \phi C_T D_i \nabla X_i / \tau^2 \f$.
51  */
52  virtual double tortuosityFactor(double porosity) {
53  return pow(porosity, expBrug_ - 1.0);
54  }
55 
56  //! The McMillan number is the ratio of the flux-like
57  //! variable to the value it would have without porous flow.
58  /**
59  * The McMillan number combines the effect of tortuosity and volume fraction
60  * of the transported phase. The net flux observed is then the product of
61  * the McMillan number and the non-porous transport rate. For a
62  * conductivity in a non-porous media, \f$ \kappa_0 \f$, the conductivity in
63  * the porous media would be \f$ \kappa = (\rm McMillan) \kappa_0 \f$.
64  */
65  virtual double McMillan(double porosity) {
66  return pow(porosity, expBrug_);
67  }
68 
69 protected:
70  //! Bruggeman exponent: power to which the tortuosity depends on the volume
71  //! fraction
72  double expBrug_;
73 };

92  return pow((porosity - percolationThreshold_)

95  }
96 
97 protected:

101  /**
102  * The McMillan number (ratio of effective conductivity
103  * to non-porous conductivity) is
104  * \f[
105  * \kappa/\kappa_0 = ( \phi - \phi_c )^\mu
106  * \f]
107  * where \f$ \mu \f$ is the conductivity exponent (typical values range from
108  * 1.6 to 2.0) and \f$ \phi_c \f$ is the percolation threshold.
109  */
110  double conductivityExponent_;
111 };

114 /**
115  * This class implements transport coefficient corrections appropriate for
116  * porous media with a dispersed phase. This model goes back to Maxwell. The
117  * formula for the conductivity is expressed in terms of the volume fraction of
118  * the continuous phase, \f$ \phi \f$, and the relative conductivities of the
119  * dispersed and continuous phases, \f$ r = \kappa_d / \kappa_0 \f$. For dilute
120  * particle suspensions the effective conductivity is
121  * \f[
122  * \kappa / \kappa_0 = 1 + 3 ( 1 - \phi ) ( r - 1 ) / ( r + 2 ) + O(\phi^2)
123  * \f]
124  */
125 class TortuosityMaxwell : public Tortuosity
126 {
127 public:
128  //! Default constructor uses Bruggeman exponent of 1.5
129  TortuosityMaxwell(double relativeConductivites = 0.0) : relativeConductivites_(relativeConductivites) {
130  }
131 
132  double tortuosityFactor(double porosity) {
133  return McMillan(porosity) / porosity;
134  }
135 
136  double McMillan(double porosity) {
137  return 1 + 3 * (1.0 - porosity) * (relativeConductivites_ - 1.0) / (relativeConductivites_ + 2);
138  }
139 
140 protected:
141  //! Relative conductivities of the dispersed and continuous phases,
142  //! `relativeConductivites_` \f$ = \kappa_d / \kappa_0 \f$.
143  double relativeConductivites_;
144 };
