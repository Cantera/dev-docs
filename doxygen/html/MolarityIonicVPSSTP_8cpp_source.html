163 void MolarityIonicVPSSTP::getPartialMolarEnthalpies(doublereal* hbar) const
164 {
165  /*
166  * Get the nondimensional standard state enthalpies
167  */
168  getEnthalpy_RT(hbar);
169  /*
170  * dimensionalize it.
171  */
172  double T = temperature();

252 void MolarityIonicVPSSTP::calcPseudoBinaryMoleFractions() const
253 {
254  size_t k;
255  size_t kCat;
256  size_t kMax;
257  doublereal sumCat;
258  doublereal sumAnion;
259  doublereal chP, chM;
260  doublereal sum = 0.0;
261  doublereal sumMax;
262  switch (PBType_) {
263  case PBTYPE_PASSTHROUGH:
264  for (k = 0; k < m_kk; k++) {
265  PBMoleFractions_[k] = moleFractions_[k];
266  }
267  break;
268  case PBTYPE_SINGLEANION:
269  sumCat = 0.0;
270  sumAnion = 0.0;
271  for (k = 0; k < m_kk; k++) {
272  moleFractionsTmp_[k] = moleFractions_[k];
273  }
274  kMax = npos;
275  sumMax = 0.0;
276  for (k = 0; k < cationList_.size(); k++) {
277  kCat = cationList_[k];
278  chP = m_speciesCharge[kCat];
279  if (moleFractions_[kCat] > sumMax) {
280  kMax = k;
281  sumMax = moleFractions_[kCat];
282  }
283  sumCat += chP * moleFractions_[kCat];
284  }
285  k = anionList_[0];
286  chM = m_speciesCharge[k];
287  sumAnion = moleFractions_[k] * chM;
288  sum = sumCat - sumAnion;
289  if (fabs(sum) > 1.0E-16) {
290  moleFractionsTmp_[cationList_[kMax]] -= sum / m_speciesCharge[kMax];
291  sum = 0.0;
292  for (k = 0; k < numCationSpecies_; k++) {
293  sum += moleFractionsTmp_[k];
294  }
295  for (k = 0; k < numCationSpecies_; k++) {
296  moleFractionsTmp_[k]/= sum;
297  }
298  }
299 
300  for (k = 0; k < numCationSpecies_; k++) {
301  PBMoleFractions_[k] = moleFractionsTmp_[cationList_[k]];
302  }
303  for (k = 0; k < numPassThroughSpecies_; k++) {
304  PBMoleFractions_[neutralPBindexStart + k] = moleFractions_[passThroughList_[k]];
305  }
306 
307  sum = std::max(0.0, PBMoleFractions_[0]);
308  for (k = 1; k < numPBSpecies_; k++) {
309  sum += PBMoleFractions_[k];
310 
311  }
312  for (k = 0; k < numPBSpecies_; k++) {
313  PBMoleFractions_[k] /= sum;
314  }
315 
316  break;
317  case PBTYPE_SINGLECATION:
318  throw CanteraError("eosType", "Unknown type");
319 
320  break;
321 
322  case PBTYPE_MULTICATIONANION:
323  throw CanteraError("eosType", "Unknown type");
324 
325  break;
326  default:
327  throw CanteraError("eosType", "Unknown type");
328  break;
329 
330  }

362 void MolarityIonicVPSSTP::initThermo()
363 {
364  GibbsExcessVPSSTP::initThermo();
365  initLengths();
366  /*
367  * Go find the list of cations and anions
368  */
369  double ch;
370  numCationSpecies_ = 0;
371  cationList_.clear();
372  anionList_.clear();
373  passThroughList_.clear();
374  for (size_t k = 0; k < m_kk; k++) {
375  ch = m_speciesCharge[k];
376  if (ch > 0.0) {
377  cationList_.push_back(k);
378  numCationSpecies_++;
379  } else if (ch < 0.0) {
380  anionList_.push_back(k);
381  numAnionSpecies_++;
382  } else {
383  passThroughList_.push_back(k);
384  numPassThroughSpecies_++;
385  }
386  }
387  numPBSpecies_ = numCationSpecies_ + numAnionSpecies_ - 1;
388  neutralPBindexStart = numPBSpecies_;
389  PBType_ = PBTYPE_MULTICATIONANION;
390  if (numAnionSpecies_ == 1) {
391  PBType_ = PBTYPE_SINGLEANION;
392  } else if (numCationSpecies_ == 1) {
393  PBType_ = PBTYPE_SINGLECATION;
394  }
395  if (numAnionSpecies_ == 0 && numCationSpecies_ == 0) {
396  PBType_ = PBTYPE_PASSTHROUGH;
397  }
398 }
