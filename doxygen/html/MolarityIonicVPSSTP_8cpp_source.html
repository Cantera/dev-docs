126 void MolarityIonicVPSSTP::getPartialMolarEnthalpies(doublereal* hbar) const
127 {
128  // Get the nondimensional standard state enthalpies
129  getEnthalpy_RT(hbar);
130 
131  // dimensionalize it.
132  double T = temperature();

198 void MolarityIonicVPSSTP::calcPseudoBinaryMoleFractions() const
199 {
200  switch (PBType_) {
201  case PBTYPE_PASSTHROUGH:
202  for (size_t k = 0; k < m_kk; k++) {
203  PBMoleFractions_[k] = moleFractions_[k];
204  }
205  break;
206  case PBTYPE_SINGLEANION:
207  {
208  double sumCat = 0.0;
209  double sumAnion = 0.0;
210  for (size_t k = 0; k < m_kk; k++) {
211  moleFractionsTmp_[k] = moleFractions_[k];
212  }
213  size_t kMax = npos;
214  double sumMax = 0.0;
215  for (size_t k = 0; k < cationList_.size(); k++) {
216  size_t kCat = cationList_[k];
217  double chP = m_speciesCharge[kCat];
218  if (moleFractions_[kCat] > sumMax) {
219  kMax = k;
220  sumMax = moleFractions_[kCat];
221  }
222  sumCat += chP * moleFractions_[kCat];
223  }
224  size_t ka = anionList_[0];
225  sumAnion = moleFractions_[ka] * m_speciesCharge[ka];
226  double sum = sumCat - sumAnion;
227  if (fabs(sum) > 1.0E-16) {
228  moleFractionsTmp_[cationList_[kMax]] -= sum / m_speciesCharge[kMax];
229  sum = 0.0;
230  for (size_t k = 0; k < cationList_.size(); k++) {
231  sum += moleFractionsTmp_[k];
232  }
233  for (size_t k = 0; k < cationList_.size(); k++) {
234  moleFractionsTmp_[k]/= sum;
235  }
236  }
237 
238  for (size_t k = 0; k < cationList_.size(); k++) {
239  PBMoleFractions_[k] = moleFractionsTmp_[cationList_[k]];
240  }
241  for (size_t k = 0; k < passThroughList_.size(); k++) {
242  PBMoleFractions_[neutralPBindexStart + k] = moleFractions_[passThroughList_[k]];
243  }
244 
245  sum = std::max(0.0, PBMoleFractions_[0]);
246  for (size_t k = 1; k < numPBSpecies_; k++) {
247  sum += PBMoleFractions_[k];
248  }
249  for (size_t k = 0; k < numPBSpecies_; k++) {
250  PBMoleFractions_[k] /= sum;
251  }
252  break;
253  }
254  case PBTYPE_SINGLECATION:
255  throw CanteraError("eosType", "Unknown type");
256  case PBTYPE_MULTICATIONANION:
257  throw CanteraError("eosType", "Unknown type");
258  default:
259  throw CanteraError("eosType", "Unknown type");
260  }
261 }

278 void MolarityIonicVPSSTP::initThermo()
279 {
280  GibbsExcessVPSSTP::initThermo();
281  initLengths();
282 
283  // Go find the list of cations and anions
284  cationList_.clear();
285  anionList_.clear();
286  passThroughList_.clear();
287  for (size_t k = 0; k < m_kk; k++) {
288  double ch = m_speciesCharge[k];
289  if (ch > 0.0) {
290  cationList_.push_back(k);
291  } else if (ch < 0.0) {
292  anionList_.push_back(k);
293  } else {
294  passThroughList_.push_back(k);
295  }
296  }
297  numPBSpecies_ = cationList_.size() + anionList_.size() - 1;
298  neutralPBindexStart = numPBSpecies_;
299  PBType_ = PBTYPE_MULTICATIONANION;
300  if (anionList_.size() == 1) {
301  PBType_ = PBTYPE_SINGLEANION;
302  } else if (cationList_.size() == 1) {
303  PBType_ = PBTYPE_SINGLECATION;
304  }
305  if (anionList_.size() == 0 && cationList_.size() == 0) {
306  PBType_ = PBTYPE_PASSTHROUGH;
307  }
308 }

315 void MolarityIonicVPSSTP::initThermoXML(XML_Node& phaseNode, const std::string& id)
316 {
317  if ((int) id.size() > 0 && phaseNode.id() != id) {
318  throw CanteraError("MolarityIonicVPSSTP::initThermoXML",
319  "phasenode and Id are incompatible");
320  }
321 
322  // Check on the thermo field. Must have one of:
323  // <thermo model="MolarityIonicVPSS" />
324  // <thermo model="MolarityIonicVPSSTP" />
325  if (!phaseNode.hasChild("thermo")) {
326  throw CanteraError("MolarityIonicVPSSTP::initThermoXML",
327  "no thermo XML node");
328  }
329  XML_Node& thermoNode = phaseNode.child("thermo");
330  std::string mStringa = thermoNode.attrib("model");
331  std::string mString = lowercase(mStringa);
332  if (mString != "molarityionicvpss" && mString != "molarityionicvpsstp") {
333  throw CanteraError("MolarityIonicVPSSTP::initThermoXML",
334  "Unknown thermo model: " + mStringa + " - This object only knows \"MolarityIonicVPSSTP\" ");
335  }
336 
337  // Go get all of the coefficients and factors in the activityCoefficients
338  // XML block
339  if (thermoNode.hasChild("activityCoefficients")) {
340  XML_Node& acNode = thermoNode.child("activityCoefficients");
341  for (size_t i = 0; i < acNode.nChildren(); i++) {
342  XML_Node& xmlACChild = acNode.child(i);
343  // Process a binary interaction
344  if (lowercase(xmlACChild.name()) == "binaryneutralspeciesparameters") {
345  readXMLBinarySpecies(xmlACChild);
346  }
347  }
348  }
349 
350  // Go down the chain
351  GibbsExcessVPSSTP::initThermoXML(phaseNode, id);
352 }
