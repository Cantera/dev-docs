66 void MolarityIonicVPSSTP::getChemPotentials(doublereal* mu) const
67 {
68  // First get the standard chemical potentials in molar form. This requires
69  // updates of standard state as a function of T and P
70  getStandardChemPotentials(mu);
71 
72  // Update the activity coefficients
73  s_update_lnActCoeff();
74  for (size_t k = 0; k < m_kk; k++) {

80 void MolarityIonicVPSSTP::getPartialMolarEnthalpies(doublereal* hbar) const
81 {
82  // Get the nondimensional standard state enthalpies
83  getEnthalpy_RT(hbar);
84 
85  // dimensionalize it.
86  double T = temperature();
87  for (size_t k = 0; k < m_kk; k++) {
88  hbar[k] *= GasConstant * T;
89  }
90 
91  // Update the activity coefficients, This also update the internally stored
92  // molalities.
93  s_update_lnActCoeff();
94  s_update_dlnActCoeff_dT();
95  for (size_t k = 0; k < m_kk; k++) {
96  hbar[k] -= GasConstant * T * T * dlnActCoeffdT_Scaled_[k];
97  }
98 }

100 void MolarityIonicVPSSTP::getPartialMolarCp(doublereal* cpbar) const
101 {
102  // Get the nondimensional standard state entropies
103  getCp_R(cpbar);
104  double T = temperature();

106  // Update the activity coefficients, This also update the internally stored

121 void MolarityIonicVPSSTP::getPartialMolarEntropies(doublereal* sbar) const
122 {

152 void MolarityIonicVPSSTP::calcPseudoBinaryMoleFractions() const
153 {
154  switch (PBType_) {
155  case PBTYPE_PASSTHROUGH:
156  for (size_t k = 0; k < m_kk; k++) {
157  PBMoleFractions_[k] = moleFractions_[k];
158  }
159  break;
160  case PBTYPE_SINGLEANION:
161  {
162  double sumCat = 0.0;
163  double sumAnion = 0.0;
164  for (size_t k = 0; k < m_kk; k++) {
165  moleFractionsTmp_[k] = moleFractions_[k];
166  }
167  size_t kMax = npos;
168  double sumMax = 0.0;
169  for (size_t k = 0; k < cationList_.size(); k++) {
170  size_t kCat = cationList_[k];
171  double chP = m_speciesCharge[kCat];
172  if (moleFractions_[kCat] > sumMax) {
173  kMax = k;
174  sumMax = moleFractions_[kCat];
175  }
176  sumCat += chP * moleFractions_[kCat];
177  }
178  size_t ka = anionList_[0];
179  sumAnion = moleFractions_[ka] * m_speciesCharge[ka];
180  double sum = sumCat - sumAnion;
181  if (fabs(sum) > 1.0E-16) {
182  moleFractionsTmp_[cationList_[kMax]] -= sum / m_speciesCharge[kMax];
183  sum = 0.0;
184  for (size_t k = 0; k < cationList_.size(); k++) {
185  sum += moleFractionsTmp_[k];
186  }
187  for (size_t k = 0; k < cationList_.size(); k++) {
188  moleFractionsTmp_[k]/= sum;
189  }
190  }
191 
192  for (size_t k = 0; k < cationList_.size(); k++) {
193  PBMoleFractions_[k] = moleFractionsTmp_[cationList_[k]];
194  }
195  for (size_t k = 0; k < passThroughList_.size(); k++) {
196  PBMoleFractions_[neutralPBindexStart + k] = moleFractions_[passThroughList_[k]];
197  }
198 
199  sum = std::max(0.0, PBMoleFractions_[0]);
200  for (size_t k = 1; k < numPBSpecies_; k++) {
201  sum += PBMoleFractions_[k];
202  }
203  for (size_t k = 0; k < numPBSpecies_; k++) {
204  PBMoleFractions_[k] /= sum;
205  }
206  break;
207  }
208  case PBTYPE_SINGLECATION:
209  throw CanteraError("eosType", "Unknown type");
210  case PBTYPE_MULTICATIONANION:
211  throw CanteraError("eosType", "Unknown type");
212  default:
213  throw CanteraError("eosType", "Unknown type");
214  }
215 }

232 void MolarityIonicVPSSTP::initThermo()
233 {
234  GibbsExcessVPSSTP::initThermo();
235  initLengths();
236 
237  // Go find the list of cations and anions
238  cationList_.clear();
239  anionList_.clear();
240  passThroughList_.clear();
241  for (size_t k = 0; k < m_kk; k++) {
242  double ch = m_speciesCharge[k];
243  if (ch > 0.0) {
244  cationList_.push_back(k);
245  } else if (ch < 0.0) {
246  anionList_.push_back(k);
247  } else {
248  passThroughList_.push_back(k);
249  }
250  }
251  numPBSpecies_ = cationList_.size() + anionList_.size() - 1;
252  neutralPBindexStart = numPBSpecies_;
253  PBType_ = PBTYPE_MULTICATIONANION;
254  if (anionList_.size() == 1) {
255  PBType_ = PBTYPE_SINGLEANION;
256  } else if (cationList_.size() == 1) {
257  PBType_ = PBTYPE_SINGLECATION;
258  }
259  if (anionList_.size() == 0 && cationList_.size() == 0) {
260  PBType_ = PBTYPE_PASSTHROUGH;
261  }
262 }

269 void MolarityIonicVPSSTP::initThermoXML(XML_Node& phaseNode, const std::string& id)
270 {
271  if ((int) id.size() > 0 && phaseNode.id() != id) {
272  throw CanteraError("MolarityIonicVPSSTP::initThermoXML",
273  "phasenode and Id are incompatible");
274  }
275 
276  // Check on the thermo field. Must have one of:
277  // <thermo model="MolarityIonicVPSS" />
278  // <thermo model="MolarityIonicVPSSTP" />
279  if (!phaseNode.hasChild("thermo")) {
280  throw CanteraError("MolarityIonicVPSSTP::initThermoXML",
281  "no thermo XML node");
282  }
283  XML_Node& thermoNode = phaseNode.child("thermo");
284  if (!caseInsensitiveEquals(thermoNode["model"], "molarityionicvpss")
285  && !caseInsensitiveEquals(thermoNode["model"], "molarityionicvpsstp")) {
286  throw CanteraError("MolarityIonicVPSSTP::initThermoXML",
287  "Unknown thermo model: " + thermoNode["model"]
288  + " - This object only knows \"MolarityIonicVPSSTP\" ");
289  }
290 
291  // Go get all of the coefficients and factors in the activityCoefficients
292  // XML block
293  if (thermoNode.hasChild("activityCoefficients")) {
294  XML_Node& acNode = thermoNode.child("activityCoefficients");
295  for (size_t i = 0; i < acNode.nChildren(); i++) {
296  XML_Node& xmlACChild = acNode.child(i);
297  // Process a binary interaction
298  if (caseInsensitiveEquals(xmlACChild.name(), "binaryneutralspeciesparameters")) {
299  readXMLBinarySpecies(xmlACChild);
300  }
301  }
302  }
303 
304  // Go down the chain
305  GibbsExcessVPSSTP::initThermoXML(phaseNode, id);
306 }
