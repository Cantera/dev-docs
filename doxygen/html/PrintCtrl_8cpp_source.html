109 double PrintCtrl::cropAbs10(const double d, int Ndec) const
110 {
111  if (!doCrop()) {
112  return d;
113  }
114  if (Ndec < -301 || Ndec > 301) {
115  return d;
116  }
117  double dfabs = fabs(d);
118  double pdec = pow(10.0, (double) Ndec);
119  if (dfabs < pdec) {
120  return 0.0;
121  }
122  double dl10 = log10(dfabs);
123  int N10 = (int) dl10;
124  if (dl10 > -0.0) {
125  N10 += 1;
126  }

131 double PrintCtrl::cropSigDigits(const double d, int nSig) const
132 {
133  if (!doCrop()) {
134  return d;
135  }
136  if (nSig <=0) {
137  nSig = 1;
138  }
139  if (nSig >=9) {
140  nSig = 9;
141  }
142  double sgn = 1.0;
143  if (d < 0.0) {
144  sgn = -1.0;
145  }
146  double dfabs = fabs(d);
147  double dl10 = log10(dfabs);
148  int N10 = (int) dl10;
149  if (dl10 > -0.0) {
150  N10 += 1;
151  }
152  int E10 = -N10 + nSig ;
153  double pfabs = dfabs * pow(10.0, (double) E10);
154  pfabs *= (1.0 + 1.0E-14);
155  long int nfabs = (long int) pfabs;
156  double remainder = pfabs - nfabs;
157  if (remainder > 0.5) {
158  nfabs++;
159  }
160  double paltabs = (double) nfabs;
161  double daltabs = paltabs * pow(10.0, (double) -E10);
162  return sgn * daltabs;
163 }
