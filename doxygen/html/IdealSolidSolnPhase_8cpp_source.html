203 void IdealSolidSolnPhase::getActivityConcentrations(doublereal* c) const
204 {
205  const doublereal* const dtmp = moleFractdivMMW();
206  const double mmw = meanMolecularWeight();
207  switch (m_formGC) {
208  case 0:
209  for (size_t k = 0; k < m_kk; k++) {
210  c[k] = dtmp[k] * mmw;
211  }
212  break;
213  case 1:
214  for (size_t k = 0; k < m_kk; k++) {
215  c[k] = dtmp[k] * mmw / m_speciesMolarVolume[k];
216  }
217  break;
218  case 2:
219  double atmp = mmw / m_speciesMolarVolume[m_kk-1];
220  for (size_t k = 0; k < m_kk; k++) {
221  c[k] = dtmp[k] * atmp;
222  }
223  break;
224  }

280 void IdealSolidSolnPhase::getChemPotentials(doublereal* mu) const
281 {
282  doublereal delta_p = m_Pcurrent - m_Pref;
283  const vector_fp& g_RT = gibbs_RT_ref();
284  for (size_t k = 0; k < m_kk; k++) {
285  double xx = std::max(SmallNumber, moleFraction(k));
286  mu[k] = RT() * (g_RT[k] + log(xx))

453 void IdealSolidSolnPhase::initThermoXML(XML_Node& phaseNode, const std::string& id_)
454 {
455  if (id_.size() > 0 && phaseNode.id() != id_) {
456  throw CanteraError("IdealSolidSolnPhase::initThermoXML",
457  "phasenode and Id are incompatible");
458  }
459 
460  // Check on the thermo field. Must have:
461  // <thermo model="IdealSolidSolution" />
462  if (phaseNode.hasChild("thermo")) {
463  XML_Node& thNode = phaseNode.child("thermo");
464  string mString = thNode.attrib("model");
465  if (lowercase(mString) != "idealsolidsolution") {
466  throw CanteraError("IdealSolidSolnPhase::initThermoXML",
467  "Unknown thermo model: " + mString);
468  }
469  } else {
470  throw CanteraError("IdealSolidSolnPhase::initThermoXML",
471  "Unspecified thermo model");
472  }
473 
474  // Form of the standard concentrations. Must have one of:
475  //
476  // <standardConc model="unity" />
477  // <standardConc model="molar_volume" />
478  // <standardConc model="solvent_volume" />
479  if (phaseNode.hasChild("standardConc")) {
480  XML_Node& scNode = phaseNode.child("standardConc");
481  string formStringa = scNode.attrib("model");
482  string formString = lowercase(formStringa);
483  if (formString == "unity") {
484  m_formGC = 0;
485  } else if (formString == "molar_volume") {
486  m_formGC = 1;
487  } else if (formString == "solvent_volume") {
488  m_formGC = 2;
489  } else {
490  throw CanteraError("IdealSolidSolnPhase::initThermoXML",
491  "Unknown standardConc model: " + formStringa);
492  }
493  } else {
494  throw CanteraError("IdealSolidSolnPhase::initThermoXML",
495  "Unspecified standardConc model");
496  }
497 
498  // Initialize all of the lengths now that we know how many species
499  // there are in the phase.
500  initLengths();
501 
502  // Now go get the molar volumes
503  XML_Node& speciesList = phaseNode.child("speciesArray");
504  XML_Node* speciesDB = get_XML_NameID("speciesData", speciesList["datasrc"],
505  &phaseNode.root());
506 
507  for (size_t k = 0; k < m_kk; k++) {
508  XML_Node* s = speciesDB->findByAttr("name", speciesName(k));
509  XML_Node* ss = s->findByName("standardState");
510  m_speciesMolarVolume[k] = getFloat(*ss, "molarVolume", "toSI");
511  }
512 
513  // Call the base initThermo, which handles setting the initial state.
514  ThermoPhase::initThermoXML(phaseNode, id_);
515 }

534 void IdealSolidSolnPhase::setToEquilState(const doublereal* lambda_RT)
535 {
536  const vector_fp& grt = gibbs_RT_ref();
537 
538  // set the pressure and composition to be consistent with the temperature
539  doublereal pres = 0.0;
540  for (size_t k = 0; k < m_kk; k++) {
541  m_pp[k] = -grt[k];
542  for (size_t m = 0; m < nElements(); m++) {
543  m_pp[k] += nAtoms(k,m)*lambda_RT[m];
544  }
545  m_pp[k] = m_Pref * exp(m_pp[k]);
546  pres += m_pp[k];
547  }
548  setState_PX(pres, &m_pp[0]);
549 }

561 void IdealSolidSolnPhase::_updateThermo() const
562 {
563  doublereal tnow = temperature();
564  if (m_tlast != tnow) {
565 
566  // Update the thermodynamic functions of the reference state.
567  m_spthermo->update(tnow, m_cp0_R.data(), m_h0_RT.data(), m_s0_R.data());
568  m_tlast = tnow;
569  doublereal rrt = 1.0 / RT();
570  for (size_t k = 0; k < m_kk; k++) {
571  double deltaE = rrt * m_pe[k];
572  m_h0_RT[k] += deltaE;
573  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
574  }
575  m_tlast = tnow;
576  }
577 }
