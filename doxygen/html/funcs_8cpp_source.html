38 doublereal linearInterp(doublereal x, const vector_fp& xpts,
39  const vector_fp& fpts)
40 {
41  if (x <= xpts[0]) {
42  return fpts[0];
43  }
44  if (x >= xpts.back()) {
45  return fpts.back();
46  }
47  vector_fp::const_iterator loc =
48  lower_bound(xpts.begin(), xpts.end(), x);
49  int iloc = int(loc - xpts.begin()) - 1;
50  doublereal ff = fpts[iloc] +
51  (x - xpts[iloc])*(fpts[iloc + 1]
52  - fpts[iloc])/(xpts[iloc + 1] - xpts[iloc]);
53  return ff;

56 //! Fits a polynomial function to a set of data points
57 /*!
58  * Given a collection of points X(I) and a set of values Y(I) which
59  * correspond to some function or measurement at each of the X(I),
60  * subroutine DPOLFT computes the weighted least-squares polynomial
61  * fits of all degrees up to some degree either specified by the user
62  * or determined by the routine. The fits thus obtained are in
63  * orthogonal polynomial form. Subroutine DP1VLU may then be
64  * called to evaluate the fitted polynomials and any of their
65  * derivatives at any point. The subroutine DPCOEF may be used to
66  * express the polynomial fits as powers of (X-C) for any specified
67  * point C.
68  *
69  * @param n The number of data points.
70  *
71  * @param x A set of grid points on which the data is specified.
72  * The array of values of the independent variable. These
73  * values may appear in any order and need not all be
74  * distinct. There are n of them.
75  *
76  * @param y array of corresponding function values. There are n of them
77  *
78  * @param w array of positive values to be used as weights. If
79  * W[0] is negative, DPOLFT will set all the weights
80  * to 1.0, which means unweighted least squares error
81  * will be minimized. To minimize relative error, the
82  * user should set the weights to: W(I) = 1.0/Y(I)**2,
83  * I = 1,...,N .
84  *
85  * @param maxdeg maximum degree to be allowed for polynomial fit.
86  * MAXDEG may be any non-negative integer less than N.
87  * Note -- MAXDEG cannot be equal to N-1 when a
88  * statistical test is to be used for degree selection,
89  * i.e., when input value of EPS is negative.
90  *
91  * @param ndeg output degree of the fit computed.
92  *
93  * @param eps Specifies the criterion to be used in determining
94  * the degree of fit to be computed.
95  * (1) If EPS is input negative, DPOLFT chooses the
96  * degree based on a statistical F test of
97  * significance. One of three possible
98  * significance levels will be used: .01, .05 or
99  * .10. If EPS=-1.0 , the routine will
100  * automatically select one of these levels based
101  * on the number of data points and the maximum
102  * degree to be considered. If EPS is input as
103  * -.01, -.05, or -.10, a significance level of
104  * .01, .05, or .10, respectively, will be used.
105  * (2) If EPS is set to 0., DPOLFT computes the
106  * polynomials of degrees 0 through MAXDEG .
107  * (3) If EPS is input positive, EPS is the RMS
108  * error tolerance which must be satisfied by the
109  * fitted polynomial. DPOLFT will increase the
110  * degree of fit until this criterion is met or
111  * until the maximum degree is reached.
112  *
113  * @param r Output vector containing the first LL+1 Taylor coefficients
114  * where LL=ABS(ndeg).
115  * P(X) = r[0] + r[1]*(X-C) + ... + r[ndeg] * (X-C)**ndeg
116  * ( here C = 0.0)
117  *
118  * @return returns the RMS error of the polynomial of degree ndeg .
119  */
120 doublereal polyfit(int n, doublereal* x, doublereal* y, doublereal* w, int maxdeg, int& ndeg, doublereal eps, doublereal* r)
121 {
122  integer nn = n;
123  integer mdeg = maxdeg;
124  integer ndg = ndeg;
125  doublereal epss = eps;
126  integer ierr;
127  int worksize = 3*n + 3*maxdeg + 3;
128  vector_fp awork(worksize,0.0);
129  vector_fp coeffs(n+1, 0.0);
130  doublereal zer = 0.0;
131 
132  _DPOLFT_(&nn, x, y, w, &mdeg, &ndg, &epss, &coeffs[0],
133  &ierr, &awork[0]);
134  if (ierr != 1) throw CanteraError("polyfit",
135  "DPOLFT returned error code IERR = " + int2str(ierr) +
136  "while attempting to fit " + int2str(n) + " data points "
137  + "to a polynomial of degree " + int2str(maxdeg));
138  ndeg = ndg;
139  _DPCOEF_(&ndg, &zer, r, &awork[0]);
140  return epss;
141 }
