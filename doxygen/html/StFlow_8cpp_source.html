214 void StFlow::eval(size_t jg, doublereal* xg,
215  doublereal* rg, integer* diagg, doublereal rdt)
216 {
217  // if evaluating a Jacobian, and the global point is outside the domain of
218  // influence for this domain, then skip evaluating the residual
219  if (jg != npos && (jg + 1 < firstPoint() || jg > lastPoint() + 1)) {
220  return;
221  }
222 
223  // if evaluating a Jacobian, compute the steady-state residual
224  if (jg != npos) {
225  rdt = 0.0;
226  }
227 
228  // start of local part of global arrays
229  doublereal* x = xg + loc();
230  doublereal* rsd = rg + loc();
231  integer* diag = diagg + loc();
232 
233  size_t jmin, jmax;
234  if (jg == npos) { // evaluate all points
235  jmin = 0;
236  jmax = m_points - 1;
237  } else { // evaluate points for Jacobian
238  size_t jpt = (jg == 0) ? 0 : jg - firstPoint();
239  jmin = std::max<size_t>(jpt, 1) - 1;
240  jmax = std::min(jpt+1,m_points-1);
241  }
242 
243  updateProperties(jg, x, jmin, jmax);
244  evalResidual(x, rsd, diag, rdt, jmin, jmax);
245 }

247 void StFlow::updateProperties(size_t jg, double* x, size_t jmin, size_t jmax)
248 {
249  // properties are computed for grid points from j0 to j1
250  size_t j0 = std::max<size_t>(jmin, 1) - 1;
251  size_t j1 = std::min(jmax+1,m_points-1);
252 
253  updateThermo(x, j0, j1);
254  if (jg == npos || m_force_full_update) {
255  // update transport properties only if a Jacobian is not being
256  // evaluated, or if specifically requested
257  updateTransport(x, j0, j1);
258  }
259  if (jg == npos) {
260  double* Yleft = x + index(c_offset_Y, jmin);
261  m_kExcessLeft = distance(Yleft, max_element(Yleft, Yleft + m_nsp));
262  double* Yright = x + index(c_offset_Y, jmax);
263  m_kExcessRight = distance(Yright, max_element(Yright, Yright + m_nsp));
264  }
265 
266  // update the species diffusive mass fluxes whether or not a
267  // Jacobian is being evaluated
268  updateDiffFluxes(x, j0, j1);
269 }

271 void StFlow::evalResidual(double* x, double* rsd, int* diag,
272  double rdt, size_t jmin, size_t jmax)
273 {
274  //----------------------------------------------------
275  // evaluate the residual equations at all required
276  // grid points
277  //----------------------------------------------------
278 
279  // calculation of qdotRadiation
280 
281  // The simple radiation model used was established by Y. Liu and B. Rogg [Y.
282  // Liu and B. Rogg, Modelling of thermally radiating diffusion flames with
283  // detailed chemistry and transport, EUROTHERM Seminars, 17:114-127, 1991].
284  // This model uses the optically thin limit and the gray-gas approximation
285  // to simply calculate a volume specified heat flux out of the Planck
286  // absorption coefficients, the boundary emissivities and the temperature.
287  // The model considers only CO2 and H2O as radiating species. Polynomial
288  // lines calculate the species Planck coefficients for H2O and CO2. The data
289  // for the lines is taken from the RADCAL program [Grosshandler, W. L.,
290  // RADCAL: A Narrow-Band Model for Radiation Calculations in a Combustion
291  // Environment, NIST technical note 1402, 1993]. The coefficients for the
292  // polynomials are taken from [http://www.sandia.gov/TNF/radiation.html].
293 
294  if (m_do_radiation) {
295  // variable definitions for the Planck absorption coefficient and the
296  // radiation calculation:
297  doublereal k_P_ref = 1.0*OneAtm;
298 
299  // polynomial coefficients:
300  const doublereal c_H2O[6] = {-0.23093, -1.12390, 9.41530, -2.99880,
301  0.51382, -1.86840e-5};
302  const doublereal c_CO2[6] = {18.741, -121.310, 273.500, -194.050,
303  56.310, -5.8169};
304 
305  // calculation of the two boundary values
306  double boundary_Rad_left = m_epsilon_left * StefanBoltz * pow(T(x, 0), 4);
307  double boundary_Rad_right = m_epsilon_right * StefanBoltz * pow(T(x, m_points - 1), 4);
308 
309  // loop over all grid points
310  for (size_t j = jmin; j < jmax; j++) {
311  // helping variable for the calculation
312  double radiative_heat_loss = 0;
313 
314  // calculation of the mean Planck absorption coefficient
315  double k_P = 0;
316  // absorption coefficient for H2O
317  if (m_kRadiating[1] != npos) {
318  double k_P_H2O = 0;
319  for (size_t n = 0; n <= 5; n++) {
320  k_P_H2O += c_H2O[n] * pow(1000 / T(x, j), (double) n);
321  }
322  k_P_H2O /= k_P_ref;
323  k_P += m_press * X(x, m_kRadiating[1], j) * k_P_H2O;
324  }
325  // absorption coefficient for CO2
326  if (m_kRadiating[0] != npos) {
327  double k_P_CO2 = 0;
328  for (size_t n = 0; n <= 5; n++) {
329  k_P_CO2 += c_CO2[n] * pow(1000 / T(x, j), (double) n);
330  }
331  k_P_CO2 /= k_P_ref;
332  k_P += m_press * X(x, m_kRadiating[0], j) * k_P_CO2;
333  }
334 
335  // calculation of the radiative heat loss term
336  radiative_heat_loss = 2 * k_P *(2 * StefanBoltz * pow(T(x, j), 4)
337  - boundary_Rad_left - boundary_Rad_right);
338 
339  // set the radiative heat loss vector
340  m_qdotRadiation[j] = radiative_heat_loss;
341  }
342  }
343 
344  for (size_t j = jmin; j <= jmax; j++) {
345  //----------------------------------------------
346  // left boundary
347  //----------------------------------------------
348 
349  if (j == 0) {
350  // these may be modified by a boundary object
351 
352  // Continuity. This propagates information right-to-left, since
353  // rho_u at point 0 is dependent on rho_u at point 1, but not on
354  // mdot from the inlet.
355  rsd[index(c_offset_U,0)] =
356  -(rho_u(x,1) - rho_u(x,0))/m_dz[0]
357  -(density(1)*V(x,1) + density(0)*V(x,0));
358 
359  // the inlet (or other) object connected to this one will modify
360  // these equations by subtracting its values for V, T, and mdot. As
361  // a result, these residual equations will force the solution
362  // variables to the values for the boundary object
363  rsd[index(c_offset_V,0)] = V(x,0);
364  if (doEnergy(0)) {
365  rsd[index(c_offset_T,0)] = T(x,0);
366  } else {
367  rsd[index(c_offset_T,0)] = T(x,0) - T_fixed(0);
368  }
369  rsd[index(c_offset_L,0)] = -rho_u(x,0);
370 
371  // The default boundary condition for species is zero flux. However,
372  // the boundary object may modify this.
373  double sum = 0.0;
374  for (size_t k = 0; k < m_nsp; k++) {
375  sum += Y(x,k,0);
376  rsd[index(c_offset_Y + k, 0)] =
377  -(m_flux(k,0) + rho_u(x,0)* Y(x,k,0));
378  }
379  rsd[index(c_offset_Y + leftExcessSpecies(), 0)] = 1.0 - sum;
380 
381  // set residual of poisson's equ to zero
382  rsd[index(c_offset_P, 0)] = x[index(c_offset_P, j)];
383  } else if (j == m_points - 1) {
384  evalRightBoundary(x, rsd, diag, rdt);
385  // set residual of poisson's equ to zero
386  rsd[index(c_offset_P, j)] = x[index(c_offset_P, j)];
387  } else { // interior points
388  evalContinuity(j, x, rsd, diag, rdt);
389  // set residual of poisson's equ to zero
390  rsd[index(c_offset_P, j)] = x[index(c_offset_P, j)];
391 
392  //------------------------------------------------
393  // Radial momentum equation
394  //
395  // \rho dV/dt + \rho u dV/dz + \rho V^2
396  // = d(\mu dV/dz)/dz - lambda
397  //-------------------------------------------------
398  rsd[index(c_offset_V,j)]
399  = (shear(x,j) - lambda(x,j) - rho_u(x,j)*dVdz(x,j)
400  - m_rho[j]*V(x,j)*V(x,j))/m_rho[j]
401  - rdt*(V(x,j) - V_prev(j));
402  diag[index(c_offset_V, j)] = 1;
403 
404  //-------------------------------------------------
405  // Species equations
406  //
407  // \rho dY_k/dt + \rho u dY_k/dz + dJ_k/dz
408  // = M_k\omega_k
409  //-------------------------------------------------
410  getWdot(x,j);
411  for (size_t k = 0; k < m_nsp; k++) {
412  double convec = rho_u(x,j)*dYdz(x,k,j);
413  double diffus = 2.0*(m_flux(k,j) - m_flux(k,j-1))
414  / (z(j+1) - z(j-1));
415  rsd[index(c_offset_Y + k, j)]
416  = (m_wt[k]*(wdot(k,j))
417  - convec - diffus)/m_rho[j]
418  - rdt*(Y(x,k,j) - Y_prev(k,j));
419  diag[index(c_offset_Y + k, j)] = 1;
420  }
421 
422  //-----------------------------------------------
423  // energy equation
424  //
425  // \rho c_p dT/dt + \rho c_p u dT/dz
426  // = d(k dT/dz)/dz
427  // - sum_k(\omega_k h_k_ref)
428  // - sum_k(J_k c_p_k / M_k) dT/dz
429  //-----------------------------------------------
430  if (m_do_energy[j]) {
431  setGas(x,j);
432 
433  // heat release term
434  const vector_fp& h_RT = m_thermo->enthalpy_RT_ref();
435  const vector_fp& cp_R = m_thermo->cp_R_ref();
436  double sum = 0.0;
437  double sum2 = 0.0;
438  for (size_t k = 0; k < m_nsp; k++) {
439  double flxk = 0.5*(m_flux(k,j-1) + m_flux(k,j));
440  sum += wdot(k,j)*h_RT[k];
441  sum2 += flxk*cp_R[k]/m_wt[k];
442  }
443  sum *= GasConstant * T(x,j);
444  double dtdzj = dTdz(x,j);
445  sum2 *= GasConstant * dtdzj;
446 
447  rsd[index(c_offset_T, j)] = - m_cp[j]*rho_u(x,j)*dtdzj
448  - divHeatFlux(x,j) - sum - sum2;
449  rsd[index(c_offset_T, j)] /= (m_rho[j]*m_cp[j]);
450  rsd[index(c_offset_T, j)] -= rdt*(T(x,j) - T_prev(j));
451  rsd[index(c_offset_T, j)] -= (m_qdotRadiation[j] / (m_rho[j] * m_cp[j]));
452  diag[index(c_offset_T, j)] = 1;
453  } else {
454  // residual equations if the energy equation is disabled
455  rsd[index(c_offset_T, j)] = T(x,j) - T_fixed(j);
456  diag[index(c_offset_T, j)] = 0;
457  }
458 
459  rsd[index(c_offset_L, j)] = lambda(x,j) - lambda(x,j-1);
460  diag[index(c_offset_L, j)] = 0;
461  }
462  }
463 }

465 void StFlow::updateTransport(doublereal* x, size_t j0, size_t j1)
466 {
467  if (m_do_multicomponent) {
468  for (size_t j = j0; j < j1; j++) {
469  setGasAtMidpoint(x,j);
470  doublereal wtm = m_thermo->meanMolecularWeight();
471  doublereal rho = m_thermo->density();
472  m_visc[j] = (m_dovisc ? m_trans->viscosity() : 0.0);
473  m_trans->getMultiDiffCoeffs(m_nsp, &m_multidiff[mindex(0,0,j)]);
474 
475  // Use m_diff as storage for the factor outside the summation
476  for (size_t k = 0; k < m_nsp; k++) {
477  m_diff[k+j*m_nsp] = m_wt[k] * rho / (wtm*wtm);
478  }
479 
480  m_tcon[j] = m_trans->thermalConductivity();
481  if (m_do_soret) {
482  m_trans->getThermalDiffCoeffs(m_dthermal.ptrColumn(0) + j*m_nsp);
483  }
484  }
485  } else { // mixture averaged transport
486  for (size_t j = j0; j < j1; j++) {
487  setGasAtMidpoint(x,j);
488  m_visc[j] = (m_dovisc ? m_trans->viscosity() : 0.0);
489  m_trans->getMixDiffCoeffs(&m_diff[j*m_nsp]);
490  m_tcon[j] = m_trans->thermalConductivity();
491  }
492  }

512 void StFlow::updateDiffFluxes(const doublereal* x, size_t j0, size_t j1)
513 {
514  if (m_do_multicomponent) {
515  for (size_t j = j0; j < j1; j++) {
516  double dz = z(j+1) - z(j);
517  for (size_t k = 0; k < m_nsp; k++) {
518  doublereal sum = 0.0;
519  for (size_t m = 0; m < m_nsp; m++) {
520  sum += m_wt[m] * m_multidiff[mindex(k,m,j)] * (X(x,m,j+1)-X(x,m,j));
521  }
522  m_flux(k,j) = sum * m_diff[k+j*m_nsp] / dz;
523  }
524  }
525  } else {
526  for (size_t j = j0; j < j1; j++) {
527  double sum = 0.0;
528  double wtm = m_wtm[j];
529  double rho = density(j);
530  double dz = z(j+1) - z(j);
531  for (size_t k = 0; k < m_nsp; k++) {
532  m_flux(k,j) = m_wt[k]*(rho*m_diff[k+m_nsp*j]/wtm);
533  m_flux(k,j) *= (X(x,k,j) - X(x,k,j+1))/dz;
534  sum -= m_flux(k,j);
535  }
536  // correction flux to insure that \sum_k Y_k V_k = 0.
537  for (size_t k = 0; k < m_nsp; k++) {
538  m_flux(k,j) += sum*Y(x,k,j);
539  }
540  }
541  }
542 
543  if (m_do_soret) {
544  for (size_t m = j0; m < j1; m++) {
545  double gradlogT = 2.0 * (T(x,m+1) - T(x,m)) /
546  ((T(x,m+1) + T(x,m)) * (z(m+1) - z(m)));
547  for (size_t k = 0; k < m_nsp; k++) {
548  m_flux(k,m) -= m_dthermal(k,m)*gradlogT;
549  }

598 void StFlow::restore(const XML_Node& dom, doublereal* soln, int loglevel)
599 {

601  vector<string> ignored;
602  size_t nsp = m_thermo->nSpecies();
603  vector_int did_species(nsp, 0);
604 
605  vector<XML_Node*> str = dom.getChildren("string");
606  for (size_t istr = 0; istr < str.size(); istr++) {
607  const XML_Node& nd = *str[istr];
608  writelog(nd["title"]+": "+nd.value()+"\n");
609  }
610 
611  double pp = getFloat(dom, "pressure", "pressure");
612  setPressure(pp);
613  vector<XML_Node*> d = dom.child("grid_data").getChildren("floatArray");
614  vector_fp x;
615  size_t np = 0;
616  bool readgrid = false, wrote_header = false;
617  for (size_t n = 0; n < d.size(); n++) {
618  const XML_Node& fa = *d[n];
619  string nm = fa["title"];
620  if (nm == "z") {
621  getFloatArray(fa,x,false);
622  np = x.size();
623  if (loglevel >= 2) {
624  writelog("Grid contains {} points.\n", np);
625  }
626  readgrid = true;
627  setupGrid(np, x.data());
628  }
629  }
630  if (!readgrid) {
631  throw CanteraError("StFlow::restore",
632  "domain contains no grid points.");
633  }
634 
635  debuglog("Importing datasets:\n", loglevel >= 2);
636  for (size_t n = 0; n < d.size(); n++) {
637  const XML_Node& fa = *d[n];
638  string nm = fa["title"];
639  getFloatArray(fa,x,false);
640  if (nm == "u") {
641  debuglog("axial velocity ", loglevel >= 2);
642  if (x.size() != np) {
643  throw CanteraError("StFlow::restore",
644  "axial velocity array size error");
645  }
646  for (size_t j = 0; j < np; j++) {
647  soln[index(c_offset_U,j)] = x[j];
648  }
649  } else if (nm == "z") {
650  ; // already read grid
651  } else if (nm == "V") {
652  debuglog("radial velocity ", loglevel >= 2);
653  if (x.size() != np) {
654  throw CanteraError("StFlow::restore",
655  "radial velocity array size error");
656  }
657  for (size_t j = 0; j < np; j++) {
658  soln[index(c_offset_V,j)] = x[j];
659  }
660  } else if (nm == "T") {
661  debuglog("temperature ", loglevel >= 2);
662  if (x.size() != np) {
663  throw CanteraError("StFlow::restore",
664  "temperature array size error");
665  }
666  for (size_t j = 0; j < np; j++) {
667  soln[index(c_offset_T,j)] = x[j];
668  }
669 
670  // For fixed-temperature simulations, use the imported temperature
671  // profile by default. If this is not desired, call
672  // setFixedTempProfile *after* restoring the solution.
673  vector_fp zz(np);
674  for (size_t jj = 0; jj < np; jj++) {
675  zz[jj] = (grid(jj) - zmin())/(zmax() - zmin());
676  }
677  setFixedTempProfile(zz, x);
678  } else if (nm == "L") {
679  debuglog("lambda ", loglevel >= 2);
680  if (x.size() != np) {
681  throw CanteraError("StFlow::restore",
682  "lambda arary size error");
683  }
684  for (size_t j = 0; j < np; j++) {
685  soln[index(c_offset_L,j)] = x[j];
686  }
687  } else if (m_thermo->speciesIndex(nm) != npos) {
688  debuglog(nm+" ", loglevel >= 2);
689  if (x.size() == np) {
690  size_t k = m_thermo->speciesIndex(nm);
691  did_species[k] = 1;
692  for (size_t j = 0; j < np; j++) {
693  soln[index(k+c_offset_Y,j)] = x[j];
694  }
695  }
696  } else {
697  ignored.push_back(nm);
698  }
699  }
700 
701  if (loglevel >=2 && !ignored.empty()) {
702  writelog("\n\n");
703  writelog("Ignoring datasets:\n");
704  size_t nn = ignored.size();
705  for (size_t n = 0; n < nn; n++) {
706  writelog(ignored[n]+" ");
707  }
708  }
709 
710  if (loglevel >= 1) {
711  for (size_t ks = 0; ks < nsp; ks++) {
712  if (did_species[ks] == 0) {
713  if (!wrote_header) {
714  writelog("Missing data for species:\n");
715  wrote_header = true;
716  }
717  writelog(m_thermo->speciesName(ks)+" ");
718  }
719  }
720  }
721 
722  if (dom.hasChild("energy_enabled")) {
723  getFloatArray(dom, x, false, "", "energy_enabled");
724  if (x.size() == nPoints()) {
725  for (size_t i = 0; i < x.size(); i++) {
726  m_do_energy[i] = (x[i] != 0);
727  }
728  } else if (!x.empty()) {
729  throw CanteraError("StFlow::restore", "energy_enabled is length {}"
730  "but should be length {}", x.size(), nPoints());
731  }
732  }
733 
734  if (dom.hasChild("species_enabled")) {
735  getFloatArray(dom, x, false, "", "species_enabled");
736  if (x.size() == m_nsp) {
737  for (size_t i = 0; i < x.size(); i++) {
738  m_do_species[i] = (x[i] != 0);
739  }
740  } else if (!x.empty()) {
741  // This may occur when restoring from a mechanism with a different
742  // number of species.
743  if (loglevel > 0) {
744  writelog("\nWarning: StFlow::restore: species_enabled is "
745  "length {} but should be length {}. Enabling all species "
746  "equations by default.", x.size(), m_nsp);
747  }
748  m_do_species.assign(m_nsp, true);
749  }
750  }
751 
752  if (dom.hasChild("refine_criteria")) {
753  XML_Node& ref = dom.child("refine_criteria");
754  refiner().setCriteria(getFloat(ref, "ratio"), getFloat(ref, "slope"),

880 void AxiStagnFlow::evalRightBoundary(doublereal* x, doublereal* rsd,
881  integer* diag, doublereal rdt)
882 {
883  size_t j = m_points - 1;
884  // the boundary object connected to the right of this one may modify or
885  // replace these equations. The default boundary conditions are zero u, V,
886  // and T, and zero diffusive flux for all species.
887  rsd[index(c_offset_U,j)] = rho_u(x,j);
888  rsd[index(c_offset_V,j)] = V(x,j);
889  if (m_do_energy[j]) {
890  rsd[index(c_offset_T,j)] = T(x,j);
891  } else {
892  rsd[index(c_offset_T, j)] = T(x,j) - T_fixed(j);
893  }
894  rsd[index(c_offset_L, j)] = lambda(x,j) - lambda(x,j-1);
895  diag[index(c_offset_L, j)] = 0;
896  doublereal sum = 0.0;
897  for (size_t k = 0; k < m_nsp; k++) {
898  sum += Y(x,k,j);
899  rsd[index(k+c_offset_Y,j)] = m_flux(k,j-1) + rho_u(x,j)*Y(x,k,j);
900  }
901  rsd[index(c_offset_Y + rightExcessSpecies(), j)] = 1.0 - sum;
902  diag[index(c_offset_Y + rightExcessSpecies(), j)] = 0;
903 }

905 void AxiStagnFlow::evalContinuity(size_t j, doublereal* x, doublereal* rsd,
906  integer* diag, doublereal rdt)
907 {
908  //----------------------------------------------
909  // Continuity equation
910  //
911  // Note that this propagates the mass flow rate information to the left
912  // (j+1 -> j) from the value specified at the right boundary. The
913  // lambda information propagates in the opposite direction.
914  //
915  // d(\rho u)/dz + 2\rho V = 0
916  //------------------------------------------------
917  rsd[index(c_offset_U,j)] =
918  -(rho_u(x,j+1) - rho_u(x,j))/m_dz[j]
919  -(density(j+1)*V(x,j+1) + density(j)*V(x,j));
920 
921  //algebraic constraint
922  diag[index(c_offset_U, j)] = 0;
923 }

958 void FreeFlame::evalContinuity(size_t j, doublereal* x, doublereal* rsd,
959  integer* diag, doublereal rdt)
960 {
961  //----------------------------------------------
962  // Continuity equation
963  //
964  // d(\rho u)/dz + 2\rho V = 0
965  //----------------------------------------------
966  if (grid(j) > m_zfixed) {
967  rsd[index(c_offset_U,j)] =
968  - (rho_u(x,j) - rho_u(x,j-1))/m_dz[j-1]
969  - (density(j-1)*V(x,j-1) + density(j)*V(x,j));
970  } else if (grid(j) == m_zfixed) {
971  if (m_do_energy[j]) {
972  rsd[index(c_offset_U,j)] = (T(x,j) - m_tfixed);
973  } else {
974  rsd[index(c_offset_U,j)] = (rho_u(x,j)
975  - m_rho[0]*0.3);
976  }
977  } else if (grid(j) < m_zfixed) {
978  rsd[index(c_offset_U,j)] =
979  - (rho_u(x,j+1) - rho_u(x,j))/m_dz[j]
980  - (density(j+1)*V(x,j+1) + density(j)*V(x,j));
981  }
982  //algebraic constraint
983  diag[index(c_offset_U, j)] = 0;
984 }

986 void FreeFlame::_finalize(const doublereal* x)
987 {
988  StFlow::_finalize(x);
989  // If the domain contains the temperature fixed point, make sure that it
990  // is correctly set. This may be necessary when the grid has been modified
991  // externally.
992  if (m_tfixed != Undef) {
993  for (size_t j = 0; j < m_points; j++) {
994  if (z(j) == m_zfixed) {

1001  // fixed temperature.

1003  m_tfixed = T(x, j+1);
1004  m_zfixed = z(j+1);
1005  return;
1006  }
1007  }
