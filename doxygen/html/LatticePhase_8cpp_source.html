151 void LatticePhase::getChemPotentials(doublereal* mu) const
152 {
153  doublereal delta_p = m_Pcurrent - m_Pref;
154  const vector_fp& g_RT = gibbs_RT_ref();
155  for (size_t k = 0; k < m_kk; k++) {
156  double xx = std::max(SmallNumber, moleFraction(k));
157  mu[k] = RT() * (g_RT[k] + log(xx))
158  + delta_p * m_speciesMolarVolume[k];
159  }
160 }

292 void LatticePhase::initThermoXML(XML_Node& phaseNode, const std::string& id_)
293 {
294  if (!id_.empty() && id_ != phaseNode.id()) {
295  throw CanteraError("LatticePhase::initThermoXML",
296  "ids don't match");
297  }
298 
299  /*
300  * Check on the thermo field. Must have:
301  * <thermo model="Lattice" />
302  */
303  if (phaseNode.hasChild("thermo")) {
304  XML_Node& thNode = phaseNode.child("thermo");
305  std::string mString = thNode.attrib("model");
306  if (lowercase(mString) != "lattice") {
307  throw CanteraError("LatticePhase::initThermoXML",
308  "Unknown thermo model: " + mString);
309  }
310  } else {
311  throw CanteraError("LatticePhase::initThermoXML",
312  "Unspecified thermo model");
313  }
314  /*
315  * Now go get the molar volumes. use the default if not found
316  */
317  XML_Node& speciesList = phaseNode.child("speciesArray");
318  XML_Node* speciesDB = get_XML_NameID("speciesData", speciesList["datasrc"], &phaseNode.root());
319 
320  for (size_t k = 0; k < m_kk; k++) {
321  m_speciesMolarVolume[k] = m_site_density;
322  XML_Node* s = speciesDB->findByAttr("name", speciesName(k));
323  if (!s) {
324  throw CanteraError(" LatticePhase::initThermoXML", "database problems");
325  }
326  XML_Node* ss = s->findByName("standardState");
327  if (ss && ss->findByName("molarVolume")) {
328  m_speciesMolarVolume[k] = getFloat(*ss, "molarVolume", "toSI");
329  }
330  }
331 
332  /*
333  * Call the base initThermo, which handles setting the initial
334  * state.
335  */
336  ThermoPhase::initThermoXML(phaseNode, id_);
337 }

339 void LatticePhase::_updateThermo() const
340 {
341  doublereal tnow = temperature();
342  if (m_tlast != tnow) {
343  m_spthermo->update(tnow, &m_cp0_R[0], &m_h0_RT[0], &m_s0_R[0]);
344  m_tlast = tnow;
345  for (size_t k = 0; k < m_kk; k++) {
346  m_g0_RT[k] = m_h0_RT[k] - m_s0_R[k];
347  }
348  m_tlast = tnow;
349  }
350 }
