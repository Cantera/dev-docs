106 {
107  _updateThermo();

285 void SurfPhase::setCoverages(const doublereal* theta)
286 {
287  double sum = 0.0;
288  for (size_t k = 0; k < m_kk; k++) {
289  sum += theta[k];
290  }
291  if (sum <= 0.0) {
292  for (size_t k = 0; k < m_kk; k++) {
293  cout << "theta(" << k << ") = " << theta[k] << endl;
294  }
295  throw CanteraError("SurfPhase::setCoverages",
296  "Sum of Coverage fractions is zero or negative");
297  }
298  for (size_t k = 0; k < m_kk; k++) {
299  m_work[k] = m_n0*theta[k]/(sum*size(k));
300  }
301  /*
302  * Call the Phase:: class function
303  * setConcentrations.
304  */
305  setConcentrations(DATA_PTR(m_work));
306 }

351 void SurfPhase::_updateThermo(bool force) const
352 {
353  doublereal tnow = temperature();
354  if (m_tlast != tnow || force) {
355  m_spthermo->update(tnow, DATA_PTR(m_cp0), DATA_PTR(m_h0),
356  DATA_PTR(m_s0));
357  m_tlast = tnow;
358  for (size_t k = 0; k < m_kk; k++) {
359  m_h0[k] *= GasConstant * tnow;
360  m_s0[k] *= GasConstant;
361  m_cp0[k] *= GasConstant;
362  m_mu0[k] = m_h0[k] - tnow*m_s0[k];
363  }
364  m_tlast = tnow;
365  }
366 }
