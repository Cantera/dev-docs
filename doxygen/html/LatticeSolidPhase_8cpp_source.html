339 void LatticeSolidPhase::installSlavePhases(Cantera::XML_Node* phaseNode)
340 {
341  size_t kk = 0;
342  size_t kstart = 0;
343  SpeciesThermoFactory* spFactory = SpeciesThermoFactory::factory();
344  SpeciesThermo* spthermo_ptr = new GeneralSpeciesThermo();
345  setSpeciesThermo(spthermo_ptr);
346  m_speciesData.clear();
347 
348  XML_Node& eosdata = phaseNode->child("thermo");
349  XML_Node& la = eosdata.child("LatticeArray");
350  std::vector<XML_Node*> lattices;
351  la.getChildren("phase",lattices);
352  for (size_t n = 0; n < m_nlattice; n++) {
353  LatticePhase* lp = m_lattice[n];
354  XML_Node* phaseNode_ptr = lattices[n];
355  size_t nsp = lp->nSpecies();
356  vector<doublereal> constArr(lp->nElements());
357  const vector_fp& aws = lp->atomicWeights();
358  for (size_t es = 0; es < lp->nElements(); es++) {
359  string esName = lp->elementName(es);
360  double wt = aws[es];
361  int an = lp->atomicNumber(es);
362  int e298 = lp->entropyElement298(es); //! @todo Why is this an int instead of a double?
363  int et = lp->elementType(es);
364  addUniqueElementAfterFreeze(esName, wt, an, e298, et);
365  }
366  const std::vector<const XML_Node*> & spNode = lp->speciesData();
367  kstart = kk;
368 
369 
370  for (size_t k = 0; k < nsp; k++) {
371  std::string sname = lp->speciesName(k);
372  std::map<std::string, double> comp;
373  lp->getAtoms(k, DATA_PTR(constArr));
374  size_t nel = nElements();
375  vector_fp ecomp(nel, 0.0);
376  for (size_t m = 0; m < lp->nElements(); m++) {
377  if (constArr[m] != 0.0) {
378  std::string oldEname = lp->elementName(m);
379  size_t newIndex = elementIndex(oldEname);
380  if (newIndex == npos) {
381  throw CanteraError("LatticeSolidPhase::installSlavePhases", "confused");
382  }
383  ecomp[newIndex] = constArr[m];
384  }
385  }
386  double chrg = lp->charge(k);
387  double sz = lp->size(k);
388  addUniqueSpecies(sname, &ecomp[0], chrg, sz);
389  spFactory->installThermoForSpecies(kk, *(spNode[k]), this, *m_spthermo, phaseNode_ptr);
390 
391  m_speciesData.push_back(new XML_Node(*(spNode[k])));
392  kk++;
393  }
394  /*
395  * Add in the lattice stoichiometry constraint
396  */
397  if (n > 0) {
398  string econ = "LC_";
399  econ += int2str(n);
400  econ += "_" + id();
401  size_t m = addUniqueElementAfterFreeze(econ, 0.0, 0, 0.0, CT_ELEM_TYPE_LATTICERATIO);
402  size_t mm = nElements();
403  LatticePhase* lp0 = m_lattice[0];
404  size_t nsp0 = lp0->nSpecies();
405  for (size_t k = 0; k < nsp0; k++) {
406  m_speciesComp[k * mm + m] = -theta_[0];
407  }
408  for (size_t k = 0; k < nsp; k++) {
409  size_t ks = kstart + k;
410  m_speciesComp[ks * mm + m] = theta_[n];
411  }
412  }
413  }
