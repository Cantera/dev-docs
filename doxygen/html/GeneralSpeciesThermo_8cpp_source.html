76 void GeneralSpeciesThermo::install_STIT(size_t index,
77  shared_ptr<SpeciesThermoInterpType> stit_ptr)
78 {
79  if (!stit_ptr) {
80  throw CanteraError("GeneralSpeciesThermo::install_STIT",
81  "zero pointer");
82  }
83  AssertThrowMsg(m_speciesLoc.find(index) == m_speciesLoc.end(),
84  "GeneralSpeciesThermo::install_STIT",
85  "Index position isn't null, duplication of assignment: {}", index);
86  int type = stit_ptr->reportType();
87  m_speciesLoc[index] = {type, m_sp[type].size()};
88  m_sp[type].emplace_back(index, stit_ptr);
89  if (m_sp[type].size() == 1) {
90  m_tpoly[type].resize(stit_ptr->temperaturePolySize());
91  }
92 
93  // Calculate max and min T
94  m_tlow_max = std::max(stit_ptr->minTemp(), m_tlow_max);
95  m_thigh_min = std::min(stit_ptr->maxTemp(), m_thigh_min);
96  markInstalled(index);
97 }

115 void GeneralSpeciesThermo::update(doublereal t, doublereal* cp_R,
116  doublereal* h_RT, doublereal* s_R) const
117 {
118  auto iter = m_sp.begin();
119  auto jter = m_tpoly.begin();
120  for (; iter != m_sp.end(); iter++, jter++) {
121  const std::vector<index_STIT>& species = iter->second;
122  double* tpoly = &jter->second[0];
123  species[0].second->updateTemperaturePoly(t, tpoly);
124  for (size_t k = 0; k < species.size(); k++) {
125  size_t i = species[k].first;
126  species[k].second->updateProperties(tpoly, cp_R+i, h_RT+i, s_R+i);
127  }
