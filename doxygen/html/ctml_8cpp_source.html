450 void getFloats(const Cantera::XML_Node& node, std::map<std::string, double>& v,

458  for (int i = 0; i < n; i++) {
459  const XML_Node& fi = *(f[i]);
460  x = atof(fi().c_str());
461  x0 = Undef;
462  x1 = Undef;
463  typ = fi["type"];
464  title = fi["title"];
465  units = fi["units"];
466  vmin = fi["min"];
467  vmax = fi["max"];
468  if (vmin != "") {
469  x0 = atof(vmin.c_str());

471  writelog("\nWarning: value "+fi()+" is below lower limit of "

473  }
474  }

567  const std::string& type)
568 {
569  doublereal x, x0, x1, fctr = 1.0;
570  string units, vmin, vmax;
571  x = atof(node().c_str());
572  x0 = Undef;
573  x1 = Undef;
574  units = node["units"];
575  vmin = node["min"];
576  vmax = node["max"];
577  if (vmin != "") {
578  x0 = atof(vmin.c_str());
579  if (x < x0 - Tiny) {
580  writelog("\nWarning: value "+node()+" is below lower limit of "

582  }
583  }
584  if (node["max"] != "") {
585  x1 = atof(vmax.c_str());
586  if (x > x1 + Tiny) {
587  writelog("\nWarning: value "+node()+" is above upper limit of "
588  +vmax+".\n");
589  }
590  }
591  // Note, most types of converters default to toSI() type atm.
592  // This may change and become more specific in the future.
593  if (type == "actEnergy" && units != "") {
594  fctr = actEnergyToSI(units);
595  } else if (type == "toSI" && units != "") {
596  fctr = toSI(units);
597  } else if (type == "temperature" && units != "") {
598  fctr = toSI(units);
599  } else if (type == "density" && units != "") {
600  fctr = toSI(units);
601  } else if (type == "pressure" && units != "") {
602  fctr = toSI(units);
603  } else if (type != "" && units != "") {
604  fctr = toSI(units);
605 #ifdef DEBUG_MODE

609  }
610  // Note, below currently produces a lot of output due to transport blocks.

612 #ifdef DEBUG_MODE_MORE
613  else if (type == "" && units != "") {

616  "but no conversion was done because the getFloat() command didn't have a type\n");
617  }
618 #endif
619  return fctr*x;
620 }
