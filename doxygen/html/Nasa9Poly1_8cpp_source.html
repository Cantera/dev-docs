51 void Nasa9Poly1::updateProperties(const doublereal* tt,
52  doublereal* cp_R, doublereal* h_RT,
53  doublereal* s_R) const
54 {
55 
56  doublereal ct0 = m_coeff[0] * tt[5]; // a0 / (T^2)
57  doublereal ct1 = m_coeff[1] * tt[4]; // a1 / T
58  doublereal ct2 = m_coeff[2]; // a2
59  doublereal ct3 = m_coeff[3] * tt[0]; // a3 * T
60  doublereal ct4 = m_coeff[4] * tt[1]; // a4 * T^2
61  doublereal ct5 = m_coeff[5] * tt[2]; // a5 * T^3
62  doublereal ct6 = m_coeff[6] * tt[3]; // a6 * T^4
63 
64  doublereal cpdivR = ct0 + ct1 + ct2 + ct3 + ct4 + ct5 + ct6;
65  doublereal hdivRT = -ct0 + tt[6]*ct1 + ct2 + 0.5*ct3 + 1.0/3.0*ct4
66  + 0.25*ct5 + 0.2*ct6 + m_coeff[7] * tt[4];
67  doublereal sdivR = -0.5*ct0 - ct1 + tt[6]*ct2 + ct3 + 0.5*ct4
68  + 1.0/3.0*ct5 + 0.25*ct6 + m_coeff[8];
69 
70  // return the computed properties for this species
71  *cp_R = cpdivR;
72  *h_RT = hdivRT;
73  *s_R = sdivR;
74 }
