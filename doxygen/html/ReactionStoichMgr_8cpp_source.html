63 void ReactionStoichMgr::add(size_t rxn, const ReactionData& r)
64 {
65  size_t k;
66  std::vector<size_t> rk;
67  doublereal frac;
68  doublereal oo, os, of;
69  bool doGlobal = false;
70  std::vector<size_t> extReactants = r.reactants;
71  vector_fp extRStoich = r.rstoich;
72  vector_fp extROrder = r.rorder;
73 
74  //
75  // If we have a complete global reaction then we need to do something more complete
76  // than the previous treatment. Basically we will use the reactant manager to calculate the
77  // global forward reaction rate of progress.
78  //
79  if (r.forwardFullOrder_.size() > 0) {
80  //
81  // Trigger a treatment where the order of the reaction and the stoichiometry
82  // are treated as different.
83  //
84  doGlobal = true;
85  size_t nsp = r.forwardFullOrder_.size();
86  //
87  // Set up a signal vector to indicate whether the species has been added into
88  // the input vectors for the stoich manager
89  //
90  vector_int kHandled(nsp, 0);
91  //
92  // Loop over the reactants which are also nonzero stoichioemtric entries
93  // making sure the forwardFullOrder_ entries take precedence over rorder entries
94  //
95  for (size_t kk = 0; kk < r.reactants.size(); kk++) {
96  k = r.reactants[kk];
97  os = r.rstoich[kk];
98  oo = r.rorder[kk];
99  of = r.forwardFullOrder_[k];
100  if (of != oo) {
101  extROrder[kk] = of;
102  }
103  kHandled[k] = 1;
104  }
105  for (k = 0; k < nsp; k++) {
106  of = r.forwardFullOrder_[k];
107  if (of != 0.0) {
108  if (kHandled[k] == 0) {
109  //
110  // Add extra entries to reactant inputs. Set their reactant stoichiometric entries to zero.
111  //
112  extReactants.push_back(k);
113  extROrder.push_back(of);
114  extRStoich.push_back(0.0);
115  }
116  }
117  }
118  }
119 
120  bool isfrac = false;
121  for (size_t n = 0; n < r.reactants.size(); n++) {
122  size_t ns = size_t(r.rstoich[n]);
123  frac = r.rstoich[n] - 1.0*int(r.rstoich[n]);
124  if (frac != 0.0) {
125  isfrac = true;
126  }
127  for (size_t m = 0; m < ns; m++) {
128  rk.push_back(r.reactants[n]);
129  }
130  }
131 
132  //
133  // If the reaction is non-mass action add it in in a general way
134  // Reactants get extra terms for the forward reaction rate of progress
135  // that may have zero stoichiometries.
136  //
137  if (doGlobal) {
138  m_reactants.add(rxn, extReactants, extROrder, extRStoich);
139  } else {
140  //
141  // this is confusing. The only issue should be whether rorder is different than rstoich!
142  //
143  if (isfrac || r.global || rk.size() > 3) {
144  m_reactants.add(rxn, r.reactants, r.rorder, r.rstoich);
145  } else {
146  m_reactants.add(rxn, rk);
147  }
148  }
149 
150  std::vector<size_t> pk;
151  isfrac = false;
152  for (size_t n = 0; n < r.products.size(); n++) {
153  size_t ns = size_t(r.pstoich[n]);
154  frac = r.pstoich[n] - 1.0*int(r.pstoich[n]);
155  if (frac != 0.0) {
156  isfrac = true;
157  }
158  for (size_t m = 0; m < ns; m++) {
159  pk.push_back(r.products[n]);
160  }
161  }
162 
163  if (r.reversible) {
164  //
165  // this is confusing. The only issue should be whether porder is different than pstoich!
166  //
167  if (pk.size() > 3 || r.isReversibleWithFrac) {
168  m_revproducts.add(rxn, r.products, r.porder, r.pstoich);
169  } else {
170  m_revproducts.add(rxn, pk);
171  }
172  } else {
173  //
174  // this is confusing. The only issue should be whether porder is different than pstoich!
175  //
176  if (isfrac || pk.size() > 3) {
177  m_irrevproducts.add(rxn, r.products, r.porder, r.pstoich);
178  } else {
179  m_irrevproducts.add(rxn, pk);
180  }
181  }
182 }

185 void ReactionStoichMgr::getCreationRates(size_t nsp, const doublereal* ropf,
186  const doublereal* ropr, doublereal* c)
187 {
188  // zero out the output array
189  fill(c, c + nsp, 0.0);
190 
191  // the forward direction creates product species
