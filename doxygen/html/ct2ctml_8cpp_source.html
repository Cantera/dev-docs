24 //! return the full path to the Python interpreter.
25 /*!
26  * Use the environment variable PYTHON_CMD if it is set. If not, return
27  * the string 'python'.
28  *
29  * Note, there are hidden problems here that really direct us to use
30  * a full pathname for the location of python. Basically the system
31  * call will use the shell /bin/sh, in order to launch python.
32  * This default shell may not be the shell that the user is employing.
33  * Therefore, the default path to python may be different during
34  * a system call than during the default user shell environment.
35  * This is quite a headache. The answer is to always set the
36  * PYTHON_CMD environmental variable in the user environment to
37  * an absolute path to locate the python executable. Then this
38  * issue goes away.
39  */
40 static string pypath()
41 {
42  string s = "python";
43  const char* py = getenv("PYTHON_CMD");
44 
45  if (py) {
46  string sp = stripws(string(py));
47  if (sp.size() > 0) {
48  s = sp;
49  }
50  }
51  return s;
52 }

58 #ifdef _WIN32

62  std::bind2nd(std::equal_to<char>(), '\\'), '/') ;
63 #endif
64  size_t idir = out_name.rfind('/');

80 #ifdef HAS_NO_PYTHON

82  * Section to bomb out if python is not

95  stringstream output_stream, error_stream;

97  args.push_back("-c");
98  args.push_back(
99  "from __future__ import print_function\n"

101  "try:\n"
102  " from cantera import ctml_writer\n"
103  "except ImportError:\n"
104  " print('sys.path: ' + repr(sys.path) + '\\n', file=sys.stderr)\n"
105  " raise\n"

107  "sys.exit(0)\n");
108 
109  python.start(pypath(), args.begin(), args.end());
110  std::string line;
111  while (true) {
112  if (python.out().good()) {
113  std::getline(python.out(), line);
114  output_stream << line << std::endl;
115  } else if (python.err().good()) {
116  std::getline(python.err(), line);
117  error_stream << line << std::endl;
118  } else {
119  break;
120  }
121  }
122  python.close();
123  python_exit_code = python.exit_code();
124  error_output = stripws(error_stream.str());

127  // Report failure to execute Python
128  stringstream message;

130  message << "Python command was: '" << pypath() << "'\n";
131  message << err.what() << std::endl;
132  throw CanteraError("ct2ctml_string", message.str());
133  }

139  message << "Python command was: '" << pypath() << "'\n";
140  message << "The exit code was: " << python_exit_code << "\n";
141  if (error_output.size() > 0) {
142  message << "-------------- start of converter log --------------\n";
143  message << error_output << std::endl;
144  message << "--------------- end of converter log ---------------";
145  } else {
146  message << "The command did not produce any output." << endl;
147  }
148  throw CanteraError("ct2ctml_string", message.str());
149  }
150 
151  if (error_output.size() > 0) {
152  // Warn if there was any output from the conversion process
153  stringstream message;
154  message << "Warning: Unexpected output from CTI converter\n";
155  message << "-------------- start of converter log --------------\n";
156  message << error_output << std::endl;
157  message << "--------------- end of converter log ---------------\n";
158  writelog(message.str());
159  }
160  return python_output;
161 }

253 void get_CTML_Tree(Cantera::XML_Node* rootPtr, const std::string& file, const int debug)
254 {
255  std::string ext = "";
256 
257  // find the input file on the Cantera search path
258  std::string inname = findInputFile(file);
259  writelog("Found file: "+inname+"\n", debug);
260 
261  if (inname == "") {
262  throw CanteraError("get_CTML_Tree", "file "+file+" not found");
263  }
264 
265  /*
266  * Check whether or not the file is XML. If not, it will be first
267  * processed with the preprocessor.
268  */
269  std::string::size_type idot = inname.rfind('.');
270  if (idot != string::npos) {
271  ext = inname.substr(idot, inname.size());
272  }
273  if (ext != ".xml" && ext != ".ctml") {
274  string phase_xml = ctml::ct2ctml_string(inname);
275  stringstream s(phase_xml);
276  rootPtr->build(s);
277  return;
278  }
279 
280  writelog("Attempting to parse xml file " + inname + "\n", debug);
281  ifstream fin(inname.c_str());
282  if (!fin) {
283  throw
284  CanteraError("get_CTML_Tree",
285  "XML file " + inname + " not found");
286  }
287  rootPtr->build(fin);
288  fin.close();
289 }
