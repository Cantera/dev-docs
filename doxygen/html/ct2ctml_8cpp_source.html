26 //! return the full path to the Python interpreter.
27 /*!
28  * Use the environment variable PYTHON_CMD if it is set. If not, return
29  * the string 'python'.
30  *
31  * Note, there are hidden problems here that really direct us to use
32  * a full pathname for the location of python. Basically the system
33  * call will use the shell /bin/sh, in order to launch python.
34  * This default shell may not be the shell that the user is employing.
35  * Therefore, the default path to python may be different during
36  * a system call than during the default user shell environment.
37  * This is quite a headache. The answer is to always set the
38  * PYTHON_CMD environmental variable in the user environment to
39  * an absolute path to locate the python executable. Then this
40  * issue goes away.
41  */
42 static string pypath()
43 {
44  string s = "python";
45  const char* py = getenv("PYTHON_CMD");
46 
47  if (py) {
48  string sp = stripws(string(py));
49  if (sp.size() > 0) {
50  s = sp;
51  }
52  }
53  return s;
54 }

56 void ct2ctml(const char* file, const int debug)
57 {
58  string xml = ct2ctml_string(file);
59  string out_name = file;
60 #ifdef _WIN32
61  // For Windows, make the path POSIX compliant so code looking for directory
62  // separators is simpler. Just look for '/' not both '/' and '\\'
63  std::replace_if(out_name.begin(), out_name.end(),
64  std::bind2nd(std::equal_to<char>(), '\\'), '/') ;
65 #endif
66  size_t idir = out_name.rfind('/');
67  if (idir != npos) {
68  out_name = out_name.substr(idir+1, out_name.size());
69  }
70  size_t idot = out_name.rfind('.');
71  if (idot != npos) {
72  out_name = out_name.substr(0, idot) + ".xml";
73  } else {
74  out_name += ".xml";
75  }
76  std::ofstream out(out_name.c_str());
77  out << xml;

80 static std::string call_ctml_writer(const std::string& text, bool isfile)
81 {
82  std::string file, arg;
83  if (isfile) {
84  file = text;

91 #ifdef HAS_NO_PYTHON
92  /*
93  * Section to bomb out if python is not
94  * present in the computation environment.
95  */
96  throw CanteraError("ct2ctml",

104  exec_stream_t python;
105  python.set_wait_timeout(exec_stream_t::s_all, 1800000); // 30 minutes
106  stringstream output_stream, error_stream;
107  std::vector<string> args;
108  args.push_back("-c");
109 
110  args.push_back(
111  "from __future__ import print_function\n"
112  "import sys\n"
113  "try:\n"
114  " from cantera import ctml_writer\n"
115  "except ImportError:\n"
116  " print('sys.path: ' + repr(sys.path) + '\\n', file=sys.stderr)\n"
117  " raise\n"

119  "sys.exit(0)\n");

121  python.start(pypath(), args.begin(), args.end());
122  std::string line;
123 
124  while (python.out().good()) {
125  std::getline(python.out(), line);
126  output_stream << line << std::endl;
127  }
128 
129 #ifdef _WIN32
130  // Sleeping for 1 ms prevents a (somewhat inexplicable) deadlock while
131  // reading from the stream.
132  Sleep(1);
133 #endif
134  while (python.err().good()) {
135  std::getline(python.err(), line);
136  error_stream << line << std::endl;
137  }
138  python.close();

140  error_output = stripws(error_stream.str());

142  } catch (std::exception& err) {

147  message << err.what() << std::endl;
148  throw CanteraError("ct2ctml_string", message.str());
149  }
150

152  // Report a failure in the conversion process
153  stringstream message;
154  message << "Error converting input file \"" << file << "\" to CTML.\n";
155  message << "Python command was: '" << pypath() << "'\n";
156  message << "The exit code was: " << python_exit_code << "\n";
157  if (error_output.size() > 0) {
158  message << "-------------- start of converter log --------------\n";
159  message << error_output << std::endl;
160  message << "--------------- end of converter log ---------------";
161  } else {
162  message << "The command did not produce any output." << endl;
163  }
164  throw CanteraError("ct2ctml_string", message.str());
165  }
166 
167  if (error_output.size() > 0) {
168  // Warn if there was any output from the conversion process
169  stringstream message;
170  message << "Warning: Unexpected output from CTI converter\n";
171  message << "-------------- start of converter log --------------\n";
172  message << error_output << std::endl;
173  message << "--------------- end of converter log ---------------\n";
174  writelog(message.str());
175  }
176  return python_output;
177 }
