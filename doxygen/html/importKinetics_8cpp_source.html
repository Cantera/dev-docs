244  // reaction order

274 /**
275  * getStick() processes the XML element called Stick that specifies
276  * the sticking coefficient reaction. This routine will
277  * translate the sticking coefficient value into a "normal"
278  * rate constant for the surface reaction.
279  *
280  * Output
281  * -----------
282  * Output is the normal Arrhenius expressions for a surface
283  * reaction rate constant.
284  *
285  * A - units such that rate of rxn has kmol/m^2/s when
286  * A is multiplied by activity concentrations of
287  * reactants in the normal manner.
288  * n - unitless
289  * E - Units 1/Kelvin
290  */
291 static void getStick(const XML_Node& node, Kinetics& kin,
292  ReactionData& r, doublereal& A, doublereal& b, doublereal& E)
293 {
294  size_t nr = r.reactants.size();
295  size_t k, klocal, not_surf = 0;
296  size_t np = 0;
297  doublereal f = 1.0;
298  doublereal order;
299  /*
300  * species is the name of the special reactant whose surface
301  * flux rate will be calculated.
302  * isp = species # in the local phase
303  * ispKinetics = species # in the kinetics object
304  * ispPhaseIndex = phase # of the special species
305  */
306  string spname = node["species"];
307  ThermoPhase& th = kin.speciesPhase(spname);
308  size_t isp = th.speciesIndex(spname);
309  size_t ispKinetics = kin.kineticsSpeciesIndex(spname);
310  size_t ispPhaseIndex = kin.speciesPhaseIndex(ispKinetics);
311 
312  doublereal ispMW = th.molecularWeights()[isp];
313  doublereal sc;
314 
315  // loop over the reactants
316  for (size_t n = 0; n < nr; n++) {
317  k = r.reactants[n];
318  order = r.rorder[n]; // stoich coeff
319 
320  // get the phase species k belongs to
321  np = kin.speciesPhaseIndex(k);
322  const ThermoPhase& p = kin.thermo(np);
323 
324  // get the local index of species k in this phase
325  klocal = p.speciesIndex(kin.kineticsSpeciesName(k));
326 
327  // if it is a surface species, divide f by the standard
328  // concentration for this species, in order to convert
329  // from concentration units used in the law of mass action
330  // to coverages used in the sticking probability
331  // expression
332  if (p.eosType() == cSurf || p.eosType() == cEdge) {
333  sc = p.standardConcentration(klocal);
334  f /= pow(sc, order);
335  }
336  // Otherwise:
337  else {
338  // We only allow one species to be in the phase
339  // containing the special sticking coefficient
340  // species.
341  if (ispPhaseIndex == np) {
342  not_surf++;
343  }
344  // Other bulk phase species on the other side
345  // of ther interface are treated like surface
346  // species.
347  else {
348  sc = p.standardConcentration(klocal);
349  f /= pow(sc, order);
350  }
351  }
352  }
353  if (not_surf != 1) {
354  throw CanteraError("getStick",
355  "reaction probabilities can only be used in "
356  "reactions with exactly 1 gas/liquid species.");
357  }
358 
359  doublereal cbar = sqrt(8.0*GasConstant/(Pi*ispMW));
360  A = 0.25 * getFloat(node, "A", "toSI") * cbar * f;
361  b = getFloat(node, "b") + 0.5;
362  E = getFloat(node, "E", "actEnergy");
363  E /= GasConstant;
364 }

568 doublereal isDuplicateReaction(std::map<int, doublereal>& r1,
569  std::map<int, doublereal>& r2)
570 {
571 
572  map<int, doublereal>::const_iterator b = r1.begin(), e = r1.end();
573  int k1 = b->first;
574  doublereal ratio = 0.0;
575  if (r1[k1] == 0.0 || r2[k1] == 0.0) {
576  goto next;
577  }
578  ratio = r2[k1]/r1[k1];
579  ++b;
580  for (; b != e; ++b) {
581  k1 = b->first;
582  if (r1[k1] == 0.0 || r2[k1] == 0.0) {
583  goto next;
584  }
585  if (fabs(r2[k1]/r1[k1] - ratio) > 1.e-8) {
586  goto next;
587  }
588  }
589  return ratio;
590 next:
591  ratio = 0.0;
592  b = r1.begin();
593  k1 = b->first;
594  if (r1[k1] == 0.0 || r2[-k1] == 0.0) {
595  return 0.0;
596  }
597  ratio = r2[-k1]/r1[k1];
598  ++b;
599  for (; b != e; ++b) {
600  k1 = b->first;
601  if (r1[k1] == 0.0 || r2[-k1] == 0.0) {
602  return 0.0;
603  }
604  if (fabs(r2[-k1]/r1[k1] - ratio) > 1.e-8) {
605  return 0.0;
606  }
607  }
608  return ratio;
609 }

617  throw CanteraError(" rxninfo::installReaction",
618  " expected xml node reaction, got " + r.name());
619  }
620 
621  // We use the ReactionData object to store initial values read in from the
622  // xml data. Then, when we have collected everything we add the reaction to
623  // the kinetics object, kin, at the end of the routine.
624  ReactionData& rdata = **m_rdata.insert(m_rdata.end(), new ReactionData());
625  rdata.validate = validate_rxn;
626 
627  // Check to see if the reaction is specified to be a duplicate of another
628  // reaction. It's an error if the reaction is a duplicate and this is not
629  // set.
630  rdata.duplicate = (r.hasAttrib("duplicate")) ? 1 : 0;
631 
632  // Check to see if the reaction rate constant can be negative. It's an
633  // error if a negative rate constant is found and this is not set.
634  rules.allowNegativeA = (r.hasAttrib("negative_A")) ? 1 : 0;
635 
636  // Use the contents of the "equation" child element as the reaction's
637  // string representation. Post-process to convert "[" and "]" characters
638  // back into "<" and ">" which cannot easily be stored in an XML file. This
639  // reaction string is used only for display purposes. It is not parsed for
640  // the identities of reactants or products.
641  rdata.equation = (r.hasChild("equation")) ? r("equation") : "<no equation>";
642  for (size_t nn = 0; nn < rdata.equation.size(); nn++) {
643  if (rdata.equation[nn] == '[') {
644  rdata.equation[nn] = '<';
645  } else if (rdata.equation[nn] == ']') {
646  rdata.equation[nn] = '>';
647  }
648  }
649 
650  // get the reactants
651  bool ok = getReagents(r, kin, 1, default_phase, rdata.reactants,
652  rdata.rstoich, rdata.rorder, rules);
653 
654  // Get the products. We store the id of products in rdata.products
655  ok = ok && getReagents(r, kin, -1, default_phase, rdata.products,
656  rdata.pstoich, rdata.porder, rules);
657 
658  // if there was a problem getting either the reactants or the products,
659  // then abort.
660  if (!ok) {
661  return false;
662  }
663 
664  // check whether the reaction is specified to be
665  // reversible. Default is irreversible.
666  string isrev = r["reversible"];
667  rdata.reversible = (isrev == "yes" || isrev == "true");
668 
669  // If reaction orders are specified, then this reaction does not follow
670  // mass-action kinetics, and is not an elementary reaction. So check that
671  // it is not reversible, since computing the reverse rate from
672  // thermochemistry only works for elementary reactions. Set the type to
673  // global, so that kinetics managers will know to process the reaction
674  // orders.
675  if (r.hasChild("order")) {
676  if (rdata.reversible == true)
677  throw CanteraError("installReaction",
678  "reaction orders may only be given for "
679  "irreversible reactions");
680  rdata.global = true;
681  }
682 
683  // Some reactions can be elementary reactions but have fractional
684  // stoichiometries wrt to some products and reactants. An example of these
685  // are solid reactions involving phase transformations. Species with
686  // fractional stoichiometries must be from single-species phases with
687  // unity activities. For these reactions set the bool isReversibleWithFrac
688  // to true.
689  if (rdata.reversible == true) {
690  for (size_t i = 0; i < rdata.products.size(); i++) {
691  doublereal po = rdata.porder[i];
692  AssertTrace(po == rdata.pstoich[i]);
693  doublereal chk = po - 1.0 * int(po);
694  if (chk != 0.0) {
695  size_t k = rdata.products[i];
696  // Special case when k is a single species phase.
697  if (kin.speciesPhase(k).nSpecies() == 1) {
698  rdata.porder[i] = 0.0;
699  }
700 
701  rdata.isReversibleWithFrac = true;
702  }
703  }
704  for (size_t i = 0; i < rdata.reactants.size(); i++) {
705  doublereal ro = rdata.rorder[i];
706  AssertTrace(ro == rdata.rstoich[i]);
707  doublereal chk = ro - 1.0 * int(ro);
708  if (chk != 0.0) {
709  size_t k = rdata.reactants[i];
710  // Special case when k is a single species phase.
711  if (kin.speciesPhase(k).nSpecies() == 1) {
712  rdata.rorder[i] = 0.0;
713  }
714  rdata.isReversibleWithFrac = true;
715  }
716  }
717  }
718 
719  /*
720  * Search the reaction element for the attribute "type".
721  * If found, then branch on the type, to fill in appropriate
722  * fields in rdata.
723  */
724  rdata.reactionType = ELEMENTARY_RXN;
725  string typ = r["type"];
726  if (typ == "falloff") {
727  rdata.reactionType = FALLOFF_RXN;
728  rdata.falloffType = SIMPLE_FALLOFF;
729  } else if (typ == "chemAct") {
730  rdata.reactionType = CHEMACT_RXN;
731  rdata.falloffType = SIMPLE_FALLOFF;
732  } else if (typ == "threeBody") {
733  rdata.reactionType = THREE_BODY_RXN;
734  } else if (typ == "plog") {
735  rdata.reactionType = PLOG_RXN;
736  } else if (typ == "chebyshev") {
737  rdata.reactionType = CHEBYSHEV_RXN;
738  } else if (typ == "surface") {
739  rdata.reactionType = SURFACE_RXN;
740  } else if (typ == "edge") {
741  rdata.reactionType = EDGE_RXN;
742  } else if (typ != "") {
743  throw CanteraError("installReaction", "Unknown reaction type: " + typ);
744  }
745 
746  // Look for undeclared duplicate reactions.
747  if (validate_rxn) {
748  vector<char> participants(kin.nTotalSpecies(), 0);
749  for (size_t nn = 0; nn < rdata.reactants.size(); nn++) {
750  rdata.net_stoich[-1 - int(rdata.reactants[nn])] -= rdata.rstoich[nn];
751  participants[rdata.reactants[nn]] += 1;
752  }
753  for (size_t nn = 0; nn < rdata.products.size(); nn++) {
754  rdata.net_stoich[int(rdata.products[nn])+1] += rdata.pstoich[nn];
755  participants[rdata.products[nn]] += 2;
756  }
757 
758  vector<size_t>& related = m_participants[participants];
759  for (size_t mm = 0; mm < related.size(); mm++) {
760  ReactionData& other = *m_rdata[related[mm]];
761  if ((rdata.reactants.size() == other.reactants.size())
762  && (rdata.reactionType == other.reactionType)) {
763  doublereal c = isDuplicateReaction(rdata.net_stoich, other.net_stoich);
764  if (c > 0.0
765  || (c < 0.0 && rdata.reversible)
766  || (c < 0.0 && other.reversible)) {
767  if ((!rdata.duplicate || !other.duplicate)) {
768  string msg = string("Undeclared duplicate reactions detected: \n")
769  +"Reaction "+int2str(other.number+1)+": "+other.equation
770  +"\nReaction "+int2str(iRxn+1)+": "+rdata.equation+"\n";
771  throw CanteraError("installReaction", msg);
772  }
773  }
774  }
775  }
776  m_participants[participants].push_back(m_rdata.size() - 1);
777  }
778 
779  rdata.number = iRxn;
780  rdata.rxn_number = iRxn;
781 
782  // Read the rate coefficient data from the XML file. Trigger an
783  // exception for negative A unless specifically authorized.
784  getRateCoefficient(r.child("rateCoeff"), kin, rdata, rules);
785 
786  // Check to see that the elements balance in the reaction.
787  // Throw an error if they don't
788  if (validate_rxn) {
789  checkRxnElementBalance(kin, rdata);
790  }
791 
792  // Ok we have read everything in about the reaction. Add it to the
793  // kinetics object by calling the Kinetics member function addReaction()
794  kin.addReaction(rdata);
795  return true;
796 }

798 bool installReactionArrays(const XML_Node& p, Kinetics& kin,
799  std::string default_phase, bool check_for_duplicates)
800 {
801  rxninfo _rxns;
802 
803  vector<XML_Node*> rarrays;
804  int itot = 0;
805  /*
806  * Search the children of the phase element for the
807  * xml element named reactionArray. If we can't find it,
808  * then return signaling having not found any reactions.
809  * Apparently, we allow multiple reactionArray elements here
810  * Each one will be processed sequentially, with the
811  * end result being purely additive.
812  */
813  p.getChildren("reactionArray",rarrays);
814  int na = static_cast<int>(rarrays.size());
815  if (na == 0) {
816  kin.finalize();
817  return false;
818  }
819  for (int n = 0; n < na; n++) {
820  /*
821  * Go get a reference to the current xml element,
822  * reactionArray. We will process this element now.
823  */
824  const XML_Node& rxns = *rarrays[n];
825  /*
826  * The reactionArray element has an attribute called,
827  * datasrc. The value of the attribute is the xml
828  * element comprising the top of the
829  * tree of reactions for the phase.
830  * Find this datasrc element starting with the root
831  * of the current xml node.
832  */
833  const XML_Node* rdata = get_XML_Node(rxns["datasrc"], &rxns.root());
834  /*
835  * If the reactionArray element has a child element named "skip", and
836  * if the attribute of skip called "species" has a value of "undeclared",
837  * we will set rxnrule.skipUndeclaredSpecies to 'true'. rxnrule is
838  * passed to the routine that parses each individual reaction so that
839  * the parser will skip all reactions containing an undefined species
840  * without throwing an error.
841  *
842  * Similarly, an attribute named "third_bodies" with the value of
843  * "undeclared" will skip undeclared third body efficiencies (while
844  * retaining the reaction and any other efficiencies).
845  */
846  ReactionRules rxnrule;
847  if (rxns.hasChild("skip")) {
848  const XML_Node& sk = rxns.child("skip");
849  string sskip = sk["species"];
850  if (sskip == "undeclared") {
851  rxnrule.skipUndeclaredSpecies = true;
852  }
853  if (sk["third_bodies"] == "undeclared") {
854  rxnrule.skipUndeclaredThirdBodies = true;
855  }
856  }
857  int i, nrxns = 0;
858  /*
859  * Search for child elements called include. We only include
860  * a reaction if it's tagged by one of the include fields.
861  * Or, we include all reactions if there are no include fields.
862  */
863  vector<XML_Node*> incl;
864  rxns.getChildren("include",incl);
865  int ninc = static_cast<int>(incl.size());
866 
867  vector<XML_Node*> allrxns;
868  rdata->getChildren("reaction",allrxns);
869  nrxns = static_cast<int>(allrxns.size());
870  // if no 'include' directive, then include all reactions
871  if (ninc == 0) {
872  for (i = 0; i < nrxns; i++) {
873  const XML_Node* r = allrxns[i];
874  if (r) {
875  if (_rxns.installReaction(itot, *r, kin,
876  default_phase, rxnrule, check_for_duplicates)) {
877  ++itot;
878  }
879  }
880  }
881  } else {
882  for (int nii = 0; nii < ninc; nii++) {
883  const XML_Node& ii = *incl[nii];
884  string imin = ii["min"];
885  string imax = ii["max"];
886 
887  string::size_type iwild = string::npos;
888  if (imax == imin) {
889  iwild = imin.find("*");
890  if (iwild != string::npos) {
891  imin = imin.substr(0,iwild);
892  imax = imin;
893  }
894  }
895 
896  for (i = 0; i < nrxns; i++) {
897  const XML_Node* r = allrxns[i];
898  string rxid;
899  if (r) {
900  rxid = (*r)["id"];
901  if (iwild != string::npos) {
902  rxid = rxid.substr(0,iwild);
903  }
904  /*
905  * To decide whether the reaction is included or not
906  * we do a lexical min max and operation. This
907  * sometimes has surprising results.
908  */
909  if ((rxid >= imin) && (rxid <= imax)) {
910  if (_rxns.installReaction(itot, *r, kin,
911  default_phase, rxnrule, check_for_duplicates)) {
912  ++itot;
913  }
914  }
915  }
916  }
917  }
918  }
919  }
920 
921  /*
922  * Finalize the installation of the kinetics, now that we know
923  * the true number of reactions in the mechanism, itot.
924  */
925  kin.finalize();
926 
927  return true;
928 }

930 bool importKinetics(const XML_Node& phase, std::vector<ThermoPhase*> th,
931  Kinetics* k)
932 {
933 
934  if (k == 0) {
935  return false;
936  }
937 
938  Kinetics& kin = *k;
939 
940  // This phase will be the owning phase for the kinetics operator
941  // For interfaces, it is the surface phase between two volumes.
942  // For homogeneous kinetics, it's the current volumetric phase.
943  string owning_phase = phase["id"];
944 
945  bool check_for_duplicates = false;
946  if (phase.parent()->hasChild("validate")) {
947  const XML_Node& d = phase.parent()->child("validate");
948  if (d["reactions"] == "yes") {
949  check_for_duplicates = true;
950  }
951  }
952 
953  // if other phases are involved in the reaction mechanism,
954  // they must be listed in a 'phaseArray' child
955  // element. Homogeneous mechanisms do not need to include a
956  // phaseArray element.
957 
958  vector<string> phase_ids;
959  if (phase.hasChild("phaseArray")) {
960  const XML_Node& pa = phase.child("phaseArray");
961  getStringArray(pa, phase_ids);
962  }
963  phase_ids.push_back(owning_phase);
964 
965  int np = static_cast<int>(phase_ids.size());
966  int nt = static_cast<int>(th.size());
967 
968  // for each referenced phase, attempt to find its id among those
969  // phases specified.
970  bool phase_ok;
971 
972  string phase_id;
973  string msg = "";
974  for (int n = 0; n < np; n++) {
975  phase_id = phase_ids[n];
976  phase_ok = false;
977 
978  // loop over the supplied 'ThermoPhase' objects representing
979  // phases, to find an object with the same id.
980  for (int m = 0; m < nt; m++) {
981  if (th[m]->id() == phase_id) {
982  phase_ok = true;
983 
984  // if no phase with this id has been added to
985  //the kinetics manager yet, then add this one
986  if (kin.phaseIndex(phase_id) == npos) {
987  kin.addPhase(*th[m]);
988  }
989  }
990  msg += " "+th[m]->id();
991  }
992  if (!phase_ok) {
993  throw CanteraError("importKinetics",
994  "phase "+phase_id+" not found. Supplied phases are:"+msg);
995  }
996  }
997 
998  // allocates arrays, etc. Must be called after the phases have
999  // been added to 'kin', so that the number of species in each
1000  // phase is known.
1001  kin.init();
1002 
1003  // Install the reactions.
1004  return installReactionArrays(phase, kin, owning_phase, check_for_duplicates);
1005 }
