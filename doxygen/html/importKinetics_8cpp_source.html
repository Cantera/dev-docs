85 void checkRxnElementBalance(Kinetics& kin,
86  const ReactionData& rdata, doublereal errorTolerance)
87 {
88  doublereal kstoich;
89 
90  map<string, double> bal, balr, balp;
91  bal.clear();
92  balp.clear();
93  balr.clear();
94  size_t np = rdata.products.size();
95 
96  // iterate over the products
97  for (size_t index = 0; index < np; index++) {
98  size_t kp = rdata.products[index]; // index of the product in 'kin'
99  size_t n = kin.speciesPhaseIndex(kp); // phase this product belongs to
100  size_t klocal = kp - kin.kineticsSpeciesIndex(0,n); // index within this phase
101  kstoich = rdata.pstoich[index]; // product stoichiometric coeff
102  const ThermoPhase& ph = kin.speciesPhase(kp);
103  for (size_t m = 0; m < ph.nElements(); m++) {
104  bal[ph.elementName(m)] += kstoich*ph.nAtoms(klocal,m);
105  balp[ph.elementName(m)] += kstoich*ph.nAtoms(klocal,m);
106  }
107  }
108  for (size_t index = 0; index < rdata.reactants.size(); index++) {
109  size_t kr = rdata.reactants[index];
110  size_t n = kin.speciesPhaseIndex(kr);
111  size_t klocal = kr - kin.kineticsSpeciesIndex(0,n);
112  kstoich = rdata.rstoich[index];
113  const ThermoPhase& ph = kin.speciesPhase(kr);
114  for (size_t m = 0; m < ph.nElements(); m++) {
115  bal[ph.elementName(m)] -= kstoich*ph.nAtoms(klocal,m);
116  balr[ph.elementName(m)] += kstoich*ph.nAtoms(klocal,m);
117  }
118  }
119 
120  map<string, double>::iterator b = bal.begin();
121  string msg = "\n\tElement Reactants Products";
122  bool ok = true;
123  doublereal err, elemsum;
124  for (; b != bal.end(); ++b) {
125  elemsum = fabs(balr[b->first]) + fabs(balp[b->first]);
126  if (elemsum > 0.0) {
127  err = fabs(b->second/elemsum);
128  if (err > errorTolerance) {
129  ok = false;
130  msg += "\n\t"+b->first+" "+ fp2str(balr[b->first])
131  +" "+ fp2str(balp[b->first]);
132  }
133  }
134  }
135  if (!ok) {
136  msg = "The following reaction is unbalanced:\n\t"
137  + rdata.equation + "\n" + msg + "\n";
138  throw CanteraError("checkRxnElementBalance",msg);
139  }

143  std::string default_phase, std::vector<size_t>& spnum,
144  vector_fp& stoich, vector_fp& order,
145  const ReactionRules& rules)

149 
150  /*
151  * The id of reactants and products are kept in child elements
152  * of reaction, named "reactants" and "products". We search
153  * the xml tree for these children based on the value of rp,
154  * and store the xml element pointer here.
155  */
156  if (rp == 1) {
157  rptype = "reactants";
158  } else {
159  rptype = "products";
160  }

162 
163  /*
164  * The species and stoichiometric coefficient for the species
165  * are stored as a colon separated pair. Get all of these
166  * pairs in the reactions/products object.
167  */
168  vector<string> key, val;
169  getPairs(rg, key, val);
170 
171  /*
172  * Loop over each of the pairs and process them
173  */
174  doublereal ord, stch;
175  string ph, sp;
176  map<string, size_t> speciesMap;
177  for (size_t n = 0; n < key.size(); n++) {
178  sp = key[n]; // sp is the string name for species
179  ph = "";
180  /*
181  * Search for the species in the kinetics object using the
182  * member function kineticsSpeciesIndex(). We will search
183  * for the species in all phases defined in the kinetics operator.
184  */
185  size_t isp = kin.kineticsSpeciesIndex(sp);
186  if (isp == npos) {
187  if (rules.skipUndeclaredSpecies) {
188  return false;

192  return false;
193  }
194  }
195 
196  /*
197  * For each reagent, we store the the species number, isp
198  * the stoichiometric coefficient, val[n], and the order
199  * species in the reaction rate expression. We assume mass
200  * action kinetics here, but will modify this below for
201  * specified species.
202  */
203  spnum.push_back(isp);
204  stch = fpValue(val[n]);
205  stoich.push_back(stch);
206  ord = doublereal(stch);
207  order.push_back(ord);

209  /*
210  * Needed to process reaction orders below.
211  */

216  * Check to see if reaction orders have been specified.

231  if (forder < 0.0) {
232  throw CanteraError("getReagents",
233  "reaction order must be non-negative");
234  }
235  // replace the stoichiometric coefficient
236  // stored above in 'order' with the specified
237  // reaction order
238  order[loc-1] = forder;
239  }
240  }

269 /**
270  * getStick() processes the XML element called Stick that specifies
271  * the sticking coefficient reaction. This routine will
272  * translate the sticking coefficient value into a "normal"
273  * rate constant for the surface reaction.
274  *
275  * Output
276  * -----------
277  * Output is the normal Arrhenius expressions for a surface
278  * reaction rate constant.
279  *
280  * A - units such that rate of rxn has kmol/m^2/s when
281  * A is multiplied by activity concentrations of
282  * reactants in the normal manner.
283  * n - unitless
284  * E - Units 1/Kelvin
285  */
286 static void getStick(const XML_Node& node, Kinetics& kin,
287  ReactionData& r, doublereal& A, doublereal& b, doublereal& E)
288 {
289  size_t nr = r.reactants.size();
290  size_t k, klocal, not_surf = 0;
291  size_t np = 0;
292  doublereal f = 1.0;
293  doublereal order;
294  /*
295  * species is the name of the special reactant whose surface
296  * flux rate will be calculated.
297  * isp = species # in the local phase
298  * ispKinetics = species # in the kinetics object
299  * ispPhaseIndex = phase # of the special species
300  */
301  string spname = node["species"];
302  ThermoPhase& th = kin.speciesPhase(spname);
303  size_t isp = th.speciesIndex(spname);
304  size_t ispKinetics = kin.kineticsSpeciesIndex(spname);
305  size_t ispPhaseIndex = kin.speciesPhaseIndex(ispKinetics);
306 
307  doublereal ispMW = th.molecularWeights()[isp];
308  doublereal sc;
309 
310  // loop over the reactants
311  for (size_t n = 0; n < nr; n++) {
312  k = r.reactants[n];
313  order = r.rorder[n]; // stoich coeff
314 
315  // get the phase species k belongs to
316  np = kin.speciesPhaseIndex(k);
317  const ThermoPhase& p = kin.thermo(np);
318 
319  // get the local index of species k in this phase
320  klocal = p.speciesIndex(kin.kineticsSpeciesName(k));
321 
322  // if it is a surface species, divide f by the standard
323  // concentration for this species, in order to convert
324  // from concentration units used in the law of mass action
325  // to coverages used in the sticking probability
326  // expression
327  if (p.eosType() == cSurf || p.eosType() == cEdge) {
328  sc = p.standardConcentration(klocal);
329  f /= pow(sc, order);
330  }
331  // Otherwise:
332  else {
333  // We only allow one species to be in the phase
334  // containing the special sticking coefficient
335  // species.
336  if (ispPhaseIndex == np) {
337  not_surf++;
338  }
339  // Other bulk phase species on the other side
340  // of ther interface are treated like surface
341  // species.
342  else {
343  sc = p.standardConcentration(klocal);
344  f /= pow(sc, order);
345  }
346  }
347  }
348  if (not_surf != 1) {
349  throw CanteraError("getStick",
350  "reaction probabilities can only be used in "
351  "reactions with exactly 1 gas/liquid species.");
352  }
353 
354  doublereal cbar = sqrt(8.0*GasConstant/(Pi*ispMW));
355  A = 0.25 * getFloat(node, "A", "toSI") * cbar * f;
356  b = getFloat(node, "b") + 0.5;
357  E = getFloat(node, "E", "actEnergy");
358  E /= GasConstant;
359 }

562 doublereal isDuplicateReaction(std::map<int, doublereal>& r1,
563  std::map<int, doublereal>& r2)
564 {
565  map<int, doublereal>::const_iterator b = r1.begin(), e = r1.end();
566  int k1 = b->first;
567  // check for duplicate written in the same direction
568  doublereal ratio = 0.0;
569  if (r1[k1] && r2[k1]) {
570  ratio = r2[k1]/r1[k1];
571  ++b;
572  bool different = false;
573  for (; b != e; ++b) {
574  k1 = b->first;
575  if (!r1[k1] || !r2[k1] || fabs(r2[k1]/r1[k1] - ratio) > 1.e-8) {
576  different = true;
577  break;
578  }
579  }
580  if (!different) {
581  return ratio;
582  }
583  }
584 
585  // check for duplicate written in the reverse direction
586  b = r1.begin();
587  k1 = b->first;
588  if (r1[k1] == 0.0 || r2[-k1] == 0.0) {
589  return 0.0;
590  }
591  ratio = r2[-k1]/r1[k1];
592  ++b;
593  for (; b != e; ++b) {
594  k1 = b->first;
595  if (!r1[k1] || !r2[-k1] || fabs(r2[-k1]/r1[k1] - ratio) > 1.e-8) {
596  return 0.0;
597  }
598  }
599  return ratio;
600 }

623  // Check to see if the reaction rate constant can be negative. It's an
624  // error if a negative rate constant is found and this is not set.
625  rules.allowNegativeA = (r.hasAttrib("negative_A")) ? 1 : 0;
626 
627  // Use the contents of the "equation" child element as the reaction's
628  // string representation. Post-process to convert "[" and "]" characters
629  // back into "<" and ">" which cannot easily be stored in an XML file. This
630  // reaction string is used only for display purposes. It is not parsed for
631  // the identities of reactants or products.
632  rdata.equation = (r.hasChild("equation")) ? r("equation") : "<no equation>";
633  static const char* delimiters[] = {" [=] ", " =] ", " = ", "[=]", "=]", "="};
634  static const char* replacements[] = {" <=> ", " => ", " = ", "<=>", "=>", "="};
635  for (size_t i = 0; i < 6; i++) {
636  size_t n = rdata.equation.find(delimiters[i]);
637  if (n != npos) {
638  size_t w = strlen(delimiters[i]);
639  rdata.reactantString = stripws(rdata.equation.substr(0, n));
640  rdata.productString = stripws(rdata.equation.substr(n+w, npos));
641  rdata.equation.replace(n, w, replacements[i]);
642  break;
643  }
644  }
645 
646  // get the reactants
647  bool ok = getReagents(r, kin, 1, default_phase, rdata.reactants,
648  rdata.rstoich, rdata.rorder, rules);
649 
650  // Get the products. We store the id of products in rdata.products
651  ok = ok && getReagents(r, kin, -1, default_phase, rdata.products,
652  rdata.pstoich, rdata.porder, rules);
653 
654  // if there was a problem getting either the reactants or the products,
655  // then abort.
656  if (!ok) {
657  return false;
658  }
659 
660  // check whether the reaction is specified to be
661  // reversible. Default is irreversible.
662  string isrev = r["reversible"];
663  rdata.reversible = (isrev == "yes" || isrev == "true");
664 
665  // If reaction orders are specified, then this reaction does not follow
666  // mass-action kinetics, and is not an elementary reaction. So check that
667  // it is not reversible, since computing the reverse rate from
668  // thermochemistry only works for elementary reactions. Set the type to
669  // global, so that kinetics managers will know to process the reaction
670  // orders.
671  if (r.hasChild("order")) {
672  if (rdata.reversible == true)
673  throw CanteraError("installReaction",
674  "reaction orders may only be given for "
675  "irreversible reactions");
676  rdata.global = true;
677  }
678 
679  // Some reactions can be elementary reactions but have fractional
680  // stoichiometries wrt to some products and reactants. An example of these
681  // are solid reactions involving phase transformations. Species with
682  // fractional stoichiometries must be from single-species phases with
683  // unity activities. For these reactions set the bool isReversibleWithFrac
684  // to true.
685  if (rdata.reversible == true) {
686  for (size_t i = 0; i < rdata.products.size(); i++) {
687  doublereal po = rdata.porder[i];
688  AssertTrace(po == rdata.pstoich[i]);
689  doublereal chk = po - 1.0 * int(po);
690  if (chk != 0.0) {
691  size_t k = rdata.products[i];
692  // Special case when k is a single species phase.
693  if (kin.speciesPhase(k).nSpecies() == 1) {
694  rdata.porder[i] = 0.0;
695  }
696 
697  rdata.isReversibleWithFrac = true;
698  }
699  }
700  for (size_t i = 0; i < rdata.reactants.size(); i++) {
701  doublereal ro = rdata.rorder[i];
702  AssertTrace(ro == rdata.rstoich[i]);
703  doublereal chk = ro - 1.0 * int(ro);
704  if (chk != 0.0) {
705  size_t k = rdata.reactants[i];
706  // Special case when k is a single species phase.
707  if (kin.speciesPhase(k).nSpecies() == 1) {
708  rdata.rorder[i] = 0.0;
709  }
710  rdata.isReversibleWithFrac = true;
711  }
712  }
713  }
714 
715  /*
716  * Search the reaction element for the attribute "type".
717  * If found, then branch on the type, to fill in appropriate
718  * fields in rdata.
719  */
720  rdata.reactionType = ELEMENTARY_RXN;
721  string typ = r["type"];
722  if (typ == "falloff") {
723  rdata.reactionType = FALLOFF_RXN;
724  rdata.falloffType = SIMPLE_FALLOFF;
725  } else if (typ == "chemAct") {
726  rdata.reactionType = CHEMACT_RXN;
727  rdata.falloffType = SIMPLE_FALLOFF;
728  } else if (typ == "threeBody") {
729  rdata.reactionType = THREE_BODY_RXN;
730  } else if (typ == "plog") {
731  rdata.reactionType = PLOG_RXN;
732  } else if (typ == "chebyshev") {
733  rdata.reactionType = CHEBYSHEV_RXN;
734  } else if (typ == "surface") {
735  rdata.reactionType = SURFACE_RXN;
736  } else if (typ == "edge") {
737  rdata.reactionType = EDGE_RXN;
738  } else if (typ != "") {
739  throw CanteraError("installReaction", "Unknown reaction type: " + typ);
740  }
741 
742  rdata.number = iRxn;
743  rdata.rxn_number = iRxn;
744 
745  // Read the rate coefficient data from the XML file. Trigger an
746  // exception for negative A unless specifically authorized.
747  getRateCoefficient(r.child("rateCoeff"), kin, rdata, rules);
748 
749  if (validate_rxn) {
750  // Look for undeclared duplicate reactions.
751  unsigned long int participants = 0;
752  for (size_t nn = 0; nn < rdata.reactants.size(); nn++) {
753  rdata.net_stoich[-1 - int(rdata.reactants[nn])] -= rdata.rstoich[nn];
754  participants += static_cast<unsigned long int>(rdata.reactants[nn]);
755  }
756  for (size_t nn = 0; nn < rdata.products.size(); nn++) {
757  rdata.net_stoich[int(rdata.products[nn])+1] += rdata.pstoich[nn];
758  participants += 1000000 * static_cast<unsigned long int>(rdata.products[nn]);
759  }
760 
761  vector<size_t>& related = m_participants[participants];
762  for (size_t mm = 0; mm < related.size(); mm++) {
763  ReactionData& other = *m_rdata[related[mm]];
764  if (rdata.reactants.size() != other.reactants.size()) {
765  continue; // different numbers of reactants
766  } else if (rdata.reactionType != other.reactionType) {
767  continue; // different reaction types
768  } else if (rdata.duplicate && other.duplicate) {
769  continue; // marked duplicates
770  }
771  doublereal c = isDuplicateReaction(rdata.net_stoich, other.net_stoich);
772  if (c == 0) {
773  continue; // stoichiometries differ (not by a multiple)
774  } else if (c < 0.0 && !rdata.reversible && !other.reversible) {
775  continue; // irreversible reactions in opposite directions
776  } else if (rdata.reactionType == FALLOFF_RXN ||
777  rdata.reactionType == THREE_BODY_RXN ||
778  rdata.reactionType == CHEMACT_RXN) {
779  bool thirdBodyOk = true;
780  for (size_t k = 0; k < kin.nTotalSpecies(); k++) {
781  if (rdata.efficiency(k) * other.efficiency(k) != 0.0) {
782  thirdBodyOk = false;
783  }
784  }
785  if (thirdBodyOk) {
786  continue; // No overlap in third body efficiencies
787  }
788  }
789  string msg = string("Undeclared duplicate reactions detected: \n")
790  +"Reaction "+int2str(other.number+1)+": "+other.equation
791  +"\nReaction "+int2str(iRxn+1)+": "+rdata.equation+"\n";
792  throw CanteraError("installReaction", msg);
793  }
794  m_participants[participants].push_back(m_rdata.size() - 1);
795 
796  // Check to see that the elements balance in the reaction.
797  // Throw an error if they don't
798  checkRxnElementBalance(kin, rdata);
799  }
800 
801  // Ok we have read everything in about the reaction. Add it to the
802  // kinetics object by calling the Kinetics member function addReaction()
803  kin.addReaction(rdata);
804  return true;
805 }

807 bool installReactionArrays(const XML_Node& p, Kinetics& kin,
808  std::string default_phase, bool check_for_duplicates)
809 {
810  rxninfo _rxns;
811 
812  vector<XML_Node*> rarrays;
813  int itot = 0;
814  /*
815  * Search the children of the phase element for the
816  * xml element named reactionArray. If we can't find it,
817  * then return signaling having not found any reactions.
818  * Apparently, we allow multiple reactionArray elements here
819  * Each one will be processed sequentially, with the
820  * end result being purely additive.
821  */
822  p.getChildren("reactionArray",rarrays);
823  if (rarrays.empty()) {
824  kin.finalize();
825  return false;
826  }
827  for (size_t n = 0; n < rarrays.size(); n++) {
828  /*
829  * Go get a reference to the current xml element,
830  * reactionArray. We will process this element now.
831  */
832  const XML_Node& rxns = *rarrays[n];
833  /*
834  * The reactionArray element has an attribute called,
835  * datasrc. The value of the attribute is the xml
836  * element comprising the top of the
837  * tree of reactions for the phase.
838  * Find this datasrc element starting with the root
839  * of the current xml node.
840  */
841  const XML_Node* rdata = get_XML_Node(rxns["datasrc"], &rxns.root());
842  /*
843  * If the reactionArray element has a child element named "skip", and
844  * if the attribute of skip called "species" has a value of "undeclared",
845  * we will set rxnrule.skipUndeclaredSpecies to 'true'. rxnrule is
846  * passed to the routine that parses each individual reaction so that
847  * the parser will skip all reactions containing an undefined species
848  * without throwing an error.
849  *
850  * Similarly, an attribute named "third_bodies" with the value of
851  * "undeclared" will skip undeclared third body efficiencies (while
852  * retaining the reaction and any other efficiencies).
853  */
854  ReactionRules rxnrule;
855  if (rxns.hasChild("skip")) {
856  const XML_Node& sk = rxns.child("skip");
857  string sskip = sk["species"];
858  if (sskip == "undeclared") {
859  rxnrule.skipUndeclaredSpecies = true;
860  }
861  if (sk["third_bodies"] == "undeclared") {
862  rxnrule.skipUndeclaredThirdBodies = true;
863  }
864  }
865  /*
866  * Search for child elements called include. We only include
867  * a reaction if it's tagged by one of the include fields.
868  * Or, we include all reactions if there are no include fields.
869  */
870  vector<XML_Node*> incl;
871  rxns.getChildren("include",incl);
872 
873  vector<XML_Node*> allrxns;
874  rdata->getChildren("reaction",allrxns);
875  // if no 'include' directive, then include all reactions
876  if (incl.empty()) {
877  for (size_t i = 0; i < allrxns.size(); i++) {
878  const XML_Node* r = allrxns[i];
879  if (r) {
880  if (_rxns.installReaction(itot, *r, kin,
881  default_phase, rxnrule, check_for_duplicates)) {
882  ++itot;
883  }
884  }
885  }
886  } else {
887  for (size_t nii = 0; nii < incl.size(); nii++) {
888  const XML_Node& ii = *incl[nii];
889  string imin = ii["min"];
890  string imax = ii["max"];
891 
892  string::size_type iwild = string::npos;
893  if (imax == imin) {
894  iwild = imin.find("*");
895  if (iwild != string::npos) {
896  imin = imin.substr(0,iwild);
897  imax = imin;
898  }
899  }
900 
901  for (size_t i = 0; i < allrxns.size(); i++) {
902  const XML_Node* r = allrxns[i];
903  string rxid;
904  if (r) {
905  rxid = (*r)["id"];
906  if (iwild != string::npos) {
907  rxid = rxid.substr(0,iwild);
908  }
909  /*
910  * To decide whether the reaction is included or not
911  * we do a lexical min max and operation. This
912  * sometimes has surprising results.
913  */
914  if ((rxid >= imin) && (rxid <= imax)) {
915  if (_rxns.installReaction(itot, *r, kin,
916  default_phase, rxnrule, check_for_duplicates)) {
917  ++itot;
918  }
919  }
920  }
921  }
922  }
923  }
924  }
925 
926  /*
927  * Finalize the installation of the kinetics, now that we know
928  * the true number of reactions in the mechanism, itot.
929  */
930  kin.finalize();
931 
932  return true;
933 }

935 bool importKinetics(const XML_Node& phase, std::vector<ThermoPhase*> th,
936  Kinetics* k)
937 {
938 
939  if (k == 0) {
940  return false;
941  }
942 
943  Kinetics& kin = *k;
944 
945  // This phase will be the owning phase for the kinetics operator
946  // For interfaces, it is the surface phase between two volumes.
947  // For homogeneous kinetics, it's the current volumetric phase.
948  string owning_phase = phase["id"];
949 
950  bool check_for_duplicates = false;
951  if (phase.parent()) {
952  if (phase.parent()->hasChild("validate")) {
953  const XML_Node& d = phase.parent()->child("validate");
954  if (d["reactions"] == "yes") {
955  check_for_duplicates = true;
956  }
957  }
958  }
959 
960  // if other phases are involved in the reaction mechanism,
961  // they must be listed in a 'phaseArray' child
962  // element. Homogeneous mechanisms do not need to include a
963  // phaseArray element.
964 
965  vector<string> phase_ids;
966  if (phase.hasChild("phaseArray")) {
967  const XML_Node& pa = phase.child("phaseArray");
968  getStringArray(pa, phase_ids);
969  }
970  phase_ids.push_back(owning_phase);
971 
972  int np = static_cast<int>(phase_ids.size());
973  int nt = static_cast<int>(th.size());
974 
975  // for each referenced phase, attempt to find its id among those
976  // phases specified.
977  bool phase_ok;
978 
979  string phase_id;
980  string msg = "";
981  for (int n = 0; n < np; n++) {
982  phase_id = phase_ids[n];
983  phase_ok = false;
984 
985  // loop over the supplied 'ThermoPhase' objects representing
986  // phases, to find an object with the same id.
987  for (int m = 0; m < nt; m++) {
988  if (th[m]->id() == phase_id) {
989  phase_ok = true;
990 
991  // if no phase with this id has been added to
992  //the kinetics manager yet, then add this one
993  if (kin.phaseIndex(phase_id) == npos) {
994  kin.addPhase(*th[m]);
995  }
996  }
997  msg += " "+th[m]->id();
998  }
999  if (!phase_ok) {
1000  throw CanteraError("importKinetics",
1001  "phase "+phase_id+" not found. Supplied phases are:"+msg);
1002  }
1003  }
1004 
1005  // allocates arrays, etc. Must be called after the phases have
1006  // been added to 'kin', so that the number of species in each
1007  // phase is known.
1008  kin.init();
1009 
1010  // Install the reactions.
1011  return installReactionArrays(phase, kin, owning_phase, check_for_duplicates);
1012 }
