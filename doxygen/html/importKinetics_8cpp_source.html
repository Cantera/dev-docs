27 bool installReactionArrays(const XML_Node& p, Kinetics& kin,
28  std::string default_phase, bool check_for_duplicates)
29 {
30  int itot = 0;
31  /*
32  * Search the children of the phase element for the
33  * XML element named reactionArray. If we can't find it,
34  * then return signaling having not found any reactions.
35  * Apparently, we allow multiple reactionArray elements here
36  * Each one will be processed sequentially, with the
37  * end result being purely additive.
38  */
39  vector<XML_Node*> rarrays = p.getChildren("reactionArray");
40  if (rarrays.empty()) {
41  kin.finalize();
42  return false;
43  }
44  for (size_t n = 0; n < rarrays.size(); n++) {
45  /*
46  * Go get a reference to the current XML element,
47  * reactionArray. We will process this element now.
48  */
49  const XML_Node& rxns = *rarrays[n];
50  /*
51  * The reactionArray element has an attribute called,
52  * datasrc. The value of the attribute is the XML
53  * element comprising the top of the
54  * tree of reactions for the phase.
55  * Find this datasrc element starting with the root
56  * of the current XML node.
57  */
58  const XML_Node* rdata = get_XML_Node(rxns["datasrc"], &rxns.root());
59  /*
60  * If the reactionArray element has a child element named "skip", and
61  * if the attribute of skip called "species" has a value of "undeclared",
62  * we will set rxnrule.skipUndeclaredSpecies to 'true'. rxnrule is
63  * passed to the routine that parses each individual reaction so that
64  * the parser will skip all reactions containing an undefined species
65  * without throwing an error.
66  *
67  * Similarly, an attribute named "third_bodies" with the value of
68  * "undeclared" will skip undeclared third body efficiencies (while
69  * retaining the reaction and any other efficiencies).
70  */
71  if (rxns.hasChild("skip")) {
72  const XML_Node& sk = rxns.child("skip");
73  if (sk["species"] == "undeclared") {
74  kin.skipUndeclaredSpecies(true);
75  }
76  if (sk["third_bodies"] == "undeclared") {
77  kin.skipUndeclaredThirdBodies(true);
78  }
79  }
80  /*
81  * Search for child elements called include. We only include
82  * a reaction if it's tagged by one of the include fields.
83  * Or, we include all reactions if there are no include fields.
84  */
85  vector<XML_Node*> incl = rxns.getChildren("include");
86  vector<XML_Node*> allrxns = rdata->getChildren("reaction");
87  // if no 'include' directive, then include all reactions
88  if (incl.empty()) {
89  for (size_t i = 0; i < allrxns.size(); i++) {
90  kin.addReaction(newReaction(*allrxns[i]));
91  ++itot;
92  }
93  } else {
94  for (size_t nii = 0; nii < incl.size(); nii++) {
95  const XML_Node& ii = *incl[nii];
96  string imin = ii["min"];
97  string imax = ii["max"];
98 
99  string::size_type iwild = string::npos;
100  if (imax == imin) {
101  iwild = imin.find("*");
102  if (iwild != string::npos) {
103  imin = imin.substr(0,iwild);
104  imax = imin;
105  }
106  }
107 
108  for (size_t i = 0; i < allrxns.size(); i++) {
109  const XML_Node* r = allrxns[i];
110  string rxid;
111  if (r) {
112  rxid = r->attrib("id");
113  if (iwild != string::npos) {
114  rxid = rxid.substr(0,iwild);
115  }
116  /*
117  * To decide whether the reaction is included or not
118  * we do a lexical min max and operation. This
119  * sometimes has surprising results.
120  */
121  if ((rxid >= imin) && (rxid <= imax)) {
122  kin.addReaction(newReaction(*r));
123  ++itot;
124  }
125  }
126  }
127  }
128  }
129  }
130 
131  if (check_for_duplicates) {
132  kin.checkDuplicates();
133  }
134  /*
135  * Finalize the installation of the kinetics, now that we know
136  * the true number of reactions in the mechanism, itot.
137  */
138  kin.finalize();
139 
140  return true;
141 }

143 bool importKinetics(const XML_Node& phase, std::vector<ThermoPhase*> th,
144  Kinetics* k)
145 {
146  if (k == 0) {
147  return false;
148  }
149 
150  // This phase will be the owning phase for the kinetics operator
151  // For interfaces, it is the surface phase between two volumes.
152  // For homogeneous kinetics, it's the current volumetric phase.
153  string owning_phase = phase["id"];
154 
155  bool check_for_duplicates = false;
156  if (phase.parent()) {
157  if (phase.parent()->hasChild("validate")) {
158  const XML_Node& d = phase.parent()->child("validate");
159  if (d["reactions"] == "yes") {
160  check_for_duplicates = true;
161  }
162  }
163  }
164 
165  // if other phases are involved in the reaction mechanism,
166  // they must be listed in a 'phaseArray' child
167  // element. Homogeneous mechanisms do not need to include a
168  // phaseArray element.
169 
170  vector<string> phase_ids;
171  if (phase.hasChild("phaseArray")) {
172  const XML_Node& pa = phase.child("phaseArray");
173  getStringArray(pa, phase_ids);
174  }
175  phase_ids.push_back(owning_phase);
176 
177  int np = static_cast<int>(phase_ids.size());
178  int nt = static_cast<int>(th.size());
179 
180  // for each referenced phase, attempt to find its id among those
181  // phases specified.
182  bool phase_ok;
183 
184  string phase_id;
185  string msg = "";
186  for (int n = 0; n < np; n++) {
187  phase_id = phase_ids[n];
188  phase_ok = false;
189 
190  // loop over the supplied 'ThermoPhase' objects representing
191  // phases, to find an object with the same id.
192  for (int m = 0; m < nt; m++) {
193  if (th[m]->id() == phase_id) {
194  phase_ok = true;
195 
196  // if no phase with this id has been added to
197  //the kinetics manager yet, then add this one
198  if (k->phaseIndex(phase_id) == npos) {
199  k->addPhase(*th[m]);
200  }
201  }
202  msg += " "+th[m]->id();
203  }
204  if (!phase_ok) {
205  throw CanteraError("importKinetics",
206  "phase "+phase_id+" not found. Supplied phases are:"+msg);
207  }
208  }
209 
210  // allocates arrays, etc. Must be called after the phases have
211  // been added to 'kin', so that the number of species in each
212  // phase is known.
213  k->init();
214 
215  // Install the reactions.
216  return installReactionArrays(phase, *k, owning_phase, check_for_duplicates);
217 }
