24 /**
25  * The Parker temperature correction to the rotational collision number.
26  *
27  * @param tr Reduced temperature \f$ \epsilon/kT \f$
28  * @param sqtr square root of tr.
29  */
30 inline doublereal Frot(doublereal tr, doublereal sqtr)
31 {
32  const doublereal c1 = 0.5*SqrtPi*Pi;
33  const doublereal c2 = 0.25*Pi*Pi + 2.0;
34  const doublereal c3 = SqrtPi*Pi;
35  return 1.0 + c1*sqtr + c2*tr + c3*sqtr*tr;
36 }

135 void MultiTransport::solveLMatrixEquation()
136 {
137  // if T has changed, update the temperature-dependent properties.
138  updateThermal_T();
139  update_C();
140  if (m_lmatrix_soln_ok) {
141  return;
142  }
143 
144  // Copy the mole fractions twice into the last two blocks of
145  // the right-hand-side vector m_b. The first block of m_b was
146  // set to zero when it was created, and is not modified so
147  // doesn't need to be reset to zero.
148  for (size_t k = 0; k < m_nsp; k++) {
149  m_b[k] = 0.0;
150  m_b[k + m_nsp] = m_molefracs[k];
151  m_b[k + 2*m_nsp] = m_molefracs[k];
152  }
153 
154  // Set the right-hand side vector to zero in the 3rd block for
155  // all species with no internal energy modes. The
156  // corresponding third-block rows and columns will be set to
157  // zero, except on the diagonal of L01,01, where they are set
158  // to 1.0. This has the effect of eliminating these equations
159  // from the system, since the equation becomes: m_a[2*m_nsp +
160  // k] = 0.0.
161 
162  // Note that this differs from the Chemkin procedure, where
163  // all *monatomic* species are excluded. Since monatomic
164  // radicals can have non-zero internal heat capacities due to
165  // electronic excitation, they should be retained.
166 
167  for (size_t k = 0; k < m_nsp; k++) {
168  if (!hasInternalModes(k)) {
169  m_b[2*m_nsp + k] = 0.0;
170  }
171  }
172 
173  // evaluate the submatrices of the L matrix
174  m_Lmatrix.resize(3*m_nsp, 3*m_nsp, 0.0);
175 
176  //! Evaluate the upper-left block of the L matrix.
177  eval_L0000(DATA_PTR(m_molefracs));
178  eval_L0010(DATA_PTR(m_molefracs));
179  eval_L0001();
180  eval_L1000();
181  eval_L1010(DATA_PTR(m_molefracs));
182  eval_L1001(DATA_PTR(m_molefracs));
183  eval_L0100();
184  eval_L0110();
185  eval_L0101(DATA_PTR(m_molefracs));
186 
187  // Solve it using GMRES or LU decomposition. The last solution
188  // in m_a should provide a good starting guess, so convergence
189  // should be fast.
190 
191  //if (m_gmres) {
192  // gmres(m_mgmres, 3*m_nsp, m_Lmatrix, m_b.begin(),
193  // m_a.begin(), m_eps_gmres);
194  // m_lmatrix_soln_ok = true;
195  // m_l0000_ok = true; // L matrix not modified by GMRES
196  //}
197  //else {
198  copy(m_b.begin(), m_b.end(), m_a.begin());
199  try {
200  solve(m_Lmatrix, DATA_PTR(m_a));
201  } catch (CanteraError& err) {
202  err.save();
203  //if (info != 0) {
204  throw CanteraError("MultiTransport::solveLMatrixEquation",
205  "error in solving L matrix.");
206  }
207  m_lmatrix_soln_ok = true;
208  m_molefracs_last = m_molefracs;
209  // L matrix is overwritten with LU decomposition

213 void MultiTransport::getSpeciesFluxes(size_t ndim, const doublereal* const grad_T,
214  size_t ldx, const doublereal* const grad_X,
215  size_t ldf, doublereal* const fluxes)
216 {
217  // update the binary diffusion coefficients if necessary
218  update_T();
219  updateDiff_T();
220 
221  // If any component of grad_T is non-zero, then get the
222  // thermal diffusion coefficients
223  bool addThermalDiffusion = false;
224  for (size_t i = 0; i < ndim; i++) {
225  if (grad_T[i] != 0.0) {
226  addThermalDiffusion = true;
227  }
228  }
229  if (addThermalDiffusion) {
230  getThermalDiffCoeffs(DATA_PTR(m_spwork));
231  }
232 
233  const doublereal* y = m_thermo->massFractions();
234  doublereal rho = m_thermo->density();
235 
236  for (size_t i = 0; i < m_nsp; i++) {
237  double sum = 0.0;
238  for (size_t j = 0; j < m_nsp; j++) {
239  m_aa(i,j) = m_molefracs[j]*m_molefracs[i]/m_bdiff(i,j);
240  sum += m_aa(i,j);
241  }
242  m_aa(i,i) -= sum;
243  }
244 
245  // enforce the condition \sum Y_k V_k = 0. This is done by replacing
246  // the flux equation with the largest gradx component in the first
247  // coordinate direction with the flux balance condition.
248  size_t jmax = 0;
249  doublereal gradmax = -1.0;
250  for (size_t j = 0; j < m_nsp; j++) {
251  if (fabs(grad_X[j]) > gradmax) {
252  gradmax = fabs(grad_X[j]);
253  jmax = j;
254  }
255  }
256 
257  // set the matrix elements in this row to the mass fractions,
258  // and set the entry in gradx to zero
259  for (size_t j = 0; j < m_nsp; j++) {
260  m_aa(jmax,j) = y[j];
261  }
262  vector_fp gsave(ndim), grx(ldx*m_nsp);
263  for (size_t n = 0; n < ldx*ndim; n++) {
264  grx[n] = grad_X[n];
265  }
266  //for (n = 0; n < ndim; n++) {
267  // gsave[n] = grad_X[jmax + n*ldx]; // save the input mole frac gradient
268  //grad_X[jmax + n*ldx] = 0.0;
269  // grx[jmax + n*ldx] = 0.0;
270  // }
271 
272  // copy grad_X to fluxes
273  const doublereal* gx;
274  for (size_t n = 0; n < ndim; n++) {
275  gx = grad_X + ldx*n;
276  copy(gx, gx + m_nsp, fluxes + ldf*n);
277  fluxes[jmax + n*ldf] = 0.0;
278  }
279 
280  // use LAPACK to solve the equations
281  int info = m_aa.factor();
282  if (info) {
283  throw CanteraError("MultiTransport::getSpeciesFluxes",
284  "Error factorizing matrix.");
285  }
286  info = m_aa.solve(fluxes, ndim, ldf);
287  if (info) {
288  throw CanteraError("MultiTransport::getSpeciesFluxes",
289  "Error solving linear system.");
290  }
291 
292  size_t offset;
293  doublereal pp = pressure_ig();
294 
295  // multiply diffusion velocities by rho * V to create
296  // mass fluxes, and restore the gradx elements that were
297  // modified
298  for (size_t n = 0; n < ndim; n++) {
299  offset = n*ldf;
300  for (size_t i = 0; i < m_nsp; i++) {
301  fluxes[i + offset] *= rho * y[i] / pp;
302  }
303  //grad_X[jmax + n*ldx] = gsave[n];
304  }
305 
306  // thermal diffusion
307  if (addThermalDiffusion) {
308  for (size_t n = 0; n < ndim; n++) {
309  offset = n*ldf;
310  doublereal grad_logt = grad_T[n]/m_temp;
311  for (size_t i = 0; i < m_nsp; i++) {
312  fluxes[i + offset] -= m_spwork[i]*grad_logt;
313  }
314  }
315  }

319  doublereal* fluxes)
320 {
321  double* x1 = DATA_PTR(m_spwork1);
322  double* x2 = DATA_PTR(m_spwork2);
323  double* x3 = DATA_PTR(m_spwork3);
324  size_t n, nsp = m_thermo->nSpecies();
325  m_thermo->restoreState(nsp+2, state1);
326  double p1 = m_thermo->pressure();
327  double t1 = state1[0];
328  m_thermo->getMoleFractions(x1);
329 
330  m_thermo->restoreState(nsp+2, state2);
331  double p2 = m_thermo->pressure();
332  double t2 = state2[0];
333  m_thermo->getMoleFractions(x2);
334 
335  double p = 0.5*(p1 + p2);
336  double t = 0.5*(state1[0] + state2[0]);
337 
338  for (n = 0; n < nsp; n++) {
339  x3[n] = 0.5*(x1[n] + x2[n]);
340  }
341  m_thermo->setState_TPX(t, p, x3);
342  m_thermo->getMoleFractions(DATA_PTR(m_molefracs));
343 
344  // update the binary diffusion coefficients if necessary

346  updateDiff_T();
347 
348  // If there is a temperature gradient, then get the
349  // thermal diffusion coefficients
350 
351  bool addThermalDiffusion = false;
352  if (state1[0] != state2[0]) {
353  addThermalDiffusion = true;
354  getThermalDiffCoeffs(DATA_PTR(m_spwork));
355  }
356 
357  const doublereal* y = m_thermo->massFractions();
358  doublereal rho = m_thermo->density();
359 
360  for (size_t i = 0; i < m_nsp; i++) {
361  doublereal sum = 0.0;
362  for (size_t j = 0; j < m_nsp; j++) {
363  m_aa(i,j) = m_molefracs[j]*m_molefracs[i]/m_bdiff(i,j);
364  sum += m_aa(i,j);
365  }
366  m_aa(i,i) -= sum;
367  }
368 
369  // enforce the condition \sum Y_k V_k = 0. This is done by
370  // replacing the flux equation with the largest gradx
371  // component with the flux balance condition.
372  size_t jmax = 0;
373  doublereal gradmax = -1.0;
374  for (size_t j = 0; j < m_nsp; j++) {
375  if (fabs(x2[j] - x1[j]) > gradmax) {
376  gradmax = fabs(x1[j] - x2[j]);
377  jmax = j;
378  }
379  }
380 
381  // set the matrix elements in this row to the mass fractions,
382  // and set the entry in gradx to zero
383 
384  for (size_t j = 0; j < m_nsp; j++) {
385  m_aa(jmax,j) = y[j];
386  fluxes[j] = x2[j] - x1[j];
387  }
388  fluxes[jmax] = 0.0;
389 
390  // Solve the equations
391  int info = m_aa.factor();
392  if (info) {
393  throw CanteraError("MultiTransport::getMassFluxes",
394  "Error in factorization. Info = "+int2str(info));
395  }
396  info = m_aa.solve(fluxes);
397  if (info) {
398  throw CanteraError("MultiTransport::getMassFluxes",
399  "Error in linear solve. Info = "+int2str(info));
400  }
401 
402  doublereal pp = pressure_ig();
403 
404  // multiply diffusion velocities by rho * Y_k to create
405  // mass fluxes, and divide by pressure
406  for (size_t i = 0; i < m_nsp; i++) {
407  fluxes[i] *= rho * y[i] / pp;
408  }
409 
410  // thermal diffusion
411  if (addThermalDiffusion) {
412  doublereal grad_logt = (t2 - t1)/m_temp;
413  for (size_t i = 0; i < m_nsp; i++) {
414  fluxes[i] -= m_spwork[i]*grad_logt;
415  }
416  }

502 void MultiTransport::updateThermal_T()
503 {
504  if (m_thermal_tlast == m_thermo->temperature()) {
505  return;
506  }
507  // we need species viscosities and binary diffusion coefficients
508  updateSpeciesViscosities();
509  updateDiff_T();
510 
511  // evaluate polynomial fits for A*, B*, C*
512  doublereal z;
513  int ipoly;
514  for (size_t i = 0; i < m_nsp; i++) {
515  for (size_t j = i; j < m_nsp; j++) {
516  z = m_logt - m_log_eps_k(i,j);
517  ipoly = m_poly[i][j];
518  if (m_mode == CK_Mode) {
519  m_om22(i,j) = poly6(z, DATA_PTR(m_om22_poly[ipoly]));
520  m_astar(i,j) = poly6(z, DATA_PTR(m_astar_poly[ipoly]));
521  m_bstar(i,j) = poly6(z, DATA_PTR(m_bstar_poly[ipoly]));
522  m_cstar(i,j) = poly6(z, DATA_PTR(m_cstar_poly[ipoly]));
523  } else {
524  m_om22(i,j) = poly8(z, DATA_PTR(m_om22_poly[ipoly]));
525  m_astar(i,j) = poly8(z, DATA_PTR(m_astar_poly[ipoly]));
526  m_bstar(i,j) = poly8(z, DATA_PTR(m_bstar_poly[ipoly]));
527  m_cstar(i,j) = poly8(z, DATA_PTR(m_cstar_poly[ipoly]));
528  }
529  m_om22(j,i) = m_om22(i,j);
530  m_astar(j,i) = m_astar(i,j);
531  m_bstar(j,i) = m_bstar(i,j);
532  m_cstar(j,i) = m_cstar(i,j);
533  }
534  }
535  m_abc_ok = true;
536 
537  // evaluate the temperature-dependent rotational relaxation rate
538  doublereal tr, sqtr;
539  for (size_t k = 0; k < m_nsp; k++) {
540  tr = m_eps[k]/ m_kbt;
541  sqtr = m_sqrt_eps_k[k] / m_sqrt_t;
542  m_rotrelax[k] = std::max(1.0,m_zrot[k]) * m_frot_298[k]/Frot(tr, sqtr);
543  }
544 
545  doublereal d;
546  doublereal c = 1.2*GasConstant*m_temp;
547  for (size_t k = 0; k < m_nsp; k++) {
548  d = c * m_visc[k] * m_astar(k,k)/m_mw[k];
549  m_bdiff(k,k) = d;
550  }
551 
552  // Calculate the internal heat capacities by subtracting off the translational contributions
553  /*
554  * HKM Exploratory comment:
555  * The translational component is 1.5
556  * The rotational component is 1.0 for a linear molecule and 1.5 for a nonlinear molecule
557  * and zero for a monatomic.
558  * Chemkin has traditionally subtracted 1.5 here (SAND86-8246).
559  * The original Dixon-Lewis paper subtracted 1.5 here.
560  */
561  const vector_fp& cp = ((IdealGasPhase*)m_thermo)->cp_R_ref();
562  for (size_t k = 0; k < m_nsp; k++) {
563  m_cinternal[k] = cp[k] - 2.5;
564  }
565 
566  // m_thermo->update_T(m_update_thermal_T);
567  m_thermal_tlast = m_thermo->temperature();
568 }

578 void MultiTransport::eval_L0000(const doublereal* const x)
579 {
580  doublereal prefactor = 16.0*m_temp/25.0;
581  doublereal sum;
582  for (size_t i = 0; i < m_nsp; i++) {
583  // subtract-off the k=i term to account for the first delta
584  // function in Eq. (12.121)
585 
586  sum = -x[i]/m_bdiff(i,i);
587  for (size_t k = 0; k < m_nsp; k++) {
588  sum += x[k]/m_bdiff(i,k);
589  }
590 
591  sum /= m_mw[i];
592  for (size_t j = 0; j != m_nsp; ++j) {
593  m_Lmatrix(i,j) = prefactor * x[j]
594  * (m_mw[j] * sum + x[i]/m_bdiff(i,j));
595  }
596  // diagonal term is zero
597  m_Lmatrix(i,i) = 0.0;
598  }
599 }
